59  // Two bounding boxes are considered to match if for every bound,
60  // the difference between the two boxes is smaller than a fraction (eps)
61  // of the largest dimension of the two boxes.
62  bool do_match(const Bbox_3& bb, double tol) const {
63  return (std::abs(xmax()-bb.xmax()) <= tol &&
64  std::abs(xmin()-bb.xmin()) <= tol &&
65  std::abs(ymax()-bb.ymax()) <= tol &&
66  std::abs(ymin()-bb.ymin()) <= tol &&
67  std::abs(zmax()-bb.zmax()) <= tol &&
