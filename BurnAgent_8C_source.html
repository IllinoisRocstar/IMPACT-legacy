93 void BurnAgent::init_module( double t, double dt) {
94  MAN_DEBUG(1, ("BurnAgent::init_module called t:%f.\n", t));
95 
96  Agent::init_module(t, dt);
97 
98  int initial_start = get_coupling()->initial_start();
99 
100  if (initial_start ||
101  COM_get_dataitem_handle_const( burnSurfIN+".bflag") <= 0)
102  {
103  COM_use_dataitem( burnSurfIN+".mesh", parentWin+".mesh", 0);
104  COM_use_dataitem( burnSurfIN+".bflag", parentWin+".bflag", 0);
105 // COM_window_init_done( burnSurfIN);
106  }
107 
108  COM_window_init_done( burnSurfIN);
109 
110  if (initial_start)
111  {
112  // Change burnVolIN to use parent windows' mesh without ghost
113  COM_use_dataitem( burnVolIN+".mesh", parentWin+".mesh", 0);
114  COM_window_init_done( burnVolIN);
115  }
116 
117  // Call initialization routine of physics module
118  // rocman.f90:INITIALIZE()
119  // call to INIT_WRAPPER() in rocburn_2D.f90
120  COM_call_function( init_handle, &t, &communicator, &ic_handle,
121  burnSurfIN.c_str(), burnVolIN.c_str(),
122  &obtain_attr_handle);
123 
124  tbl_flag = *(int *)option_data;
125  MAN_DEBUG(3, ("BurnAgent: tbl_flag = %d.\n", tbl_flag));
126 
127  // Delete input buffer windows
128  COM_delete_window( burnSurfIN);
129  COM_delete_window( burnVolIN);
130 }

132 void BurnAgent::create_buffer_all()
133 {
134  Agent::create_buffer_all();
135 
136  // Precondition: The window burnWin should have been defined and contain
137  // the following variables: pf_alp, qc_alp, qr_alp, rhos_alp, Tf_alp,
138  // rb, and Tflm. It should have uses the dataitems rb,
139  // pf, qc, qr, rhos_alp, Tf, and Tflm_alp from parent window.
140 
141  int with_qc = COM_get_dataitem_handle_const( surf_window+".qc_alp") > 0;
142  int with_qr = COM_get_dataitem_handle_const( surf_window+".qr_alp") > 0;
143  int with_Tf = COM_get_dataitem_handle_const( surf_window+".Tf_alp") > 0;
144  int with_Tv = COM_get_dataitem_handle_const( surf_window+".Tv_alp") > 0;
145  int with_dn = COM_get_dataitem_handle_const( surf_window+".dn_alp") > 0;
146  int with_rhos = COM_get_dataitem_handle_const( surf_window+".rhos_alp") > 0;
147 
148  MAN_DEBUG(3, ("with_qc=%d with_qr=%d with_Tf=%d with_rhos=%d\n with_Tv=%d\n with_dn=%d\n",
149  with_qc, with_qr, with_Tf,with_rhos,with_Tv,with_dn));
150 
151  COM_use_dataitem( iburn_all, parentWin+".Tflm_alp");
152  COM_clone_dataitem( iburn_all+".Tflm_old", surf_window+".Tflm");
153 
154  COM_use_dataitem( iburn_all, parentWin+".pf");
155  COM_use_dataitem( iburn_all, parentWin+".rhos");
156  if (with_qc) COM_use_dataitem( iburn_all, parentWin+".qc");
157  if (with_qr) COM_use_dataitem( iburn_all, parentWin+".qr");
158  if (with_Tf) COM_use_dataitem( iburn_all, parentWin+".Tf");
159  if (with_Tv) COM_use_dataitem( iburn_all, parentWin+".Tv");
160  if (with_dn) COM_use_dataitem( iburn_all, parentWin+".dn");
161 
162  COM_clone_dataitem( iburn_all+".rb_alp", surf_window+".rb");
163  COM_clone_dataitem( iburn_all+".rb_old", surf_window+".rb");
164  COM_clone_dataitem( iburn_all+".rb_grad", surf_window+".rb");
165 
166  // Create back-up data fields if predictor-corrector iteration is on
167  COM_clone_dataitem( iburn_all+".pf_old", surf_window+".pf_alp");
168  if ( with_qc ) {
169  COM_clone_dataitem( iburn_all+".qc_old", surf_window+".qc_alp");
170  COM_clone_dataitem( iburn_all+".qc_grad", surf_window+".qc_alp");
171  }
172  if ( with_qr) {
173  COM_clone_dataitem( iburn_all+".qr_old", surf_window+".qr_alp");
174  COM_clone_dataitem( iburn_all+".qr_grad", surf_window+".qr_alp");
175  }
176  if ( with_rhos)
177  COM_clone_dataitem( iburn_all+".rhos_old", surf_window+".rhos_alp");
178  if (with_Tf)
179  COM_clone_dataitem( iburn_all+".Tf_old", surf_window+".Tf_alp");
180  if (with_Tv)
181  COM_clone_dataitem( iburn_all+".Tv_old", surf_window+".Tv_alp");
182  if (with_dn)
183  COM_clone_dataitem( iburn_all+".dn_old", surf_window+".dn_alp");
184 
185  create_registered_window_dataitems( iburn_all);
186  COM_window_init_done( iburn_all);
187 
188  // no ghost
189  COM_new_window( iburn_ng);
190  COM_use_dataitem( iburn_ng+".all", surf_window+".all", 0);
191  COM_use_dataitem( iburn_ng, iburn_all+".data");
192  create_registered_window_dataitems( iburn_ng);
193  COM_window_init_done( iburn_ng);
194 
195  // Create windows for writing interface data
196  COM_new_window( burnBufOUT);
197  COM_use_dataitem( burnBufOUT, surf_window+".all");
198 
199  COM_use_dataitem( burnBufOUT, iburn_all+".rb_old");
200  COM_use_dataitem( burnBufOUT, iburn_all+".pf_old");
201  COM_use_dataitem( burnBufOUT, iburn_all+".Tflm_old");
202  if ( with_qc)
203  COM_use_dataitem( burnBufOUT, iburn_all+".qc_old");
204  if ( with_qr)
205  COM_use_dataitem( burnBufOUT, iburn_all+".qr_old");
206  if (with_Tf)
207  COM_use_dataitem( burnBufOUT, iburn_all+".Tf_old");
208  if (with_Tv)
209  COM_use_dataitem( burnBufOUT, iburn_all+".Tv_old");
210  if (with_dn)
211  COM_use_dataitem( burnBufOUT, iburn_all+".dn_old");
212  COM_window_init_done( burnBufOUT);
213 
214  // Create a window for backing up the internal data of Rocburn
215  // if predictor-corrector iteration is on.
216  int maxPredCorr = get_coupling()->get_max_ipc();
217  if ( maxPredCorr>1) {
218  // Create window for backing up Rocburn's internal data
219  COM_new_window( burnIntBak);
220  COM_use_dataitem( burnIntBak+".mesh", vol_window+".mesh");
221  COM_clone_dataitem( burnIntBak, vol_window+".data");
222 
223  COM_window_init_done( burnIntBak);
224 
225  pc_hdls[0][0] = COM_get_dataitem_handle( vol_window+".data");
226  pc_hdls[1][0] = COM_get_dataitem_handle( burnIntBak+".data");
227  pc_count = 1;
228  }
229 
230  if (!get_coupling()->initial_start()) read_restart_data();
231 }
