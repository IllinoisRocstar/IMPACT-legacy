59 DataItem *ComponentInterface::
60 new_dataitem( const std::string &aname, const char loc, const int type,
61  int ncomp, const std::string &unit) throw(COM_exception)
62 {
63  // Special handing for connectivity
64  if ( Connectivity::is_element_name( aname) ||
65  DataItem::is_digit(aname[0]))
66  throw COM_exception( COM_ERR_INVALID_DATAITEM_NAME,append_frame
67  (_name+"."+aname,ComponentInterface::new_dataitem));
68  if ( ncomp<=0)
69  throw COM_exception( COM_ERR_INVALID_DIMENSION,append_frame
70  (_name+"."+aname,ComponentInterface::new_dataitem));
71 
72  if ( loc != 'w' && loc != 'p' && loc != 'e' && loc != 'n' && loc != 'c')
73  throw COM_exception( COM_ERR_UNKNOWN_KEYWORD,append_frame
74  (_name+"."+aname,ComponentInterface::new_dataitem));
75 
76  if ( type<COM_MIN_TYPEID || type > COM_MAX_TYPEID)
77  throw COM_exception( COM_ERR_UNKNOWN_DATATYPE,append_frame
78  (_name+"."+aname,ComponentInterface::new_dataitem));
79 
80  // Redefine an dataitem if it pre-exists, but ncomp cannot be increased.
81  Attr_map::const_iterator it = _attr_map.find( aname);
82 
83  if ( it != _attr_map.end() && ncomp > it->second->size_of_components()) {
84  COM_assertion_msg( it->second->id()>=COM_NUM_KEYWORDS,
85  "Cannot increase size of keywords");
86  delete_dataitem( it->first);
87  it = _attr_map.end();
88  }
89 
90  int id = (it==_attr_map.end())?_last_id:it->second->id();
91 
92  // Insert the object into both the set and the map.
93  DataItem *a = ((Pane_friend&)_dummy).
94  new_dataitem( aname, id, loc, type, ncomp, unit);
95  _attr_map[aname] = a;
96 
97  // Propagete onto all existing panes.
98  for (Pane_map::iterator it=_pane_map.begin(); it!=_pane_map.end(); ++it) {
99  Pane_friend *pn = (Pane_friend*)it->second;
100  pn->new_dataitem( aname, id, loc, type, ncomp, unit);
101  }
102 
103  // Update last available id
104  if ( it == _attr_map.end()) _last_id = id+ncomp+(ncomp>1);
105 
106  if ( _status == STATUS_NOCHANGE) _status = STATUS_CHANGED;
107  return a;
108 }

110 void ComponentInterface::
111 delete_dataitem( const std::string &aname) throw(COM_exception)
112 {
113  Attr_map::iterator it = _attr_map.find( aname);
114  if ( it == _attr_map.end())
115  throw COM_exception( COM_ERR_DATAITEM_NOTEXIST,append_frame
116  (_name+"."+aname,ComponentInterface::delete_dataitem));
117 
118  DataItem *attr = it->second;
119  int id = attr->id(), ncomp=attr->size_of_components();
120 
121  if ( id < COM_NUM_KEYWORDS && id != COM_DATA)
122  throw COM_exception( COM_ERR_INVALID_DATAITEM_NAME,append_frame
123  (_name+"."+aname,ComponentInterface::delete_dataitem));
124 
125  // Remove the object from both the set
126  ((Pane_friend&)_dummy).delete_dataitem( id);
127 
128  // Remove from all panes.
129  for (Pane_map::iterator it=_pane_map.begin(); it!=_pane_map.end(); ++it) {
130  Pane_friend *pn = (Pane_friend*)it->second;
131  pn->delete_dataitem( id);
132  }
133 
134  // Update last id
135  if (id != COM_DATA) {
136  _attr_map.erase( it); // Remove from _attr_map
137  if ( _last_id == id + ncomp+(ncomp>1)) {
138  _last_id = id;
139  if ( _status==STATUS_NOCHANGE) _status = STATUS_CHANGED;
140  }
141  else
142  _status = STATUS_SHRUNK;
143  }
144  else if (_last_id>COM_NUM_KEYWORDS) {
145  _attr_map.clear();
146  for ( int i=0; i<COM_NUM_KEYWORDS; ++i) {
147  DataItem *a = _dummy.dataitem(i);
148  if ( !DataItem::is_digit(a->name()[0])) _attr_map[ a->name()] = a;
149  }
150 
151  _status = STATUS_CHANGED;
152  _last_id = COM_NUM_KEYWORDS;
153  }

308 DataItem *ComponentInterface::inherit( DataItem *from, const std::string &aname,
309  int mode, bool withghost,
310  const DataItem *cond, int val)
311  throw(COM_exception)
312 {
313  DataItem *a = ((Pane_friend&)_dummy).inherit(from, aname, mode, withghost);
314  if ( from->is_windowed()) return a;
315 
316  // Copy the panes
317  Pane_map &ps = from->window()->_pane_map;
318 
319  bool with_conn = from->id()==COM_CONN || from->id()==COM_MESH ||
320  from->id()==COM_PMESH || from->id()==COM_ALL;
321 
322  bool insert_pane = (mode != Pane::INHERIT_COPY) && with_conn;
323 
324  if ( with_conn && aname.size() && from->name() != aname)
325  throw COM_exception( COM_ERR_INCOMPATIBLE_DATAITEMS, append_frame
326  (from->fullname()+" and " +aname,
327  ComponentInterface::inherit));
328 
329  if ( cond && cond->window()!=from->window()) {
330  throw COM_exception( COM_ERR_INCOMPATIBLE_DATAITEMS, append_frame
331  (from->fullname()+" and "+cond->fullname(),
332  ComponentInterface::inherit));
333  }
334 
335  for ( Pane_map::iterator ppit = ps.begin(); ppit!=ps.end(); ++ppit) {
336  Pane *ppn=ppit->second;
337  const int *cnd = NULL;
338  if ( cond) {
339  const DataItem *cond_pn = ppn->dataitem( cond->id());
340  cnd = (const int*)cond_pn->pointer();
341  }
342 
343  // Inherit the pane if the condition was set to an variable (cnd!=NULL),
344  // or set to to pane ID (cnd==NULL && val!=0), or not set.
345  if ( !cond && (val==0 || val==ppn->id()) || cnd && *cnd == val) {
346  // Create new panes only when we are inheriting the mesh.
347  Pane_friend *pn;
348  if ( insert_pane)
349  pn = (Pane_friend*)( &pane( ppit->first, true));
350  else {
351  Pane_map::iterator it = _pane_map.find( ppit->first);
352  pn = ( it!=_pane_map.end())?(Pane_friend*)(it->second):NULL;
353  }
354 
355  if ( pn)
356  pn->inherit( ppn->dataitem( from->id()), aname, mode, withghost);
357  }
358  }
359  return a;

362 void ComponentInterface::init_done( bool pane_changed) throw(COM_exception) {
363  // Loop through the dataitems.
364  if ( _status == STATUS_SHRUNK) {
365  int max_id=0;
366  for (Attr_map::iterator it=_attr_map.begin(); it!=_attr_map.end(); ++it) {
367  int id = it->second->id(), ncomp=it->second->size_of_components();
368  max_id = std::max( max_id, id+ncomp+(ncomp>1));
369  }
370  if (max_id < _last_id) _last_id = max_id;
371  }
372 
373  int npanes = _pane_map.size();
374  std::vector< int> pane_ids; pane_ids.reserve(npanes);
375 
376  for (Pane_map::iterator it=_pane_map.begin(); it!=_pane_map.end(); ++it) {
377  it->second->init_done();
378  pane_ids.push_back( it->second->id());
379  }
380 
381  _status = STATUS_NOCHANGE;
382 
383  if ( !pane_changed) {
384  if ( npanes>int(_pane_map.size())) {
385  throw COM_exception( COM_ERR_INIT_DONE_PANEMAP, append_frame
386  (_name,ComponentInterface::init_done));
387  }
388  return;
389  }
390 
391  // communicate pane mapping
392  int flag; MPI_Initialized( &flag);
393  if ( _comm == MPI_COMM_NULL) flag = 0;
394 
395  // Compute proc_map
396  int nprocs;
397  if ( flag) MPI_Comm_size( _comm, &nprocs); else nprocs = 1;
398 
399  // Obtain the number of panes.
400  std::vector<int> npanes_all(nprocs);
401  if ( flag)
402  MPI_Allgather( &npanes, 1, MPI_INT, &npanes_all[0], 1, MPI_INT, _comm);
403  else
404  npanes_all[0] = npanes;
405 
406  std::vector<int> disps(nprocs+1);
407  disps[0]=0;
408  for ( int i=0; i<nprocs; ++i) disps[i+1] = disps[i]+npanes_all[i];
409 
410  std::vector<int> pane_ids_all( disps[nprocs]);
411  if ( flag)
412  MPI_Allgatherv( &pane_ids[0], npanes, MPI_INT, &pane_ids_all[0],
413  &npanes_all[0], &disps[0], MPI_INT, _comm);
414  else
415  pane_ids_all = pane_ids;
416 
417  // Build process map
418  _proc_map.clear();
419  for ( int p=0; p<nprocs; ++p) {
420  for ( int j=disps[p], jn=disps[p+1]; j<jn; ++j)
421  _proc_map[ pane_ids_all[j]] = p;
422  }
423 }
