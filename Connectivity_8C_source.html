78 Connectivity::Size Connectivity::size_of_elements() const
79 {
80  if ( !is_structured())
81  return size_of_items();
82  else {
83  COM_assertion_msg( _ngitems==0 || !_pane->ignore_ghost(),
84  "Only unstructured meshes supports inheritance without ghost");
85 
86  const int *sizes=pointer();
87  if ( sizes==NULL) return 0;
88  int n=0;
89  switch ( _size_info[TYPE_ID]) {
90  case ST1: n=sizes[0]-1; break;
91  case ST2: n=(sizes[0]-1)*(sizes[1]-1); break;
92  case ST3: n=(sizes[0]-1)*(sizes[1]-1)*(sizes[2]-1); break;
93  default: COM_assertion(false); // Should never reach here.
94  }
95  return std::max(0,n);
96  }
97 }

216 const int* Connectivity::get_size_info( const std::string &aname)

231  if ( aname[idx]=='6')

233  break;
234  case 'q':

236  if ( aname[idx]=='4')

242  break;

248  { COM_assertion( _sizes[TET10][TYPE_ID]==TET10); return _sizes[TET10]; }
249  break;

270  return _sizes[PRISM6];
271  }
272 
273  if ( aname[idx]=='1') {

275  COM_assertion( _sizes[PYRIMID14][TYPE_ID]==PYRIMID14);
276  return _sizes[PYRIMID14];
277  }
278 
279  if ( aname[idx+1] == '5') {

281  return _sizes[PRISM15];

284  if ( aname[idx+1] == '8') {

286  return _sizes[PRISM18];
287  }
288  }
289  break;

292  if ( aname[idx]=='6') {
