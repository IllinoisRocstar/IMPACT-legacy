222 double Coupling::run( double t, double dt, int pred, double zoom)
223 {
224  iPredCorr = pred;
225 
226  // Change dt to the maximum time setp allowed.
227  for ( int i=0, n=agents.size(); i<n; ++i) {
228  dt = std::min( dt, agents[i]->max_timestep(t, dt));
229  }
230 
231  MAN_DEBUG(1, ("[%d] Rocman: Coupling::run with t: %f dt: %f.\n", comm_rank, t, dt));
232 
233  scheduler.run_actions( t, dt);
234 
235  // Return the new time as t + dt
236  if (zoom > 0.0)
237  return t + dt*zoom;
238  else
239  return t + dt;
240 }

314  }
315  else {

492  case 2:
493  //

508  break;
509  case 3:

545  default:
