230 void DataItem::
231 copy_array( void *buf, int strd, int n,
232  int offset, int direction) throw (COM_exception)
233 {
234  if (direction==COPY_IN) {
235  if ( is_const())
236  throw COM_exception( COM_ERR_DATAITEM_CONST,
237  append_frame( fullname(), DataItem::copy_array));
238 
239  if ( _status==STATUS_NOT_INITIALIZED && _parent)
240  { _cap = root()->_cap; _strd = root()->_strd; _nbytes_strd=root()->_nbytes_strd; }
241  }
242 
243  int nitems=size_of_items(), ncomp=size_of_components();
244 
245  if ( n ==0) { if ( direction==COPY_IN) return; else n = nitems; }
246  if ( strd==0) strd=ncomp;
247 
248  if ( !buf && n)
249  throw COM_exception( COM_ERR_NULL_POINTER,
250  append_frame( fullname(), DataItem::copy_array));
251 
252  // strd must be either 1 or at least as large as _strd
253  if ( strd!=1 && strd<ncomp)
254  throw COM_exception( COM_ERR_INVALID_STRIDE,
255  append_frame( fullname(), DataItem::copy_array));
256 
257  if ( offset<0 || n+offset>_cap)
258  throw COM_exception( COM_ERR_INVALID_SIZE,
259  append_frame( fullname(), DataItem::copy_array));
260 
261  int basesize = get_sizeof( data_type());
262 
263  char *ptr0 = (char*)pointer();
264  if (offset) ptr0 += offset*ncomp*basesize;
265 
266  if ( _strd == strd && (_strd == ncomp || n==_cap && strd==1) ) {
267  // two arrays have the same layout
268  if ( direction == COPY_IN)
269  std::memcpy( ptr0, buf, n*ncomp*basesize);
270  else
271  std::memcpy( buf, ptr0, n*ncomp*basesize);
272  }
273  else if ( _strd>1 && strd>1) {
274  // the components are stored contiguously in both arrays
275  char *p_buf=(char*)buf;
276  char *p_att = ptr0;
277  int strd_att_in_bytes = _nbytes_strd;
278  int strd_buf_in_bytes = strd*basesize;
279  int vecsize = ncomp*basesize;
280 
281  for ( int i=0, ni=std::min(n, nitems); i<ni; ++i) {
282  if ( direction == COPY_IN)
283  std::memcpy( p_att, p_buf, vecsize);
284  else
285  std::memcpy( p_buf, p_att, vecsize);
286  p_buf += strd_buf_in_bytes;
287  p_att += strd_att_in_bytes;
288  }
289  }
290  else {
291  // layouts of two arrays are very different from each
292  char *p_buf=(char*)buf;
293  char *p_att = ptr0;
294 
295  int strd_att_in_bytes = _nbytes_strd;
296  int strd_buf_in_bytes = strd*basesize;
297  int step_att_in_bytes= (_strd==1?_cap:1)*basesize;
298  int step_buf_in_bytes = (strd==1?n:1)*basesize;
299  for ( int i=0, ni=std::min(n, nitems); i<ni; ++i) {
300  for ( int j=0, offset_buf=0, offset_att=0; j<ncomp; ++j) {
301  if ( direction == COPY_IN)
302  std::memcpy( p_att+offset_att, p_buf+offset_buf, basesize);
303  else
304  std::memcpy( p_buf+offset_buf, p_att+offset_att, basesize);
305  offset_buf += step_buf_in_bytes;
306  offset_att += step_att_in_bytes;
307  }
308  p_buf += strd_buf_in_bytes;
309  p_att += strd_att_in_bytes;
310  }
311  }

327 void *DataItem::allocate( int strd, int cap, bool force) throw( COM_exception)
328 {
329  int nitems=size_of_items(), ncomp=size_of_components();
330 
331  if ( strd!=1 && strd<ncomp)
332  throw COM_exception( COM_ERR_INVALID_STRIDE,
333  append_frame( fullname(), DataItem::allocate));
334  if ( cap<nitems)
335  throw COM_exception( COM_ERR_INVALID_CAPACITY,
336  append_frame( fullname(), DataItem::allocate));
337 
338  if ( !force && (_parent || _status==STATUS_SET || _status==STATUS_SET_CONST))
339  throw COM_exception( COM_ERR_RESIZE,
340  append_frame( fullname(), DataItem::allocate));
341 
342  try {
343  int type = data_type();
344  // Go ahead to allocate for the dataitem if it was not initialized
345  // and not inherited or it is forced to overwrite previously set address.
346  int nold, old_cap;
347  if ( _status == STATUS_NOT_INITIALIZED || force)
348  { nold = -1; old_cap = 0; }
349  else {
350  old_cap = _cap;
351  nold = old_cap*get_sizeof(type, std::max(_strd,ncomp));
352  }
353 
354  int nnew = cap*get_sizeof(type, std::max(strd,ncomp));
355 
356  // if the capacity is not big enough or the stride is changed.
357  if ( nold < nnew || strd != _strd) {
358  // Deallocate the old array and copy values to the new one
359  char *old_ptr = (char*)_ptr;
360  int old_strd = _strd;
361 
362  if ( nnew) {
363  _ptr = new char[nnew];
364  std::fill_n( (char*)_ptr, nnew, 0);
365  if (_ptr==NULL)
366  throw COM_exception( COM_ERR_OUT_OF_MEMORY,
367  append_frame( fullname(), DataItem::allocate));
368  }
369  else _ptr = NULL;
370  _cap = cap; _strd = strd; _nbytes_strd = get_sizeof( data_type(), strd);
371 
372  // Copy data from old array to the new.
373  if ( old_cap && _ptr) {
374  if ( _ncomp ==1 || _id<0) // Copy for connectivity and for scalars
375  copy_array( old_ptr, old_strd, std::min(old_cap,_cap));
376  else { // loop through individual components
377  for ( int i=1; i<=ncomp; ++i) {
378  DataItem *ai = this+i;
379  char *old_ptr_i = (char*)ai->_ptr; old_strd=ai->_strd;
380  old_cap = ai->_cap;
381  ai->set_pointer( _ptr, _strd, _cap, i-1, false);
382  ai->copy_array( old_ptr_i, old_strd, std::min(old_cap,_cap));
383 
384  // Delete the individual components
385  if ( ai->_status == STATUS_ALLOCATED && old_ptr_i)
386  delete [] old_ptr_i;
387  }
388  }
389  }
390  else {
391  if ( _ncomp >1 && _id>=0) for ( int i=1; i<=ncomp; ++i) {
392  this[i].set_pointer( _ptr, _strd, _cap, i-1, false);
393  }
394  }
395 
396  // Delete the old array for all components
397  if ( _status == STATUS_ALLOCATED && old_ptr) delete [] old_ptr;
398 
399  _status = STATUS_ALLOCATED;
400  if ( _parent) _parent=NULL; // Break inheritance.
401  }
402  }
403  CATCHEXP_APPEND(DataItem::allocate) CATCHBADALLOC_APPEND(DataItem::allocate);
404  return _ptr;
405 }
