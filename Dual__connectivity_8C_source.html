33 void Pane_dual_connectivity::construct_connectivity_str_2() {
34  COM_assertion_msg( _with_ghost || _pane.size_of_ghost_layers()==0,
35  "Not yet implemented for structured mesh with ghosts");
36 
37  int n=_with_ghost ? _pane.size_of_nodes() : _pane.size_of_real_nodes();
38  { // First, count the number of incident elements of each .
39  std::vector<char> nielems(n);
40  std::fill_n( nielems.begin(), n, 4);
41 
42  int ni=_pane.size_i(), nj=_pane.size_j();
43 
44  std::fill_n( nielems.begin(), ni, 2);
45  std::fill_n( nielems.begin()+ni*(nj-1), ni, 2);
46 
47  for ( int i=ni; i<ni*(nj-1); i+=ni)
48  nielems[i] = nielems[i+ni-1] = 2;
49 
50  nielems[0] = nielems[ni-1] = 1;
51  nielems[ni*nj-ni] = nielems[ni*nj-1] = 1;
52 
53  // Now construct a vector for storing the offsets
54  _offsets.clear(); _offsets.resize(n+1, 0);
55 
56  for ( int i=1; i<=n; ++i) _offsets[i] = _offsets[i-1]+nielems[i-1];
57  _eids.resize( _offsets[n]);
58  }
59 
60  { // Fill in the connectivity table
61  const int nn = 4;
62  Element_node_enumerator_str_2 ene( &_pane, 1);
63  for ( int i=1,size=_pane.size_of_elements(); i<=size; ++i, ene.next())
64  for ( int k=0; k<nn; ++k)
65  _eids[ _offsets[ene[k]-1]++] = i;
66 
67  // Recover _offsets
68  for ( int i=n; i>0; --i) _offsets[i] = _offsets[i-1];
69  _offsets[0] = 0;
70  }
71 }
