91 Element_node_enumerator::vertex( int lvid, bool level) const {
92  if ( _pane->is_structured()) { // Structured mesh
93  if ( lvid == _base) return 0;
94  if ( lvid == _base+1) return 1;
95  const int nr = _pane->size_i();
96  if ( lvid == int(_base+nr)) return 3;
97  if ( lvid == int(_base+nr+1)) return 2;
98 
99  // There is a branch cut in the pane. Resort to more expensive step.
100  // Check which boundary lvid is on.
101  if ( !level) {
102  COM_assertion( nr>2 && _pane->size_j()>2);
103  const int c1 = (_base-1)/nr, r1 = (_base-1)%nr;
104  const int c2 = (lvid-1)/nr, r2 = (lvid-1)%nr;
105 
106  if ( std::abs( r1-r2)<=1) {
107  const int nc = _pane->size_j();
108  if ( c2 == 0) return vertex( (nc-1)*nr+r2+1, true);
109  if ( c2 == nc-1) return vertex( r2+1, true);
110  }
111  else {
112  COM_assertion( std::abs( c1-c2)<=1);
113  if ( r2 == 0) return vertex( (c2+1)*nr, true);
114  if ( r2 == nr-1) return vertex( c2*nr+1, true);
115  }
116  }
117  COM_assertion( false); return -1;
118  }
119  else { // Unstructured
120  const int *p=_start;
121  for ( int i=0, size=_conn->size_of_nodes_pe(); i<size; ++i,++p) {
122  if ( lvid == *p) return i;
123  }
124 
125  return -1; // The node is not in the table
126  }

136 Facet_node_enumerator::
137 Facet_node_enumerator( const Element_node_enumerator *ene, int k)
138  : _ene(ene), _k(k), _ne(0) {
139  if ( ene->dimension() == 2) {
140  const_cast<int&>(_ne) = 1;
141  _fn_list = NULL;
142  }
143  else {
144  switch ( ene->size_of_nodes()) {
145  case 4: // Tetrahedra
146  case 10: {
147  const_cast<int&>(_ne) = 3;
148  _fn_list = _face_node_lists_tets[_k];
149  break;
150  }
151  case 5: // Pyramids
152  case 14: {
153  const_cast<int&>(_ne) = (_k==0)? 4 : 3;
154  _fn_list = _face_node_lists_pyra[_k];
155  break;
156  }
157  case 6: // Prisms (Pentahedra)
158  case 15:
159  case 18: {
160  const_cast<int&>(_ne) = (_k<3)? 4 : 3;
161  _fn_list = _face_node_lists_pris[_k];
162  break;
163  }
164  case 8: // Hexahedra
165  case 20:
166  case 27: {
167  const_cast<int&>(_ne) = 4;
168  _fn_list = _face_node_lists_hexa[_k];
169  break;
170  }
171  default: COM_assertion_msg(false, "Not supported yet.");
172  }
173  }
