221 template < class Field, class Value>
222 void Generic_element_2::interpolate( const Field &f,
223  const Nat_coor &nc,
224  Value *v) const throw(int) {
225  const Real xi = nc[0], eta = nc[1];
226 
227  if ( eta == 0) // Interpolate on the edge.
228  { interpolate( f, xi, v); return; }
229 
230  *v = f[0];
231  switch ( _nnodes) {
232  case 3:
233  // *v = f[0] + xi*(f[1]-f[0]) + eta*(f[2]-f[0]);
234  *v += ((f[1]-f[0])*=xi) += ((f[2]-f[0])*=eta);
235  return;
236  case 4: {
237  // *v = f[0] + xi * (1.-eta) * (f[1]-f[0])
238  // + eta * ( f[3]-f[0] + xi *(f[2]-f[3]));
239  *v += ((f[1]-f[0]) *= (xi * (1.-eta)))
240  += ((f[3]-f[0]) *= eta)
241  += ((f[2]-f[3]) *= xi*eta);
242  return;
243  }
244  case 6: {
245  const Real zeta=1.-xi-eta;
246 
247  // *v = f[0] + xi*( (2.*xi-1.)*(f[1]-f[0]) + 4.*zeta*(f[3]-f[0])) +
248  // eta * ( (2.*eta-1.)*(f[2]-f[0]) + 4.*zeta*(f[5]-f[0])) +
249  // 4.*xi*eta*(f[4]-f[0]);
250  *v += ((f[1]-f[0]) *= xi * (2.*xi-1.))
251  += ((f[3]-f[0]) *= 4.* xi * zeta)
252  += ((f[2]-f[0]) *= eta * (2.*eta-1.))
253  += ((f[5]-f[0]) *= 4. * eta * zeta)
254  += ((f[4]-f[0]) *= 4.*xi*eta);
255  return;
256  }
257  case 8: {
258  const Real xi_minus = 1. - xi;
259  const Real eta_minus = 1. - eta;
260 
261  // *v = f[0] + xi * eta_minus * (f[1]-f[0]) +
262  // eta * ( f[3]-f[0] + xi *(f[2]-f[3])) -
263  // 2.*xi*xi_minus*eta_minus*( (f[0]-f[4])+(f[1]-f[4])) -
264  // 2.*xi*xi_minus*eta*( (f[2]-f[6])+(f[3]-f[6])) -
265  // 2.*xi*eta*eta_minus*( (f[1]-f[5])+(f[2]-f[5])) -
266  // 2.*xi_minus*eta*eta_minus*( (f[0]-f[7])+(f[3]-f[7]));
267 
268  *v += ((f[1]-f[0]) *= eta * (2.*eta-1.))
269  += ((f[3]-f[0]) *= eta)
270  += ((f[2]-f[3]) *= xi * eta)
271  -= ((((f[0]-f[4])+=(f[1]-f[4])) *= 2.*xi*xi_minus*eta_minus)
272  += (((f[2]-f[6])+=(f[3]-f[6])) *= 2.*xi*xi_minus*eta)
273  += (((f[1]-f[5])+=(f[2]-f[5])) *= 2.*xi*eta*eta_minus)
274  += (((f[0]-f[7])+=(f[3]-f[7])) *= 2.*xi_minus*eta*eta_minus));
275  return;
276  }
277  default:
278  throw(-1);
279  }

398 template < class Field>
399 void Generic_element_2::Jacobian( const Field &f,
400  const Nat_coor &nc,
401  Vector_3 J[2]) const {
402  switch (_nnodes) {
403  case 3:
404  J[0] = f[1]-f[0]; J[1] = f[2]-f[0];
405  return;
406  case 4: {
407  const Real xi = nc[0], xi_minus = 1. - xi;
408  const Real eta = nc[1], eta_minus = 1. - eta;
409 
410  // J[0] = eta_minus * ( f[1] - f[0]) + eta * ( f[2] - f[3]);
411  J[0] = ((f[1]-f[0]) *= eta_minus) += (( f[2]-f[3]) *= eta );
412  // J[1] = xi_minus * ( f[3] - f[0]) + xi * ( f[2] - f[1]);
413  J[1] = ((f[3]-f[0]) *= xi_minus) += (( f[2]-f[1]) *= xi);
414  return;
415  }
416  case 6: {
417  const Real xi=nc[0], eta=nc[1], zeta=1.-xi-eta;
418 
419  // J[0] = (4.*xi -1.)*(f[1]-f[0]) +
420  // (4.*zeta-4.*xi)*(f[3]-f[0]) + 4.*eta*(f[4] -f[5]);
421  J[0] = ((f[1]-f[0]) *= 4.*xi -1.)
422  += ((f[3]-f[0]) *= 4.*zeta-4*xi)
423  += ((f[4]-f[5]) *= 4.*eta);
424  // J[1] = (4.*eta -1.)*(f[2]-f[0]) +
425  // (4.*zeta-4.*eta)*(f[5]-f[0]) + 4.*xi*(f[4] -f[3]);
426  J[1] = ((f[2]-f[0]) *= 4.*eta -1.)
427  += ((f[5]-f[0]) *= 4*zeta-4*eta)
428  += ((f[4]-f[3]) *= 4.*xi);
429  return;
430  }
431  case 8: {
432  const Real xi = nc[0], xi_minus = 1. - xi;
433  const Real eta = nc[1], eta_minus = 1. - eta;
434 
435  // J[0] = eta_minus * ( f[1] - f[0]) + eta * ( f[2] - f[3]) -
436  // 2.*eta_minus*(xi_minus-xi)*( (f[0]-f[4])+(f[1]-f[4])) -
437  // 2.*eta*(xi_minus-xi)*( (f[2]-f[6])+(f[3]-f[6])) -
438  // 2.*eta*eta_minus*( (f[1]-f[5])+(f[2]-f[5])-(f[0]-f[7])-(f[3]-f[7]));
439  J[0] = ((f[1]-f[0]) *= eta_minus) += ((f[2]-f[3]) *= eta)
440  -= ((((f[0]-f[4])+=(f[1]-f[4])) *= 2.*eta_minus*(xi_minus-xi))
441  += (((f[2]-f[6])+=(f[3]-f[6])) *= 2.*eta*(xi_minus-xi))
442  += (((f[1]-f[5])+=(f[2]-f[5])-=((f[0]-f[7])+=(f[3]-f[7])))
443  *= 2.*eta*eta_minus));
444 
445  // J[1] = xi_minus * ( f[3] - f[0]) + xi * ( f[2] - f[1]) -
446  // 2.*xi*(eta_minus-eta)*( (f[1]-f[5])+(f[2]-f[5]))-
447  // 2.*xi_minus*(eta_minus-eta)*( (f[0]-f[7])+(f[3]-f[7])) -
448  // 2.*xi*xi_minus*( (f[2]-f[6])-(f[3]-f[6])-(f[0]-f[4])+(f[1]-f[4]));
449  ((J[1] = f[3]-f[0]) *= xi_minus) += ((f[2]-f[1]) *= xi)
450  -= ((((f[1]-f[5])+=(f[2]-f[5])) *= 2.*xi*(eta_minus-eta))
451  += (((f[0]-f[7])+=(f[3]-f[7])) *= 2.*xi_minus*(eta_minus-eta))
452  += (((f[2]-f[6])+=(f[1]-f[4])-=((f[3]-f[6])+=(f[0]-f[4])))
453  *= 2.*xi*xi_minus));
454  return;
455  }
456  default:
457  abort(); // Should never reach here
458  }

509 // Evaluates the determinant of the Jacobian (dx/dxi,dx/deta), where
510 // x is (1-alpha)*f1+alpha*f2. Used by Rocface.
511 template < class Field>
512 Generic_element_2::Real
513 Generic_element_2::Jacobian_det( const Field &f1, const Field &f2,
514  const Real alpha, const Nat_coor &nc) const {
515  Vector_3 J1[2],J2[2];
516  if ( alpha!=1.) {
517  Jacobian( f1, nc, J1);
518  }
519 
520  if ( alpha!=0.) {
521  Jacobian( f2, nc, J2);
522  if ( alpha!=1) {
523  J2[0] = (1-alpha)*J1[0]+alpha*J2[0];
524  J2[1] = (1-alpha)*J1[1]+alpha*J2[1];
525  }
526  }
527  else {
528  J2[0] = J1[0]; J2[1] = J1[1];
529  }
530 
531  return std::sqrt( Vector_3::cross_product( J2[0], J2[1]).squared_norm());
532 }
