188 class INode : private In_place_list_base<INode,2> {
189 public:

191  typedef In_place_list_base<INode,2> Base;
192 
193  using Base::next_link;
194  using Base::prev_link;
195 protected:
196  Host_face bp; // Blue parent object
197  Host_face gp; // Green parent object
198  Point_2S b_nc; // The natrual coordinate in the original blue mesh
199  Point_2S g_nc; // The natrual coordinate in the original green mesh
200  int _id;
201 
202 public:
203  INode() : _id(-1) {}
204  ~INode() { RFC_assertion_code(bp=gp=Host_face()); }
205 
206  bool operator==( const INode &x)
207  { return ( b_nc == x.b_nc && g_nc == x.g_nc && bp == x.bp && gp == x.gp); }
208 
209  const Point_2S &nat_coor( const int color) const {
210  if ( color == BLUE) return b_nc;
211  else { RFC_assertion(color == GREEN); return g_nc; }
212  }
213 
214  void nat_coor( const int color, Point_2 &p) const {
215  if ( color == BLUE) { p[0]=b_nc[0]; p[1]=b_nc[1]; }
216  else { RFC_assertion(color == GREEN); p[0]=g_nc[0]; p[1]=g_nc[1]; }
217  }
218 
219  void set_parent( HEdge h, const Point_2& p, int color) {
220  if ( color == BLUE) {
221  bp = Host_face( h, pt(p)); b_nc[0]=p[0]; b_nc[1]=p[1];
222  }
223  else {
224  RFC_assertion( color == GREEN);
225  gp = Host_face( h, pt(p)); g_nc[0]=p[0]; g_nc[1]=p[1];
226  }
227  }
228 
229  int id() const { return _id; }
230  void set_id( int i) { _id = i; }
231 
232  Parent_type parent_type( const int color) const {
233  if ( color == BLUE) { return bp.parent_type(); }
234  else { RFC_assertion( color == GREEN); return gp.parent_type(); }
235  }
236  Parent_type blue_parent_type() const { return bp.parent_type(); }
237  Parent_type green_parent_type() const { return gp.parent_type(); }
238 
239  HEdge halfedge( const int color) const {
240  if ( color == BLUE) { return bp.halfedge(); }
241  else { RFC_assertion( color == GREEN); return gp.halfedge(); }
242  }
243 
244 private:
245  Parent_type pt( const Point_2 &p) {
246  if ( p[1] != 0) return PARENT_FACE;
247  else if ( p[0] != 0) return PARENT_EDGE;
248  else return PARENT_VERTEX;
249  }
250 };
