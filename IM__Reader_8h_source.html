42  int read_winmesh( const char *fname, const std::string &wname,
43  bool del=true) {
44 
45  const char *lastdot=std::strrchr( fname, '.');
46  if ( lastdot && (std::strcmp( lastdot, ".hdf")==0 ||
47  std::strcmp( lastdot, ".cgns")==0 ||
48  std::strcmp( lastdot, ".txt")==0)) {
49  // Read in HDF format
50  COM_LOAD_MODULE_STATIC_DYNAMIC( SimIN, "IN");
51 
52  int IN_read;
53  if ( std::strcmp( lastdot, ".hdf") == 0 ||
54  std::strcmp( lastdot, ".cgns") == 0)
55  IN_read = COM_get_function_handle( "IN.read_window");
56  else
57  IN_read = COM_get_function_handle( "IN.read_by_control_file");
58 
59  COM_call_function( IN_read, fname, wname.c_str());
60 
61  int IN_obtain = COM_get_function_handle( "IN.obtain_dataitem");
62  int mesh_hdl = COM_get_dataitem_handle((wname+".mesh").c_str());
63  COM_call_function( IN_obtain, &mesh_hdl, &mesh_hdl);
64  COM_UNLOAD_MODULE_STATIC_DYNAMIC( SimIN, "IN");
65 
66  // Change the memory layout to contiguous.
67  COM_resize_array( (wname+".mesh").c_str(), 0, NULL, 0);
68 
69  if ( del) {
70  // Delete all dataitems and leave alone the mesh.
71  COM_delete_dataitem( (wname+".atts").c_str());
72  }
73 
74  int npanes; COM_get_panes( wname.c_str(), &npanes, NULL);
75  return npanes;
76  }
77  else if (!lastdot || std::strcmp( lastdot, ".im")) {
78  std::cerr << "Unknown file format with suffix " << lastdot << std::endl;
79  return -1;
80  }
81  return read_mesh( fname, wname, NULL);
82  }

84  // Assuming the window has been created, read in the local panes from
85  // given file and return the number of local panes.
86  int read_mesh( const char *fname, const std::string &wname,
87  Func_ptr is_local) {
88  std::ifstream is( fname);
89  if ( is == NULL) {
90  std::cerr << "Error: Could not open file " << fname << std::endl;
91  exit(-1);
92  }
93 
94  // Create the window if not yet exist
95  int h=COM_get_window_handle(wname.c_str());
96  if ( h<=0) COM_new_window( wname.c_str());
97 
98  std::cout << "Reading file " << fname << std::endl;
99 
100  get_nextline( is, buf);
101  int num_panes, mesh_type;
102  char t;
103 
104  std::sscanf( buf, "%d%c%d", &num_panes, &t, &mesh_type);
105  if ( num_panes<=0) {
106  std::cerr << "Error: The number of panes must be positive." << std::endl;
107  exit(-1);
108  }
109 
110  int n=0;
111  switch (mesh_type) {
112  case 2: // Structured mesh
113  for ( int i=0; i<num_panes; ++i) {
114  get_nextline( is, buf);
115  int pid;
116  std::sscanf( buf, "%d", &pid);
117  if ( pid<=0) {
118  std::cerr << "Error: Found nonpositive pane id " << pid << std::endl;
119  exit(-1);
120  }
121 
122  bool local=(!is_local) || is_local( pid, _rank, _size);
123  n+=local;
124  read_pane_ij( is, wname, pid, local);
125  }
126  break;
127  case 3: // Unstrured mesh with 3-node triangles
128  case 4: // Unstrured mesh with 4-node quadrilaterals
129  case 6: // Unstrured mesh with 6-node triangles
130  for ( int i=0; i<num_panes; ++i) {
131  get_nextline( is, buf);
132  int pid;
133  std::sscanf( buf, "%d", &pid);
134  if ( pid<=0) {
135  std::cerr << "Error: Found nonpositive pane id " << pid << std::endl;
136  exit(-1);
137  }
138 
139  bool local=(!is_local) || is_local( pid, _rank, _size);
140  n+=local;
141  read_pane_uns( is, wname, pid, mesh_type, local);
142  }
143  break;
144  case 5: // Mixed mesh with 3-node triangles and 4-node quadrilaterals
145  for ( int i=0; i<num_panes; ++i) {
146  get_nextline( is, buf);
147  int pid;
148  std::sscanf( buf, "%d", &pid);
149  if ( pid<=0) {
150  std::cerr << "Error: Found nonpositive pane id " << pid << std::endl;
151  exit(-1);
152  }
153 
154  bool local=(!is_local) || is_local( pid, _rank, _size);
155  n+=local;
156  read_pane_mixed( is, wname, pid, local);
157  }
158  break;
159  default:
160  std::cerr << "Error: File: " << fname
161  << " has unknown mesh type" << mesh_type << std::endl;
162  exit(-1);
163  }
164  std::cout << "Finished reading file " << fname << std::endl;
165  return n;
166  }

182  // Read in coordinates of a pane into a window if local
183  void read_pane_coors( std::istream &is, const std::string &wname,
184  int pid, int n, bool local) {
185  SURF::Point_3<double> *coors=NULL;
186 
187  if ( local) {
188  COM_set_size( (wname+".nc").c_str(), pid, n);
189  COM_allocate_array( (wname+".nc").c_str(), pid, &(void*&)coors);
190  }
191 
192  for ( int i=0; i<n; ++i) {
193  get_nextline( is, buf);
194  if ( local) {
195  std::sscanf( buf, "%lf %lf %lf", &coors[i][0], &coors[i][1], &coors[i][2]);
196  if (!isfinite( coors[i][0]) || !isfinite(coors[i][1]) ||
197  !isfinite( coors[i][2])) {
198  std::cerr << "Error: Got invalid coordinates " << coors[i]
199  << " for node " << i+1 << " on pane " << pid
200  << " of window " << wname << std::endl;
201  exit(-1);
202  }
203  }
204  }
205 
206  if ( local && _alpha!=1.)
207  for ( int i=0; i<n; ++i) {
208  coors[i][0]*=_alpha; coors[i][1]*=_alpha; coors[i][2]*=_alpha;
209  }
210  }
