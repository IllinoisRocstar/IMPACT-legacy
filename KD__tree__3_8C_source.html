23 /*************************************************************
24  *
25  * FUNCTION: select_entry
26  *
27  * Reorders an array for a given integer k, so that the kth entry
28  * partitions the smaller and larger values after reordering.
29  * It takes an integer k, a real array arr, and an integer array iprm of
30  * length n with a starting index start, and reorders iprm so that
31  * arr(iprm(k)) is the kth largest element in the set {arr(iprm(i)),
32  * i=start,..,n}, and in addition we have the partitioning properties:
33  * i<k ==> arr(iprm(i)) <= arr(iprm(k)) and
34  * i>k ==> arr(iprm(i)) >= arr(iprm(k)).
35  *************************************************************/
36 static void select_entry( const double *arr, int k, int **iprm,
37  int start, int n, int n2) {
38  int start_1, il, ir, itemp, mid;
39  int i, j, ia;
40  double a;
41 
42  /* Note: This routine is similar to the Numerical Recipes subroutine */
43  /* ``select'', except Numerical Recipes do not have the permutation */
44  /* array IPRM; instead they reorder the array directly. */
45  start_1 = start - 1;
46  il = 1;
47  ir = n;
48  for (;;) {
49  if (1 + il >= ir) {
50  if ((1 + il == ir)&&( arr[ 3 * (*iprm)[ start_1 - 1 + ir] - 4 + n2] <
51  arr[ 3 * (*iprm)[ start_1 - 1 + il] - 4 + n2])) {
52  itemp = (*iprm)[ start_1 - 1 + il];
53  (*iprm)[ start_1 - 1 + il] = (*iprm)[ start_1 - 1 + ir];
54  (*iprm)[ start_1 - 1 + ir] = itemp;
55  }
56  return;
57  }
58  else {
59  mid = (il + ir) >> 1;
60  itemp = (*iprm)[ start_1 - 1 + mid];
61  (*iprm)[ start_1 - 1 + mid] = (*iprm)[start_1 + il];
62  (*iprm)[start_1 + il] = itemp;
63  if ( arr[ 3 * (*iprm)[ start_1 - 1 + il] - 4 + n2] >
64  arr[ 3 * (*iprm)[ start_1 - 1 + ir] - 4 + n2]) {
65  itemp = (*iprm)[ start_1 - 1 + il];
66  (*iprm)[ start_1 - 1 + il] = (*iprm)[ start_1 - 1 + ir];
67  (*iprm)[ start_1 - 1 + ir] = itemp;
68  }
69  if ( arr[ 3 * (*iprm)[start_1 + il] - 4 + n2] >
70  arr[ 3 * (*iprm)[ start_1 - 1 + ir] - 4 + n2]) {
71  itemp = (*iprm)[start_1 + il];
72  (*iprm)[start_1 + il] = (*iprm)[ start_1 - 1 + ir];
73  (*iprm)[ start_1 - 1 + ir] = itemp;
74  }
75  if ( arr[ 3 * (*iprm)[ start_1 - 1 + il] - 4 + n2] >
76  arr[ 3 * (*iprm)[start_1 + il] - 4 + n2]) {
77  itemp = (*iprm)[start_1 + il];
78  (*iprm)[start_1 + il] = (*iprm)[ start_1 - 1 + il];
79  (*iprm)[ start_1 - 1 + il] = itemp;
80  }
81  i = 1 + il;
82  j = ir;
83  ia = (*iprm)[start_1 + il];
84  a = arr[ n2 - 4 + 3 * ia];
85  for (;;) {
86  i = 1 + i;
87  while ( arr[ 3 * (*iprm)[ start_1 - 1 + i] - 4 + n2] < a) {
88  i = 1 + i;
89  }
90  j = j - 1;
91  while ( arr[ 3 * (*iprm)[ start_1 - 1 + j] - 4 + n2] > a) {
92  j = j - 1;
93  }
94  if (j < i) {
95  break;
96  }
97  itemp = (*iprm)[ start_1 - 1 + i];
98  (*iprm)[ start_1 - 1 + i] = (*iprm)[ start_1 - 1 + j];
99  (*iprm)[ start_1 - 1 + j] = itemp;
100  }
101  (*iprm)[start_1 + il] = (*iprm)[ start_1 - 1 + j];
102  (*iprm)[ start_1 - 1 + j] = ia;
103  if (j >= k) {
104  ir = j - 1;
105  }
106  if (j <= k) {
107  il = i;
108  }
109  }
110  }
111 }

113 void KD_tree_3::kdtree_build_3( const double *xs, int npoints) {
114  int *points;
115  int nextt, itop, node, imn;
116  /* size of the stack is bounded by the depth of tree. So 32 is plenty. */
117  int imin[32], imax[32], ict[32], istack[32];
118  double dimx, dimy, dimz;
119  int imx, icut, imd, v;
120  int i;
121 
122  /*.... If the number of points is not positive, give an error. */
123  assert(npoints > 0);
124 
125  /*.... Let the bounding box associated with the root node */
126  /*.... [bboxes(1,:)] exactly contain all points. */
127  /*.... We use this bounding box to help choose the cutting direction. */
128 
129  _bboxes[0] = HUGE_VAL;
130  _bboxes[1] = HUGE_VAL;
131  _bboxes[2] = HUGE_VAL;
132  _bboxes[3] = -HUGE_VAL;
133  _bboxes[4] = -HUGE_VAL;
134  _bboxes[5] = -HUGE_VAL;
135  for (i=0; i<npoints; i++) {
136  _bboxes[0] = min(xs[3 * i], _bboxes[0]);
137  _bboxes[1] = min(xs[1 + 3 * i], _bboxes[1]);
138  _bboxes[2] = min(xs[2 + 3 * i], _bboxes[2]);
139  _bboxes[3] = max(xs[3 * i], _bboxes[3]);
140  _bboxes[4] = max(xs[1 + 3 * i], _bboxes[4]);
141  _bboxes[5] = max(xs[2 + 3 * i], _bboxes[5]);
142  }
143 
144  /*.... If there is only one point, the root node is a leaf. */
145  /*.... (Our convention is to set the link corresponding to a leaf */
146  /*.... equal to the negative of the unique point contained in */
147  /*.... that leaf.) If the root is a leaf, our work is done. */
148 
149  if (npoints == 1) {
150  _tree[0] = -1;
151  return;
152  }
153  /*.... DIMX, DIMY, DIMZ are equal to the x, y, and z dimensions */
154  /*.... of the bounding box corresponding to the current node */
155  /*.... under consideration. */
156 
157  dimx = _bboxes[3] - _bboxes[0];
158  dimy = _bboxes[4] - _bboxes[1];
159  dimz = _bboxes[5] - _bboxes[2];
160  /*.... points will contain a permutation of the integers */
161  /*.... {1,...,npoints}. This permutation will be altered as we */
162  /*.... create our balanced binary tree. NEXTT contains the */
163  /*.... address of the next available node to be filled. */
164 
165  nextt = 2;
166  points = new int[npoints];
167  for (i=0; i<npoints; i++) {
168  points[i] = 1 + i;
169  }
170  /*.... Use a stack to create the tree. Put the root node */
171  /*.... "1" on the top of the stack (ICRSTACK). The array */
172  /*.... subset of points (i.e., subset of points associated */
173  /*.... with this node) is recorded using the arrays */
174  /*.... IMIN and IMAX. */
175 
176  itop = 1;
177  istack[0] = 1;
178  imin[0] = 1;
179  imax[0] = npoints;
180  /*.... Finally, we record in ICT the appropriate "cutting */
181  /*.... direction" for bisecting the set of points */
182  /*.... corresponding to this node. This direction is either */
183  /*.... the x, y, or z direction, depending on which dimension */
184  /*.... of the bounding box is largest. */
185 
186  if (max(dimz, dimy) <= dimx) {
187  ict[0] = 1;
188  }
189  else if (dimy >= dimz) {
190  ict[0] = 2;
191  }
192  else {
193  ict[0] = 3;
194  }
195  /*.... Pop nodes off stack, create children nodes and put them */
196  /*.... on stack. Continue until k-D tree has been created. */
197 
198  while (itop > 0) {
199  /*.Pop top node off stack. */
200 
201  node = istack[itop - 1];
202  /*.Make this node point to next available node location (NEXTT). */
203  /*.This link represents the location of the FIRST CHILD */
204  /*.of the node. The adjacent location (NEXTT+1) */
205  /*.is implicitly taken to be the location of the SECOND */
206  /*.child of the node. */
207 
208  _tree[node - 1] = nextt;
209  imn = imin[itop - 1];
210  imx = imax[itop - 1];
211  icut = ict[itop - 1];
212  itop = itop - 1;
213  /*.Partition point subset associated with this node. */
214  /*.Using the appropriate cutting direction, use SELECT_ENTRY to */
215  /*.reorder points so that the point with median */
216  /*.coordinate is points(imd), while the */
217  /*.points {points(i),i<imd} have SMALLER (or equal) */
218  /*.coordinates, and the points with */
219  /*.{points(i),i>imd} have GREATER coordinates. */
220 
221  imd = (imn + imx) >> 1;
222  select_entry(xs, 1 - imn + imd, &points, imn, 1 - imn + imx, icut);
223  /*.If the first child's subset of points is a singleton, */
224  /*.the child is a leaf. Set the child's link to point to the */
225  /*.negative of the point number. Set the child's bounding */
226  /*.box to be equal to a box with zero volume located at */
227  /*.the coordinates of the point. */
228 
229  if (imn == imd) {
230  v = points[imn - 1];
231  _tree[nextt - 1] = -v;
232  nextt = 1 + nextt;
233  }
234  else {
235  /*.In this case, the subset of points corresponding to the */
236  /*.first child is more than one point, and the child is */
237  /*.not a leaf. Approximate the bounding box of this child. */
238 
239  for (i=0; i<=5; i++) {
240  _bboxes[ 6 * nextt - 6 + i] = _bboxes[ 6 * node - 6 + i];
241  }
242  _bboxes[ 6 * nextt - 4 + icut] = xs[ 3 * points[imd - 1] - 4 + icut];
243  /*...Put the first child onto the stack, noting the */
244  /*...associated point subset in IMIN and IMAX, and */
245  /*...putting the appropriate cutting direction in ICT. */
246 
247  dimx = _bboxes[ 6 * nextt - 3] - _bboxes[ 6 * nextt - 6];
248  dimy = _bboxes[ 6 * nextt - 2] - _bboxes[ 6 * nextt - 5];
249  dimz = _bboxes[ 6 * nextt - 1] - _bboxes[ 6 * nextt - 4];
250  itop = 1 + itop;
251  assert (itop <= 32);
252 
253  istack[itop - 1] = nextt;
254  imin[itop - 1] = imn;
255  imax[itop - 1] = imd;
256  if ((dimx >= dimy)&&(dimx >= dimz)) {
257  ict[itop - 1] = 1;
258  }
259  else if (dimy >= dimz) {
260  ict[itop - 1] = 2;
261  }
262  else {
263  ict[itop - 1] = 3;
264  }
265  nextt = 1 + nextt;
266  }
267  /*.If the first child's subset of points is a singleton, */
268  /*.the child is a leaf. Set the child's link to point to the */
269  /*.negative of the point number. Set the child's bounding */
270  /*.box to be equal to a box with zero volume located at */
271  /*.the coordinates of the point. */
272 
273  if (!(1 - imx + imd)) {
274  v = points[imx - 1];
275  _tree[nextt - 1] = -v;
276  nextt = 1 + nextt;
277  }
278  else {
279  /*.. In this case, the subset of points corresponding to the */
280  /*...second child is more than one point, and the child is */
281  /*...not a leaf. Approximate the bounding box of this child. */
282 
283  for (i=0; i<=5; i++) {
284  _bboxes[ 6 * nextt - 6 + i] = _bboxes[ 6 * node - 6 + i];
285  }
286  _bboxes[ 6 * nextt - 7 + icut] = xs[ 3 * points[imd] - 4 + icut];
287  /*.. Put the second child onto the stack, noting the */
288  /*...associated point subset in IMIN and IMAX, and */
289  /*...putting the appropriate cutting direction in ICT. */
290 
291  dimx = _bboxes[ 6 * nextt - 3] - _bboxes[ 6 * nextt - 6];
292  dimy = _bboxes[ 6 * nextt - 2] - _bboxes[ 6 * nextt - 5];
293  dimz = _bboxes[ 6 * nextt - 1] - _bboxes[ 6 * nextt - 4];
294  itop = 1 + itop;
295  assert (itop <= 32);
296 
297  istack[itop - 1] = nextt;
298  imin[itop - 1] = 1 + imd;
299  imax[itop - 1] = imx;
300  if ((dimx >= dimy)&&(dimx >= dimz)) {
301  ict[itop - 1] = 1;
302  }
303  else if (dimy >= dimz) {
304  ict[itop - 1] = 2;
305  }
306  else {
307  ict[itop - 1] = 3;
308  }
309  nextt = 1 + nextt;
310  }
311  }
312 
313  delete [] points;
314 }

316 void KD_tree_3::kdtree_bboxes_3( const double *xs) {
317  static int istack[64];
318  int n = 2*_npnts-1;
319  int itop, node, ind, child, v;
320  int i;
321 
322  /* size of stack is twice of the depth of tree. So 64 is plenty. */
323  for (i=0; i<n; i++) {
324  _bboxes[6 * i] = HUGE_VAL;
325  }
326  itop = 1;
327  istack[0] = 1;
328  while (itop > 0) {
329  node = istack[itop - 1];
330  ind = _tree[node - 1];
331  if ( _bboxes[ 6 * ind - 6] != HUGE_VAL) {
332  assert( _bboxes[6 * ind] != HUGE_VAL);
333  /* Pop the node from stack if its children have been visited. */
334 
335  itop = itop - 1;
336  /* Assign bounding box */
337 
338  _bboxes[ 6 * node - 6] = min(_bboxes[6 * ind], _bboxes[ 6 * ind - 6]);
339  _bboxes[ 6 * node - 5] = min(_bboxes[1 + 6 * ind], _bboxes[ 6 * ind - 5]);
340  _bboxes[ 6 * node - 4] = min(_bboxes[2 + 6 * ind], _bboxes[ 6 * ind - 4]);
341  _bboxes[ 6 * node - 3] = max(_bboxes[3 + 6 * ind], _bboxes[ 6 * ind - 3]);
342  _bboxes[ 6 * node - 2] = max(_bboxes[4 + 6 * ind], _bboxes[ 6 * ind - 2]);
343  _bboxes[ 6 * node - 1] = max(_bboxes[5 + 6 * ind], _bboxes[ 6 * ind - 1]);
344  }
345  else {
346  /* Vist children */
347 
348  for (i=0; i<=1; i++) {
349  child = ind + i;
350  /*.... If child is a leaf, add point to the indices */
351 
352  if ( _tree[child - 1] < 0) {
353  v = - _tree[child - 1];
354  _bboxes[ 6 * child - 6] = xs[ 3 * v - 3];
355  _bboxes[ 6 * child - 3] = xs[ 3 * v - 3];
356  _bboxes[ 6 * child - 5] = xs[ 3 * v - 2];
357  _bboxes[ 6 * child - 2] = xs[ 3 * v - 2];
358  _bboxes[ 6 * child - 4] = xs[ 3 * v - 1];
359  _bboxes[ 6 * child - 1] = xs[ 3 * v - 1];
360  }
361  else {
362  itop = 1 + itop;
363  assert(itop <= 64);
364  istack[itop - 1] = child;
365  }
366  }
367  }
368  }

371 int KD_tree_3::kdtree_search_3( const double range[6], int start) {
372  int nfound_out, itop, node, ind, child;
373  int istack[32];
374  bool wholesubtree[32];
375  bool wst;
376  int i;
377  int offset = start-1;
378 
379  /* size of the stack is bounded by the depth of tree. So 32 is plenty. */
380  /* size of istack is bounded by the depth of tree. So MAXDEPTH is plenty. */
381 
382  for (i=0; i<=31; i++) {
383  wholesubtree[i] = false;
384  }
385  /*.... If the bounding boxes do not intersect, then return directly. */
386 
387  nfound_out = 0;
388  if (( !_bboxes || _bboxes[2] > range[5]) || ( _bboxes[1] > range[4]) ||
389  ( _bboxes[3] < range[0]) || ( _bboxes[0] > range[3]) ||
390  ( _bboxes[4] < range[1]) || ( _bboxes[5] < range[2])) {
391  return nfound_out;
392  }
393  /*.... If root node is a leaf, return leaf. */
394 
395  if ( _tree[0] < 0) {
396  nfound_out = 1;
397  if (_maxout >= 1) {
398  _indices[0] = -_tree[0]+offset;
399  }
400 
401  return nfound_out;
402  }
403  itop = 1;
404  istack[0] = 1;
405  /*.... Traverse (relevant part of) k-D tree using stack */
406 
407  while (itop > 0) {
408  /*.... pop node off of stack */
409 
410  node = istack[itop - 1];
411  ind = _tree[node - 1];
412  wst = wholesubtree[itop - 1];
413  if (!wst) {
414  wst = range[0] <= _bboxes[ 6 * node - 6] &&
415  range[3] >= _bboxes[ 6 * node - 3] &&
416  range[1] <= _bboxes[ 6 * node - 5] &&
417  range[4] >= _bboxes[ 6 * node - 2] &&
418  range[2] <= _bboxes[ 6 * node - 4] &&
419  range[5] >= _bboxes[ 6 * node - 1];
420  }
421  itop = itop - 1;
422  /*.... check if either child of NODE is a leaf or should be */
423  /*.... put on stack. */
424 
425  for (i=0; i<=1; i++) {
426  child = ind + i;
427  /*.... If child is a leaf, add point to the indices */
428 
429  if ( wst || _bboxes[ 6 * child - 4] <= range[5] &&
430  _bboxes[ 6 * child - 5] <= range[4] &&
431  _bboxes[ 6 * child - 3] >= range[0] &&
432  _bboxes[ 6 * child - 6] <= range[3] &&
433  _bboxes[ 6 * child - 2] >= range[1] &&
434  _bboxes[ 6 * child - 1] >= range[2]) {
435  if ( _tree[child - 1] < 0) {
436  nfound_out = 1 + nfound_out;
437  if (_maxout >= nfound_out)
438  _indices[nfound_out - 1] = -_tree[child - 1]+offset;
439  }
440  else {
441  itop = 1 + itop;
442  assert(itop <= 32);
443 
444  istack[itop - 1] = child;
445  wholesubtree[itop - 1] = wst;
446  }
447  }
448  }
449  }
450 
451  return nfound_out;
452 }
