28  // Assuming the window has been created, read in the local panes from
29  // given file and return the number of local panes.
30  int read_mesh( const char *fname, const std::string &wname) {
31  std::ifstream is( fname);
32  if ( is == NULL) {
33  std::cerr << "Error: Could not open file " << fname << std::endl;
34  exit(-1);
35  }
36 
37  // Create the window if not yet exist

40 
41  std::cout << "Reading file " << fname << std::endl;
42 
43  read_pane_coors( is, wname);
44  read_pane_elems( is, wname);
45 
46  return 1;
47  }

56  // Read in coordinates of a pane into a window if local
57  void read_pane_coors( std::istream &is, const std::string &wname) {
58  std::vector<SURF::Point_3<double> > coors;
59 
60  // First, look for number of vertices
61  for ( ;;) {
62  get_nextline( is, buf);
63  std::string s(buf);
64 
65  if ( s.find("Vertices",0)!=std::string::npos) break;
66  }
67 
68  int n=0;
69  is >> n; get_nextline(is, buf);
70  coors.reserve(n);
71 
72  for ( int i=0; i<n; ++i) {
73  get_nextline( is, buf);
74 
75  SURF::Point_3<double> p;
76 
77  std::sscanf( buf, "%lf %lf %lf", &p[0], &p[1], &p[2]);
78  coors.push_back( p);
79  }
80 
81  COM_assertion_msg( n>0, "Did not find any nodes");
82  std::cout << "Read in " << n << " nodes" << std::endl;
83 
84  COM_set_size( (wname+".nc").c_str(), 1, n);
85  SURF::Point_3<double> *p;
86  COM_allocate_array( (wname+".nc").c_str(), 1, &(void*&)p);
87 
88  std::copy( coors.begin(), coors.end(), p);
89  }
