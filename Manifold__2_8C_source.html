53 Halfedge Pane_manifold_2::
54 get_prev_edge( Edge_ID eID, Access_Mode mode) const
55 {
56  Edge_ID prv;
57  switch ( mode) {
58  case REAL_PANE:
59  prv = get_prev_real_edge( eID); break;
60  case WHOLE_PANE:
61  prv = Simple_manifold_2::get_prev_edge( eID); break;
62  case ACROSS_PANE: {
63  // In the across-pane mode, it is illegal to access halfedges that are
64  // not on physical boundaries but on pane boundaries. So a pane-border
65  // halfedge must be a physical border edge.
66  if ( !eID.is_border())
67  { prv = get_prev_real_edge( eID); break; }
68  else { // the edge is on physical boundary.
69  Halfedge h = get_opposite_edge( eID, ACROSS_PANE);
70  while ( !h.is_border()) h = h.next().opposite();
71  return h;
72  }
73  }
74  default: COM_assertion_msg(false, "Should never reach here");
75  }
76 
77  return Halfedge( this, prv, mode);
78 }

235 void Window_manifold_2::determine_counterparts(const COM::DataItem *pconn_e) {
236 
237  const int pconn_offset=MAP::Pane_connectivity::pconn_offset();
238 
239  typedef std::map< std::pair<int,int>,
240  std::pair< const int*, const int*> > B2e_map;
241  B2e_map bm;
242 
243  // First, loop through the pane-connectivity arrays in all local panes
244  // to construct a mapping from the internal local pane ids (between 0 and
245  // #local_panes-1) of each pair of abutting local panes to the addresses
246  // in the pane-connectivity arrays corresponding to the communication
247  // patterns of the two panes. This mapping is stored in B2e_map, and
248  // will allow convenient construction of the counterparts in the second step.
249  // Note: As a side-product, this step also fills in _cnt_pn for the edges.
250  for ( int i=0, n=_pms.size(); i<n; ++i) {
251  // Pane_manifold_2 &pm = _pms[i];
252  Pane_manifold_2 &pm = *_pms[i];
253  const COM::DataItem *pconn_l = pm.pane()->dataitem(pconn_e->id());
254  const int *b2e_l = (const int*)pconn_l->pointer();
255 
256  int pane_id_l = i;
257 
258  for ( int j=pconn_offset, nj=pconn_l->size_of_real_items();
259  j<nj; j+=b2e_l[j+1]+2) {
260 
261  std::map<int,int>::iterator it= _pi_map.find( b2e_l[j]);
262  bool is_remote = it == _pi_map.end();
263 
264  // Loop through the edges in the pconn to assign _cnt_pn.
265  int pnid_other = b2e_l[j];
266  if ( is_remote) pnid_other = -pnid_other;
267 
268  const Edge_ID *es = (const Edge_ID*)(&b2e_l[j+2]);
269  for ( int k=0, nk=b2e_l[j+1]; k<nk; ++k) {
270  pm._cnt_pn[pm.get_border_edgeID(pm.get_opposite_real_edge(es[k]))-1]
271  = pnid_other;
272  // Note: this handles branch-cut as well.
273  }
274 
275  if ( is_remote) continue; // skip remote panes
276 
277  // Insert addresses of the pconn arrayes into B2e_map
278  int pane_id_r = it->second;
279 
280  if ( pane_id_l <= pane_id_r)
281  bm[ std::make_pair( pane_id_l, pane_id_r)].first = &b2e_l[j+1];
282  if ( pane_id_l >= pane_id_r)
283  bm[ std::make_pair( pane_id_r, pane_id_l)].second = &b2e_l[j+1];
284  }
285  }
286 
287  // Second, loop through the B2e_map to insert *local* counterparts of
288  // border-edges into _bd_cnt (i.e., if the counterpart of an edge is
289  // on a remote process, the counterpart is ignored).
290  for ( B2e_map::const_iterator it=bm.begin(); it!=bm.end(); ++it) {
291  // Pane_manifold_2 *p1 = &_pms[it->first.first];
292  // Pane_manifold_2 *p2 = &_pms[it->first.second];
293  Pane_manifold_2 *p1 = _pms[it->first.first];
294  Pane_manifold_2 *p2 = _pms[it->first.second];
295  const int *v1 = it->second.first;
296  const int *v2 = it->second.second;
297 
298  int is_branchcut = (it->first.first == it->first.second);
299  COM_assertion_msg( ! is_branchcut || *v1%2==0,
300  "Branch-cut must list nodes in pairs.");
301 
302  for ( int k=1, nk=*v1; k<=nk; ++k) {
303  const Edge_ID e1 = (Edge_ID&)v1[k];
304  const Edge_ID e2 = (Edge_ID&)v2[k+is_branchcut];
305 
306  p1->_bd_cnt[p1->get_border_edgeID(p1->get_opposite_real_edge(e1))-1] =
307  Halfedge(p2, e2, ACROSS_PANE);
308 
309  p2->_bd_cnt[p2->get_border_edgeID(p2->get_opposite_real_edge(e2))-1] =
310  Halfedge(p1, e1, ACROSS_PANE);
311 
312  if (is_branchcut) ++k;
313  }
314  }
315 }

317 void Window_manifold_2::determine_primaries( const COM::DataItem *pconn_n) {
318  // Using the pane connectivity to construct a helper map bm, which
319  // maps from the pane IDs of each pair of incident panes to its
320  // correponding arrays in b2v.
321  // Note in each pane-ID pair, the first ID is no greater than the second.
322  typedef std::map< std::pair<int,int>,
323  std::pair< const int*, const int*> > B2v_map;
324  B2v_map bm;
325 
326  const int pconn_offset=MAP::Pane_connectivity::pconn_offset();
327 
328  for ( int i=0, n=_pms.size(); i<n; ++i) {
329  // const COM::DataItem *pconn_l = _pms[i].pane()->dataitem(pconn_n->id());
330  const COM::DataItem *pconn_l = _pms[i]->pane()->dataitem(pconn_n->id());
331  const int *b2v_l = (const int*)pconn_l->pointer();
332  int pane_id_l = i;
333 
334  for ( int j=pconn_offset, nj=pconn_l->size_of_real_items();
335  j<nj; j+=b2v_l[j+1]+2) {
336 
337  std::map<int,int>::iterator it= _pi_map.find( b2v_l[j]);
338  if ( it == _pi_map.end()) continue; // skip remote panes
339 
340  int pane_id_r = it->second;
341 
342  if ( pane_id_l <= pane_id_r)
343  bm[ std::make_pair( pane_id_l, pane_id_r)].first = &b2v_l[j+1];
344  if ( pane_id_l >= pane_id_r)
345  bm[ std::make_pair( pane_id_r, pane_id_l)].second = &b2v_l[j+1];
346  }
347  }
348 
349  // Now loop through bm to compute _nd_prm within the process.
350  for ( B2v_map::const_iterator it=bm.begin(); it!=bm.end(); ++it) {
351  // Pane_manifold_2 *p1 = &_pms[it->first.first];
352  // Pane_manifold_2 *p2 = &_pms[it->first.second];
353  Pane_manifold_2 *p1 = _pms[it->first.first];
354  Pane_manifold_2 *p2 = _pms[it->first.second];
355  const int *v1 = it->second.first, *v2=it->second.second;
356 
357  int is_branchcut = (it->first.first == it->first.second);
358  COM_assertion_msg( ! is_branchcut || *v1%2==0,
359  "Branch-cut must list nodes in pairs.");
360 
361  for ( int k=1, nk=*v1; k<=nk; ++k) {
362  Node node1( p1, v1[k], ACROSS_PANE);
363  Node node2( p2, v2[k+is_branchcut], ACROSS_PANE);
364 
365  int i1 = p1->get_bnode_index( v1[k]);
366  if ( i1>=0 && !node2.is_isolated() &&
367  ( node2.halfedge().is_border() ||
368  node1.is_isolated() && p1->_nd_prm[i1].pane()==0 )) {
369  p1->_nd_prm[i1] = node2;
370  }
371  else {
372  int i2 = p2->get_bnode_index( v2[k+is_branchcut]);
373  if ( i2 >= 0 && !node1.is_isolated() &&
374  (node1.halfedge().is_border() || p2->_nd_prm[i2].pane()==0)) {
375  p2->_nd_prm[i2] = node1;
376  }
377  }
378 
379  if (is_branchcut) ++k;
380  }
381  }

420 void Window_manifold_2::
421 compute_shortest_edgelen_nodes( COM::DataItem *lens) {
422  std::vector< COM:: Pane*> panes;
423  lens->window()-> panes( panes);
424  std::vector< COM::Pane*>::const_iterator it = panes.begin();
425 
426  // Initialize to HUGE_VAL
427  double inf=HUGE_VAL;
428  Rocblas::copy_scalar( &inf, lens);
429 
430  for (int i=0, local_npanes = panes.size(); i<local_npanes; ++i, ++it){
431  const COM::Pane &pane = **it;
432  const Point_3<Real> *pnts =
433  reinterpret_cast<const Point_3<Real>*>(pane.coordinates());
434  Real *lp = (Real *)(pane.dataitem( lens->id())->pointer());
435 
436  // Loop through elements of the pane
437  Element_node_enumerator ene( &pane, 1);
438  for ( int j=pane.size_of_elements(); j>0; --j, ene.next()) {
439  int nn=ene.size_of_nodes();
440  int uindex = ene[0]-1;
441  for (int k = 0, ne=ene.size_of_edges(); k<ne; k++){
442  int vindex=ene[(k+1)%ne]-1;
443  double sqlen = (pnts[ uindex]-pnts[ vindex]).squared_norm();
444 
445  lp[uindex] = std::min( lp[uindex], sqlen);
446  lp[vindex] = std::min( lp[vindex], sqlen);
447  if ( nn>ne) lp[uindex+ne] = std::min( lp[uindex+ne], sqlen);
448  if ( nn>ne+ne) lp[uindex+ne+ne] = std::min( lp[uindex+ne+ne], sqlen);
449 
450  uindex = vindex;
451  }
452  }
453  }
454 
455  // Reduce on shared nodes and then compute squared roots.
456  reduce_on_shared_nodes( lens, OP_MIN);
457  Rocblas::sqrt( lens, lens);
458 }

914  Real *p;

918  strd = weights_strds[i] = a->stride();

920  // Initialize values to 0.
921  for ( int k=0; k<nn; ++k, p+=strd) *p = 0.;
922  }
923 
924  Vector_3<Real> J[2];
925  Vector_2<Real> nc(0.5,0.5);
926 
927  Element_node_vectors_k_const<Point_3<Real> > ps;
928  Element_vectors_k_const<Real> elem_vals_evk;
929  Element_node_vectors_k<Real> nodal_vals_evk;
930  Element_vectors_k_const<Real> elem_weights_evk;
931  Element_node_vectors_k<Real> nodal_weights_evk;

934  std::vector< COM::Pane*>::const_iterator it=_cc->panes().begin();
935  for (int i=0; i<local_npanes; ++i, ++it) { // Loop through the panes
936  COM::Pane &pane = **it;

939  (pane.coordinates());
940  const COM::DataItem *elem_vals_pane =
941  pane.dataitem( elem_vals->id());
942  const COM::DataItem *elem_weights_pane =
943  elem_weights ? pane.dataitem( elem_weights->id()) : NULL;
944  COM::DataItem *nodal_vals_pane =
945  pane.dataitem( nodal_vals->id());
946 
947  // Initialize values of nodal_vals_pane to 0.
948  for ( int d=1; d<=ncomp; ++d) {
949  COM::DataItem *nvpi = ncomp==1?nodal_vals_pane:(nodal_vals_pane+d);
950  Real *p=reinterpret_cast<Real*>(nvpi->pointer());
951  for ( int j=0,s=nvpi->stride(),n=nvpi->size_of_real_items()*s; j<n; j+=s)
952  p[j] = 0.;
953  }

955  // Loop through the elements of the pane
956  Element_node_enumerator ene( &pane, 1);
957  for ( int j=pane.size_of_real_elements(); j>0; --j, ene.next()) {

959  elem_vals_evk.set( elem_vals_pane, ene);

961  nodal_weights_evk.set( weights_ptrs[i], ene, weights_strds[i]);
962

964  int nn=ene.size_of_nodes();

966  switch ( scheme) {
967  case E2N_USER:
968  case E2N_AREA:
969  if ( scheme == E2N_USER) {
970  // Use user specified weights.
971  elem_weights_evk.set( elem_weights_pane, ene);
972  w = elem_weights_evk[0];
973  } // Continue to the case of E2N_ONE
974  else {
975  Generic_element_2 e(ne, nn);
976  e.Jacobian( ps, nc, J);
977 
978  const Vector_3<Real> v = Vector_3<Real>::cross_product( J[0], J[1]);
979  w = std::sqrt(v.squared_norm());
980  if ( ne==3) w*=0.5;
981  } // Continue to the case of E2N_ONE
982  case E2N_ONE: {
983  // Update nodal weights
984  for ( int k=0; k<nn; ++k)
985  nodal_weights_evk[k] += w;
986 
987  // Update nodal sums
988  for ( int d=0; d<ncomp; ++d) {
989  Real t = w*elem_vals_evk(0,d);
990  for ( int k=0; k<nn; ++k)
991  nodal_vals_evk(k,d) += t;
992  }
993  break;
994  }
995  case E2N_ANGLE:
996  case E2N_SPHERE: {
997  for ( int k=0; k<ne; ++k) {
998  J[0] = ps[k==ne-1?0:k+1]-ps[k]; J[1] = ps[k?k-1:ne-1]-ps[k];
999  double s = std::sqrt((J[0]*J[0])*(J[1]*J[1]));
1000  if ( s>0) {
1001  double cosw = J[0]*J[1]/s;
1002  if (cosw>1) cosw=1; else if ( cosw<-1) cosw=-1;
1003  w = std::acos( cosw);
1004 
1005  if ( scheme==SURF::E2N_SPHERE)
1006  w = std::sin(w)/s;
1007 
1008  // Update nodal weights
1009  nodal_weights_evk[k] += w;
1010 
1011  // Update nodal sums
1012  for ( int d=0; d<ncomp; ++d)
1013  nodal_vals_evk(k,d) += w*elem_vals_evk(0,d);
1014  }
1015  }
1016  for ( int k=ne; k<nn; ++k) {
1017  // Update nodal weights
1018  nodal_weights_evk[k] += 1;
1019 
1020  // Update nodal sums
1021  for ( int d=0; d<ncomp; ++d)
1022  nodal_vals_evk(k,d) += elem_vals_evk(0,d);
1023  }
1024  break;
1025  }
1026 
1027  default: COM_assertion_msg(false, "Should never reach here");
1028  }
1029  }
1030  }
