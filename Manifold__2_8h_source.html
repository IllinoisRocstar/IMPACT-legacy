705  Edge_ID opp;

767 Halfedge Node::halfedge() const
768 {
769  Edge_ID eID;
770 
771  switch ( _mode) {
772  case REAL_PANE:
773  eID = _pm->get_incident_real_edge( _vID); break;
774  case WHOLE_PANE:
775  eID = _pm->get_incident_edge( _vID); break;
776  default: {
777  eID =_pm->get_incident_real_edge( _vID);
778  // the current node (vertex) is in general the origin of the edge or
779  // the edge center, unless the node is isolated.
780 
781  if ( eID.is_border()) {
782  // Halfedge h=_pm->get_counterpart( eID);
783  // if ( h.pane_manifold()) return h;
784  Halfedge h=_pm->get_counterpart( eID), h0=h;
