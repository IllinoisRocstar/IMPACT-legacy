27  // Assuming the window has been created, read in the local panes from
28  // given file and return the number of local panes.
29  int read_mesh( const char *fname, const std::string &wname) {
30  std::ifstream is( fname);
31  if ( is == NULL) {
32  std::cerr << "Error: Could not open file " << fname << std::endl;
33  exit(-1);
34  }
35 
36  // Create the window if not yet exist
37  int h=COM_get_window_handle(wname.c_str());
38  if ( h<=0) COM_new_window( wname.c_str());
39 
40  std::cout << "Reading file " << fname << std::endl;
41 
42  // Search for NUMNP
43  do {
44  get_nextline( is, buf);
45  } while ( !is.eof() && std::string(buf).find( "NUMNP")==std::string::npos);
46 
47  std::cerr << "Buf is now \"" << buf << "\" before getting nn" << std::endl;
48  get_nextline( is, buf);
49  int t1, t2, t3, t4, nn, ne;
50  std::sscanf( buf, "%d %d %d %d %d %d", &nn, &ne, &t1, &t2, &t3, &t4);
51 
52  // Search for NODAL COORDINAES
53  do {
54  get_nextline( is, buf);
55  } while ( !is.eof() && std::string(buf).find( "NODAL")==std::string::npos);
56 
57  std::cout << "Reading " << nn << " nodes and "
58  << ne << " triangles " << std::endl;
59  read_pane_coors( is, wname, nn);
60 
61  // Search for ELEMENTS
62  do {
63  get_nextline( is, buf);
64  } while ( !is.eof() && std::string(buf).find( "ELEMENTS")==std::string::npos);
65  read_pane_elems( is, wname, ne);
66 
67  return 1;
68  }

77  // Read in coordinates of a pane into a window if local
78  void read_pane_coors( std::istream &is, const std::string &wname, int nn) {
79  COM_set_size( (wname+".nc").c_str(), 1, nn);
80  SURF::Point_3<double> *ps;
81  COM_allocate_array( (wname+".nc").c_str(), 1, &(void*&)ps);
82 
83  for ( int i=0; i<nn; ++i) {
84  get_nextline( is, buf);
85  int t1;
86 
87  std::sscanf( buf, "%d %lf %lf %lf", &t1, &ps[i][0], &ps[i][1], &ps[i][2]);
88  COM_assertion( t1==i+1);
89  }
90  }

92  // Read in element connectivity of a pane into a window if local

98 
99  for (int i=0; i<ne; ++i) {
100  int t1, t2, t3;
101 
102  get_nextline( is, buf);
103  std::sscanf( buf, "%d %d %d %d %d %d", &t1, &t2, &t3,
104  &es[i][0], &es[i][1], &es[i][2]);
105  COM_assertion( t1 == i+1 && t2==3 && t3==3);
106  }
