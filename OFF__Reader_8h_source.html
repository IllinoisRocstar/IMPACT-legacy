21 // This class provides an interface for reading in a surface mesh
22 // from the ".off" file. It can read files with mixed meshes with

29  // Assuming the window has been created, read in the local panes from
30  // given file and return the number of local panes.
31  // Return 0 if successful.
32  int read_mesh( const char *fname, const std::string &wname) {
33  std::ifstream is( fname);
34  if ( is == NULL) {
35  std::cerr << "Error: Could not open file " << fname << std::endl;
36  exit(-1);
37  }
38 
39  // Create the window if not yet exist
40  int h=COM_get_window_handle(wname.c_str());
41  if ( h<=0) COM_new_window( wname.c_str());
42  char cmt[MAXLEN];
43 
44  for ( int pane_id=1; ; ++pane_id) {
45  int nn, ne, tmp, pid=0;
46  get_nextline( is, buf);
47  // First line of the file tells that it is a .off file
48  if ( std::string( buf, 0, 3) != "OFF") {
49  if ( pane_id==1)
50  std::cerr << "Input file does not have a valid OFF header."
51  << std::endl;
52  else {
53  if ( pane_id>2)
54  std::cout << "Read in " << pane_id-1 << " panes." << std::endl;
55  return 0;
56  }
57  }
58  else
59  // 2nd line of the file gives number of nodes and elements
60  get_nextline( is, buf);
61 
62  std::sscanf( buf, "%d %d %d %s %d", &nn, &ne, &tmp, cmt, &pid);
63  if ( std::strcmp(cmt, "#!") || !pid) pid = pane_id;
64 
65  std::cout << "Reading " << nn << " nodes and "
66  << ne << " faces for patch " << pid << std::endl;
67  read_pane_coors( is, wname, pid, nn);
68  read_pane_elems( is, wname, pid, ne);
69  }
70 
71  return 0;
72  }

90  // Read in element connectivity of a pane into a window if local
91  void read_pane_elems( std::istream &is, const std::string &wname,
92  int pane_id, int ne){
93  if ( ne==0) return;
94 
95  // Note: Supports triangular and quadrilateral meshes now.
96  int egroups=0,num_ele=0;
97  std::string str;
98 
99  int numEdges1=0, numEdges2=0,vs[9];
100  get_nextline(is,buf);
101  std::istringstream sin(buf);
102  sin >> numEdges1;
103  for ( int i=0; i<numEdges1; ++i) sin >> vs[i];
104  numEdges2 = numEdges1;
105 
106  while ( !is.eof() && buf[0] && num_ele<ne) {
107  std::vector<int> conns[9];
108 
109  do { // read a group
110  //read through the file for connectivity
111  for ( int i=0; i<numEdges2; ++i) conns[i].push_back( vs[i]);
112  num_ele++;
113 
114  if ( num_ele==ne) break;
115  get_nextline(is,buf); numEdges2=0;
116  std::istringstream sin(buf);
117  sin >> numEdges2;
118  for ( int i=0; i<numEdges2; ++i) sin >> vs[i];
119  } while (numEdges1==numEdges2);
120 
121  char ext1[10];
122  sprintf( ext1, "%d", ++egroups);
123  switch (numEdges1) {
124  case 3: str= wname+".:t3:"+ext1 ; break;
125  case 6: str= wname+".:t6:"+ext1 ; break;
126  case 4: str= wname+".:q4:"+ext1 ; break;
127  case 8: str= wname+".:q8:"+ext1 ; break;
128  case 9: str= wname+".:q9:"+ext1 ; break;
129  default: ;
130  }
131 
132  COM_set_size( str.c_str(), pane_id, conns[0].size());
133  int *es; COM_allocate_array( str.c_str(), pane_id, &(void*&)es);
134 
135  for (int j=0, nconn=conns[0].size(); j<nconn;j++) {
136  for ( int i=0; i<numEdges1; ++i)
137  *(es++) = conns[i][j]+1;
138  }
139 
140  numEdges1=numEdges2;
141  }
142 
143  if ( num_ele<ne) {
144  std::cerr << "Incomplete element connectivity table. File corrupted?"
145  << std::endl;
146  exit(-1);
147  }
148  }

150  // Read in the next non-blank line from the file

152  str[0] = '\0';

162  /* Checks whether the line pointed to by the file handle is empty or not.
163  * return 1 : if line is empty

180  /* checks whether a line is a comment or not.
