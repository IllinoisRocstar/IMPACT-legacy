102 // This function is the main interface for the overlay algorithm.
103 int Overlay::overlay() {
104  Bbox_3 bbox=B->get_bounding_box(), gbox = G->get_bounding_box();
105 
106  std::cout << "\tMesh " << B->name() << ": "
107  << B->size_of_nodes() << " vertices, "
108  << B->size_of_faces() << " faces, with bounding box "
109  << bbox << "\n\tMesh " << G->name() << ": "
110  << G->size_of_nodes() << " vertices, "
111  << G->size_of_faces() << " faces, with bounding box "
112  << gbox << std::endl;
113 
114  double tol_high = 3.e-1, tol_low=1.e-1;
115 
116  // First, check whether the scales of the two bounding boxes are
117  // the same by comparing the total lengths of bounding boxes and
118  // the individual dimensions.
119  double bl = (bbox.xmax()-bbox.xmin()) +
120  (bbox.ymax()-bbox.ymin()) + (bbox.zmax()-bbox.zmin());
121  double gl = (gbox.xmax()-gbox.xmin()) +
122  (gbox.ymax()-gbox.ymin()) + (gbox.zmax()-gbox.zmin());
123 
124  if ( bl>gl*(1+tol_high) || gl>bl*(1+tol_high) ||
125  !bbox.do_match( gbox, std::min(bl,gl)*tol_high)) {
126  std::cerr << "ERROR: The bounding boxes differ by more than "
127  << tol_high*100 << "%. Please check the geometries. Stopping..."
128  << std::endl;
129  abort();
130  }
131 
132  // If the bounding boxes do not match at all sides within a fraction
133  // of the longest dimension, then stop the code.
134  // the difference between the two boxes is smaller than a fraction (eps)
135  // of the largest dimension of the two boxes.
136  if ( !bbox.do_match( gbox, std::min(bl,gl)*tol_low)) {
137  std::cerr << "WARNING: The bounding boxes differ by more than "
138  << tol_low*100 << "% but less than "
139  << tol_high*100 << "%. Continuing anyway." << std::endl;
140  }
141 
142  double t0 = get_wtime();
143 
144  // Create helper data in two input meshes.
145  B->create_overlay_data();
146  G->create_overlay_data();
147 
148  // Detect the 0-features of the two meshes.
149  std::cerr << "Detecting features in " << B->name() << "..." << std::endl;
150  B->detect_features();
151 
152  std::cerr << "Detecting features in " << G->name() << "..." << std::endl;
153  G->detect_features();
154 
155  match_features_0();
156 
157  // Evaluate normals for the nodes.
158  B->evaluate_normals();
159  G->evaluate_normals();
160 
161  std::cout << "Performing mesh overlay..." << std::flush;
162  // Step 1: Project the blue vertices onto G and
163  // compute the intersection points of blue edges with green
164  // edges, and insert the intersections into the blue edges on the fly.
165  intersect_blue_with_green();
166  std::cout << "..." << std::flush;
167 
168  // Step 2: Sort the intersection points corresponding to each green
169  // edge and insert them into green edges (in linear time).
170  sort_on_green_edges();
171  std::cout << "..." << std::flush;
172 
173  // Step 3: Compute the projection of the green vertices on B.
174  associate_green_vertices();
175  std::cout << "..." << std::flush;
176 
177  if ( verbose) {
178  std::cout << "\n\tLocated " << inodes.size()
179  << " edge intersections.\n\tDone in "
180  << get_wtime()-t0 << " sec.\n"
181  << "Exporting the subdivision..." << std::flush;
182  t0 = get_wtime();
183  }
184 
185  // Finally, assign local ids to the subnodes within panes and
186  number_subnodes();
187 
188  // subdivide the faces and assign local ids to these subfaces,and
189  // create the nodal and face lists for the panes.
190  number_subfaces();
191 
192  // Now, clean up the overlay data
193  // Destroy helper data in the input windows
194  B->delete_overlay_data(); G->delete_overlay_data();
195  while ( !inodes.empty()) {
196  INode* x = inodes.front(); inodes.pop_front();
197  delete x;
198  }
199 
200  std::cout << "Done";
201  if ( verbose) {
202  std::cout << " in " << get_wtime()-t0 << " sec.";
203  }
204  std::cout << std::endl;
205  return 0;
206 }

212 // This subroutine corrects the green parent of an inode i by intersecting
213 // it with the previous assigned green parent.
214 // TODO: Currently does not resolve inconsistencies involving multiple

216 void Overlay::
217 insert_node_in_blue_edge( INode &x, const HEdge &b1) {
218  HEdge b = x.halfedge( BLUE);
219  HEdge g = x.halfedge( GREEN); RFC_assertion( !g.is_border_g());
220 
221  // Step 1, verify that x is not at a green vertex that already has
222  // an inode associated with. If so, merge the two inodes by making
223  // its blue parent the intersection of the two blue edges/vertices,
224  // and delete the old one.
225  if ( x.parent_type( GREEN) == PARENT_VERTEX) {
226  INode *i = acc.get_inode( g.origin_g());
227  if (i) {
228  if ( i->parent_type( BLUE) == PARENT_VERTEX) {
229  std::cerr << "ERROR: One-to-many mapping at green vertex "
230  << g.pane()->get_index(g.origin_l())+1
231  << " of pane " << g.pane()->id()
232  << " at " << g.origin_l().point()
233  << ".\n It is projected onto from blue vertices \n\t"
234  << b.pane()->get_index(b.origin_l())+1 << " of pane "
235  << b.pane()->id() << " at " << b.origin_l().point() << " and\n\t";
236  b = i->halfedge(BLUE);
237  std::cerr << b.pane()->get_index(b.origin_l())+1
238  << " of pane " << b.pane()->id()
239  << " at " << b.origin_l().point() << std::endl;
240  RFC_assertion(i->parent_type( BLUE) != PARENT_VERTEX); abort();
241  }
242  if ( !contains( i->halfedge( BLUE), i->parent_type( BLUE),
243  b, PARENT_VERTEX)) {
244  b = b.next_g();
245  RFC_assertion( contains( i->halfedge( BLUE), i->parent_type( BLUE),
246  b, PARENT_VERTEX));
247  }
248  x.set_parent( b, Point_2(0,0), BLUE);
249  }
250  }
251 
252  if ( x.parent_type( BLUE) == PARENT_VERTEX) {
253  // Step 2a, assign blue parent
254  INode *i = acc.get_inode( b.origin_g());
255  if ( i != NULL) {
256  if ( !contains( i->halfedge( GREEN), i->parent_type( GREEN),
257  g, x.parent_type( GREEN))) {
258  // Otherwise, we must correct the green parent of i by computing
259  // the intersection of the two computed parents
260  if ( x.parent_type( GREEN) == PARENT_FACE) {
261  Parent_type pg = i->parent_type(GREEN);
262 
263  bool found=false;
264  if ( pg ==PARENT_FACE) {
265  // Find the intersection of the two faces
266  HEdge h = g; RFC_assertion( !h.is_border_g());
267  do {
268  if ( contains( i->halfedge( GREEN), pg, h, PARENT_EDGE)) {
269  // the edge is the intersection of the two
270  Point_2 nc = get_nat_coor( x, Generic_element(count_edges(g)),
271  h, GREEN);
272  RFC_assertion( nc[0]>0 && nc[0]<1 && nc[1]>0 && nc[1]<0.5);
273  x.set_parent( g, Point_2(nc[0],0), GREEN);
274 
275  found = true;
276  break;
277  }
278  } while ( (h=h.next_g()) != g);
279  }
280 
281  if ( !found) {
282  // Search for the common vertex
283  HEdge h = g;
284  do {
285  if ( contains( i->halfedge( GREEN), pg, h, PARENT_VERTEX)) {
286  g = h; x.set_parent( g, Point_2(0,0), GREEN);
287 
288  found = true;
289  break;
290  }
291  } while ( (h=h.next_g()) != g);
292  RFC_assertion( found);
293  }
294  }
295  else { // green parent of x is not a face.
296  if ( x.parent_type(GREEN)==PARENT_EDGE &&
297  x.parent_type( GREEN) == i->parent_type( GREEN) &&
298  !contains( i->halfedge( GREEN), i->parent_type( GREEN),
299  g, PARENT_VERTEX)) {
300  g = g.next_g();
301  RFC_assertion(contains(i->halfedge( GREEN), i->parent_type( GREEN),
302  g, PARENT_VERTEX));
303  }
304  RFC_assertion( acc.get_inode(g.origin_g())==NULL ||
305  acc.get_inode(g.origin_g())->parent_type(BLUE)
306  == PARENT_EDGE);
307 
308  x.set_parent( g, Point_2(0,0), GREEN);
309 
310  if ( !contains( i->halfedge( GREEN), i->parent_type( GREEN),
311  g, x.parent_type( GREEN))) {
312  std::cerr << "ERROR: One-to-many mapping at blue vertex "
313  << b.pane()->get_index(b.origin_l())+1
314  << " of pane " << b.pane()->id()
315  << " at " << b.origin_l().point()
316  << ". It projects onto \n\tgreen vertex "
317  << b.pane()->get_index(g.origin_l())+1 << " of pane "
318  << g.pane()->id() << " at " << g.origin_l().point()
319  << " and\n\tgreen ";
320  switch (i->parent_type(GREEN)) {
321  case PARENT_VERTEX: std::cerr << "vertex"; break;
322  case PARENT_EDGE: std::cerr << "edge"; break;
323  case PARENT_FACE: std::cerr << "face incident on"; break;
324  default: ;
325  }
326  g = i->halfedge(GREEN);
327  std::cerr << " ("
328  << g.pane()->get_index(g.origin_l())+1 << ","
329  << g.pane()->get_index(g.destination_l())+1
330  << ") of pane " << g.pane()->id() << " at ("
331  << g.origin_l().point() << ","
332  << g.destination_l().point() << ")." << std::endl;
333 
334  RFC_assertion( contains( i->halfedge( GREEN), i->parent_type( GREEN),
335  g, x.parent_type( GREEN))); abort();
336  }
337  }
338  }
339  delete i; i=NULL;
340  }
341  acc.set_inode( b.origin_g(), &x);
342 
343  // Step 3a
344  // Delete the inconsistent inodes on adjacent edges of b
345  HEdge h = b;
346  do {
347  INode_list &il = acc.get_inode_list( h);
348  if ( !il.empty()) {
349  // Throw away the false intersection points.
350  while ( ! il.empty()) {
351  INode *i=&il.front();
352 
353  if ( contains( i->halfedge( GREEN), i->parent_type( GREEN),
354  g, x.parent_type( GREEN))) {
355  il.pop_front();
356  delete i; i=NULL;
357  }
358  else
359  break;
360  }
361  }
362  else {
363  INode_list &ilr = acc.get_inode_list( h.opposite_g());
364  // Throw away the false intersection points.
365  while ( ! ilr.empty()) {
366  INode *i=&ilr.back();
367 
368  if ( contains( i->halfedge( GREEN), i->parent_type( GREEN),
369  g, x.parent_type( GREEN))) {
370  ilr.pop_back();
371  delete i; i=NULL;
372  }
373  else
374  break;
375  }
376  }
377  } while ( ( h = acc.get_next_around_origin( h)) != b);
378  }
379  else {
380  // If x is on a blue edge, remove the inconsistent ones
381  RFC_assertion( x.parent_type( BLUE) == PARENT_EDGE && b1==b);
382 
383  // Step 3b
384  INode_list &il = acc.get_inode_list( b);
385  // Throw away the false intersection points.
386  while ( !il.empty()) {
387  INode *i=&il.back();
388  if ( i->nat_coor(BLUE)[0] >= x.nat_coor(BLUE)[0]) {
389  std::cerr << "WARNING: Intersections are out-of-order on blue edge ("
390  << b.pane()->get_index(b.origin_l())+1 << ","
391  << b.pane()->get_index(b.destination_l())+1
392  << ") on pane " << b.pane()->id() << " at ("
393  << b.origin_l().point() << "," << b.destination_l().point()
394  << ").\n\tPrevious intersection was "
395  << i->nat_coor(BLUE)[0]
396  << " but current one is at " << x.nat_coor(BLUE)[0]
397  << " and green parameterization is "
398  << x.nat_coor(GREEN)[0] << "." << std::endl;
399 
400  RFC_assertion( std::abs(i->nat_coor(0)[0] - x.nat_coor(0)[0]) < 0.1);
401  RFC_assertion( i->parent_type( GREEN) != PARENT_VERTEX);
402 
403  if ( !contains( i->halfedge( GREEN), i->parent_type( GREEN),
404  g, PARENT_VERTEX) &&
405  !contains( i->halfedge( GREEN), i->parent_type( GREEN),
406  g.next_l(), PARENT_VERTEX)) {
407  std::cerr << "Cannot be continued. Stopping..." << std::endl;
408  abort();
409  }
410 
411  // Otherwise, we must correct the green parent of i by computing
412  // the intersection of the two computed parents
413  if ( !contains( i->halfedge( GREEN), i->parent_type( GREEN),
414  g, PARENT_VERTEX))
415  g = g.next_g();
416 
417  x.set_parent( g, Point_2(0,0), GREEN);
418 
419  { // Determine the blue parameterization of x based on closest point
420  // of the green vertex on the blue edge.
421  Vector_3 t = b.destination_l().point()-b.origin_l().point();
422  Real c = (g.origin_l().point()-b.origin_l().point())*t / (t*t);
423 
424  // If snapped onto a blue vertex, then call recursively.
425  if ( c>=1-eps_e) {
426  x.set_parent( b.next_l(), Point_2(0,0), BLUE);
427  insert_node_in_blue_edge( x, x.halfedge(BLUE)); return;
428  }
429  else if ( c<=eps_e) {
430  x.set_parent( b, Point_2(0,0), BLUE);
431  insert_node_in_blue_edge( x, x.halfedge(BLUE)); return;
432  }
433 
434  // Otherwise, pop the neighbor vertex.
435  x.set_parent( b, Point_2(c,0), BLUE);
436  }
437  }
438 
439  if ( contains( i->halfedge( GREEN), i->parent_type( GREEN),
440  g, x.parent_type( GREEN))) {
441  il.pop_back();
442  delete i; i=NULL;
443  }
444  else
445  break;
446  }
447  il.push_back( x);
448  }
449 
450  // If x is a vertex and b is border, change to a nonborder edge.
451  if ( x.parent_type(BLUE) == PARENT_VERTEX && b.is_border_g()) {
452  b = acc.get_next_around_origin(b);
453  x.set_parent( b, Point_2(0,0), BLUE);
454  }
455 
456  // If the green parent of x is a vertex, associate x with the green vertex.
457  if ( x.parent_type(GREEN) == PARENT_VERTEX) {
458  RFC_assertion(g==x.halfedge(GREEN));
459  acc.set_inode( g.origin_g(), &x);
460  }
461 
462  // Output some debugging information.
463  static int count=0, countmax=10;
464  if ( verbose && count<countmax) {
465  if ( x.parent_type(BLUE) == PARENT_VERTEX) {
466  b = x.halfedge(BLUE);
467  double dist = ( b.origin_l().point() -
468  op.get_point( x.halfedge(GREEN), x.nat_coor(GREEN))).squared_norm();
469

471  b.origin_l().point()).squared_norm()) {

474  << b.pane()->get_index(b.origin_l())+1
475  << " of pane " << b.pane()->id()
476  << " at " << b.origin_l().point()
477  << " to green mesh is relatively large: " << std::sqrt(dist) << std::endl;
478 
479  if ( count++==0) {
480  std::cout << "Note: Only " << countmax << " such warnings "
481  << " will be printed." << std::endl;
482  }
483  }
484  }
485  else if ( x.parent_type(GREEN) == PARENT_VERTEX) {
486  g = x.halfedge( GREEN);
487  double dist = ( g.origin_l().point() -
488  op.get_point( x.halfedge(BLUE), x.nat_coor(BLUE))).squared_norm();
489 
490  if ( dist > 0.1*( g.destination_l().point()-
491  g.origin_l().point()).squared_norm()) {
492 
493  if ( count == 0) std::cout << std::endl;
494  std::cout << "WARNING: Distance from green vertex "

496  << " of pane " << g.pane()->id()

509 // Move an edge from ridge- or corner-queue into main queue.
510 // We use q_rdg to store edges that point from ridge vertices to
511 // non-feature vertices and use q_crn for edges from corner
512 // vertices to non-corner vertices. Move edge from q_rdg into
513 // q only if q is empty; and move edge from q_crn to q if both q
514 // and q_rdg are empty.
515 void Overlay::
516 insert_edge_into_queue( const HEdge &h, INode *v,
517  std::queue<HEdge> &q,
518  std::queue<HEdge> &q_rdg,
519  std::queue<HEdge> &q_crn) {
520  // If the edge or its opposite has been marked, then skip.
521  if ( acc.marked( h)) return;
522 
523  // First, determine feature rank of origin and destination vertices.
524  int frank_org = acc.is_on_feature( h.origin_l());
525  if ( frank_org) frank_org += acc.is_feature_0( ( h.origin_l()));
526 
527  if ( frank_org != 2) { // Consider feature type of green parent.
528  Parent_type pt = v->parent_type(GREEN);
529  HEdge g = v->halfedge(GREEN);
530  if ( pt == PARENT_VERTEX) {
531  Node gv;
532  if ( acc.is_feature_0( gv=g.origin_l()))
533  frank_org = 2; // If mapped onto a corner, consider vertex as corner
534  else if ( frank_org == 0 && acc.is_on_feature( gv))
535  frank_org = 1;
536  }
537  else if ( frank_org==0 && pt==PARENT_EDGE && acc.is_feature_1( g)) {
538  frank_org = 1; // Ridge vertex
539  }
540  }
541 
542  int frank_dst = 0;
543  if ( frank_org) { // Determine type of destination vertex
544  frank_dst = acc.is_on_feature( ( h.destination_l()));
545  if ( frank_dst) frank_dst += acc.is_feature_0( h.destination_l());
546  }
547 
548  HEdge hopp = h.opposite_g();
549  acc.mark( h);
550  if ( frank_org == 0 || frank_org < frank_dst || hopp.is_border_g() ||
551  frank_dst != 2 && !hopp.is_border_g() && acc.marked( hopp)) {
552  // If push edge into main queue, then mark both the edge and its opposite
553  q.push( h); acc.mark( hopp);
554  }
555  else if ( frank_org == 2 && !hopp.is_border_g() && acc.marked( hopp)) {
556  // If I am a corner and the other edge is candidate, move into main queue
557  q.push( hopp);
558  }
559  else if ( frank_org == 1) {
560  // Otherwise, if frank_org is ridge, then push into ridge queue
561  q_rdg.push( h);
562  }
563  else {
564  // Otherwise frank_org is corner, don't push it.
565  q_crn.push(h);
566  }
567 }

571 // non-feature vertices and use q_crn for edges from corner
572 // vertices to non-corner vertices. Move edge from q_rdg into

575 bool Overlay::
576 is_queue_empty( std::queue<HEdge> &q,
577  std::queue<HEdge> &q_rdg,
578  std::queue<HEdge> &q_crn) {

581  for ( int i=0; i<2; ++i) {

584  HEdge h = qbuf.front(); qbuf.pop();
585  HEdge hopp = h.opposite_g();
586  if ( !acc.marked( hopp)) {
587  q.push( h); acc.mark( h.opposite_g());
588  return false;
589  }
590  }
591  }

593  return true;

596 // This is step 1 of the overlay algorithm. It projects the blue vertices
597 // and computes the intersections of the blue edges with the green
598 // edges. These intersections are stored in the lists associated with
599 // with the blue edges.
600 // At input, x is an inode corresponding to a blue vertex.
601 void Overlay::
602 intersect_blue_with_green() {
603  // Precondition: None of the blue halfedges is marked.
604  std::queue<HEdge> q, q_rdg, q_crn;
605 
606  for ( int ncomp=1; ;++ncomp) { // Top loop over connected components.
607  INode *seed = overlay_init();
608  if ( seed==NULL) return; // If all halfedges are marked, then stop.
609  RFC_assertion( seed->parent_type( BLUE) == PARENT_VERTEX);
610  HEdge b = seed->halfedge( BLUE), h=b;
611 
612  if ( verbose) {
613  if ( ncomp>1)
614  std::cout << "There appear to be multiple connected components "
615  << "in the input mesh.\n\tWorking on component "
616  << ncomp << std::endl;
617  else
618  std::cout << "\nThe input meshes have "
619  << ( is_opposite ? "opposite orientations" :
620  "the same orientation") << std::endl;
621 
622  std::cout << "Starting with seed vertex "
623  << b.pane()->get_index(b.origin_l())+1
624  << " in pane " << b.pane()->id() << " of blue mesh at "
625  << b.origin_l().point() << std::endl;
626  }
627 
628  // Embed the seed into the blue vertex, (and the green vertex if its
629  // green parent is a vertex.)
630  insert_node_in_blue_edge( *seed, seed->halfedge(BLUE));
631  do {
632  insert_edge_into_queue( h, seed, q, q_rdg, q_crn);
633  } while ( (h = acc.get_next_around_origin(h)) != b);
634 
635  while ( !is_queue_empty( q, q_rdg, q_crn)) {
636  HEdge b = q.front(); q.pop();
637  HEdge bopp=b.opposite_g();
638 
639  INode_list &il = acc.get_inode_list( b);
640  INode *x = acc.get_inode( b.origin_g()); assert(x);
641  Node dst = b.destination_g();
642  INode *idst= acc.get_inode( dst);
643 
644  bool is_feature = b.pane()->is_feature_1( b);
645  bool is_border = b.is_border_g() || b.opposite_g().is_border_g();
646  bool is_dst_feature = is_feature || acc.is_on_feature( dst);
647  bool is_dst_border = is_border || dst.is_border_g();
648 
649  // This is to snap open-ends of features.
650  const Real tol_snap_fea = 0.2, tol_snap_border = 2./3.;
651  Real cb=0, cg;
652 
653  for (;;) {
654  if (x->parent_type( BLUE)==PARENT_VERTEX &&
655  x->halfedge( BLUE).origin_g()==dst ||
656  idst && get_edge_parent(*x,*idst,GREEN)!=Host_face())
657  break; // The last two inodes on the edge are adjacent
658 
659  HEdge g1, g0, g2;
660  Parent_type t1, t0, t2;
661 
662  if ( il.empty()) {
663  INode *iorg = acc.get_inode( b.origin_g());
664  g1 = iorg->halfedge( GREEN); t1 = iorg->parent_type( GREEN);
665  g0 = HEdge(); t0 = PARENT_NONE;
666  }
667  else {
668  INode_list::iterator it=--il.end();
669  g1 = it->halfedge( GREEN); t1 = it->parent_type( GREEN);
670  if ( it==il.begin()) {
671  INode *iorg = acc.get_inode( b.origin_g());
672  g0 = iorg->halfedge( GREEN); t0 = iorg->parent_type( GREEN);
673  }
674  else {
675  --it; g0 = it->halfedge( GREEN); t0 = it->parent_type( GREEN);
676  }
677  }
678 
679  // Compute the intersection of b with the edges in the link
680  // of <g1,t1> but not in the star of <g0, t0>.
681  Real start = cb;
682  intersect_link( x, b, g0, t0, g1, t1, start, &cb, &cg, &g2, &t2);
683 
684  // If current edge is a ridge edge, then attempt to snap it onto ridge.
685  if ( is_feature && cb < 1 && cb > 0) {
686  op.snap_blue_ridge_edge( b, g2, &cb, &cg, &t2,
687  is_border ? tol_snap_border : tol_snap_fea);
688  }
689  else if ( !is_feature && cb < 1 &&
690  ( is_dst_feature && cb>1-tol_snap_fea &&
691  (t2 == PARENT_VERTEX && acc.is_on_feature(g2.origin_l()) ||
692  t2 == PARENT_EDGE && acc.is_feature_1(g2)) ||
693  is_dst_feature && cb>1-tol_snap_border &&
694  (t2 == PARENT_VERTEX && g2.origin_l().is_border_g() ||
695  t2 == PARENT_EDGE && g2.is_border_g()))) {
696  // Enforce snapping vertex onto feature/border
697  cb = 1 + tol_snap_fea;
698  t2 = PARENT_EDGE;
699  }
700 
701  Point_2 nc(cg,0.);
702  RFC_assertion( t2 != PARENT_NONE);
703  if ( cb > 1) { // If intersects beyond the end point, do projection.
704  RFC_assertion( t2 != PARENT_VERTEX);
705  bool onto=op.project_onto_element( dst.point(),&g2,&t2,
706  Vector_3(0,0,0), &nc, eps_e, 0.2);
707  RFC_assertion( onto);
708  }
709 
710  if ( is_dst_feature && t2 != PARENT_VERTEX && cb>=1) {
711  op.snap_blue_ridge_vertex( dst, &g2, &t2, &nc, is_dst_border ?
712  tol_snap_border : tol_snap_fea);
713  }
714 
715  // Create an inode for the intersection point.
716  x = new INode();
717  if ( cb < 1)
718  x->set_parent( b, Point_2( cb, 0), BLUE);
719  else
720  x->set_parent( bopp, Point_2(0,0), BLUE);
721 
722  if ( nc[0] == 1.) {
723  nc[0] = 0;
724  g2 = !g2.is_border_g() ? g2.next_l() : g2.opposite_g();
725  }
726 
727  static int count=0, countmax = 10;
728  if ( verbose && count < countmax) {
729  if ( dst.is_border_g() && cb>=1 && acc.get_inode( dst)==NULL &&
730  !( t2 == PARENT_EDGE && g2.opposite_g().is_border_g() ||
731  t2 == PARENT_VERTEX && g2.origin_l().is_border_g())) {
732  if ( count == 0) std::cerr << std::endl;
733  std::cerr << "WARNING: Blue border vertex " << b.destination_l().point()
734  << " is not matched with green boundary" << std::endl;
735  ++count;
736  }
737  else if ( is_border && cb>0 && cb<1 &&
738  !( t2 == PARENT_EDGE && g2.opposite_g().is_border_g() ||
739  t2 == PARENT_VERTEX && g2.origin_l().is_border_g())) {
740 
741  if ( count == 0) std::cerr << std::endl;
742  std::cerr << "WARNING: Blue border edge " << b.origin_l().point()
743  << b.destination_l().point()
744  << " is not matched with green boundary " << std::endl;
745  ++count;
746  }
747 
748  if ( is_dst_feature && cb>=1 && acc.get_inode( dst)==NULL &&
749  !( t2 == PARENT_EDGE && g2.pane()->is_feature_1(g2) ||
750  t2 == PARENT_VERTEX && acc.is_on_feature(g2.origin_l()))) {
751  std::cerr << "WARNING: Blue ridge vertex " << b.destination_l().point()
752  << " is not matched with green ridge. cb is " << cb
753  << " and nc is " << nc << std::endl;
754  ++count;
755  }
756  else if ( is_feature && cb<1 &&
757  !( t2 == PARENT_EDGE && g2.pane()->is_feature_1(g2) ||
758  t2 == PARENT_VERTEX && acc.is_on_feature(g2.origin_l()))) {
759  std::cerr << "WARNING: Blue ridge edge " << b.origin_l().point()
760  << b.destination_l().point()
761  << " is not matched with green ridge. cb is " << cb
762  << " and nc is " << nc << std::endl;
763  ++count;
764  }
765  }
766 
767  x->set_parent( g2, nc, GREEN);
768 
769  insert_node_in_blue_edge( *x, b);
770  }
771 
772  // Copy inode from border edge into neighbor edge
773  if ( b.is_border_g() && !il.empty()) {
774  INode_list &ilr = acc.get_inode_list( bopp);
775 
776  while ( !il.empty()) {
777  INode &i = il.front(); il.pop_front();
778  ilr.push_front( i);
779  i.set_parent( bopp, Point_2(1-i.nat_coor(BLUE)[0],0), BLUE);
780  }
781  }
782 
783  // Push unvisited incident halfedges of b.destination_l() into q.
784  HEdge h; h = b.next_g();
785 
786  do {
787  // Push the edge into queue and mark it as nonzero.
788  insert_edge_into_queue( h, x, q, q_rdg, q_crn);
789  } while ( ( h = acc.get_next_around_origin( h)) != bopp);
790  }
791  } // end for
792 
793  // Unmark all the halfedges of B
794  B->unmark_alledges();
795 }

798 // corresponding to each green edge and insert them into green edges
799 // in linear time.
800 void Overlay::

803  B->panes( ps);
804

807  for ( std::vector<RFC_Pane_overlay*>::iterator
808  pit=ps.begin(); pit!=ps.end(); ++pit){

810  for (int i=1, s=(*pit)->size_of_nodes(); i<=s; ++i) {

812  if ( !v.is_isolated() && v.halfedge_l().destination_l() == v

815  i->prev_link[GREEN]=i->next_link[GREEN]=NULL;

832  INode_list &il = acc.get_inode_list( lopp);
833  for ( INode_list::iterator it=il.begin(); it!=il.end(); ++it) {
834  INode *i = &*it; RFC_assertion( i);
835  i->prev_link[GREEN]=i->next_link[GREEN]=NULL;
836  inodes.push_back( i);
837  }
838  }
839  } while ( (h=h.next_l()) != h0);

841  }

872  }

880  if ( il.empty()) {
881  while ( i->prev_link[GREEN]) { i = i->prev_link[GREEN]; }
882 
883  for (;;) {
884  INode *j = i->next_link[GREEN];
885  il.push_back(*i);
886  if ( j) i=j; else break;

888  }
889  }

893 // The helper for sort_on_green_edges. It inserts a node v into the green
894 // edge v->green_halfedge(). Tag is for marking the buffer. We assume
895 // that each green(blue) edge intersects a blue(green) face at most twice.
896 void Overlay::
897 insert_node_in_green_edge( INode* v, int tag) {
898  switch ( v->parent_type( GREEN)) {
899  case PARENT_VERTEX: {
900  RFC_assertion(acc.get_inode(v->halfedge( GREEN).origin_g())==v);
901  return;
902  }
903  case PARENT_EDGE: {
904  HEdge g = v->halfedge( GREEN), gopp=g.opposite_g();
905  if ( !gopp.is_border_g() && gopp<g) {
906  g = gopp;
907  v->set_parent( gopp, Point_2( 1.-v->nat_coor(GREEN)[0],0.), GREEN);
908  }
909  INode *y = g.pane()->get_buffered_inode( g, tag);
910 
911  if ( y ) {
912  INode *x = v;
913 
914  if ( y->nat_coor(GREEN)[0] > x->nat_coor(GREEN)[0]) {
915  RFC_assertion(x->next_link[GREEN] == NULL || x->next_link[GREEN] == y);
916  RFC_assertion(y->prev_link[GREEN] == NULL || y->prev_link[GREEN] == x);
917  x->next_link[GREEN] = y; y->prev_link[GREEN] = x;
918  }
919  else {
920  RFC_assertion(x->prev_link[GREEN] == NULL || x->prev_link[GREEN] == y);
921  RFC_assertion(y->next_link[GREEN] == NULL || y->next_link[GREEN] == x);
922  x->prev_link[GREEN] = y; y->next_link[GREEN] = x;
923  }
924  }
925  // Otherwise, store the node
926  else
927  g.pane()->set_buffered_inode( g, tag, v);
928  }
929  case PARENT_FACE:
930  return;
931  default:
932  RFC_assertion(false);
933  }

977 void Overlay::
978 project_adjacent_green_vertices( const INode *i, const HEdge &b) {
979  std::queue< INode*> q;
980 
981  // From every known subnode, projects its adjacent green vertices
982  // onto blue faces and creates new subnodes for the green vertices.
983  do {
984  HEdge g = i->halfedge(GREEN), gopp, g0=g;
985  Parent_type igp = i->parent_type( GREEN);
986  // Loop through all the incident halfedges.
987  do {
988  gopp = g.opposite_g();
989  Node dst = g.destination_g();
990 
991  INode_list &il = acc.get_inode_list( g);
992  INode_list &ilr = acc.get_inode_list( gopp);
993 
994  // If there is any subnode between i and dst, skip the edge.
995  if ( igp == PARENT_VERTEX) {
996  if (!il.empty() || !ilr.empty())
997  continue;
998  }
999  else if ( igp == PARENT_EDGE) {
1000  INode *j;
1001  if ( !il.empty()) {
1002  if ( &il.back() != i) continue;
1003  j = ( il.size() > 1 ? i->prev_link[GREEN] : acc.get_inode( g.origin_g()));
1004  }
1005  else {
1006  if ( &ilr.front() != i) continue;
1007  j = ( ilr.size() > 1 ? i->next_link[GREEN] : acc.get_inode( g.origin_g()));
1008  }
1009  if ( j && contains( b, PARENT_FACE,
1010  j->halfedge( BLUE), j->parent_type( BLUE)))
1011  continue;
1012  }
1013 
1014  // If dst has already been projected, skip the edge.
1015  INode *x=NULL;
1016  if ( (x=acc.get_inode( dst)) != NULL) {
1017  RFC_assertion( x->parent_type( BLUE) != PARENT_FACE ||
1018  contains( x->halfedge(BLUE), PARENT_FACE,
1019  i->halfedge(BLUE), i->parent_type(BLUE)));
1020  continue;
1021  }
1022 
1023  // Determine whether the vertex projects onto the blue face, and if so,
1024  // create a new subnode at it and insert the subnode into the queue.
1025  HEdge b1 = b; Parent_type t=PARENT_FACE;
1026  Point_2 nc;
1027  const RFC_Pane_overlay *gpane = g.pane();
1028  Vector_3 v1 = gpane->get_normal( g, g.destination_l());
1029  if (op.project_onto_element( g.destination_l().point(),
1030  &b1, &t, v1, &nc, eps_p)) {
1031  Vector_3 v2 = op.get_face_normal( b1, nc);
1032  if (v2 != Vector_3(0,0,0)) v2 = v2 / std::sqrt( v2*v2);
1033 
1034  if ( logical_xor( is_opposite, v1 * v2 < 0.15)) continue;
1035  RFC_assertion( nc[0] != 0. && nc[1] != 0.);
1036 
1037  x = new INode();
1038 
1039  x->set_parent( b1, nc, BLUE);
1040  x->set_parent( gopp, Point_2(0,0), GREEN);
1041  // Check whether consistency rules are satisfied
1042  if ( verify_inode( x)) {
1043  inodes.push_back( x);
1044  acc.set_inode( dst, x);
1045  q.push( x);
1046  }
1047  else
1048  delete x;
1049  }
1050  RFC_assertion( igp != PARENT_FACE); // Must have been projected.
1051  } while ( ( g = ( igp==PARENT_VERTEX ? gopp.next_g() : gopp)) != g0);
1052  if ( !q.empty()) { i = q.front(); q.pop(); } else i = NULL;
1053  } while (i);
1054 
1055 }

1061  HEdge g = i->halfedge(GREEN), gopp, g0=g;
1062  // Loop through all the incident halfedges.
1063  do {

1067  INode_list &ilr = acc.get_inode_list( gopp);

1073  j = ( il.size() > 1 ? i->prev_link[GREEN] : acc.get_inode( g.origin_g()));

1090 // The return value indicates whether this call is conclusive.
1091 bool Overlay::
1092 intersect_link_helper( const HEdge &b,
1093  const HEdge &g2,
1094  Real start,
1095  Real *cb,
1096  Real *cg,
1097  HEdge *g,
1098  Parent_type *t,
1099  bool *found,
1100  int snapcode,
1101  const Node &anchor, bool panic) {
1102  RFC_assertion( !g2.is_border_l());
1103 
1104  Real tcb, tcg; // Parameterization of intersection on b and g2
1105  Real orient_match; // To what degree [0..1] b and g2 are counterclockwise
1106  Real normal_match; // To what degree [0..1] the intersection have same normal
1107 
1108  // Tolerance for normal matching
1109  const Real tol_nm_large = 0.6, tol_nm_small=0.3; // about 50 and 80 degrees
1110  const Real tol_nm = panic ? tol_nm_small : tol_nm_large;

1778 // sub faces. The input face is given by a list of inodes in CCW order,
1779 // and the output subfaces are given by a list of faces, each of which
1780 // is represented by a vectors of inodes.

1782 // last_checked (inclusively) are known belong to the same sub-face.
1783 // Return true if the face is not closed.
1784 bool
1785 Overlay::subdivide( const INode_const_list &face,
1786  INode_const_list::const_iterator last_checked,
1787  Subface_list &sub_faces,
1788  int color, int depth) const {
1789  if ( face.size() < 3) return true; // Do nothing
1790  RFC_assertion( last_checked != face.begin() && last_checked != face.end() && depth < 1000);
1791 
1792  INode_const_list::const_iterator it=last_checked, iend=face.end(),inext;
1793  const INode *v0 = *(--last_checked), *v1 = *it, *v2;
1794  // v0 and v1 are the last pair of adjacent nodes that are known belonging
1795  // to the sub-face of previous nodes in the node list.
1796 
1797  for ( RFC_assertion_code(unsigned int c=0);;RFC_assertion(++c<face.size())) {
1798  inext = it; ++inext; // Equivalent to inext = it + 1;
1799  if ( inext == iend) { // All nodes in the list are processed.
1800  if ( face.front() == get_next_inode( v0, v1, color)) {
1801  // Insert a new item in the sub-face list.

1805  vec.insert( vec.end(),face.begin(), face.end());

1807  }
1808  return true;
1809  }
1810  else {
1811  v2 = get_next_inode( v0, v1, color);
1812  if ( !v2) {
1813  INode_const_list sub2( it, face.end());
1814  subdivide( sub2, ++sub2.begin(), sub_faces, color, depth+1);
1815  return true;
1816  }
1817  v0 = v1; v1 = v2;
1818  if ( v2 != *inext) break; // the face is split at v2
1819 
1820  it = inext;
1821  }
1822  }

1825  // Create a binary tree structure for the nodes.

1827  IMap imap;
1828  for ( INode_const_list::const_iterator i = face.begin(); i!=iend; ++i)
1829  imap.insert( std::make_pair( *i, i));
1830

1833  std::list< const INode *> sub1, sub2;
1834  sub1.insert( sub1.end(), face.begin(), inext);

1838  if ( (mit=imap.find( v1)) == imap.end()) {

1843  }
1844  else {
1845  break;
1846  }
1847  }
