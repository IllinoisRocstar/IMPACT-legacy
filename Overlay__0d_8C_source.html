14 // Match 0-dimensional features.
15 void Overlay::
16 match_features_0() {
17 
18  const Real w2e_ratio = 1;
19  const Real d2e_ratio_sq = 1;
20 
21  std::list<Feature_0> &bf0 = B->flist_0();
22  std::list<Feature_0> &gf0 = G->flist_0();
23 
24  std::vector< Point_3> gf0_1;
25  std::vector< Node> verts;
26 
27  gf0_1.reserve( gf0.size());
28  verts.reserve( gf0.size());
29 
30  for ( std::list<Feature_0>::iterator it=gf0.begin(); it!=gf0.end(); ++it) {
31  gf0_1.push_back( it->point());
32  verts.push_back( it->vertex());
33  }
34 
35  // Create a 3-dimensional KD-tree for the 0-features of green mesh
36  KD_tree_3 kdtree( &gf0_1[0][0], gf0.size());
37 
38  int dropped=0;
39  // Query every 0-feature of the blue mesh in the KD-tree to find
40  // whether there is a unique corresponding one in the other mesh.
41  for ( std::list<Feature_0>::iterator
42  it=bf0.begin(), iend=bf0.end(); it != iend; ++it) {
43  Node v = it->vertex();
44  // Find the corresponding green feature of *it in the KD-tree
45  const Point_3 &p = v.point();
46 
47  // Compute the tolerance.
48  Real tol=HUGE_VAL;
49  HEdge b=v.halfedge_g(), b0=b;
50  do {
51  Real t = sq_length( b);
52  if ( t<tol) tol=t;
53  } while ( (b=acc.get_next_around_destination( b))!=b0);
54  tol = std::sqrt( tol);
55 
56  int *indices;
57  int nfound = kdtree.search( &p[0], tol*w2e_ratio, &indices);
58 
59  // Find the closest corner in the list
60  Real dist_min = HUGE_VAL;
61  int k=-1;
62 
63  for ( int i=0; i<nfound; ++i) {
64  Real d=(p-gf0_1[indices[i]]).squared_norm();
65  if ( d <= dist_min) {
66  dist_min = d; k=i;
67  }
68  }
69 
70  if ( dist_min<HUGE_VAL) {
71  HEdge g = verts[indices[k]].halfedge_g();
72  if ( g.is_border_g()) g = g.opposite_g();
73  else g = g.next_g();
74 
75  Real s=tol*tol;
76  HEdge h=g, h0=h;
77  do {
78  Real t = sq_length( h);
79  if ( t<s) s=t;
80  } while ( (h=acc.get_next_around_origin( h))!=h0);
81 
82  if ( dist_min < s*d2e_ratio_sq) {
83  if ( verbose2) {
84  std::cout << "Matched\t blue vertex "
85  << b.pane()->get_index(b.origin_l())+1
86  << " in pane " << b.pane()->id() << " at " << p
87  << "\n with\t green vertex "
88  << g.pane()->get_index(g.origin_l())+1
89  << " in pane " << g.pane()->id() << " at "
90  << g.origin_l().point() << std::endl;
91  }
92 
93  HEdge b = v.halfedge_g();
94  if ( b.is_border_g()) b = b.opposite_g();
95  else b = b.next_g();
96 
97  // Create an inode for the o-feature
98  INode *x = new INode();
99  x->set_parent( b, Point_2(0,0), BLUE);
100  x->set_parent( g, Point_2(0,0), GREEN);
101 
102  if ( B->snap_on_features()) {
103  // Update the coordinates for the blue point
104  Point_3 &pnt = const_cast<Point_3&>( b.origin_g().point());
105  pnt = g.origin_g().point();
106  }
107 
108  insert_node_in_blue_edge( *x, b);
109  continue;
110  }
111  }
112 
113  if ( verbose) {
114  std::cout << "\nRocface Warning: Dropping blue corner vertex "
115  << b.pane()->get_index(b.origin_l())+1
116  << " in pane " << b.pane()->id()
117  << " at " << p << std::endl;
118  }
119 
120  // Remove the false blue 0-feature
121  it = --B->remove_feature_0( it);
122  ++dropped;
123  }
124 
125  if ( verbose && dropped) {
126  std::cout << "Dropped " << dropped << " corners in \"" << B->name()
127  << "\" after feature matching" << std::endl;
128  }
129 
130  dropped=0;
131  // Remove the false green 0-features
132  for ( std::list<Feature_0>::iterator
133  it=gf0.begin(), iend=gf0.end(); it != iend; ++it) {
134  Node v = it->vertex();
135  if ( acc.get_inode( v) == NULL) {
136 
137  if ( verbose) {
138  std::cout << "\nRocface Warning: Dropping green corner vertex "
139  << v.pane()->get_index(v)+1
140  << " in pane " << v.pane()->id()
141  << " at " << v.point() << std::endl;
142  }
143 
144  // Unmark the 0-features in the green window
145  it = --G->remove_feature_0( it);
146  ++dropped;
147  }
148  }
149 
150  if ( verbose && dropped) {
151  std::cout << "Dropped " << dropped << " corners in \"" << G->name()
152  << "\" after feature matching" << std::endl;
153  }
154 }
