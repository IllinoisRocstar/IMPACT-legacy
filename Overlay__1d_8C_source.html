14 // Find the next vertex on b.
15 INode *Overlay::
16 project_next_vertex( Halfedge *b, Halfedge *g) {
17  const RFC_Pane_overlay *gpane = acc.get_pane(g);
18  const Vertex *borg = b->origin(), *bdst = b->destination();
19  const Vertex *gorg = g->origin(), *gdst = g->destination();
20 
21  double eps_e;
22 
23  if ( B->snap_on_features()) eps_e = 5.e-3; // Increase the threshold for RSRM
24  else eps_e = Overlay::eps_e;
25 
26  Real s = op.project_green_feature( gpane->get_normal( g, gdst),
27  gpane->get_tangent( g, gdst),
28  borg->point(), bdst->point(),
29  gdst->point(), eps_e);
30  INode *x = new INode();
31  if ( s < 1) {
32  if ( s < 0) s = 0; // Adjust the origin of the blue edge.
33  acc.set_parent( x, b, Point_2( s,0), BLUE);
34  acc.set_parent( x, g, Point_2( 1,0), GREEN);
35  }
36  else if ( s>= 1) {
37  // The blue parent is bdst, so we locate a non-border
38  // blue halfedge originated from b.
39  acc.set_parent( x, b, Point_2( 1,0), BLUE);
40  // Project green vertex onto blue edge
41  Real t;
42  if ( s > 1) {
43  // Project the blue vertex onto the green halfedge
44  t=op.project_blue_feature( gpane->get_normal(g, gorg),
45  gpane->get_normal(g, gdst),
46  gpane->get_tangent(g, gorg),
47  gpane->get_tangent(g, gdst),
48  gorg->point(), gdst->point(),
49  bdst->point(), eps_e);
50  if ( t>1) t=1;
51  }
52  else
53  t = 1;
54  acc.set_parent( x, g, Point_2(t,0), GREEN);
55 
56  if ( B->snap_on_features()) {
57  // Update the coordinates for the blue point
58  Point_3 &pnt = const_cast<Point_3&>(bdst->point());
59  pnt = gorg->point()+t*(gdst->point()-gorg->point());
60  }
61  }
62 
63  // Insert the new edge into the blue edge.
64  insert_node_in_blue_edge( *x, b);
65 
66  return x;
67 }

69 // Compare whether two bounding boxes are the same within certain tolerance
70 static bool equal( const Bbox_3 &b1, const Bbox_3 &b2, Real tol_r) {
71  Bbox_3 u = b1+b2;
72  Real tol_a = std::max(std::max(u.xmax()-u.xmin(), u.ymax()-u.ymin()),
73  u.zmax()-u.zmin()) * tol_r;
74  return ( std::max( std::fabs(b1.xmin()-b2.xmin()),
75  std::fabs(b1.xmax()-b2.xmax())) <= tol_a &&
76  std::max( std::fabs(b1.ymin()-b2.ymin()),
77  std::fabs(b1.ymax()-b2.ymax())) <= tol_a &&
78  std::max( std::fabs(b1.zmin()-b2.zmin()),
79  std::fabs(b1.zmax()-b2.zmax())) <= tol_a);
80 }

82 int Overlay::
83 overlay_features_1() {
84  Feature_list_1 &lf_b = B->flist_1(), &lf_g = G->flist_1();
85 
86  bool determined_direction=false;
87  int is_opp=-1;
88  const Real bbox_tol = 1.e-1, d2e_ratio_sq=0.5, d2e_ratio_sq_l=1.e-4;
89 
90  int dropped=0;
91 
92  // Loop through blue features
93  for ( Feature_list_1::iterator it_f_b = lf_b.begin(), iend_f_b=lf_b.end();
94  it_f_b != iend_f_b; ++it_f_b) {
95  std::cout << "\nProcessing blue ridge with bounding box "
96  << it_f_b->bbox << std::endl;
97  Feature_1 *f_b = &*it_f_b;
98  Feature_1::iterator it_b_mid = it_f_b->begin();
99  // Find a vertex in middle of the blue feature
100  std::advance( it_b_mid, (f_b->size()-1)/2);
101 
102  Halfedge *b = *it_b_mid;
103  Vertex *bdst = acc.get_destination(b);
104  const Point_3 &bpdst = bdst->point();
105  INode *bnode=acc.get_inode( bdst);
106  INode *bn_frt=acc.get_inode( acc.get_origin(it_f_b->front()));
107  INode *bn_bck=acc.get_inode( acc.get_destination(it_f_b->back()));
108 
109  Feature_list_1::iterator f_g = lf_g.end();
110  Feature_1::iterator it_g_mid;
111  Real param=-HUGE_VAL, sqd=HUGE_VAL;
112 
113  // Loop through green 1-features
114  for ( Feature_list_1::iterator it_f_g = lf_g.begin(), iend_f_g=lf_g.end();
115  it_f_g != iend_f_g; ++it_f_g) {
116  // Pre-filter out severely mismatching ones
117  if ( bn_frt && bn_frt!=acc.get_inode( acc.get_origin(it_f_g->front())) &&
118  bn_frt != acc.get_inode( acc.get_destination(it_f_g->back())) ||
119  bn_bck && bn_bck!=acc.get_inode( acc.get_origin(it_f_g->front())) &&
120  bn_bck != acc.get_inode( acc.get_destination(it_f_g->back())) ||
121  !equal( it_f_b->bbox, it_f_g->bbox, bbox_tol))
122  continue;
123 
124  std::cout << "\tComparing with green ridge with bounding box "
125  << it_f_g->bbox << std::endl;
126  if ( bnode != NULL) {
127  f_g = it_f_g; RFC_assertion( bn_frt && bn_bck == bnode);
128  if ( bnode == acc.get_inode( acc.get_origin(it_f_g->front())))
129  { it_g_mid = it_f_g->begin(); param = 0.; }
130  else
131  { it_g_mid = --it_f_g->end(); param = 1.; }
132  break;
133  }
134  // Identify the green edge closest to the blue point
135  for ( Feature_1::iterator tmp_it_g=it_f_g->begin(), iend=it_f_g->end();
136  tmp_it_g != iend; ++tmp_it_g) {
137  Halfedge *g = *tmp_it_g;
138  const Vertex *gorg = g->origin(), *gdst = g->destination();
139  const Point_3 &gporg = gorg->point(), &gpdst = gdst->point();
140 
141  Real gsqlen = (gpdst-gporg).squared_norm(), sd=gsqlen*d2e_ratio_sq;
142  if ( (bpdst-gporg).squared_norm()<sd ||
143  (bpdst-gpdst).squared_norm()<sd) {
144  const RFC_Pane_overlay *gpane = acc.get_pane(g);
145  Real t = op.project_blue_feature( gpane->get_normal(g, gorg),
146  gpane->get_normal(g, gdst),
147  gpane->get_tangent(g, gorg),
148  gpane->get_tangent(g, gdst),
149  gporg, gpdst,
150  bpdst, eps_e);
151  if ( t>=0 && t <= 1) {
152  Point_3 gp = gporg+t*(gpdst-gporg);
153  Real sd = (gp-bpdst).squared_norm();
154  if (sd < d2e_ratio_sq*(bpdst-b->origin()->point()).squared_norm() && sd<sqd) {
155  f_g = it_f_g; it_g_mid = tmp_it_g; param=t; sqd = sd;
156  if ( sqd < d2e_ratio_sq_l*gsqlen) break;
157  }
158  }
159  }
160  }
161  }
162 
163  // If did not find a match, remove the blue feature.
164  if ( f_g == lf_g.end()) {
165  std::cout << "Warning: Could not find match for the blue ridge!" << std::endl;
166 
167  it_f_b = --lf_b.erase( it_f_b);
168  ++dropped;
169  continue;
170  }
171  else {
172  std::cout << "\tMatched with green ridge with bounding box "
173  << f_g->bbox << std::endl;
174  }
175 
176  // Determine the orientation of the blue and green features and
177  // reverse the green feature if they have different orientation
178  Vector_3 dir_b = acc.get_pane(b)->get_tangent( b,b->destination());
179  Halfedge *g=*it_g_mid;
180  Vector_3 dir_g = acc.get_pane(g)->get_tangent( g,g->destination());
181  bool dir_match = dir_b*dir_g > 0;
182 
183  // If no inode has yet been created at the blue vertex, create one now.
184  if ( bnode == NULL) {
185  // Create a new inode for the vertex
186  bnode = new INode();
187  acc.set_parent( bnode, b, Point_2(1,0), BLUE);
188  acc.set_parent( bnode, *it_g_mid, Point_2( param, 0), GREEN);
189 
190  insert_node_in_blue_edge( *bnode, b);
191 
192  if ( !determined_direction) {
193  const Halfedge *bopp = acc.get_opposite( b);
194  const Halfedge *g=*it_g_mid, *gopp = acc.get_opposite( g);
195  is_opp = ((acc.get_pane(b)->get_normal( b, b->destination())+
196  acc.get_pane(bopp)->get_normal( bopp, bopp->origin())) *
197  (acc.get_pane(g)->get_normal( g, g->destination()) +
198  acc.get_pane(gopp)->get_normal( gopp, gopp->origin())) < 0);
199  determined_direction = true;
200  }
201  }
202 
203  // Project the blue vertex to the green 1-feature.
204  // Starting from the blue vertex, move forward along the blue 1-feature
205  // and intersect with the normal planes of the green 1-features to
206  // locate all the inodes.
207  INode *x = bnode;
208  Feature_1::iterator it_g=it_g_mid;
209  if ( param == 0. && dir_match)
210  { if ( it_g == f_g->begin()) it_g = f_g->end(); --it_g; }
211  else if ( param == 1. && !dir_match)
212  { ++it_g; if ( it_g == f_g->end()) it_g = f_g->begin(); }
213 
214  for (Feature_1::iterator it_b=it_b_mid, iend=--f_b->begin();
215  it_b!=iend; --it_b) {
216  // Loop until the target of the green halfedge projects beyond the
217  // halfedge.
218  Halfedge *b=acc.get_opposite(*it_b);
219  RFC_assertion( acc.get_pane(b)->is_feature_1(b));
220  INode *idst= acc.get_inode(acc.get_destination(b));
221 
222  for (;;) {
223  Halfedge *g = dir_match?acc.get_opposite(*it_g):*it_g;
224  RFC_assertion( acc.get_pane(g)->is_feature_1( g));
225 
226  if ( x->parent_type( BLUE)==PARENT_VERTEX &&
227  acc.get_origin(x->halfedge( BLUE))==acc.get_destination(b) ||
228  idst && get_edge_parent(*x,*idst,GREEN)!=Host_face(0,PARENT_NONE))
229  break;
230  x = project_next_vertex(b, g);
231 
232  if ( x->nat_coor( GREEN)[0] == 0. &&
233  acc.get_origin(x->halfedge( GREEN)) != acc.get_origin(g)) {
234  // Done with the green edge
235  if ( dir_match)
236  { if ( it_g == f_g->begin()) it_g = f_g->end(); --it_g; }
237  else
238  { ++it_g; if ( it_g == f_g->end()) it_g = f_g->begin(); }
239  }
240  else
241  break;
242  }
243  }
244  x = bnode;
245  it_g=it_g_mid;
246  if ( param == 0. && !dir_match)
247  { if ( it_g == f_g->begin()) it_g = f_g->end(); --it_g; }
248  else if ( param == 1. && dir_match)
249  { ++it_g; if ( it_g == f_g->end()) it_g = f_g->begin(); }
250 
251  for (Feature_1::iterator it_b=it_b_mid, iend=f_b->end(); ++it_b!=iend; ) {
252  // Loop until the target of the green halfedge projects beyond the
253  // halfedge.
254  Halfedge *b=*it_b; RFC_assertion( acc.get_pane(b)->is_feature_1(b));
255  INode *idst= acc.get_inode(acc.get_destination(b));
256 
257  for (;;) {
258  Halfedge *g = dir_match?*it_g:acc.get_opposite(*it_g);
259  RFC_assertion( acc.get_pane(g)->is_feature_1( g));
260 
261  if ( x->parent_type( BLUE)==PARENT_VERTEX &&
262  acc.get_origin(x->halfedge( BLUE))==acc.get_destination(b) ||
263  idst && get_edge_parent(*x,*idst,GREEN)!=Host_face(0,PARENT_NONE))
264  break;
265  x = project_next_vertex(b, g);
266 
267  if ( x->nat_coor( GREEN)[0] == 0. &&
268  acc.get_origin(x->halfedge( GREEN)) != acc.get_origin(g)) {
269  // Done with the green edge
270  if ( dir_match)
271  { ++it_g; if ( it_g == f_g->end()) it_g = f_g->begin(); }
272  else
273  { if ( it_g == f_g->begin()) it_g = f_g->end(); --it_g; }
274  }
275  else
276  break;
277  }
278  }
279  }
280 
281  if ( dropped)
282  std::cout << "Ignored " << dropped << " 1-features in \"" << B->name()
283  << "\" during matching" << std::endl;
284 
285  if ( (dropped=lf_g.size()-lf_b.size()))
286  std::cout << "Ignored " << dropped << " 1-features in \"" << G->name()
287  << "\" after matching" << std::endl;
288 
289  return is_opp;
290 }
