114

118  RFC_assertion_code( int count=0);

121  h = h.next_l(); RFC_assertion( ++count <= 4);
122  }
123  if ( i->parent_type(color) == PARENT_VERTEX)
124  eid.edge_id = pane.get_lvid( ene, pane.get_lid( h.destination_l()));

127  nc = Point_2( 1-nc[0], 0);
128  }
129  }
130  else {
131  eid.edge_id = pane.get_lvid( ene, pane.get_lid( b.origin_l()));

134  pane.normalize_nat_coor( eid.edge_id, ene.size_of_edges(), nc);

140 // Assign numbers to an S-vertex within all its host panes
141 void Overlay::
142 number_a_subnode( INode *i, int color,
143  std::map<int,std::pair<int,int> > &cnts) {
144  Parent_type t = i->parent_type( color);
145  HEdge h = i->halfedge( color);
146 
147  switch ( t) {
148  case PARENT_FACE: {
149  int pane_id = h.pane()->id();
150  set_subnode_id( i, color, pane_id, ++(cnts[ pane_id].second));
151  break;
152  }
153  case PARENT_EDGE: {
154  HEdge h0 = h;
155  int last_id = -1;
156  do {
157  int pane_id = h.pane()->id();
158  if ( pane_id == last_id) continue;
159  set_subnode_id( i, color, pane_id, ++(cnts[ pane_id].second));
160  last_id = pane_id;
161  } while ( (h=h.opposite_g()) != h0);
162  break;
163  }
164  default: {
165  RFC_assertion( t == PARENT_VERTEX);
166 
167  HEdge h0 = h;
168  int last_id = -1;
169  std::set<int> panes;
170  bool first_iter = true;
171  do {
172  int pane_id = h.pane()->id();
173  if ( !first_iter && (pane_id==last_id||panes.find(pane_id)!=panes.end()))
174  continue;
175  set_subnode_id( i, color, pane_id, ++(cnts[ pane_id].first));
176  panes.insert(last_id=pane_id);
177 
178  // Optimization assuming the halfedge structure is normalized
179  if ( first_iter && !h.origin_l().is_border_l()) break;
180  first_iter = false;
181  } while ( (h=acc.get_next_around_origin(h)) != h0);
182  break;
183  }
184  }
185 }

211 // Determine the numbering system for the S-vertices within panes.
212 void Overlay::
213 number_subnodes() {
214  std::vector<RFC_Pane_overlay*> b_ps, g_ps;
215  B->panes( b_ps); G->panes( g_ps);
216 
217  // Maps from pane ids to the number of S-vertices processed
218  std::map< int, std::pair<int,int> > b_vertex_counts;
219  std::map< int, std::pair<int,int> > g_vertex_counts;
220 
221  std::vector<RFC_Pane_overlay*>::iterator pi;
222  for ( pi=b_ps.begin(); pi!=b_ps.end(); ++pi)
223  b_vertex_counts[(*pi)->id()] = std::make_pair(0,0);
224  for ( pi=g_ps.begin(); pi!=g_ps.end(); ++pi)
225  g_vertex_counts[(*pi)->id()] = std::make_pair(0,0);
226 
227  // Loop through the S-vertices
228  // First, count the number of subvertices host at vertices
229  for ( std::list< INode*>::const_iterator
230  it=inodes.begin(); it!=inodes.end(); ++it) {
231  count_subnodes( *it, BLUE, b_vertex_counts);
232  count_subnodes( *it, GREEN, g_vertex_counts);
233  }
234 
235  // Second, assign IDs for subnodes
236  int n=inodes.size();
237  _subnode_ids_b.resize( n, Node_ID( 0,-1));
238  _subnode_ids_g.resize( n, Node_ID( 0,-1));
239  _subnode_copies_b.resize( n, 0);
240  _subnode_copies_g.resize( n, 0);
241  int i=0;
242  for ( std::list< INode*>::const_iterator
243  it=inodes.begin(); it!=inodes.end(); ++it,++i) {
244  (*it)->set_id( i);
245  number_a_subnode( *it, BLUE, b_vertex_counts);
246  number_a_subnode( *it, GREEN, g_vertex_counts);
247  }
248 
249  // Allocate space for subnodes.
250  for ( pi=b_ps.begin(); pi!=b_ps.end(); ++pi)
251  (*pi)->allocate_subnodes( b_vertex_counts[(*pi)->id()].second);
252  for ( pi=g_ps.begin(); pi!=g_ps.end(); ++pi)
253  (*pi)->allocate_subnodes( g_vertex_counts[(*pi)->id()].second);
254 }

256 // Determine the number system for the S-faces within panes
257 void Overlay::
258 number_subfaces() {
259  std::vector<RFC_Pane_overlay*> b_ps, g_ps;
260  B->panes( b_ps); G->panes( g_ps);
261 
262  typedef std::map< int, std::vector< int> > Subface_counts;
263  Subface_counts cnts_b, cnts_g;
264 
265  std::vector<RFC_Pane_overlay*>::iterator pi;
266  for ( pi=b_ps.begin(); pi!=b_ps.end(); ++pi)
267  cnts_b[(*pi)->id()].resize( (*pi)->size_of_faces(), 0);
268  for ( pi=g_ps.begin(); pi!=g_ps.end(); ++pi)
269  cnts_g[(*pi)->id()].resize( (*pi)->size_of_faces(), 0);
270 
271  // First, count the number of S-faces by looping through all the blue faces
272  for ( pi=b_ps.begin(); pi!=b_ps.end(); ++pi) {
273  RFC_Pane_overlay *pane_b = *pi;
274  Subface_counts::iterator cnts_it_b = cnts_b.find(pane_b->id());
275  Subface_counts::iterator cnts_it_g = cnts_g.begin();
276 
277  for (int i=1; i<=pane_b->size_of_faces(); ++i) {
278  Face fi(pane_b, i);
279 
280  // Construct a list of nodes for the blue face
281  INode_const_list nodes;
282  get_inodes_of_face( fi, nodes);
283  RFC_assertion( nodes.size() > 2);
284 
285  Subface_list sub_faces;
286  // subdivide the face
287  bool ret = subdivide( nodes, ++nodes.begin(), sub_faces, BLUE);
288  if ( ret) {
289  std::cerr << "ERROR: Got error code " << ret
290  << " when subdividing face " << i << " with node "
291  << fi.halfedge().origin_l().id()
292  << " in pane " << pane_b->id() << std::endl;
293 
294  sub_faces.clear();
295  subdivide( nodes, ++nodes.begin(), sub_faces, BLUE);
296  RFC_assertion(!ret); abort();
297  }
298  if (sub_faces.empty()) {
299  std::cerr << "ERROR: Error in enumerating subface in face "
300  << i << " in pane " << pane_b->id() << std::endl;
301  RFC_assertion(!sub_faces.empty()); abort();
302  }
303 
304  // Count the subfaces in the blue face
305  for (Subface_list::iterator
306  si=sub_faces.begin(), send=sub_faces.end(); si != send; ++si) {
307  int num_tris = si->size()-2;
308  HEdge s = get_parent_face( *si, GREEN);
309  RFC_Pane_overlay *pane_g = s.pane();
310  if ( cnts_it_g->first != pane_g->id())
311  cnts_it_g=cnts_g.find(pane_g->id());
312 
313  cnts_it_b->second[fi.id()-1] += num_tris;
314  cnts_it_g->second[s.face().id()-1] += num_tris;
315  }
316  }
317  }
318 
319  // Allocate space for subfaces and convert counts into offsets.
320  for ( pi=b_ps.begin(); pi!=b_ps.end(); ++pi) {
321  std::vector< int> &cnts = cnts_b[ (*pi)->id()];
322  (*pi)->allocate_subfaces( cnts);
323  if ( cnts.empty()) continue;
324  // Convert counts into offsets
325  int t1=cnts[0];
326  cnts[0] = 0;
327  for ( int i=1, n=cnts.size(); i<n; ++i) {
328  int t2 = cnts[i];
329  cnts[i] = cnts[i-1] + t1;
330  t1 = t2;
331  }
332  }
333 
334  for ( pi=g_ps.begin(); pi!=g_ps.end(); ++pi) {
335  std::vector< int> &cnts = cnts_g[ (*pi)->id()];
336  (*pi)->allocate_subfaces( cnts);
337  if ( cnts.empty()) continue;
338  // Convert counts into offsets
339  int t1=cnts[0];
340  cnts[0] = 0;
341  for ( int i=1, n=cnts.size(); i<n; ++i) {
342  int t2 = cnts[i];
343  cnts[i] = cnts[i-1] + t1;
344  t1 = t2;
345  }
346  }
347 
348  Subface_counts &offsets_b=cnts_b, &offsets_g=cnts_g;
349 
350  // Third, we fill up the arrays for face-list and etc.
351  for ( pi=b_ps.begin(); pi!=b_ps.end(); ++pi) {
352  RFC_Pane_overlay *pane_b = *pi;
353  const int pane_id_b = pane_b->id();
354 
355  Subface_counts::iterator offsets_it_b = offsets_b.find(pane_b->id());
356  Subface_counts::iterator offsets_it_g = offsets_g.begin();
357 
358  for (int i=1; i<=(*pi)->size_of_faces(); ++i) {
359  Face fi(*pi, i);
360 
361  // Construct a list of nodes for the blue face
362  INode_const_list nodes;
363  get_inodes_of_face( fi, nodes);
364  RFC_assertion( nodes.size() > 2);
365 
366  Subface_list sub_faces;
367  // subdivide the face
368  RFC_assertion_code( bool ret = )
369  subdivide( nodes, ++nodes.begin(), sub_faces, BLUE);
370  RFC_assertion( !ret);
371 
372  HEdge b=fi.halfedge();
373  Generic_element e_b( count_edges(b));
374 
375  // output the subfaces of the blue face
376  for (Subface_list::iterator
377  si=sub_faces.begin(), send=sub_faces.end(); si != send; ++si) {
378  int n = si->size(), num_tris = n-2;
379 
380  HEdge g = get_parent_face( *si, GREEN);
381  Generic_element e_g( count_edges(g));
382 
383  std::vector< Point_2> pnts_b( n), pnts_g( n);
384  for ( int k=0; k<n; ++k) {
385  pnts_b[k] = get_nat_coor( *(*si)[k], e_b, b, BLUE);
386  pnts_g[k] = get_nat_coor( *(*si)[k], e_g, g, GREEN);
387  }
388 
389  std::vector< Three_tuple<int> > tris( num_tris);
390  Triangulation triangulation;
391  // Triangulate the sub-face in both B and G.
392  triangulation.triangulate( &pnts_b[0], n, &tris);
393 
394  RFC_Pane_overlay *pane_g = g.pane();
395  const int pane_id_g = pane_g->id();
396  // Insert the triangles into the blue and green panes
397  for ( int k=0; k<num_tris; ++k) {
398  int lids_b[3], lids_g[3];
399  RFC_Pane_overlay::ParentEdge_ID eids_b[3], eids_g[3];
400  Point_2 ncs_b[3], ncs_g[3];
401 
402  for ( int j=0; j<3; ++j) {
403  COM_assertion_msg( (*si)[tris[k][j]], "Empty intersection.");
404 
405  convert_nat_coordinates( (*si)[tris[k][j]], b, BLUE, pane_id_b,
406  lids_b[j], eids_b[j], ncs_b[j]);
407  convert_nat_coordinates( (*si)[tris[k][j]], g, GREEN, pane_id_g,
408  lids_g[j], eids_g[j], ncs_g[j]);
409  }
410 
411  if ( offsets_it_g->first != pane_g->id())
412  offsets_it_g=offsets_g.find(pane_g->id());
413 
414  int idx_b = offsets_it_b->second[fi.id()-1]++;
415  int idx_g = offsets_it_g->second[g.face().id()-1]++;
416 
417  pane_b->insert_subface(idx_b, pane_b->get_lid( b.face()),
418  lids_b, eids_b, ncs_b,
419  pane_id_g, idx_g+1, lids_g);
420  pane_g->insert_subface(idx_g, pane_g->get_lid( g.face()),
421  lids_g, eids_g, ncs_g,
422  pane_id_b, idx_b+1, lids_b);
423  }
424  }
425  }
426  }
