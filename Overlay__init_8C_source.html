25 // The initialization for the overlay algorithm. It locates the
26 // green parent of a blue vertex and create an inode for it.
27 // This initialization step takes linear time.
28 // Returns NULL if unsuccessful. Otherwise, creates a new INode.
29 INode *Overlay::overlay_init() {
30  INode *v=NULL;
31 
32  // get an unmarked halfedge from the blue mesh and take its origin as x.
33  HEdge b=B->get_an_unmarked_halfedge(), g;
34  if ( b.pane()==NULL) return NULL;
35 
36  Node x = b.origin_l();
37 
38  // locate the green parent of x.
39  Parent_type t;
40  Point_2 nc;
41  get_green_parent( x, b, &g, &t, &nc);
42  RFC_assertion ( t != PARENT_NONE && g.pane() != NULL);
43 
44  // create a new inode for x
45  v = new INode();
46  v->set_parent( b, Point_2(0,0), BLUE);
47  v->set_parent( g, nc, GREEN);
48 
49  if ( verbose2) {
50  std::cout << "\nFound a seed at blue vertex ("
51  << x.pane()->get_point(x) << ") \nin green ";
52  const RFC_Pane_overlay *gpane=g.pane();
53  switch ( t) {
54  case PARENT_VERTEX:
55  std::cout << "vertex (" << gpane->get_point( g.origin_l()) << ")\n";
56  break;
57  case PARENT_EDGE:
58  std::cout << "edge (" << gpane->get_point( g.origin_l()) << "), ("
59  << gpane->get_point( g.destination_l()) << ")\n";
60  break;
61  case PARENT_FACE: {
62  std::cout << "face ";
63  HEdge gt = g;
64  do {
65  std::cout << "\t(" << gpane->get_point( gt.origin_l()) << "), ";
66  } while ( (gt = gt.next_l()) != g);
67  std::cout << std::endl;
68  break;
69  }
70  default:
71  RFC_assertion( false);
72  }
73  std::cout << std::endl;
74  }
75 
76  return v;
77 }

79 // Get the green parent of a vertex v. This subroutine takes
80 // linear time in terms of the number of vertices in G.
81 void Overlay::
82 get_green_parent( const Node &v,
83  const HEdge &b,
84  HEdge *h_out,
85  Parent_type *t_out,
86  Point_2 *nc)
87 {
88  const Point_3 &p = v.pane()->get_point(v);
89  Real sq_dist = 1.e30;
90  Node w;
91 
92  //=================================================================
93  // Locate the closest green vertex
94  //=================================================================
95  std::vector<RFC_Pane_overlay*> ps;
96  G->panes( ps);
97  // Loop through all the panes of G
98  for ( std::vector<RFC_Pane_overlay*>::iterator
99  pit=ps.begin(); pit != ps.end(); ++pit) {
100  RFC_Pane_overlay *pane = *pit;
101  // loop through all the green vertices that are complete
102  for ( int i=1; i<=pane->size_of_nodes(); ++i) {
103  Node n(pane, i);
104 
105  if ( !n.is_isolated() && n.halfedge_l().destination_l() == n) {
106  // if the distance is closer than previous ones, save it
107  Real sq_d = ( p- pane->get_point(n)).squared_norm();
108  if ( sq_d < sq_dist) { sq_dist = sq_d; w = n; }
109  }
110  }
111  }
112 
113  if ( w.pane() == NULL) return;
114 
115  RFC_assertion( w.is_primary()); // Because we start from smaller ids.
116  // Let h be a nonborder incident halfedge of w
117  HEdge h = w.halfedge_g();
118  h = !h.is_border_g() ? h.next_g() : h.opposite_g();
119 
120  // We perform breadth-first search starting from h
121  std::queue< HEdge> q;
122  std::list< HEdge> hlist;
123 
124  q.push( h);
125  // Mark the halfedges in the same face as h
126  HEdge h0 = h;
127  do { RFC_assertion( !acc.marked(h)); acc.mark( h); hlist.push_back( h); }
128  while ( (h=h.next_g()) != h0);
129 
130  Vector_3 vec(0.,0.,0.);
131  while ( !q.empty()) {
132  h = q.front(); q.pop();
133 
134  *t_out = PARENT_NONE; *h_out=h;
135  if ( op.project_onto_element( p, h_out, t_out, vec, nc, eps_e) &&
136  std::abs(acc.get_normal(b)* acc.get_normal(h)) >= 0.6)
137  break;
138 
139  // Insert the incident faces of h into the queue
140  h0 = h;
141  do {
142  HEdge hopp=h.opposite_g();
143  if ( !hopp.is_border_g() && ! acc.marked( hopp)) {
144  HEdge h1 = hopp;
145  q.push( h1);
146  do {
147  acc.mark(h1); hlist.push_back( h1);
148  } while ((h1=h1.next_g())!=hopp);
149  }
150  } while ( (h=h.next_g()) != h0);
151  }
152  // Unmark the halfedges
153  while ( !hlist.empty())
154  { acc.unmark( hlist.front()); hlist.pop_front(); }
155 
156  // If v is too far from p_out, return NULL.
157  vec = op.get_point( *h_out, *nc) - p;
158  if ( vec * vec > sq_length( *h_out) + sq_length( v.halfedge_l())) {
159  *h_out = HEdge(); *t_out = PARENT_NONE;
160  }
161  else {
162  // Determine whether the two surfaces are facing each other
163  is_opposite = ( b.pane()->get_normal( b, v) *
164  op.get_face_normal( *h_out, *nc)) < 0;
165  }
166 }
