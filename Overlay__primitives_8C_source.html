62 // This function computes the intersection of the edge [b.org,b.dst]
63 // with [g.org,g.dst]. It assumes that the two edges are not
64 // parallel. The output c1 and c2 are parameterization of
65 // the locations of the intersection. It snaps the intersections
66 // onto vertices if their parameterizations are close to 0 or 1.
67 // Returns true iff both |*c1| and |*c2| are in [0,1].
68 bool Overlay_primitives::
69 intersect( HEdge hb,
70  const HEdge &hg,
71  Real start,
72  Real *c1,
73  Real *c2,
74  Real *orien_match,
75  Real *normal_match,
76  bool is_opposite,
77  Real eps_e) const {
78  RFC_assertion( !hg.is_border_l());
79 
80  // Switch blue edge if it is a border edge to use Point_set.
81  bool is_b_border = hb.is_border_g();
82  if ( is_b_border) hb = hb.opposite_g();
83 
84  const Point_set pbs(hb), pgs(hg);
85  const Normal_set ngs(hg);
86 
87  *c1 = start; // At input, set *c1 to start
88  intersect( pbs, pgs, ngs, c1, c2, is_b_border);
89 
90  if ( std::fabs( *c1) < 1/eps_e && std::fabs( *c2) < 1/eps_e) {
91  Vector_3 v2 = acc.get_normal( hg);
92  Vector_3 v1 = acc.get_normal( hb);
93  double prod1 = v1 * v2;
94 
95  v1 = acc.get_normal( hb.opposite_g());
96  double prod2 = v1 * v2;
97 
98  // Reverse the sign if the two projdirs having different directions.
99  if ( is_opposite)
100  { prod1 = -prod1; prod2 = -prod2; }
101 
102  *normal_match = std::max(prod1, prod2);
103 
104  // Finally, we determine c1 and c2.
105  if ( std::fabs(*c1-1) <= eps_e) *c1 = 1.;
106  else if ( std::fabs(*c1) <= eps_e) *c1 = 0.;
107 
108  if ( std::fabs(*c2-1) <= eps_e) *c2 = 1.;
109  else if ( std::fabs(*c2) <= eps_e) *c2 = 0.;
110 
111  if ( orien_match) {
112  Vector_3 J[3];
113  J[0] = get_tangent( hb, *c1);
114  J[1] = get_tangent( hg, *c2);
115  J[2] = get_projdir( hg, *c2);
116  *orien_match = Vector_3::cross_product( J[0], J[1]) * J[2] /
117  std::sqrt( (J[0]*J[0]) * (J[1]*J[1]) * (J[2]*J[2]));
118 
119  if (is_b_border) *orien_match = -*orien_match;
120  }
121 
122  // Switch edge back
123  if ( is_b_border) { *c1 = 1-*c1; }
124 
125  return *c1 >= 0. && *c1 <= 1. && *c2 >= 0. && *c2 <= 1.;
126  }
127  else
128  return false;
129 }

131 // The intersection is a quadratic function and in general it has two
132 // solutions. We pick the solution based on s (the parameterization on blue
133 // edge) and t (the parameterization on green edge) as follows:
134 // If any t is far less than 0 or far greater than 1, ignore this solution.
135 // If both s1 and s2 are greater than start, then take smaller one.
136 // If both are smaller than the start, then no intersection.
137 // If one is greater and the other is smaller, then return the greater one.
138 // In addition, it safeguards errors by comparing the residual against
139 // the longer of the green and blue edges.
140 void Overlay_primitives::
141 intersect( const Point_set &pbs,
142  const Point_set &pgs,
143  const Normal_set &ngs,
144  Real *c1,
145  Real *c2, bool reversed) const {
146 
147  Vector_3 vb = pbs[1]-pbs[0], vg=pgs[1]-pgs[0], vn=ngs[1]-ngs[0];
148  Vector_3 vd = pgs[0]-pbs[0];
149 
150  Real c = Vector_3::cross_product( vb, ngs[0])*vd;
151  Real d = Vector_3::cross_product( vb, ngs[1])*( pgs[1]-pbs[0]);
152  Real a = Vector_3::cross_product( vb, vn) * vg;
153  Real b = d - c - a;
154 
155  // We now need to solve a*t^2 + b*t + c =0 to get t. First scale a, b, and c.
156  Real max_abc = std::max( std::max(std::fabs(a), std::fabs(b)), std::fabs(c));
157  if ( max_abc > 0)
158  { a /= max_abc; b /= max_abc; c /= max_abc; d /= max_abc; }
159 
160  Real s, t;
161  Real eps = 0.1, tol_fea = 1;
162 
163  if ( a == 0) {
164  // The two edges are parallel.
165  if ( b == 0)
166  { *c1 = *c2 = HUGE_VAL; return; }
167 
168  t = -c/b;
169 
170  if ( t >= -tol_fea && t <= 1+tol_fea) {
171  Real denominator = (1-t) * (Vector_3::cross_product( vg, ngs[0]) * vb) +
172  t * (Vector_3::cross_product( vg, ngs[1]) * vb);
173  Real nominator = (1-t) * (Vector_3::cross_product( vg, ngs[0]) * vd) +
174  t * (Vector_3::cross_product( vg, ngs[1]) * vd);
175  s = nominator / denominator;
176  }
177  else
178  { *c1 = *c2 = HUGE_VAL; return; }
179  }
180  else {
181  Real det = b*b - 4*a*c;
182  if ( det < 0)
183  { *c1 = *c2 = HUGE_VAL; return; }
184 
185  Real sqrt_d = std::sqrt( det);
186  Real s1, s2, t1, t2;
187 
188  if ( b>0) {
189  Real temp = -b - sqrt_d;
190  t1 = 2*c / temp;
191  t2 = temp / 2/ a;
192  }
193  else {
194  Real temp = -b + sqrt_d;
195  t1 = temp / 2 / a;
196  t2 = 2*c / temp;
197  }
198 
199  if ( std::fabs(t1 -1.) < eps || std::fabs(t2 -1.) < eps ) {
200  // Use alternative formula to compute intersection
201  Real b = -(c - d - a);
202  Real sqrt_d = std::sqrt( b*b - 4*a*d);
203  if ( det < 0)
204  { *c1 = *c2 = HUGE_VAL; return; }
205  if ( b>0) {
206  Real temp = -b - sqrt_d;
207  if (std::fabs(t1 - 1.) < eps) t1 = 1 + 2*d / temp;
208  if (std::fabs(t2 - 1.) < eps) t2 = 1 + temp / 2 / a;
209  }
210  else {
211  Real temp = -b + sqrt_d;
212  if (std::fabs(t1 - 1.) < eps) t1 = 1 + temp / 2 / a;
213  if (std::fabs(t2 - 1.) < eps) t2 = 1 + 2*d / temp;
214  }
215  }
216 
217  if ( t1 >= -tol_fea && t1 <= 1+tol_fea) {
218  Real denominator = (1-t1) * (Vector_3::cross_product( vg, ngs[0]) * vb) +
219  t1 * (Vector_3::cross_product( vg, ngs[1]) * vb);
220  Real numerator = (1-t1) * (Vector_3::cross_product( vg, ngs[0]) * vd) +
221  t1 * (Vector_3::cross_product( vg, ngs[1]) * vd);
222  s1 = numerator / denominator;
223  }
224  else
225  s1 = reversed ? HUGE_VAL : -HUGE_VAL;
226 
227  if ( t2 >= -tol_fea && t2 <= 1+tol_fea) {
228  Real denominator = (1-t2) * (Vector_3::cross_product( vg, ngs[0]) * vb) +
229  t2 * (Vector_3::cross_product( vg, ngs[1]) * vb);
230  Real numerator = (1-t2) * (Vector_3::cross_product( vg, ngs[0]) * vd) +
231  t2 * (Vector_3::cross_product( vg, ngs[1]) * vd);
232  s2 = numerator / denominator;
233  }
234  else
235  s2 = reversed ? HUGE_VAL : -HUGE_VAL;
236 
237  double start = *c1-eps;
238 
239  if ( !reversed) {
240  if ( s1 < start && s2 < start)
241  { *c1 = *c2 = HUGE_VAL; return; }
242  else if ( s2 < start || s1 >= start && s1 <= s2)
243  { s = s1; t = t1; }
244  else
245  { s = s2; t = t2; }
246  }
247  else {
248  if ( 1-s1 < start && 1-s2 < start)
249  { *c1 = *c2 = -HUGE_VAL; return; }
250  else if ( 1-s2 < start || 1-s1 >= start && 1-s1 <= 1-s2)
251  { s = s1; t = t1; }
252  else
253  { s = s2; t = t2; }
254  }
255  }
256 
257  // Check residule of the solution and make sure the direction between
258  // the two points is reasonably close to the projection direction
259  Generic_element e(3);
260  Vector_3 v2 = e.interpolate( ngs, t);
261  Vector_3 v3 = e.interpolate( pbs, s)-e.interpolate( pgs, t);
262  Real error = (v3-(v3*v2)/(v2*v2)*v2).squared_norm() / std::max( vb*vb,vg*vg);
263  if ( error >= _eps_p*_eps_p) {
264 #ifdef DEBUG
265  std::cerr << "WARNING: Rejecting intersection due to too large error "
266  << error << ". Solution was " << s << "," << t << std::endl;
267 #endif
268  *c1 = *c2 = HUGE_VAL;
269  }
270  else
271  { *c1 = s; *c2 = t; }
272 }

274 // Project a point p onto an element along a given direction.
275 // The projection direction is given by the input parameter dir.
276 // If the input is Vector_3(0,0,0), evaluate the direction by
277 // sum N_i*n_i. Note that g and pt are in-out parameters.
278 bool
279 Overlay_primitives::
280 project_onto_element( const Point_3 &p,
281  HEdge *g,
282  Parent_type *pt,
283  Vector_3 dir,
284  Point_2 *nc_out,
285  const Real eps_p,
286  Real eps_np) const {
287  if ( eps_np < 0) eps_np = eps_p;
288  RFC_assertion( eps_p>_eps_c);
289  RFC_assertion( *pt != PARENT_VERTEX);
290 
291  // Construct two elements.
292  HEdge h(*g);
293  Point_set ps(h);
294  Normal_set ns(h);
295 
296  Point_2 nc(0.5,0.5);
297  Generic_element e(ps.size_of_edges()); // We ignore the midpoints
298  const bool comp_dir = ( dir == Vector_3(0,0,0));
299 
300  // Solve the equation using Newton's method for nonlinear equations.
301  Real t(0.);
302  RFC_assertion_code( Size i=0);
303  RFC_assertion_code( const Size max_nsteps = 20);
304  double err = 1.e30, eps_c2=_eps_c*_eps_c;
305  for ( ;;RFC_assertion(++i<max_nsteps)) {
306  if ( comp_dir) e.interpolate( ns, nc, &dir);
307 
308  // evaluate f
309  Vector_3 f = e.interpolate( ps, nc) - p + t*dir;
310  double et = f*f / e.Jacobian_det( ps,nc);
311  if ( et <= eps_c2 || et >= err) break;
312  err = et;
313 
314  Vector_3 J[3];
315  e.Jacobian( ps, nc, J);
316  if ( comp_dir) {
317  Vector_3 J2[2];
318  e.Jacobian( ns, nc, J2);
319  J[0] = J[0] + t*J2[0];
320  J[1] = J[1] + t*J2[1];
321  }
322  J[2] = dir;
323 
324  // Solve the linear system Ju=f
325  Real u[3];
326  _solve_3x3( J[0][0], J[0][1], J[0][2], J[1][0], J[1][1], J[1][2],
327  J[2][0], J[2][1], J[2][2], f[0], f[1], f[2], u[0], u[1], u[2]);
328 
329  nc = Point_2( nc[0] - u[0], nc[1] - u[1]); t -= u[2];
330 
331  if ( std::fabs( nc[0]) > 1.e3 || std::fabs( nc[1]) > 1.e3 ) {
332  // System is diverging
333  break;
334  }
335  }
336 
337  // Perturb nc by eps_p.
338  nc = Point_2( ( nc[0] <= eps_p && nc[0] >= -eps_np)
339  ? 0. : ( ( nc[0]>=1.-eps_p && nc[0]<=1.+eps_np) ? 1 : nc[0]),
340  ( nc[1] <= eps_p && nc[1] >= -eps_np)
341  ? 0. : ( ( nc[1]>=1.-eps_p && nc[1]<=1.+eps_np) ? 1 : nc[1]));
342 
343  // if the projection is restricted onto the edge, return it now.
344  if ( *pt == PARENT_EDGE) {
345  if ( nc[0] < 0.) { nc[0] = nc[1] = 0; }
346  else if ( nc[0] > 1.) { nc[0] = nc[1] = 0; *g = (*g).next_l(); }
347  else nc = Point_2( nc[0], 0.);
348 
349  if ( nc[0] == 0 || nc[0] == 1) *pt = PARENT_VERTEX;
350  if ( nc_out) *nc_out = nc;
351  return true;
352  }
353 
354  // Otherwise, determine whether the projection is in the interior,
355  // on the edge, or at a vertex.
356  switch ( ps.size_of_edges()) {
357  case 3: {
358  if ( (nc[0] + nc[1]) >= 1-eps_p && (nc[0] + nc[1]) <= 1+eps_np) {
359  // Projects onto the diagonal edge
360  if ( nc[0] < eps_p) {
361  *g = (*g).prev_l(); nc[0] = nc[1] = 0; *pt = PARENT_VERTEX;
362  }
363  else {
364  *g = (*g).next_l(); nc = Point_2( nc[1], 0.);
365  *pt = (nc[1] == 0.) ? PARENT_VERTEX : PARENT_EDGE;
366  }
367  }
368  else if ( nc[0] == 0.) {
369  // Projects onto the horizontal edge
370  if ( nc[1] == 0.)
371  { nc[0] = nc[1] = 0; *pt = PARENT_VERTEX; }
372  else
373  { *g = (*g).prev_l(); nc = Point_2( 1.-nc[1], 0.); *pt = PARENT_EDGE; }
374  }
375  else if ( nc[1] == 0.)
376  // Projects onto the vertical edge
377  { nc = Point_2( nc[0], 0.); *pt = PARENT_EDGE; }
378  else
379  { *pt = PARENT_FACE; }
380  if ( nc_out) *nc_out = nc;
381 
382  return ( nc[0] >= 0. && nc[1] >= 0. && nc[0]+nc[1] <= 1.);
383  }
384  case 4: {
385  // Projects onto the left edge
386  if ( nc[0] == 0.) {
387  if ( nc[1] == 0.)
388  { *pt = PARENT_VERTEX; }
389  else {
390  *g = (*g).prev_l(); nc = Point_2( 1.-nc[1], 0.);
391  *pt = (nc[1] == 1.) ? PARENT_VERTEX : PARENT_EDGE;
392  }
393  }
394  else if ( nc[0] == 1.) {
395  // Projects onto the right edge
396  if ( nc[1] == 1.)
397  { *g = (*g).next_l().next_l(); nc[0] = nc[1] = 0; *pt = PARENT_VERTEX;}
398  else {
399  *g = (*g).next_l(); nc = Point_2( nc[1], 0.);
400  *pt = (nc[1] == 0.) ? PARENT_VERTEX : PARENT_EDGE;
401  }
402  }
403  else if ( nc[1] == 0.)
404  // Projects onto the bottom edge
405  { *pt = PARENT_EDGE; }
406  else if ( nc[1] == 1.)
407  // Projects onto the top edge
408  { *g=(*g).next_l().next_l(); nc=Point_2(1.-nc[0],0.); *pt=PARENT_EDGE; }
409  else
410  { *pt = PARENT_FACE; }
411 
412  if ( nc_out) *nc_out = nc;
413 
414  return ( nc[0] >= 0. && nc[0] <= 1. && nc[1] >= 0. && nc[1] <= 1.);
415  }
416  default:
417  break;
418  }
419 
420  return false;
421 }

450 // Solve the equation p=q1+s*(q2-q1)+u*(n1+s*(n2-n1))+v*(b1+s*(b2-b1))
451 // for s, u, and v and return s;
452 Real
453 Overlay_primitives::
454 project_blue_feature( const Vector_3 &n1, const Vector_3 &n2,
455  const Vector_3 &t1, const Vector_3 &t2,
456  const Point_3 &q1, const Point_3 &q2,
457  const Point_3 &p,
458  const Real eps_e) const {
459  RFC_assertion( eps_e>_eps_c);
460  Real s=0, u=0, v=0;
461  Vector_3 b1 = Vector_3::cross_product( n1, t1);
462  Vector_3 b2 = Vector_3::cross_product( n2, t2);
463  Vector_3 qdiff=q2-q1, bdiff=b2-b1, ndiff=n2-n1;
464 
465  RFC_assertion_code( Size i=0);
466  RFC_assertion_code( const Size max_nsteps = 20);
467  double err = 1.e30, eps_c2=_eps_c*_eps_c;
468  for ( ;;RFC_assertion(++i<max_nsteps)) {
469  Vector_3 c2 = (s*ndiff += n1);
470  Vector_3 c3 = (s*bdiff += b1);
471 
472  Vector_3 f = ((q1-p) += s*qdiff += u*c2 += v*c3);
473  double et = f*f / (qdiff*qdiff);
474 
475  if ( et <= eps_c2 || et >= err) break;
476  err = et;
477 
478  // Vector_3 c1 = (q2-q1) + u*(n2-n1) + v*(b2-b1);
479  Vector_3 c1 = (u*ndiff += v*bdiff += qdiff);
480  Real x[3];
481  _solve_3x3( c1[0], c1[1], c1[2], c2[0], c2[1], c2[2],
482  c3[0], c3[1], c3[2], f[0], f[1], f[2], x[0], x[1], x[2]);
483  s -= x[0]; u -= x[1]; v -= x[2];
484  }
485 
486  if ( std::fabs(s) <= eps_e)
487  s = 0;
488  else if ( std::fabs(1.-s) <= eps_e)
489  s = 1.;
490 
491  return s;
492 }
