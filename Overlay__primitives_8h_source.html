85  for ( unsigned int i=0, s=hs.size(); i<s; ++i)
86  { hs[i] = h; h = h.next_l(); }

172  // Helpers for evaluating physical coordinates, tangents and projdirs
173  Point_3 get_point( const HEdge &b, const Point_2S &nc) const {
174  if ( b.is_border_l())

189  Point_3 get_point( const HEdge &b, Real a) const {
190  // Here, we assume edge is straight line.

192  return b.origin_l().point() + a*(b.destination_l().point()-

208  enum { AREA_WEIGHTED=0, UNIT_WEIGHT=1, ANGLE_WEIGHTED=2 };
209 
210  // Note that the normal vector returned is not normalized.
211  // Use area-weighted by default.
212  Vector_3 get_face_normal( const HEdge &b, const Point_2 &nc,
213  int scheme=0) const {
214  // Evaluate the normal as the cross product of the colume vectors
215  // of its Jacobian matrix.
216  Point_set ps(b);
217  Size ne = ps.size_of_edges(), nv=ne;
218  Vector_3 J[2];
219 
220  Generic_element( ne, nv).Jacobian( ps, nc, J);
221 
222  Vector_3 nrm = Vector_3::cross_product( J[0], J[1]);
223  if ( scheme == AREA_WEIGHTED) return nrm;
224  nrm.normalize();
225  if ( scheme == UNIT_WEIGHT) return nrm;
226 
227  double cosa = J[0]*J[1] / std::sqrt((J[0]*J[0])*(J[1]*J[1]));
228  if ( cosa>1) cosa = 1; else if ( cosa<-1) cosa = -1;
229 
230  return std::acos(cosa) * nrm;
231  }
