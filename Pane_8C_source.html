247 void Pane::
248 reinit_dataitem( int aid, OP_Init op, void **addr,
249  int strd, int cap) throw (COM_exception)
250 {
251  switch ( aid) {
252  case COM_CONN: {
253  COM_assertion( op != OP_SET && op != OP_SET_CONST);
254  // loop through the connectivity
255  for ( int i=0, n=_cnct_set.size(); i<n; ++i) {
256  if ( !_cnct_set[i]->is_structured())
257  reinit_conn( _cnct_set[i], op, NULL, strd, cap);
258  }
259 
260  if ( addr) *addr = NULL;
261  return;
262  }
263  case COM_MESH: {
264  COM_assertion( op != OP_SET && op != OP_SET_CONST);
265  reinit_dataitem( COM_NC, op, NULL, strd, cap);
266  reinit_dataitem( COM_CONN, op, NULL, strd, cap);
267  reinit_dataitem( COM_RIDGES, op, NULL, strd, cap);
268  if ( addr) *addr = NULL;
269  return;
270  }
271  case COM_PMESH: {
272  COM_assertion( op != OP_SET && op != OP_SET_CONST);
273  reinit_dataitem( COM_NC, op, NULL, strd, cap);
274  reinit_dataitem( COM_CONN, op, NULL, strd, cap);
275  reinit_dataitem( COM_RIDGES, op, NULL, strd, cap);
276  reinit_dataitem( COM_PCONN, op, NULL, strd, cap);
277  if ( addr) *addr = NULL;
278  return;
279  }
280  case COM_DATA: {
281  COM_assertion( op != OP_SET && op != OP_SET_CONST);
282  // Loop through the dataitems
283  for ( int i=COM_NUM_KEYWORDS, n=_attr_set.size(); i<n; ++i) {
284  DataItem *a = _attr_set[i];
285  if ( a==NULL || a->pane()==NULL) continue;
286  if ( !a->is_windowed() || _id==0)
287  reinit_dataitem( i, op, NULL, strd, cap);
288  int ncomp= a->size_of_components();
289  if ( ncomp>1) i+= ncomp;
290  }
291  return;
292  }
293  case COM_ALL: {
294  COM_assertion( op != OP_SET && op != OP_SET_CONST);
295  reinit_dataitem( COM_PMESH, op, NULL, strd, cap);
296  reinit_dataitem( COM_DATA, op, NULL, strd, cap);
297  return;
298  }
299  default: ;
300  }
301 
302  DataItem *a = dataitem( aid);
303  if ( a==NULL)
304  throw COM_exception( COM_ERR_DATAITEM_NOTEXIST,
305  append_frame(a->fullname(), Pane::reinit_dataitem));
306 
307  // If _id is 0, then handle only window dataitems.
308  if ( !a->is_windowed() && !_id) {
309  if (op == OP_SET || op == OP_SET_CONST)
310  throw COM_exception( COM_ERR_PANE_NOTEXIST, append_frame
311  ( _window->name(), Pane::reinit_dataitem));
312  else
313  return;
314  }
315 
316  int errcode;
317  void *p;
318  int ncomp = a->size_of_components();
319 
320  if ( op==OP_DEALLOC) {
321  // Deallocate if was allocated by Roccom
322  errcode = a->deallocate();
323  p = NULL;
324  }
325  else {
326  COM_assertion( op==OP_SET_CONST||op==OP_SET||op==OP_ALLOC||op==OP_RESIZE);
327 
328  // If size is not yet set
329  if ( !a->size_set()) {
330  if ( a->is_windowed())
331  a->set_size(1); // Change the default size of window dataitems to 1
332  else // Throw an exception for other types of dataitems
333  throw COM_exception( COM_ERR_INVALID_SIZE,
334  append_frame(a->fullname(), Pane::reinit_dataitem));
335  }
336 
337  // Assign default value for cap and strd
338  if ( cap==0) {
339  // Default value for capacity is the number of items for set_array
340  // but the large of the current capacity and the the number of items.
341  if ( op==OP_SET || op==OP_SET_CONST) cap = a->size_of_items();
342  else {
343  if ( a->capacity()==0)
344  cap = a->size_of_items();
345  else if ( a->size_of_items()<=a->capacity())
346  cap = a->capacity();
347  else
348  cap = a->size_of_items()+a->size_of_items()/5; // Current size pluse 20% more
349  }
350  }
351 
352  if ( (op == OP_SET || op == OP_SET_CONST) && *addr && cap==0)
353  *addr = NULL; // Enforce the array to be a NULL pointer
354 
355  // Make the dataitem itself set to NULL if operating on its components.
356  if ( DataItem::is_digit(a->name()[0])) {
357  int icomp=std::atoi( a->name().c_str());
358  if ( dataitem(aid-icomp)->initialized())
359  throw COM_exception(COM_ERR_WAS_INITIALIZED,
360  append_frame(a->fullname(), Pane::reinit_dataitem));
361  }
362 
363  if ( strd==0 || strd<0 && a->stride()==0)
364  // Default value for stride is number of components
365  strd = ncomp;
366  else if ( strd<0)
367  strd = a->stride();
368 
369  if ( op == OP_SET || op == OP_SET_CONST) {
370  ((DataItem_friend*)a)->set_pointer(*addr, strd, cap,
371  0, op==OP_SET_CONST);
372  p = *addr;
373  }
374  else {
375  p = a->allocate( strd, cap, op == OP_ALLOC);
376  if ( addr) *addr = p;
377  }
378  errcode=0;
379  }

382 void Pane::
383 reinit_conn( Connectivity *con, OP_Init op, int **addr,
384  int strd, int cap) throw (COM_exception)
385 {
386  // Assign default value for cap and strd
387  if ( op!=OP_DEALLOC) {
388  if ( cap==0) {
389  // Default value for capacity is the number of items for set_array
390  // but the large of the current capacity and the the number of items plus some buffer.
391  if ( op==OP_SET || op==OP_SET_CONST) cap = con->size_of_items();
392  else {
393  if ( con->capacity()==0)
394  cap = con->size_of_items();
395  else if ( con->size_of_items()<=con->capacity())
396  cap = con->capacity();
397  else
398  cap = con->size_of_items()+con->size_of_items()/5; // Current size pluse 20% more
399  }
400  }
401 
402  if ( strd==0 || strd<0 && con->stride()==0)
403  // Default value for stride is number of components
404  strd = con->size_of_components();
405  else if ( strd<0)
406  strd = con->stride();
407  }
408 
409  if ( op == OP_SET || op == OP_SET_CONST) {
410  if ( !id())
411  throw COM_exception( COM_ERR_PANE_NOTEXIST,
412  append_frame(con->fullname(),Pane::reinit_conn));
413 
414  if ( *addr && cap==0 && !con->is_structured())
415  *addr = NULL; // Enforce the array to be a NULL pointer
416 
417  ((Connectivity_friend*)con)->set_pointer( *addr, strd, cap,
418  op==OP_SET_CONST);
419  }
420  else {
421  if ( con->is_structured())
422  throw COM_exception( COM_ERR_ALLOC_STRUCTURED,
423  append_frame(con->fullname(),Pane::reinit_conn));
424  void *p=NULL;
425  if ( op==OP_DEALLOC)
426  { con->deallocate(); p=NULL; }
427  else
428  p = con->allocate( strd, cap, op==OP_ALLOC);
429 
430  if ( addr) *addr = (int*)p;
431  }
432 }

434 DataItem *Pane::inherit( DataItem *from, const std::string &aname,
435  int mode, bool withghost) throw(COM_exception)
436 {
437  if ( from==NULL)
438  throw COM_exception( COM_ERR_DATAITEM_NOTEXIST, append_frame
439  (_window->name()+"."+aname,Pane::inherit));
440 
441  switch (from->id()) { // Process keywords
442  case COM_ALL:
443  inherit( from->pane()->dataitem( COM_PMESH), "", mode, withghost);
444  inherit( from->pane()->dataitem( COM_DATA), "", mode, withghost);
445  return dataitem( COM_ALL);
446  case COM_MESH:
447  inherit( from->pane()->dataitem( COM_NC), "", mode, withghost);
448  inherit( from->pane()->dataitem( COM_CONN), "", mode, withghost);
449  inherit( from->pane()->dataitem( COM_RIDGES), "", mode, withghost);
450  return dataitem( COM_MESH);
451  case COM_PMESH:
452  inherit( from->pane()->dataitem( COM_NC), "", mode, withghost);
453  inherit( from->pane()->dataitem( COM_CONN), "", mode, withghost);
454  inherit( from->pane()->dataitem( COM_RIDGES), "", mode, withghost);
455  inherit( from->pane()->dataitem( COM_PCONN), "", mode, withghost);
456  return dataitem( COM_PMESH);
457  case COM_DATA: {
458  // Loop through all dataitems
459  std::vector<DataItem*> &atts=from->pane()->_attr_set;
460 
461  for ( int i=COM_NUM_KEYWORDS,n=atts.size(); i<n; ++i) {
462  DataItem *a = atts[i];
463  if ( a==NULL || a->pane()==NULL) continue;
464  if ( !a->is_windowed() || _id==0)
465  inherit( a, "", mode, withghost);
466  int ncomp= a->size_of_components();
467  if ( ncomp>1) i+= ncomp;
468  }
469  return dataitem( COM_DATA);
470  }
471  case COM_CONN: {
472  if ( _id==0) return dataitem(COM_CONN);
473 
474  if (mode != INHERIT_COPY) {
475  // Clean up the current connectivity tables
476  for ( Size i=0; i<_cnct_set.size(); ++i) delete _cnct_set[i];
477  _cnct_set.clear();
478 
479  if ( mode==INHERIT_USE) {
480  ((DataItem_friend*)_attr_set[COM_CONN])->inherit( from, false, withghost);
481  // Check that if structured meshes, the nodal coordinates
482  // was also used.
483  COM_assertion( !is_structured() || _attr_set[COM_NC]->parent() &&
484  _attr_set[COM_NC]->parent()->pane() == from->pane());
485  }
486 
487  // Inherit individual connectivity tables
488  std::vector<Connectivity*>::iterator it=from->pane()->_cnct_set.begin();
489  std::vector<Connectivity*>::iterator iend=from->pane()->_cnct_set.end();
490  for ( ; it != iend; ++it) {
491  _cnct_set.push_back( new Connectivity( this, *it, (*it)->name(),
492  -_cnct_set.size()-1));
493  ((Connectivity_friend*)_cnct_set.back())->inherit(*it, mode, withghost);
494  }
495 
496  // Set _ignore_ghost
497  _ignore_ghost = !withghost;
498  if ( _ignore_ghost && is_structured() && size_of_ghost_layers())
499  throw COM_exception(COM_ERR_GHOST_LAYERS, append_frame
500  (_window->name()+"."+aname,Pane::inherit));
501 
502  if ( mode == INHERIT_USE) return dataitem( COM_DATA);
503  }
504 
505  // Continue to copy mode
506  // Do not copy connectivity tables for structured meshes.
507  const std::vector<Connectivity*> &es = from->pane()->_cnct_set;
508  COM_assertion_msg( es.size() == _cnct_set.size(),
509  "Number of connectivity tables do not match");
510 
511  // Loop over the connectivity tables to copy each table
512  for ( int i=0, ni=_cnct_set.size(); i<ni; ++i) {
513  const Connectivity *conn=es[i];
514  int n=withghost?conn->size_of_items():conn->size_of_real_items();
515  _cnct_set[i]->copy_array( const_cast<int*>(conn->pointer()),
516  conn->stride(), n);
517  }
518 
519  refresh_connectivity();
520  return dataitem( COM_DATA);
521  }
522  default: ;
523  }
524 
525  std::string str = aname.size()?aname:from->name();
526 
527  // If from is a keyword and target is the same, then the current
528  // dataitem must have the same ID. Otherwise, must find from name.
529  DataItem *a=(from->id()<COM_NUM_KEYWORDS && aname.empty())?
530  dataitem(from->id()) : _window->dataitem(str);
531 
532  if ( mode != INHERIT_COPY) {
533  if ( _id == 0) {
534  if ( a==NULL || a->size_of_components()!=from->size_of_components()) {
535  // New dataitem only if it does not exist or the sizes do not match
536  try {
537  a=_window->new_dataitem( str, from->location(), from->data_type(),
538  from->size_of_components(), from->unit());
539 
540  if ( from->is_windowed())
541  ((DataItem_friend*)a)->inherit( from, mode, withghost);
542 
543  } CATCHEXP_APPEND(Pane::inherit);
544  }
545  else try {
546  ((DataItem_friend*)a)->inherit( from, mode, withghost);
547  } CATCHEXP_APPEND(Pane::inherit) CATCHBADALLOC_APPEND(Pane::inherit);
548 
549  if ( mode == INHERIT_USE || !from->is_windowed()) return a;
550  }
551  else {
552  COM_assertion( a);
553  a = _attr_set[a->id()];
554  try {
555  ((DataItem_friend*)a)->inherit( from, mode, withghost);
556  } CATCHEXP_APPEND(Pane::inherit) CATCHBADALLOC_APPEND(Pane::inherit);
557 
558  if ( mode == INHERIT_USE) return a;
559  }
560  }
561  else {
562  // Copy an dataitem only if it exists in the target window.
563  if ( a) a = _attr_set[a->id()];
564  else return a;
565 
566  if ( !DataItem::compatible_types( from->data_type(), a->data_type()))
567  throw COM_exception( COM_ERR_INCOMPATIBLE_TYPES, append_frame
568  (from->fullname()+" and "+a->fullname(),
569  Pane::inherit));
570  }
571 
572  if ( (_id==0) != (from->is_windowed())) return a;
573 
574  // Continue to copy dataset
575  if ( !withghost && is_structured() && size_of_ghost_layers())
576  throw COM_exception(COM_ERR_GHOST_LAYERS, append_frame
577  (_window->name()+"."+aname,Pane::inherit));
578 
579  // count is the number of panes, nodes, or elements to loop through
580  int count = withghost?from->size_of_items():from->size_of_real_items();
581  int s_nc = from->size_of_components();
582  const Pane *src_pane = from->pane();
583 
584  COM_assertion_msg( from->location() == a->location(),
585  (std::string("Location of dataitems ")+
586  from->fullname()+" and " +a->fullname()+
587  " do not match during copying.").c_str());
