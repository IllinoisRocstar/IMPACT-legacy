42 // A helper class for comparing two faces.
43 struct Four_tuple {
44  Four_tuple( int a, int b, int c, int d)
45  : _a(a), _b(b), _c(c), _d(d) {}
46  int &operator[](int i) { return (&_a)[i]; }
47  const int &operator[](int i) const { return (&_a)[i]; }
48  bool operator<( const Four_tuple &x) const {
49  return _a<x._a || (_a==x._a && (_b<x._b || _b==x._b &&
50  (_c<x._c || _c==x._c && _d<x._d)));
51  }
52 private:
53  int _a, _b, _c, _d;
54 };

58 // Check whether a given face exists in the facemap. If so, remove the
59 // face from the map; otherwise, insert it into the map. This function
60 // uses a vector of Corners2Face_Map to reduce worst-case time complexity.
61 static void
62 check_face_unique( const Four_tuple &face_corners,
63  const Facet_ID &fid,
64  std::vector< Corners2Face_Map > &c2f_vec,
65  int &num_external_face, int &counter) {
66 
67  // Sort the node ids of the face
68  Four_tuple sorted_ns( face_corners);
69  std::sort( &sorted_ns[0], &sorted_ns[4]);
70 
71  // Check whether the sorted node-list is in vec2facemap_vec
72  // by trying to insert it
73  Corners2Face_Map &c2f = c2f_vec[sorted_ns[1]-1];
74 
75  std::pair<Corners2Face_Map::iterator, bool>
76  ans = c2f.insert( std::make_pair(sorted_ns, fid));
77 
78  // If the node-list already exists, then remove it from vecFaceMap
79  if ( !ans.second) {
80  counter++;
81  c2f.erase( ans.first);
82  --num_external_face;
83  }
84  // Otherwise, insertion succeeded.
85  else ++num_external_face;
86 }

88 void Pane_boundary::
89 determine_border_nodes_3( std::vector<bool> &is_border,
90  std::vector<Facet_ID > *b,
91  int ghost_level ) throw(int) {
92  int num_nodes, num_elmts;
93 
94  if (ghost_level == 0){
95  num_nodes=_pane.size_of_real_nodes();
96  num_elmts=_pane.size_of_real_elements();
97  }
98  else {
99  COM_assertion_msg(ghost_level > 0, "Ghost level must be positive");
100  num_nodes=_pane.size_of_nodes();
101  num_elmts=_pane.size_of_elements();
102  }
103 
104  // Initialize is_border to false for all nodes.
105  is_border.clear(); is_border.resize( num_nodes, false);
106 
107  // Handle structured meshes
108  if ( _pane.is_structured()) {
109 
110  const int ni=_pane.size_i(), nj=_pane.size_j(), nk=_pane.size_k();
111 
112  // s = the first index on the level above the bottom
113  // t = the first index on the top level
114  // h = an offset for the first index on a layer with ghosts
115  // w = the length of the i dimension with ghosts
116  // buffer = number of layers we 'skip' for determining borders
117 
118  int buffer = _pane.size_of_ghost_layers() - ghost_level;
119  buffer = buffer < 0 ? 0 : buffer;
120  int s = ni*nj;
121  int t = ni*nj*(nk-1);
122  int h = buffer*ni + buffer;
123  int w = ni - 2*buffer;
124 
125  //bottom
126  for ( int j=s*(buffer) + h, j_end = s*(1+buffer) - ni*buffer; j< j_end; j += ni){
127  std::fill_n( is_border.begin() +j, w, true);
128  }
129  //top
130  for ( int j= t+h-s*buffer, end = s*nk - s*buffer- ni*buffer; j< end; j += ni){
131  std::fill_n( is_border.begin() +j, w, true);
132  }
133 
134  //left and right
135  for ( int j= s*(1+buffer)+h, stop = t-s*buffer+h; j< stop; j += ni*nj){
136  std::fill_n( is_border.begin() +j, w, true);
137  std::fill_n( is_border.begin() +j + ni*((nj-1) - 2*buffer), w, true);
138  }
139 
140  //upper and lower
141  for ( int j= s*(1+buffer)+h; j< t-s*buffer+h; j += ni*nj){
142  for (int k=ni, k_end = ni*(nj-(1+2*buffer)); k< k_end; k += ni){
143  is_border[j+k] = true;
144  is_border[j+k + (ni - (2*buffer+ 1)) ] = true;
145  }
146  }
147 
148  return;
149  }
150 
151  // Now consider unstructured panes
152  std::vector< Corners2Face_Map> c2f_vec(num_nodes);
153  int num_external_face=0;
154 
155 
156  // Determine all border faces
157  Element_node_enumerator ene( &_pane, 1);
158  int counter = 0;
159  int count2 = 0;
160 
161  for (int i=1; i<=num_elmts; ++i, ene.next()) {
162  //for (int i=1; i<=6; ++i, ene.next()) {
163  for (int j=0, nf=ene.size_of_faces(); j<nf; ++j ){
164  Facet_node_enumerator fne( &ene, j) ;
165 
166  Four_tuple ns( fne[0], fne[1], fne[2], fne.size_of_edges()>3?fne[3]:-1);
167  count2++;
168  check_face_unique( ns, Facet_ID(i, j), c2f_vec, num_external_face, counter);
169  }
170  }
171 
172  // Mark all nodes of the border faces as border nodes, and
173  // insert their edges into b.
174  if ( b)
175  { b->clear(); b->reserve( num_external_face); }
176 
177  std::vector<int> nodes; nodes.reserve(9);
178 
179  for (int i=0; i<num_nodes; ++i){
180  Corners2Face_Map::const_iterator vit = c2f_vec[i].begin();
181  Corners2Face_Map::const_iterator v_end = c2f_vec[i].end();
182 
183  for ( ; vit != v_end; ++vit) {
184  // Get all nodes of the face into vector nodes.
185  const Facet_ID &fid = vit->second;
186  Element_node_enumerator_uns ene( &_pane, fid.eid());
187  Facet_node_enumerator fne( &ene, fid.lid());
188  fne.get_nodes( nodes, true);
189 
190  for ( int i=0, n=nodes.size(); i<n; ++i)
191  is_border[ nodes[i] - 1 ] = true;
192 
193  if (b) b->push_back( fid);
194  }
195  }
196

199 void Pane_boundary::
200 determine_isolated_nodes( std::vector<bool> &is_isolated,
201  int ghost_level) throw(int) {
202  // Initialize to true
203  is_isolated.clear();
204 
205  int nnodes = ghost_level ?
206  _pane.size_of_nodes() : _pane.size_of_real_nodes();
207 
208  if ( _pane.is_structured()) {
209  // A structured mesh have no isolated nodes.
210  is_isolated.resize( nnodes, false);
211  return;
212  }
213 
214  is_isolated.resize( nnodes, true);
215 
216  // Loop through the connectivity table to mark their nodes as false.
217  Element_node_enumerator_uns ene(&_pane, 1);
218  int nelems = ghost_level ?
219  _pane.size_of_elements() : _pane.size_of_real_elements();
220  for ( int i=0,size=nelems; i<size; ++i, ene.next()) {
221  for ( int j=ene.size_of_nodes()-1; j>=0; --j)
222  is_isolated[ ene[j]-1] = false;
223  }
224 }

228 double Pane_boundary::
229 min_squared_edge_len( const std::vector<Facet_ID > &be) throw(int) {
230  const COM::DataItem *attr = _pane.dataitem( COM::COM_NC);
231 
232  double sql = HUGE_VAL;
233 
234  // Loop through all the edges of border facets to compute the
235  // shortest squared edge length.
236  std::vector<Facet_ID >::const_iterator it;
237  for ( it=be.begin(); it!=be.end(); ++it) {
238  Element_node_enumerator ene( &_pane, it->eid());
239  Facet_node_enumerator fne( &ene, it->lid());
240  int ne = fne.size_of_corners();
241 
242  // Loop through all edges of the facet.
243  for ( int j=0; j<fne.size_of_edges(); ++j) {
244  int n1 = fne[j], n2 = fne[(j+1)%ne];
245  const double *x1 = (const double*)attr->get_addr( n1-1, 0);
246  const double *x2 = (const double*)attr->get_addr( n2-1, 0);
247 
248  double sqdiff;
249  if ( attr->stride() >= 3) {
250  // Contiguous layout of coordinates
251  sqdiff = square(x1[0]-x2[0])+square(x1[1]-x2[1])+square(x1[2]-x2[2]);
252  }
253  else {
254  // This supports R^2 points.
255  sqdiff=square(x1[0]-x2[0]);
256  for ( int i=1, d=attr->size_of_components(); i<d; ++i) {
257  sqdiff += square((const double*)attr->get_addr( n1-1, i)-
258  (const double*)attr->get_addr( n2-1, i));
259  }
260  }
261 
262  sql = std::min( sql, sqdiff);
263  }
264  }
265 
266  return sql;
267 }
