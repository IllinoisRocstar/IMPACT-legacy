79 collect_nodes( const std::vector<std::vector<int> > &ns,
80  std::vector<int> &nodes,
81  std::vector<Point_3> &pnts,
82  bool for_facet) {
83  int count=0;
84  for (unsigned int i=0; i<ns.size(); ++i) count+=ns[i].size();
85 
86  // Collect the nodes
87  nodes.clear(); nodes.reserve( count+2*ns.size());
88  pnts.clear(); pnts.reserve( count+2*ns.size());
89 
90  for ( int i=0, s=_panes.size(); i<s; ++i) {
91  nodes.push_back( _panes[i]->id());
92  nodes.push_back( ns[i].size());
93 
94  nodes.insert( nodes.end(), ns[i].begin(), ns[i].end());
95 
96  pnts.push_back( Point_3(HUGE_VAL,HUGE_VAL,HUGE_VAL));
97  Point_3 &bnd_min = pnts.back();
98  pnts.push_back( Point_3(-HUGE_VAL,-HUGE_VAL,-HUGE_VAL));
99  Point_3 &bnd_max = pnts.back();
100 
101  const COM::DataItem *attr = _panes[i]->dataitem( COM::COM_NC);
102  const int d = attr->size_of_components();
103 
104  for (std::vector<int>::const_iterator it=ns[i].begin(), iend=ns[i].end();
105  it!=iend;++it) {
106  Point_3 p(0,0,0);
107  if ( !for_facet) {
108  int j=*it-1;
109 
110  for ( int k=0; k<d; ++k)
111  p[k] = *(const double*)attr->get_addr( j, k);
112  }
113  else {
114  Facet_ID eID=(Facet_ID&)(*it);
115  Element_node_enumerator ene( _panes[i], eID.eid());
116  int vdst=(ene[eID.lid()+1==ene.size_of_edges()?0:eID.lid()+1]);
117  int j1=ene[eID.lid()]-1, j2=vdst-1;
118 
119  for ( int k=0; k<d; ++k) {
120  p[k] = 0.5*(*(const double*)attr->get_addr( j1, k) +
121  *(const double*)attr->get_addr( j2, k));
122  }
123  }
124 
125  pnts.push_back( p);
126 
127  for (int k=0; k<3; ++k) {
128  bnd_min[k] = std::min( bnd_min[k], p[k]);
129  bnd_max[k] = std::max( bnd_max[k], p[k]);
130  }
131  }

206 double Pane_connectivity::
207 get_local_boundary_nodes( std::vector<int> &nodes,
208  std::vector<Point_3> &pnts,
209  bool for_facet) throw(int) {
210 
211  std::vector< std::vector<int> > ns(_panes.size());
212  std::vector< std::vector<int> > iso_ns(_panes.size());
213 
214  double sql = HUGE_VAL;
215  int iso_local=0;
216  for ( int i=0, s=_panes.size(); i<s; ++i) {
217  std::vector<bool> is_border;
218  std::vector<bool> is_isolated;
219  std::vector<Facet_ID> facets;
220 
221  Pane_boundary pb =
222  _mani2 ? Pane_boundary( _mani2[i]) : Pane_boundary( _panes[i]);
223 
224  pb.determine_border_nodes( is_border, is_isolated, &facets);
225  sql = std::min( sql, pb.min_squared_edge_len( facets));
226 
227  if ( !for_facet) {
228  convert_nodelist( is_border, ns[i]);
229  convert_nodelist( is_isolated, iso_ns[i]);
230 
231  iso_local += iso_ns[i].size();
232  }
233  else {
234  ns[i].clear();
235  ns[i].insert( ns[i].end(), (int*)&*facets.begin(), (int*)&*facets.end());
236  }
237  }
238 
239  // Compute minimum squared edge length.
240  double g_sql=sql;
241  if ( _comm != MPI_COMM_NULL)
242  MPI_Allreduce( &sql, &g_sql, 1, MPI_DOUBLE, MPI_MIN, _comm);
243 
244  if ( g_sql==HUGE_VAL) g_sql=0;
245  double tol = std::sqrt( g_sql)*0.03;
246 
247  if ( !for_facet) {
248  std::vector<int> iso_counts;
249 
250  // Is there any isolated nodes?
251  int comm_size, comm_rank;
252  if ( _comm != MPI_COMM_NULL) {
253  MPI_Comm_size( _comm, &comm_size);
254  MPI_Comm_rank( _comm, &comm_rank);
255 
256  iso_counts.resize(comm_size);
257  MPI_Allgather(&iso_local, 1, MPI_INT, &iso_counts[0], 1, MPI_INT, _comm);
258 
259  iso_local=0;
260  for (int i=0; i<comm_size; ++i) iso_local+=iso_counts[i];
261  }
262  else {
263  comm_size=1;
264  comm_rank=0;
265  }
266 
267  if ( iso_local) { // Determine all nodes incident on isolated nodes.
268  collect_points( iso_ns, pnts);
269 
270  // Collect all isolated nodes onto the processor
271  std::vector< Point_3> pnts_g( iso_local);
272 
273  if ( _comm != MPI_COMM_NULL) {
274  for (int i=0; i<comm_size; ++i)
275  iso_counts[i]*=3;
276 
277  std::vector<int> iso_disps(comm_size, 0);
278  for (int i=1; i<comm_size; ++i)
279  iso_disps[i]=iso_disps[i-1]+iso_counts[i-1];
280 
281  MPI_Allgatherv( &pnts[0], iso_counts[comm_rank], MPI_DOUBLE,
282  &pnts_g[0], &iso_counts[0], &iso_disps[0],
283  MPI_DOUBLE, _comm);
284  }
285  else
286  pnts_g = pnts;
287 
288  if ( pnts_g.size()>0) {
289  // Build a kd-tree from the points in pnts_g
290  // KD_tree_3 ktree;
291  // ktree.build( pnts_g);
292  KD_tree_3 tree(&pnts_g[0][0], pnts_g.size());
293 
294  for ( int i=0, s=_panes.size(); i<s; ++i) {
295  std::vector<bool> is_co;
296  determine_coisolated_nodes( *_panes[i], pnts_g, tol, is_co, &tree);
297  convert_nodelist( is_co, iso_ns[i]);
298  }
299  }
300  }
301 
302  // Merge the lists
303  for ( unsigned int i=0; i<ns.size(); ++i) {
304  std::vector<int> &nsi = ns[i];
305  nsi.insert( ns[i].end(), iso_ns[i].begin(), iso_ns[i].end());
306 
307  // sort and unique the entries in ns[i]
308  std::sort(nsi.begin(), nsi.end());
309  std::vector<int>::iterator new_end=std::unique(nsi.begin(), nsi.end());
310  nsi.erase( new_end, nsi.end());
311  }
312  }
313 
314  // Copy the nodes from ns into nodes and pnts
315  collect_nodes( ns, nodes, pnts, for_facet);
316 
317  return tol;
318 }

783 int Pane_connectivity::pconn_nblocks( const COM::DataItem *pconn) {
784  const COM::Window *win = pconn->window();
785 
786  int nblocks = 0;
787  // Loop through all panes
788  std::vector<const COM::Pane*> panes; win->panes(panes);
789  for ( int p=0, np=panes.size(); p<np; ++p) {
790  const COM::DataItem *pconn_pn = panes[p]->dataitem( pconn->id());
791 
792  // Obtain the address and sizes of the map
793  const int *pconn_ptr = (const int *)pconn_pn->pointer();
794 
795  if ( pconn_ptr==NULL) continue;
796 
797  const int len_total = pconn_pn->size_of_items()-_pconn_offset;
798 
799  const int len_ghost = pconn_ptr ? pconn_pn->size_of_ghost_items() : 0;
800  if ( len_ghost == 0) { nblocks = std::max( nblocks, 1); continue; }
801 
802  pconn_ptr += len_total-len_ghost+_pconn_offset;
803 
804  // Skip real nodes
805  int index=1, ncpanes=pconn_ptr[0];
806  for ( int i=1; i<=ncpanes; ++i, index+=pconn_ptr[index+1]+2);
807  // Skip ghost nodes
808  ncpanes=pconn_ptr[index++];
809  for ( int i=1; i<=ncpanes; ++i, index+=pconn_ptr[index+1]+2);
810 
811  nblocks = std::max( nblocks, 2+(index<len_ghost));
812  }
813 
814  // Communicate to get the maximum of nblocks.
815  if ( COMMPI_Initialized()) {
816  int nblocks_local = nblocks;
817  MPI_Allreduce( &nblocks_local, &nblocks, 1, MPI_INT, MPI_MAX,
818  win->get_communicator());
819  }
820 
821  return nblocks;
822 }
