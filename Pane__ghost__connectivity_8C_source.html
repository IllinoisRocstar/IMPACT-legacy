115 // Get a total ordering of nodes in the form of a pair <P,N> where
116 // P is the "owner pane" and N is the nodes id on the owner pane.
117 void Pane_ghost_connectivity::
118 get_node_total_order(){
119 
120  // Resize per-pane data structures
121  _p_gorder.resize(_npanes);
122  _local_nodes.resize(_npanes);
123 
124  // On each node determine P, the pane responsible for numbering the node
125  for(int i=0; i < (int)(_npanes); ++i){
126 
127  int pane_id = _panes[i]->id();
128  int nrnodes = _panes[i]->size_of_real_nodes();
129 
130  // Initialize P values of complete ordering to the current pane
131  _p_gorder[i].clear();
132  _p_gorder[i].resize(nrnodes,pane_id);
133 
134  // Obtain the pane connectivity of the local pane.
135  const DataItem *pconn = _panes[i]->dataitem(COM::COM_PCONN);
136 
137  // The pconn MAY/MAY not list the number of communicating-panes
138  // in the shared nodes section.Handle as though its not there.
139  const int *vs
140  = (const int*)pconn->pointer()+MAP::Pane_connectivity::pconn_offset();
141 
142  // Loop through communicating panes for shared nodes.
143  for ( int j=0, index=0, nj=_cpanes[i].size();
144  j<nj; ++j, index+=vs[index+1]+2) {
145 
146  // Skip panes which the pconn refers to, but which are not
147  // in the current window. May result from partial inheritance.
148  while ( _buf_window->owner_rank(vs[index])<0)
149  index+=vs[index+1]+2;
150 
151  // Update P values for the current list of shared nodes
152  for(int k=0; k<vs[index+1]; ++k){
153  // EDIT
154  if(vs[index] > _p_gorder[i][vs[index+2+k]-1])
155  _p_gorder[i][vs[index+2+k]-1] = vs[index];
156  }
157  }
158  }
159 
160  // Set the values of N on nodes for which this pane is responsible
161  for(int i=0; i < (int)(_npanes); ++i){
162 
163  int pane_id = _panes[i]->id();
164  int nrnodes = _panes[i]->size_of_real_nodes();
165  int n_gorder_id = _w_n_gorder->id();
166 
167  // There are window-level and pane-level dataitems, we need the
168  // pane level dataitems to obtain a pointer to data.
169  DataItem* p_n_gorder =
170  const_cast<DataItem*>(_panes[i]->dataitem(n_gorder_id));
171  int * n_gorder_ptr =
172  reinterpret_cast<int*>(p_n_gorder->pointer());
173 
174  for(int j=0; j< nrnodes; ++j){
175  if(_p_gorder[i][j] == pane_id)
176  n_gorder_ptr[j] = j+1;
177  else
178  n_gorder_ptr[j] = 0;
179  }
180  }
181 
182  // Update shared nodes across all panes using a max reduce operation
183  MAP::Pane_communicator pc(_buf_window, _buf_window->get_communicator());
184  pc.init(_w_n_gorder);
185  pc.begin_update_shared_nodes();
186  pc.reduce_on_shared_nodes(MPI_MAX);
187  pc.end_update_shared_nodes();
188 
189  // Store a mapping from the total node-ordering to the local node id:
190  // NOTE: can I make _local_nodes linear and just sort it once?
191  for(int i=0; i < (int)(_npanes); ++i){
192 
193  int nrnodes = _panes[i]->size_of_real_nodes();
194  int n_gorder_id = _w_n_gorder->id();
195 
196  DataItem* p_n_gorder =
197  const_cast<DataItem*>(_panes[i]->dataitem(n_gorder_id));
198  int * n_gorder_ptr =
199  reinterpret_cast<int*>(p_n_gorder->pointer());
200 
201  for(int j=0; j< nrnodes; ++j)
202  _local_nodes[i].
203  insert(make_pair(make_pair(_p_gorder[i][j],
204  n_gorder_ptr[j])
205  ,j+1));
206  }

209 // Determine elements/nodes to be ghosted on adjacent panes.
210 void Pane_ghost_connectivity::
211 get_ents_to_send(vector<pane_i_vector > &gelem_lists,
212  vector<vector<map<pair<int,int>,int> > > &nodes_to_send,
213  vector<vector<deque<int> > > &elems_to_send,
214  pane_i_vector &comm_sizes){
215 
216  // sets of adjacent elements and nodes
217  vector<vector<set<int> > > adj_eset;
218  vector<vector<set<int> > > adj_nset;
219  set<int>::iterator eset_pos;
220 
221  // resize per-local-pane data structures
222  gelem_lists.resize(_npanes);
223  adj_eset.resize(_npanes);
224  adj_nset.resize(_npanes);
225  comm_sizes.resize(_npanes);
226  nodes_to_send.resize(_npanes);
227  elems_to_send.resize(_npanes);
228 
229  for(int i=0; i < _npanes; ++i){
230 
231  int n_comm_panes = _cpanes[i].size();
232  adj_eset[i].resize(n_comm_panes);
233  adj_nset[i].resize(n_comm_panes);
234  comm_sizes[i].resize(n_comm_panes,0);
235  nodes_to_send[i].resize(n_comm_panes);
236  elems_to_send[i].resize(n_comm_panes);
237 
238  set<int> cur_eset;
239 
240  // Obtain the pane connectivity of the local pane.
241  const DataItem *pconn = _panes[i]->dataitem(COM::COM_PCONN);
242 
243  MAP::Pane_dual_connectivity dc(_panes[i],0);
244 
245  const int *vs
246  = (const int*)pconn->pointer() +
247  MAP::Pane_connectivity::pconn_offset();
248 
249  int vs_size
250  = pconn->size_of_real_items() -
251  MAP::Pane_connectivity::pconn_offset();
252 
253  // Loop through communicating panes for shared nodes.
254  for ( int j=0, index=0; j<n_comm_panes; ++j, index+=vs[index+1]+2) {
255 
256  // skiping nonexistent panes to get to next communication pane
257  while ( _buf_window->owner_rank(vs[index])<0) {
258  index+=vs[index+1]+2;
259  COM_assertion_msg( index<=vs_size, "Invalid communication map");
260  }
261 
262  for(int k=0; k<vs[index+1]; ++k){
263 
264  // get elements incident on the shared node:
265  vector<int> elist;
266  adj_nset[i][j].insert(vs[index+2+k]);
267  dc.incident_elements(vs[index+2+k],elist);
268 
269  // remember elements adj. to this pane
270  for(unsigned int ii=0; ii<elist.size(); ++ii)
271  cur_eset.insert(elist[ii]);
272  }
273 
274  adj_eset[i][j] = cur_eset;
275 
276  // Calculate size of data to send. For every element, send its type
277  // and a list of its nodes in complete ordering format
278  for(eset_pos = cur_eset.begin(); eset_pos != cur_eset.end(); ++eset_pos){
279  comm_sizes[i][j] += 1 + 2*((_panes[i]->connectivity(*eset_pos))
280  ->size_of_nodes_pe());
281  }
282  cur_eset.clear();
283  }
284  }
285 
286  int n_gorder_id = _w_n_gorder->id();
287 
288  // fill in pane comm data
289  for(int i=0; i < _npanes; ++i){
290 
291  DataItem* p_n_gorder =
292  const_cast<DataItem*>(_panes[i]->dataitem(n_gorder_id));
293  int * n_gorder_ptr =
294  reinterpret_cast<int*>(p_n_gorder->pointer());
295 
296  gelem_lists[i].resize(_cpanes[i].size());
297 
298  // fill in the comm buffers
299  for(int j=0, nj = comm_sizes[i].size(); j<nj; ++j){
300 
301  int index = 0;
302 
303  gelem_lists[i][j].resize(comm_sizes[i][j]);
304 
305  for(eset_pos = adj_eset[i][j].begin();
306  eset_pos != adj_eset[i][j].end();
307  ++eset_pos){
308 
309  vector<int> nodes;
310  COM::Element_node_enumerator ene(_panes[i],*eset_pos);
311  elems_to_send[i][j].push_back(*eset_pos);
312  ene.get_nodes(nodes);
313 
314  // store type
315  gelem_lists[i][j][index++] = ene.type();
316 
317  for(int k=0, nk = ene.size_of_nodes(); k<nk; ++k){
318  // store nodes in (P,N) format
319  int P = _p_gorder[i][nodes[k]-1];
320  int N = n_gorder_ptr[nodes[k]-1];
321  gelem_lists[i][j][index++] = P;
322  gelem_lists[i][j][index++] = N;
323 
324  // Send nodes which aren't shared w/ this processor
325  if(adj_nset[i][j].find(nodes[k]) == adj_nset[i][j].end())
326  nodes_to_send[i][j].insert(make_pair(make_pair(P,N),nodes[k]));
327  }
328  }
329  }
330  }
331 
332  // We are finished w/ the total ordering at this point, free up some space
333  _buf_window->delete_dataitem("n_gorder");
334  _buf_window->init_done();
335 
336  _p_gorder.clear();
337 }

339 // Determine # of ghost nodes to receive and map (P,N) to ghost node ids
340 // Also determine # ghost elements of each type to receive
341 void Pane_ghost_connectivity::
342 process_received_data(
343  vector<pane_i_vector> &recv_info,
344  vector<vector<int> > &elem_renumbering,
345  vector<vector<map<pair<int,int>,int> > > &nodes_to_recv){
346 
347  map<pair<int,int>,int>::iterator pos1, pos2;
348 
349  elem_renumbering.resize(_npanes);
350  nodes_to_recv.resize(_npanes);
351 
352  for(int i=0; i<_npanes; ++i){
353 
354  int n_real_nodes = _panes[i]->size_of_real_nodes();
355  int next_node_id = n_real_nodes + 1;
356  int comm_npanes = recv_info[i].size();
357  elem_renumbering[i].resize((int)Connectivity::TYPE_MAX_CONN+1,0);
358  nodes_to_recv[i].resize(comm_npanes);
359 
360  for(int j=0; j< comm_npanes; ++j){
361 
362  int recv_size = recv_info[i][j].size();
363  int index = 0;
364 
365  while(index < recv_size){
366  int type = recv_info[i][j][index];
367  int nnodes = Connectivity::size_of_nodes_pe(type);
368  ++elem_renumbering[i][type+1];
369 
370  // Examine element's nodes, labeling those seen for the first time.
371  for(int k=1; k<=2*nnodes; k+=2){
372 
373  int P = recv_info[i][j][index+k];
374  int N = recv_info[i][j][index+k+1];
375 
376  pos1 =
377  nodes_to_recv[i][j].find(make_pair(P,N));
378 
379  pos2 =
380  _local_nodes[i].find(make_pair(P,N));
381 
382  // If we haven't seen this node at all, give it an id, otherwise
383  // use the existing id
384  int cur_node_id = next_node_id;
385  if(pos2 == _local_nodes[i].end())
386  _local_nodes[i].insert(make_pair(make_pair(P,N),next_node_id++));
387  else
388  cur_node_id = pos2->second;
389 
390  // If we haven't seen this node from the current adjacent pane,
391  // and we don't have a local real copy of the node, then
392  // remember to receive it
393 
394  if(pos1 == nodes_to_recv[i][j].end()
395  && cur_node_id > n_real_nodes)
396  nodes_to_recv[i][j].insert(make_pair(make_pair(P,N),cur_node_id));
397  }
398  index += 2*nnodes+1;
399  }
400  }
401  }
402 }

420 void Pane_ghost_connectivity::
421 finalize_pconn(vector<vector<map<pair<int,int>,int> > > &nodes_to_send,
422  vector<vector<map<pair<int,int>,int> > > &nodes_to_recv,
423  vector<vector<deque<int> > > &elems_to_send,
424  vector<vector<int> > &elem_renumbering,

428  vector<vector<int> > node_pos;
429

431  vector<vector<int> > n_elem;
432 
433  // Get the name of the buffer window
434  string buf_name(_buf_window->name());

439

441  n_elem.resize(_npanes);
442 
443  // Determine buffer space required:
444  // 1 (#comm panes) + 2 per adj pane (comm pane id and #entities)
445  // + total #entries in entity list (ie node lists for nodes to receive)
446  for(int i=0; i< _npanes; ++i){
447

451

454  rns_size += 2+nodes_to_send[i][j].size();

458  gnr_size += 2+nodes_to_recv[i][j].size();
459 
460  // Real cells to send
461  for(int j=0, nj = (int)elems_to_send[i].size(); j<nj; ++j)
462  rcs_size += 2+elems_to_send[i][j].size();
463 
464  // Ghost cells to receive
465  n_elem[i].resize(n_comm_panes,0);
466  for(int j=0, nj = (int)_cpanes[i].size(); j<nj; ++j){
467  gcr_size += 2;
468  for(int ind=0, size = (int)recv_info[i][j].size();
469  ind < size;
470  ind += 1+2*Connectivity::size_of_nodes_pe(recv_info[i][j][ind])){
471  gcr_size++;
472  n_elem[i][j]++;
473  }
474  }
475 
476  node_pos[i].assign(elem_renumbering[i].begin(),elem_renumbering[i].end());
477

479  conn_ptr[i].resize(Connectivity::TYPE_MAX_CONN,NULL);
480

482  for(int j=0; j<Connectivity::TYPE_MAX_CONN; ++j){
483 
484  int nelems = elem_renumbering[i][j+1];
485  elem_renumbering[i][j+1] += elem_renumbering[i][j];
486 
487  if(nelems >0){
488 
489  const string conn_name = _etype_str[j]+"virtual";
490 
491  int nnodes = Connectivity::size_of_nodes_pe(j);
492 
493  // Resize connectivity table and keep a pointer to its buffer
494  void* addr;
495  _buf_window->set_size(conn_name.c_str(), pane_id, nelems,nelems);
496  _buf_window->resize_array(conn_name.c_str(), pane_id, &addr,nnodes,nelems);
497

499  COM_assertion_msg(addr!= NULL, "Could not allocate space for connectivity table");
500  }
501  }
502 
503  // Resize pconn
504  DataItem *pconn = NULL;
505 
506  pconn = _panes[i]->dataitem(COM::COM_PCONN);
507  int rsize = pconn->size_of_real_items();
508  int gsize = rns_size + gnr_size + rcs_size + gcr_size;
509 
510  void *addr;
511  _buf_window->set_size("pconn", pane_id, rsize+gsize,gsize);
512  _buf_window->resize_array("pconn",pane_id,&addr);
513 
514  pconn = _panes[i]->dataitem(COM::COM_PCONN);
515 
516  int* pconn_ptr = (int*)addr;
517 
518  int rns_ind = rsize;
519  int gnr_ind = rns_ind + rns_size;
520  int rcs_ind = gnr_ind + gnr_size;
521  int gcr_ind = rcs_ind + rcs_size;
522 
523  // each block begins w/ # of communicating blocks
524  pconn_ptr[rns_ind++] = n_comm_panes;
525  pconn_ptr[gnr_ind++] = n_comm_panes;
526  pconn_ptr[rcs_ind++] = n_comm_panes;
527  pconn_ptr[gcr_ind++] = n_comm_panes;
528 
529  // Offset to start of ghost element_ids
530  int real_offset = _panes[i]->size_of_real_elements()+1;
531 
532  // My current implementation only sends nodes to panes w/ a ghost
533  // copy of a local element, so there are the same number of communicating
534  // panes in each pconn block.
535  // If the code is generalized in the future, there may need to be
536  // separate loops for some pconn blocks.
537  for(int j=0; j <n_comm_panes; ++j){
538 
539  // Write communicating-pane id to buffer
540  int comm_pane_id = _cpanes[i][j];
541  pconn_ptr[rns_ind++] = comm_pane_id;
542  pconn_ptr[gnr_ind++] = comm_pane_id;
543  pconn_ptr[rcs_ind++] = comm_pane_id;
544  pconn_ptr[gcr_ind++] = comm_pane_id;
545 
546  // Write number of enties to buffer
547  pconn_ptr[rns_ind++] = nodes_to_send[i][j].size();
548  pconn_ptr[gnr_ind++] = nodes_to_recv[i][j].size();
549  pconn_ptr[rcs_ind++] = elems_to_send[i][j].size();
550  pconn_ptr[gcr_ind++] = n_elem[i][j];
551 
552  // Write entities to ghost pconn buffers
553  for(rns_pos = nodes_to_send[i][j].begin();
554  rns_pos != nodes_to_send[i][j].end(); ++rns_pos)
555  pconn_ptr[rns_ind++] = rns_pos->second;
556 
557  for(gnr_pos = nodes_to_recv[i][j].begin();
558  gnr_pos != nodes_to_recv[i][j].end(); ++gnr_pos)
559  pconn_ptr[gnr_ind++] = gnr_pos->second;
560 
561  for(int k=0, nk = (int)elems_to_send[i][j].size(); k<nk; ++k)
562  pconn_ptr[rcs_ind++] = elems_to_send[i][j][k];
563 
564  // The GCR block is more complicated because we want all ghost elements
565  // of a single type to have contiguous element ids, which is required
566  // by Roccom if we want to register one connectivity table per type

568  int index = 0;

688 // send_info = data to send
689 // recv_info = buffer for receiving data
690 // comm_sizes = amount of data to receive
691 // cpanes = list of communicating panes
692 void Pane_ghost_connectivity::
693 send_pane_info(vector<pane_i_vector> &send_info,
694  vector<pane_i_vector> &recv_info,
695  pane_i_vector &comm_sizes){
696 
697  const Window::Proc_map &proc_map = _buf_window->proc_map();
698  int total_npanes = proc_map.size();
699  int tag_max = total_npanes*total_npanes;
700  recv_info.resize(_npanes);
701 
702  // Determine a mapping from user-defined pane ids to a set of
703  // internal pane IDs, which are unique and contiguous across all processes
704  // to be used for defining unique tags for MPI messages.
705  map<int,int> lpaneid_map;
706  map<int,int>::const_iterator it=proc_map.begin();
707  for ( int i=0; i<total_npanes; ++i, ++it)
708  lpaneid_map[ it->first] = i;
709 
710  vector<MPI_Request> reqs_send, reqs_recv;
711  int int_size = sizeof(int);
712  MPI_Request req;
713  MPI_Comm mycomm = _buf_window->get_communicator();
714  int myrank = COMMPI_Initialized() ?
715  COMMPI_Comm_rank(mycomm) : 0;
716 
717  // initiate mpi sends and recieves
718  for(int i=0; i< _npanes; ++i){
719 
720  recv_info[i].resize(_cpanes[i].size());
721  int lpid = lpaneid_map.find(_panes[i]->id())->second;
722 
723  for(int j=0, nj = _cpanes[i].size(); j<nj; ++j){
724 
725  recv_info[i][j].resize(comm_sizes[i][j],0);
726 
727  const int lqid = lpaneid_map.find(_cpanes[i][j])->second;
728  int adjrank = _buf_window->owner_rank(_cpanes[i][j]);
729 
730  int stag = 100 + ((lpid > lqid) ?
731  lpid*total_npanes+lqid : lqid*total_npanes+lpid);
732  int rtag = stag;
733  MPI_Comm comm = mycomm;
734 
735  // pane sending to self, copy data directly
736  if(myrank == adjrank && lpid == lqid)
737  std::memcpy(&send_info[i][j][0], &recv_info[i][j][0],
738  int_size*send_info[i][j].size());
739 
740  // sending to other panes, use COMMPI
741  else{
742  // process sending to self
743  if (myrank == adjrank){
744  // make tags unique
745  if(lpid > lqid)
746  stag += tag_max;
747  else
748  rtag += tag_max;
749  comm = MPI_COMM_SELF;
750  adjrank = 0;
751  }
752 
753  int ierr =
754  COMMPI_Isend(&send_info[i][j][0],int_size*send_info[i][j].size(),
755  MPI_BYTE, adjrank, stag, comm, &req);
756  COM_assertion(ierr==0);
757  reqs_send.push_back(req);
758 
759  ierr = COMMPI_Irecv(&recv_info[i][j][0],int_size*comm_sizes[i][j],
760  MPI_BYTE, adjrank, rtag, comm, &req);
761  COM_assertion(ierr==0);
762  reqs_recv.push_back(req);
763  }
764  }
765  }
766  // wait for MPI communication to finish
767  if(mycomm != MPI_COMM_NULL){
768 
769  int ierr, index;
770  MPI_Status status;
771 
772  while(!reqs_recv.empty()){
773  ierr = MPI_Waitany(reqs_recv.size(),&reqs_recv[0],
774  &index, &status);
775  COM_assertion(ierr == 0);
776  reqs_recv.erase(reqs_recv.begin()+index);
777  }
778 
779  if(reqs_send.size()){
780  ierr = MPI_Waitany(reqs_send.size(),&reqs_send[0],
781  &index, &status);
782  COM_assertion(ierr == 0);
783  reqs_send.erase(reqs_send.begin()+index);
784  }
785  }
786 }
