50 // Note: This is not a general routine for detecting whether two nodes are
51 // coincident. It assumes that the two nodes belong to the same pane and
52 // the primary copy of the nodes also belongs to the same pane.
53 bool
54 RFC_Pane_base::coincident( int v1, int v2) const {
55  if ( v1==v2) return true;
56  V2b_table::const_iterator it1 = _v2b_table.find( v1);
57  V2b_table::const_iterator it2 = _v2b_table.find( v2);
58 
59  if ( it1 == _v2b_table.end() && it2 != _v2b_table.end())
60  return it2->second.first == id() &&
61  _b2v_table.find( id())->second[it2->second.second] == v1;
62  else if ( it1 != _v2b_table.end() && it2 == _v2b_table.end())
63  return it1->second.first == id() &&
64  _b2v_table.find( id())->second[it1->second.second] == v2;
65  else if ( it1 != _v2b_table.end() && it2 != _v2b_table.end()) {
66  RFC_assertion( it1->second.first==id() && it2->second.first==id());
67  B2v_table::const_iterator bit=_b2v_table.find( id());
68  return bit->second[it1->second.second]==bit->second[it2->second.second];
69  }
70  else
71  return false;
72 }

74 void RFC_Pane_base::
75 get_nat_coor_in_element( const int sn_id,
76  Element_node_enumerator &ene, Point_2 &nc) const {
77  RFC_assertion( sn_id >= 1 && sn_id <= int(_subnode_parents.size()));
78 
79  const Point_2S &p = _subnode_nat_coors[ sn_id-1];
80  const ParentEdge_ID &eid = _subnode_parents[ sn_id-1];
81 
82  if ( eid.face_id == ene.id()) {
83  nc[0]=p[0]; nc[1]=p[1];
84  normalize_nat_coor( eid.edge_id, ene.size_of_edges(), nc);
85  }
86  else {
87  nc[0] = 1.-p[0]; nc[1]=0;
88  int vid = Element_node_enumerator(_base,eid.face_id)[eid.edge_id];
89 
90  int i = get_lvid(ene, vid)-1;
91 
92  int num_edges = ene.size_of_edges();
93  if ( i<0) i=num_edges-1;
94  normalize_nat_coor( i, num_edges, nc);
95  }
96 }

142 RFC_Pane_base::ParentEdge_ID
143 RFC_Pane_base::get_edge_id( const int face_lid, const int vertex_lid) const {
144  if ( _base->is_structured()) {
145  Element_node_enumerator ene( _base, face_lid, (COM::Connectivity *)NULL);
146  for ( int k=0; k<4; ++k)
147  if ( ene[k] == vertex_lid) return ParentEdge_ID( face_lid, k);
148  RFC_assertion(false); abort();
149  }
150  else {
151  for ( std::vector<const COM::Connectivity*>::const_iterator
152  it = _elem_conns.begin(); it != _elem_conns.end(); ++it) {
153  const int ne = (*it)->size_of_edges_pe();
154  if ( face_lid > int((*it)->index_offset()) &&
155  face_lid <= int((*it)->index_offset()+(*it)->size_of_elements())) {
156  Element_node_enumerator ene(_base, face_lid-(*it)->index_offset(), *it);
157 
158  for ( int k=0; k<ne; ++k)
159  if ( ene[k] == vertex_lid) return ParentEdge_ID(face_lid,k);
160  RFC_assertion(false); abort();
161  }
162  }
163  }
164  return ParentEdge_ID(face_lid,-1);
165 }

172  // Resort to a more expensive method.

202 void
203 RFC_Pane_base::build_v2b_from_b2v( const RFC_Window_base *w) {
204  //`Loop through the B2v_table
205  for ( B2v_table::iterator
206  it=_b2v_table.begin(), iend=_b2v_table.end(); it != iend; ++it) {
207  for ( int i=0, size=it->second.size(); i<size; ++i) {
208  if ( it->first > id() && !is_isolated_node(it->second[i])) continue;
209 
210  std::pair<int,int> p(it->first, i);
211  if ( it->first==id()) {
212  RFC_assertion( (i&1)==1 && it->second[i-1] < it->second[i] ||
213  (i&1)==0 && it->second[i] < it->second[i+1]);
214  if ( (i&1)==0 || is_isolated_node( it->second[i-1]))
215  continue;
216  --p.second;
217  }
218  else {
219  const RFC_Pane_base &pane = w->pane( it->first);
220  if ( pane.is_isolated_node( pane._b2v_table.find(id())->second[i]))
221  continue;
222  }
223 
224  if ( _v2b_table.find( it->second[i]) == _v2b_table.end())
225  _v2b_table[it->second[i]] = p;
226  }
227  }
228 }

273 RFC_Window_base::build_pc_tables() {
274  MAP::Pane_connectivity pc( _base->dataitem(COM::COM_MESH),
275  _map_comm.mpi_comm());
276  pc.compute_pconn( _base->dataitem(COM::COM_PCONN));
277 
278  // Now copy from b2v into the tables of each pane
279  Pane_set::iterator pane_it, pane_iend=_pane_set.end();
280 
281  for ( pane_it=_pane_set.begin(); pane_it != pane_iend; ++pane_it) {
282  const COM::DataItem *pconn =
283  pane_it->second->_base->dataitem(COM::COM_PCONN);
284 
285  int count=pc.pconn_offset(), s = pconn->size_of_real_items();
286  const int *buf= ((const int*)pconn->pointer())+count;
287 
288  while ( count < s) {
289  const int r_pid = *buf;
290  const int r_size = *(buf+1);
291  count += 2; buf += 2;
292 
293  Pane::B2v &b2 = pane_it->second->_b2v_table[ r_pid];
294  b2.clear();
295  b2.insert( b2.end(), buf, buf+r_size);
296 
297  count += r_size;
298  buf += r_size;
299  }
300  }
301 
302  // Construct v2b
303  for ( pane_it=_pane_set.begin(); pane_it != pane_iend; ++pane_it) {
304  Pane &pane = *pane_it->second;
305  pane.build_v2b_from_b2v( this);
306  }

309 void
310 RFC_Pane_base::comp_nat_coors() {
311  int nsn=size_of_subnodes();
312  _subnode_normalized_nc.resize( nsn);
313  _subnode_subID.resize( size_of_nodes(), -1);
314 
315  // Initialize _subnode_normalized_nc.
316  for ( int i=0; i<nsn; ++i) {
317  int f=_subnode_parents[i].face_id;
318  Element_node_enumerator ene( _base, f);
319 
320  Point_2 p; get_nat_coor_in_element( i+1, ene, p);
321  _subnode_normalized_nc[i] = Point_2S(p[0], p[1]);
322 
323  // Assign sub-node ID for each vertex in input mesh
324  if (parent_type_of_subnode( i+1)==PARENT_VERTEX) {
325  int vid = ene[_subnode_parents[i].edge_id];
326  _subnode_subID[ vid-1] = i+1;
327  }
328  }
329 
330  int nsf=size_of_subfaces();
331  _subface_nat_coors.resize( nsf);
332  // Initialize _subface_nat_coors.
333  Element_node_enumerator ene( base(), 1);
334  for ( int i=0; i<nsf; ++i) {
335  if ( ene.id() != _subface_parents[i]) {
336  ene.next();
337  if ( ene.pane()==NULL || ene.id() != _subface_parents[i])
338  get_host_element_of_subface( i+1, ene);
339  }
340  Point_2 p;
341  for ( int j=0; j<3; ++j) {
342  get_nat_coor_in_element( _subfaces[i][j], ene, p);
343  _subface_nat_coors[i][j] = Point_2S(p[0], p[1]);
344  }
345  }
