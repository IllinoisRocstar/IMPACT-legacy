708 RFC_Window_derived<_Pane>::get_primary( Pane *p, int vid) {
709  RFC_Pane_base::V2b_table::iterator it;
710 
711  // points to a border halfedge.
712  if ( !p->is_border_node( vid) ||
713  (it = p->_v2b_table.find( vid))==p->_v2b_table.end()) {
714  RFC_assertion( !p->is_isolated_node( vid));
715  return std::make_pair(p, vid);

717 
718  RFC_Pane_base::V2b &v2b = it->second;
719 
720  Pane &p2 = pane( v2b.first);

722 
723  RFC_assertion( v2 < Node_ID( p->id(), vid) || p->is_isolated_node(vid));
