57 // Register the dataitems for the pane for output using Rocout.
58 void RFC_Pane_base::
59 register_sdv_dataitems( const std::string &wname) const {
60  // Register the mesh
61  const int pid = id();
62 
63  // Pack coordinates into double precision buffers
64  int nnodes = size_of_subnodes();
65  COM_set_size( (wname+".nc").c_str(), pid, nnodes);
66 
67  Point_3 *coor;
68  COM_allocate_array( (wname+".nc").c_str(), pid, &(void*&)coor);
69  for ( int i=0; i<nnodes; ++i)
70  coor[i] = get_point_of_subnode( i+1);
71 
72  // Register subface connectivity, parents, and counterparts
73  COM_set_size( (wname+".:t3").c_str(), pid, _subfaces.size());
74  COM_set_array_const( (wname+".:t3").c_str(), pid, &_subfaces[0]);
75 
76  // Pack pane connectivity v2b of the pane into integer buffers
77  // allocated in COM.
78  int *v2b_buf;
79  COM_set_size( (wname+".v2b").c_str(), pid, _v2b_table.size()*3);
80  COM_allocate_array( (wname+".v2b").c_str(), pid, &(void*&)v2b_buf);
81 
82  for ( V2b_table::const_iterator
83  it = _v2b_table.begin(), iend=_v2b_table.end(); it!=iend; ++it) {
84  *(v2b_buf++) = it->first;
85  *(v2b_buf++) = it->second.first;
86  *(v2b_buf++) = it->second.second;
87  }
88 
89  // Pack b2v into an integer buffer.
90  std::string b2v_name=wname+".b2v";
91  for ( B2v_table::const_iterator
92  it=_b2v_table.begin(), iend=_b2v_table.end(); it != iend; ++it) {
93  const B2v &b2v = it->second;
94  int n=b2v.size();
95 
96  COM_append_array( b2v_name.c_str(), pid, &it->first, 0, 1);
97  COM_append_array( b2v_name.c_str(), pid, &n, 0, 1);
98  COM_append_array( b2v_name.c_str(), pid, &b2v[0], 0, b2v.size());
99  }
100 
101  COM_set_array_const( (wname+".sn_parent_fcID").c_str(),
102  pid, &_subnode_parents[0].face_id, 2);
103  COM_set_array_const( (wname+".sn_parent_ncs").c_str(),
104  pid, &_subnode_normalized_nc[0], 2);
105 
106  // Permuted local coordinates within parent face
107  COM_set_array_const( (wname+".sn_permu_edID").c_str(),
108  pid, &_subnode_parents[0].edge_id, 2);
109  COM_set_array_const( (wname+".sn_permu_ncs").c_str(),
110  pid, &_subnode_nat_coors[0][0], 2);
111 
112  // Register the counterpart of each subnode
113  COM_set_array_const( (wname+".sn_cntpt_pnID").c_str(),
114  pid, &_subnode_counterparts[0].pane_id, 2);
115  COM_set_array_const( (wname+".sn_cntpt_ndID").c_str(),
116  pid, &_subnode_counterparts[0].node_id, 2);
117 
118  // Register the parent of each subface
119  COM_set_array_const( (wname+".sf_parent").c_str(),
120  pid, &_subface_parents[0]);
121 
122  // Register the natural coordinates of subnodes in each subface
123  COM_set_array_const( (wname+".sf_ncs").c_str(),
124  pid, &_subface_nat_coors[0]);
125 
126  // Register the counterpart in the other mesh of each subface
127  COM_set_array_const( (wname+".sf_cntpt_pnID").c_str(),
128  pid, &_subface_counterparts[0].pane_id, 2);
129  COM_set_array_const( (wname+".sf_cntpt_fcID").c_str(),
130  pid, &_subface_counterparts[0].face_id, 2);
131 
132  // Register the offset of the subface ID for each input facet
133  COM_set_size( (wname+".sn_subID").c_str(), pid, _subnode_subID.size());
134  COM_set_array_const( (wname+".sn_subID").c_str(),
135  pid, &_subnode_subID[0]);
136 
137  // Register the offset of the subface ID for each input facet
138  COM_set_size( (wname+".sf_offset").c_str(), pid, size_of_faces());
139  COM_set_array_const( (wname+".sf_offset").c_str(),
140  pid, &_subface_offsets[0]);
141 
142  COM_window_init_done( wname.c_str());
143 }

151 void RFC_Pane_base::
152 write_off( std::FILE *fhandle) const {
153 
154  SD_set_option("start_vertexID",1);
155  SD_set_option("start_faceID",1);
156  SD_set_option("start_subvertexID",1);
157  SD_set_option("start_subfaceID",1);
158 
159  // Pack coordinates into double precision buffers
160  int nnodes = size_of_subnodes();
161  int nsf = _subfaces.size();
162 
163  std::vector<float> coor(nnodes*3);
164  std::vector<int> sn_parents(nnodes);
165  std::vector<int> sn_cntpt_pnID(nnodes), sn_cntpt_ndID(nnodes);
166 
167  for ( int i=0; i<nnodes; ++i) {
168  Point_3 p = get_point_of_subnode( i+1);
169  coor[3*i] = p[0]; coor[3*i+1] = p[1]; coor[3*i+2] = p[2];
170 
171  sn_parents[i] = _subnode_parents[i].face_id;
172  sn_cntpt_pnID[i] = _subnode_counterparts[i].pane_id;
173  sn_cntpt_ndID[i] = _subnode_counterparts[i].node_id;
174  }
175 
176  std::vector<int> sf_cntpt_pnID(nsf), sf_cntpt_fcID(nsf);
177  for ( int i=0; i<nsf; ++i) {
178  sf_cntpt_pnID[i] = _subface_counterparts[i].pane_id;
179  sf_cntpt_fcID[i] = _subface_counterparts[i].face_id;
180  }
181 
182 
183  // Construct a mapping from subnodes to real nodes
184  std::vector<int> sn_parent_vID(nnodes, -1);
185  for ( int i=0, ni=_subnode_subID.size(); i<ni; ++i) {
186  if ( _subnode_subID[i]>0)
187  sn_parent_vID[ _subnode_subID[i]-1] = i;
188  }
189 
190  SD_write_off_comref( fhandle, id(), nnodes, _subfaces.size(),
191  &coor[0], &sn_parents[0], &_subnode_normalized_nc[0][0],
192  &sn_cntpt_ndID[0], &_subfaces[0][0], &_subface_parents[0],
193  &_subface_nat_coors[0][0][0], &sf_cntpt_fcID[0],&sn_parent_vID[0],
194  &sn_cntpt_pnID[0], &sf_cntpt_pnID[0]);
195 }

198 // Read in the subdivision using Rocin.
199 void RFC_Pane_base::
200 read_rocin( const std::string &sdv_wname,
201  const std::string &parent_wname,
202  COM::Pane *pn_base) {
203 
204  const int pid = id();
205 
206  // If pn_base is not NULL, then inherit from parent pane.
207  if ( pn_base) {
208  // Obtain the mesh object in the corresponding pane in the parent window
209  COM::COM_base *rcom = COM_get_com();
210  int win_handle = rcom->get_window_handle( parent_wname.c_str());
211  COM::Window *parent_win = rcom->get_window_object(win_handle);
212 
213  COM::DataItem *mesh = parent_win->pane(pid).dataitem( COM::COM_PMESH);
214 
215  // Clone the mesh from parent pane onto pn_base.
216  ((Pane_friend*)pn_base)->inherit( mesh, "", COM::Pane::INHERIT_CLONE, 0);
217  }
218 
219  // Obtain v2b_table
220  int nv2b; const int *v2b_ptr;
221  COM_get_size( (sdv_wname+".v2b").c_str(), pid, &nv2b);
222  COM_get_array_const( (sdv_wname+".v2b").c_str(), pid, &v2b_ptr);
223 
224  for ( int i=0; i<nv2b; i+=3)
225  _v2b_table[ v2b_ptr[i]] = std::make_pair( v2b_ptr[i+1], v2b_ptr[i+2]);
226 
227  // Obtain b2v mapping
228  int nb2v; const int *b2v_ptr;
229  COM_get_size( (sdv_wname+".b2v").c_str(), pid, &nb2v);
230 
231  COM_get_array_const( (sdv_wname+".b2v").c_str(), pid, &b2v_ptr);
232 
233  int nsubn, nsubf;
234  COM_get_size( (sdv_wname+".nc").c_str(), pid, &nsubn);
235  COM_get_size( (sdv_wname+".:t3").c_str(), pid, &nsubf);
236 
237  _subfaces.resize( nsubf);
238  COM_copy_array( (sdv_wname+".:t3").c_str(), pid, &_subfaces[0]);
239 
240  if ( COM_get_dataitem_handle((sdv_wname+".sn_parent_fcID").c_str())>0) {
241  // This is the new HDF file format!
242 
243  // Obtain subnode_parents, subnode_natcoor, and subnode_counterparts
244  _subnode_parents.resize( nsubn);
245  COM_copy_array( (sdv_wname+".sn_parent_fcID").c_str(),
246  pid, &_subnode_parents[0].face_id, 2);
247  _subnode_normalized_nc.resize( nsubn);
248  COM_copy_array( (sdv_wname+".sn_parent_ncs").c_str(),
249  pid, &_subnode_normalized_nc[0], 2);
250  _subnode_subID.resize( size_of_nodes());
251  COM_copy_array( (sdv_wname+".sn_subID").c_str(),
252  pid, &_subnode_subID[0], 1);
253 
254  COM_copy_array( (sdv_wname+".sn_permu_edID").c_str(),
255  pid, &_subnode_parents[0].edge_id, 2);
256  _subnode_nat_coors.resize( nsubn);
257  COM_copy_array( (sdv_wname+".sn_permu_ncs").c_str(),
258  pid, &_subnode_nat_coors[0][0], 2);
259 
260  _subnode_counterparts.resize( nsubn);
261  COM_copy_array( (sdv_wname+".sn_cntpt_pnID").c_str(),
262  pid, &_subnode_counterparts[0].pane_id, 2);
263  COM_copy_array( (sdv_wname+".sn_cntpt_ndID").c_str(),
264  pid, &_subnode_counterparts[0].node_id, 2);
265 
266  _subface_parents.resize( nsubf);
267  COM_copy_array( (sdv_wname+".sf_parent").c_str(), pid,
268  &_subface_parents[0]);
269 
270  _subface_counterparts.resize( nsubf);
271  COM_copy_array( (sdv_wname+".sf_cntpt_pnID").c_str(), pid,
272  &_subface_counterparts[0].pane_id, 2);
273  COM_copy_array( (sdv_wname+".sf_cntpt_fcID").c_str(), pid,
274  &_subface_counterparts[0].face_id, 2);
275 
276  _subface_nat_coors.resize( nsubf);
277  COM_copy_array( (sdv_wname+".sf_ncs").c_str(), pid,
278  &_subface_nat_coors[0], 6);
279 
280  // Obtain subface_offsets
281  _subface_offsets.resize( size_of_faces()+1);
282  COM_copy_array( (sdv_wname+".sf_offset").c_str(), pid,
283  &_subface_offsets[0]);
284  _subface_offsets[size_of_faces()] = nsubf;
285  }
286  else { // This is old file format
287  // Obtain subnode_parents, subnode_natcoor, and subnode_counterparts
288  _subnode_parents.resize( nsubn);
289  COM_copy_array( (sdv_wname+".sn_parents_fcID").c_str(),
290  pid, &_subnode_parents[0].face_id, 2);
291  COM_copy_array( (sdv_wname+".sn_parents_edID").c_str(),
292  pid, &_subnode_parents[0].edge_id, 2);
293 
294  _subnode_nat_coors.resize( nsubn);
295  COM_copy_array( (sdv_wname+".sn_xi").c_str(),
296  pid, &_subnode_nat_coors[0][0], 2);
297  COM_copy_array( (sdv_wname+".sn_eta").c_str(),
298  pid, &_subnode_nat_coors[0][1], 2);
299 
300  _subnode_counterparts.resize( nsubn);
301  COM_copy_array( (sdv_wname+".sn_cntpt_pnID").c_str(),
302  pid, &_subnode_counterparts[0].pane_id, 2);
303  COM_copy_array( (sdv_wname+".sn_cntpt_ndID").c_str(),
304  pid, &_subnode_counterparts[0].node_id, 2);
305 
306  _subface_parents.resize( nsubf);
307  COM_copy_array( (sdv_wname+".sf_parents").c_str(), pid,
308  &_subface_parents[0]);
309 
310  _subface_counterparts.resize( nsubf);
311  COM_copy_array( (sdv_wname+".sf_cntpt_pnID").c_str(), pid,
312  &_subface_counterparts[0].pane_id, 2);
313  COM_copy_array( (sdv_wname+".sf_cntpt_fcID").c_str(), pid,
314  &_subface_counterparts[0].face_id, 2);
315 
316  // Obtain subface_offsets
317  _subface_offsets.resize( size_of_faces()+1);
318  COM_copy_array( (sdv_wname+".sf_offsets").c_str(), pid, &_subface_offsets[0]);
319 
320  // Compute the natural coordinates.
321  comp_nat_coors();
322  }

334 void
335 RFC_Window_base::read_sdv( const char *prefix, const char *format) {
336 
337  // Determine the format
338  int IO_format = get_sdv_format( format);
339 
340  if ( IO_format == SDV_BINARY) {
341  // Read in native format binary format.
342  for ( Pane_set::iterator
343  it=_pane_set.begin(), iend=_pane_set.end();it!=iend; ++it) {
344  std::string fname = get_sdv_fname( prefix, it->first, IO_format);
345 
346  std::ifstream is( fname.c_str());
347  if ( !is) {
348  std::cerr << "Rocface: Could not open input file " << fname
349  << "\nAbortting..." << std::endl;
350  RFC_assertion(is); exit(-1);
351  }
352 
353  RFC_Pane_base &pane = *it->second;
354  pane.read_binary( is);
355  }
356  }
357  else {
358  // Read in using Rocin.
359  SimIN_load_module( "RFC_IN");
360  int hdl_read = COM_get_function_handle( "RFC_IN.read_windows");
361  int hdl_obtain = COM_get_function_handle( "RFC_IN.obtain_dataitem");
362 
363  // Define the base-window and sdv-window names
364  std::string base_material = get_prefix_base( prefix);
365  std::string sdv_material = base_material+"_sdv";
366 
367  std::string buf_wname(_bufwin_prefix); buf_wname.append( base_material);
368  std::string sdv_wname=buf_wname+"_sdv";
369  MPI_Comm comm_self = MPI_COMM_SELF;
370 
371  // Loop through the panes to obtain the dataitems.
372  for ( Pane_set::iterator
373  it=_pane_set.begin(), iend=_pane_set.end(); it!=iend; ++it) {
374 
375  int pane_id = it->first;
376  std::string fname = get_sdv_fname( prefix, pane_id, IO_format);
377 
378  // Read the pane from the given file. Note that the file contains both
379  // the parent and the subdivided windows. Read only the subdivided one.
380  COM_call_function( hdl_read, fname.c_str(), _bufwin_prefix,
381  sdv_material.c_str(), &comm_self);
382 
383  int hdl_all = COM_get_dataitem_handle( (sdv_wname+".all").c_str());
384  COM_call_function( hdl_obtain, &hdl_all, &hdl_all, &pane_id);
385  it->second->read_rocin( sdv_wname);
386 
387  // Delete the window created by Rocin.
388  COM_delete_window( sdv_wname.c_str());
389  }
390 
391  // Unload Rocin
392  SimIN_unload_module( "RFC_IN");
393  }

405 void
406 RFC_Window_base::write_sdv( const char* prefix, const char *format) const {
407 
408  // Determine the format
409  int IO_format = get_sdv_format( format);
410 
411  if ( IO_format == SDV_OFF) {
412  std::string fname = get_sdv_fname( prefix, 0, IO_format);
413  std::FILE *fhandle = std::fopen( fname.c_str(), "w");
414  if ( !fhandle) {
415  std::cerr << "Rocface: Could not open output file " << fname
416  << "\nAbortting..." << std::endl;
417  RFC_assertion(fhandle); exit(-1);
418  }
419 
420  // Output using OFF file format
421  for ( Pane_set::const_iterator
422  it=_pane_set.begin(), iend=_pane_set.end(); it!=iend; ++it) {
423  RFC_Pane_base &pane = *it->second;
424  pane.write_off( fhandle);
425  }
426 
427  std::fclose( fhandle);
428  }
429  else if ( IO_format == SDV_BINARY) {
430  // Output using native file format
431  for ( Pane_set::const_iterator
432  it=_pane_set.begin(), iend=_pane_set.end(); it!=iend; ++it) {
433  std::string fname = get_sdv_fname( prefix, it->first, IO_format);
434 
435  std::ofstream os( fname.c_str());
436  if ( !os) {
437  std::cerr << "Rocface: Could not open output file " << fname
438  << "\nAbortting..." << std::endl;
439  RFC_assertion(os); exit(-1);
440  }
441 
442  RFC_Pane_base &pane = *it->second;
443  pane.write_binary( os);
444  }
445  }
446  else {
447  // Output using Rocout's format
448  SimOUT_load_module( "RFC_OUT");
449  int hdl_set_option = COM_get_function_handle( "RFC_OUT.set_option");
450  int hdl_write = COM_get_function_handle( "RFC_OUT.write_dataitem");
451 
452  COM_call_function( hdl_set_option, "rankwidth", "0");
453  COM_call_function( hdl_set_option, "format", format);
454 
455  int parent_mesh = COM_get_dataitem_handle( (name()+".pmesh").c_str());
456 
457  // Define the base-window and sdv-window names
458  std::string base_material = get_prefix_base( prefix);
459  std::string sdv_material = base_material+"_sdv";
460 
461  std::string buf_wname(_bufwin_prefix); buf_wname.append( base_material);
462  std::string sdv_wname=buf_wname+"_sdv";
463 
464  // Create a window for calling Rocout.
465  COM_new_window( sdv_wname.c_str());
466  new_sdv_dataitems( sdv_wname);
467  int win_all = COM_get_dataitem_handle( (sdv_wname+".all").c_str());
468 
469  const char *time_level = "000";
470  for ( Pane_set::const_iterator
471  it=_pane_set.begin(), iend=_pane_set.end(); it!=iend; ++it) {
472  int pane_id = it->first;
473 
474  // Define a file name name
475  std::string fname = get_sdv_fname( prefix, pane_id, IO_format);
476 
477  // Initialize the dataitems for the subdivision pane
478  it->second->register_sdv_dataitems( sdv_wname);
479 
480  // write parent mesh of the given pane into the file
481  COM_call_function( hdl_set_option, "mode", "w");
482  COM_call_function( hdl_write, fname.c_str(), &parent_mesh,
483  base_material.c_str(), time_level,
484  NULL, NULL, &pane_id);
485 
486  // append the subdivision pane into the file.
487  COM_call_function( hdl_set_option, "mode", "a");
488  COM_call_function( hdl_write, fname.c_str(), &win_all,
489  sdv_material.c_str(), time_level,
490  NULL, NULL, &pane_id);
491 
492  // remove the subdivision pane after output it.
493  COM_delete_pane( sdv_wname.c_str(), pane_id);
494  COM_window_init_done( sdv_wname.c_str());
495  }
496 
497  // Unload Rocout
498  COM_delete_window( sdv_wname.c_str());
499  SimOUT_unload_module( "RFC_OUT");
500  }

503 // Get the basename of the prefix without the directory part and no suffix.
504 const char *RFC_Window_base::
505 get_prefix_base(const char *prefix)
506 {
507  const char *last_slash = std::strrchr( prefix, '/');
