52 // Write the subdivision in binary format.
53 void RFC_Pane_base::
54 write_binary( std::ostream &os) const {
55  RFC_assertion( sizeof(int)==4);
56  RFC_assertion( sizeof(Real)==8);
57 
58  write( os, int(1)); // For detecting big or small endian
59  Real three = 3.; // Version number (New version changed on 04/11/04)
60  os.write( (char *)&three, sizeof(Real));
61 
62  // Write out color and the id of the pane.
63  write( os, _color); write( os, id());
64 
65  // #nodes and #faces
66  write( os, size_of_nodes()); write( os, size_of_faces());
67 
68  // Nodal coordinates of the pane
69  const Real *p = _base->coordinates();
70  os.write( (const char*)p, 3*_base->size_of_nodes()*sizeof(Real));
71 
72  if ( _base->is_structured()) {
73  write( os, 1);
74  write( os, 2); write( os, 1);
75  write( os, _base->size_i()); write( os, _base->size_j());
76  }
77  else {
78  std::vector< const COM::Connectivity*> elems;
79  _base->elements( elems);
80  write( os, elems.size());
81  // Output sizes of the faces
82  for ( std::vector<const COM::Connectivity*>::const_iterator
83  it = elems.begin(); it != elems.end(); ++it) {
84  write( os, (*it)->size_of_nodes_pe());
85  write( os, (*it)->size_of_elements());
86 
87  const int *e = (*it)->pointer();
88  const int nn = (*it)->size_of_nodes_pe();
89  os.write( (const char*)e, (*it)->size_of_elements()*nn*sizeof(int));
90  }
91  }
92 
93  // Pane connectivity of input mesh
94  write( os, _b2v_table.size());
95  for ( B2v_table::const_iterator
96  it=_b2v_table.begin(), iend=_b2v_table.end(); it != iend; ++it) {
97  const B2v &b2v = it->second;
98 
99  int n=b2v.size();
100  write( os, it->first); write( os, n);
101  os.write( (const char*)&b2v[0], n*sizeof(b2v[0]));
102  }
103  write ( os, _v2b_table.size());
104  for ( V2b_table::const_iterator
105  it = _v2b_table.begin(), iend=_v2b_table.end(); it!=iend; ++it) {
106  write( os, it->first);
107  write( os, it->second.first);
108  write( os, it->second.second);
109  }
110 
111  write(os, _subnode_parents.size()); write(os, _subfaces.size());
112 
113  // Output subnodes.
114  {
115  int n=_subnode_parents.size();
116  RFC_assertion( sizeof(_subnode_parents[0])==8);
117  os.write( (const char*)&_subnode_parents[0],
118  n*sizeof(_subnode_parents[0]));
119  os.write( (const char*)&_subnode_nat_coors[0],
120  n*sizeof(_subnode_nat_coors[0]));
121  os.write( (const char*)&_subnode_counterparts[0],
122  n*sizeof(_subnode_counterparts[0]));
123  }
124 
125  // Output subfaces.
126  {
127  int n = _subfaces.size();
128  os.write( (const char*)&_subfaces[0],
129  n*sizeof(_subfaces[0]));
130  os.write( (const char*)&_subface_parents[0],
131  n*sizeof(_subface_parents[0]));
132  os.write( (const char*)&_subface_counterparts[0],
133  n*sizeof(_subface_counterparts[0]));
134  }
135 
136  // Offsets of sub-faces.
137  RFC_assertion( int(_subface_offsets.size()) == int(size_of_faces())+1);
138  os.write( (const char*)&_subface_offsets[0],
139  sizeof(int)*_subface_offsets.size());
140 
141  write( os, _subnode_parents.size()+_subfaces.size()); // Checksum
142 }

144 // Read in RFC_Pane_base in binary format.
145 void RFC_Pane_base::
146 read_binary( std::istream &is, std::vector <int> *b2v_all, COM::Pane *pn_in) {
147 
148  COM_Pane_friend *pn = (COM_Pane_friend*)pn_in;
149  RFC_assertion( sizeof(int)==4);
150  RFC_assertion( sizeof(Real)==8);
151 
152  int t1, t2;
153 
154  read( is, t1); // For detecting big or small endian
155  bool need_swap = (t1 != 1);
156  if ( need_swap) swap_endian( t1);
157  RFC_assertion( t1 == 1);
158  Real version;
159  is.read( (char*)&version, sizeof(Real));
160  if ( need_swap) swap_endian( version);
161  RFC_assertion( version == 3.);
162 
163  // Read in the color and the id of the pane.
164  read( is, t1); read( is, t2);
165  if ( need_swap) { swap_endian( t1); swap_endian( t2); }
166 
167  // #nodes and #faces
168  read( is, t1); read( is, t2);
169  if ( need_swap) { swap_endian( t1); swap_endian( t2); }
170  if ( pn != NULL) pn->set_size( pn->dataitem(COM::COM_NC), t1, 0);
171  else
172  RFC_assertion( t1 == size_of_nodes() && t2 == size_of_faces());
173  _subface_offsets.resize( t2);
174 
175  // Nodal coordinates of the pane
176  {
177  int n=3*_base->size_of_nodes();
178  std::vector< Real> buf( n);
179  is.read( (char*)&buf.front(), n*sizeof(Real));
180  }
181 
182  // Read in number of element types.
183  int ntypes, is_structured=false;
184  read( is, ntypes); if ( need_swap) { swap_endian( ntypes); }
185  for ( int i=0; i<ntypes; ++i) {
186  read( is, t1); read( is, t2);
187  if ( need_swap) { swap_endian( t1); swap_endian( t2); }
188 
189  if ( t1 == 2) { // Structured mesh
190  is_structured=true;
191  RFC_assertion( t2==1);
192  int dims[2];
193  read( is, dims[0]); read( is, dims[1]);
194  if ( need_swap) { swap_endian( dims[0]); swap_endian( dims[1]); }
195  if ( pn != NULL) {
196  int *t = &dims[0];
197  COM::Connectivity *conn = pn->connectivity( ":st2:");
198  pn->reinit_conn( conn, COM::Pane::OP_SET, &t, 0, 0);
199  }
200  }
201  else { // Unstructured mesh
202  int n=t1*t2;
203  if ( pn) {
204  int *buf=NULL;
205  std::string elem;
206  // Create a new Connectivity object in the Pane
207  switch (t1) {
208  case 3: elem=":t3:"; break;
209  case 4: elem=":q4:"; break;
210  case 6: elem=":t6:"; break;
211  case 8: elem=":q8:"; break;
212  case 9: elem=":q9:"; break;
213  default: RFC_assertion(false);
214  }
215  // Insert a connectivity
216  COM::Connectivity *conn=pn->connectivity( elem);
217  pn->set_size( conn, t2, 0);
218  pn->reinit_conn( conn, COM::Pane::OP_RESIZE, &buf, 0, 0);
219 
220  is.read( (char*)buf, n*sizeof(int));
221  if ( need_swap) for ( int j=0; j<n; ++j) swap_endian( buf[j]);
222  }
223  else {
224  int t;
225  for ( int i=0; i<n; ++i) is.read( (char*)&t, sizeof(int));
226  }
227  }
228  }
229 
230  // Pane connectivity of input mesh
231  read( is, t1); t2 = size_of_nodes();
232  if ( need_swap) { swap_endian( t1); }
233  if ( b2v_all) b2v_all->reserve(t1);
234  for ( int i=0; i<t1; ++i) {
235  int pane_id, n;
236  read( is, pane_id); read( is, n);
237  if ( need_swap) { swap_endian( pane_id); swap_endian( n); }
238  B2v &b2v = _b2v_table[pane_id];
239  b2v.resize( n);
240  is.read( (char*)&b2v[0], n*sizeof(b2v[0]));
241  if ( need_swap) for ( int j=0; j<n; ++j) swap_endian( b2v[j]);
242  RFC_assertion_code( for ( int j=0; j<n; ++j)
243  RFC_assertion( b2v[j]>=1 && b2v[j]<=t2));
244  if ( b2v_all) {
245  b2v_all->push_back( pane_id); b2v_all->push_back( n);
246  b2v_all->insert( b2v_all->end(), b2v.begin(), b2v.end());
247  }
248  }
249 
250  read( is, t1); if ( need_swap) { swap_endian( t1); }
251  for ( int i=0; i<t1; ++i) {
252  int n1,n2,n3;
253  read( is, n1); if ( need_swap) { swap_endian( n1); }
254  read( is, n2); if ( need_swap) { swap_endian( n2); }
255  read( is, n3); if ( need_swap) { swap_endian( n3); }
256 
257  _v2b_table[n1] = std::make_pair( n2, n3);
258  }
259 
260  read( is, t1); read( is, t2);
261  if ( need_swap) { swap_endian( t1); swap_endian( t2); }
262  _subnode_parents.resize( t1);
263  _subnode_nat_coors.resize( t1);
264  _subnode_counterparts.resize( t1);
265  _subfaces.resize( t2);
266  _subface_parents.resize( t2);
267  _subface_counterparts.resize( t2);
268 
269  // Read in the subnodes.
270  int n=_subnode_parents.size();
271  RFC_assertion( sizeof(_subnode_parents[0])==8);
272  is.read( (char*)&_subnode_parents[0], n*sizeof(_subnode_parents[0]));
273  if ( need_swap)
274  for ( int j=0; j<n; ++j) {
275  swap_endian( _subnode_parents[j].face_id);
276  swap_endian( _subnode_parents[j].edge_id);
277  }
278 
279  is.read( (char*)&_subnode_nat_coors[0],
280  n*sizeof(_subnode_nat_coors[0]));
281  if ( need_swap)
282  for ( int j=0; j<n; ++j) {
283  swap_endian( _subnode_nat_coors[j][0]);
284  swap_endian( _subnode_nat_coors[j][1]);
285  }
286 
287  is.read( (char*)&_subnode_counterparts[0],
288  n*sizeof(_subnode_counterparts[0]));
289  if ( need_swap)
290  for ( int j=0; j<n; ++j) {
291  swap_endian( _subnode_counterparts[j].pane_id);
292  swap_endian( _subnode_counterparts[j].node_id);
293  }
294 
295  n = t2;
296  // Read in the subfaces.
297  is.read( (char*)&_subfaces[0], n*sizeof(_subfaces[0]));
298  if ( need_swap)
299  for ( int j=0; j<n; ++j) {
300  swap_endian( _subfaces[j][0]);
301  swap_endian( _subfaces[j][1]);
302  swap_endian( _subfaces[j][2]);
303  }
304  is.read( (char*)&_subface_parents[0], n*sizeof(_subface_parents[0]));
305  if ( need_swap)
306  for ( int j=0; j<n; ++j) {
307  swap_endian( _subface_parents[j]);
308  }
309  is.read( (char*)&_subface_counterparts[0],
310  t2*sizeof(_subface_counterparts[0]));
311  if ( need_swap)
312  for ( int j=0; j<n; ++j) {
313  swap_endian( _subface_counterparts[j].pane_id);
314  swap_endian( _subface_counterparts[j].face_id);
315  }
316 
317  // Read in the offset of sub-faces.
318  _subface_offsets.resize( size_of_faces()+1);
319  n = _subface_offsets.size();
320  is.read( (char*)&_subface_offsets[0], sizeof(int)*n);
321  if ( need_swap) for ( int j=0; j<n; ++j) swap_endian( _subface_offsets[j]);
322 
323  RFC_assertion( _subface_offsets[size_of_faces()] ==
324  int(_subfaces.size()));
325 
326  read( is, t1);
327  if ( need_swap) { swap_endian( t1); }
328  RFC_assertion( t1 ==int(_subnode_parents.size()+_subfaces.size())); // Checksum
329 
330  comp_nat_coors();
331 }
