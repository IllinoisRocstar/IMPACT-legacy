17 // The main interface for output a pane in Tecplot format.
18 void RFC_Pane_base::
19 write_tec_ascii( std::ostream &os, const COM::DataItem *attr) const {
20  if ( _base->size_of_nodes()==0) return;
21 
22  if ( _base->is_structured())
23  write_tec_ij( os, attr); // Structured mesh
24  else { // Unstructured mesh
25  std::vector< const COM::Connectivity*> elems;
26  _base->elements( elems);
27 
28  if ( elems.size() == 1 && elems[0]->size_of_edges_pe() == 3)
29  // Triangular finite element mesh
30  write_tec_tri( os, *elems[0], attr);
31  else
32  // Quadrilateral/mixed finite element mesh
33  write_tec_mixed( os, elems, attr);
34  }
35 }

37 static void write_dataitem( std::ostream &os, const COM::DataItem *attr, int i) {
38  const int dim = attr->size_of_components();
39  switch ( attr->data_type()) {
40  case COM_DOUBLE:
41  case COM_DOUBLE_PRECISION: {
42  const double *values = reinterpret_cast<const double*>(attr->pointer());
43  for ( int j=0; j<dim; ++j) os << ' ' << values[i*dim+j];
44  break;
45  }
46  case COM_FLOAT:
47  case COM_REAL: {
48  const float *values = reinterpret_cast<const float*>(attr->pointer());
49  for ( int j=0; j<dim; ++j) os << ' ' << values[i*dim+j];
50  break;
51  }
52  case COM_INT:
53  case COM_INTEGER: {
54  const int *values = reinterpret_cast<const int*>(attr->pointer());
55  for ( int j=0; j<dim; ++j) os << ' ' << values[i*dim+j];
56  break;
57  }
58  default:
59  assert(false); abort();
60  }
61 }

63 // Write out an IJ-ordered data in POINT format of Tecplot.
64 void RFC_Pane_base::
65 write_tec_ij( std::ostream &os, const COM::DataItem *attr) const {
66  RFC_assertion( _base != NULL);
67 
68  // Write out the head
69  if ( attr)
70  os << "TITLE = \"" << attr->name() << "\" \n";
71  else
72  os << "TITLE = \"Structured mesh\" \n";
73  os << "VARIABLES = \"X\", \"Y\", \"Z\"";
74  if ( attr) {
75  const int MAXDIM=4, dim=attr->size_of_components();
76  const char vname[MAXDIM+1]="UVWT";
77  RFC_assertion( dim<=MAXDIM);
78  for ( int i=0; i<dim; ++i)
79  os << ", \"" << vname[i] << "\"";
80  }
81  os << "\nZONE I=" << _base->size_i()
82  << ", J=" << _base->size_j() << ", F=POINT\n";
83 
84  // Write out the coordinates
85  const Real *p = _base->coordinates();
86  for ( int i=0, n=_base->size_of_nodes(); i<n; ++i) {
87  os << p[3*i] << ' ' << p[3*i+1] << ' ' << p[3*i+2];
88  if ( attr) write_dataitem( os, attr, i);
89  os << std::endl;
90  }
91  os << std::endl;
92 }

131 // Write out a triangular finite element mesh in FEPOINT format of Tecplot
132 void RFC_Pane_base::
133 write_tec_mixed( std::ostream &os,
134  const std::vector<const COM::Connectivity*> &elems,
135  const COM::DataItem *attr) const {
136  RFC_assertion( _base != NULL);
137 
138  if ( attr)
139  os << "TITLE = \"" << attr->name() << "\" \n";
140  else
141  os << "TITLE = \"Triangular finite element mesh\" \n";
142  os << "VARIABLES = \"X\", \"Y\", \"Z\"";
143  if ( attr) {
144  const int MAXDIM=4, dim=attr->size_of_components();
145  const char vname[MAXDIM+1]="UVWT";
146  RFC_assertion( dim<=MAXDIM);
147  for ( int i=0; i<dim; ++i)
148  os << ", \"" << vname[i] << "\"";
149  }
150  os << "\nZONE N=" << _base->size_of_nodes() << ", E="
151  << _base->size_of_elements() << ", F=FEPOINT, ET=QUADRILATERAL\n";
152 
153  // Write out the coordinates
154  const Real *p = _base->coordinates();
155  for ( int i=0, n=_base->size_of_nodes(); i<n; ++i) {
156  os << p[3*i] << ' ' << p[3*i+1] << ' ' << p[3*i+2];
157  if ( attr) write_dataitem( os, attr, i);
158  os << std::endl;
159  }
160  os << std::endl;
161 
162  // Write out the elements
163  for ( std::vector<const COM::Connectivity*>::const_iterator
164  it = elems.begin(); it != elems.end(); ++it) {
165  const int *e = (*it)->pointer();
166  int nn = (*it)->size_of_nodes_pe();
167  switch ( (*it)->size_of_edges_pe()) {
168  case 3:
169  for ( int i=0,n=(*it)->size_of_elements(); i<n; ++i)
170  { os << e[nn*i] << ' ' << e[nn*i+1] << ' '
171  << e[nn*i+2] << ' ' << e[nn*i+2] << std::endl; }
172  break;
173  case 4:
174  for ( int i=0, n=(*it)->size_of_elements(); i<n; ++i)
175  { os << e[nn*i] << ' ' << e[nn*i+1] << ' '
176  << e[nn*i+2] << ' ' << e[nn*i+3] << std::endl; }
177  break;
178  default:
179  RFC_assertion( false);
180  }
181  }
182 }

185 // Write out the subdivision in FEPOINT format of Tecplot
186 void RFC_Pane_base::
187 write_tec_sub( std::ostream &os) const {
188  RFC_assertion( _base != NULL);
189  if ( size_of_subnodes() == 0) return;
190 
191  os << "TITLE = \"Triangular subdivision mesh\" \n";
192  os << "VARIABLES = \"X\", \"Y\", \"Z\"";
193  os << "\nZONE N=" << size_of_subnodes() << ", E="
194  << size_of_subfaces() << ", F=FEPOINT, ET=TRIANGLE\n";
195 
196  // Write out the coordinates, and the node ids start from 1.
197  for ( int i=1, size=size_of_subnodes(); i<=size; ++i) {
198  Point_3 p = get_point_of_subnode( i);
199 
200  os << p.x() << ' ' << p.y() << ' ' << p.z() << std::endl;
201  }
202  os << std::endl;
203 
204  // Write out the elements. We directly access the connectivity table
205  // and their indices start from 0.
206  for ( int i=0, size=size_of_subfaces(); i<size; ++i) {
207  os << _subfaces[i][0] << ' ' << _subfaces[i][1]
208  << ' ' << _subfaces[i][2] << std::endl;
209  }
210 }
