94 // Evaluate normals for every vertex within a pane as the average of
95 // face normals of the faces of its incident faces.
96 void
97 RFC_Pane_overlay::evaluate_normals() {
98  _nrmls.resize( size_of_nodes(), Vector_3(0.,0.,0.));
99 
100  for ( int vid=1, s=size_of_nodes(); vid<=s; ++vid) {
101  Node v( this, vid);
102  if (v.is_isolated()) continue;
103  Vector_3 d(0.,0.,0.);
104 
105  HEdge h = v.halfedge_l(), i = h;
106  if ( h.destination_l() != v) continue;
107 
108  do {
109  RFC_assertion( i.destination_l() == v);
110  if ( !i.is_border_l())
111  d += Overlay_primitives().get_face_normal( i, Point_2(0,0), scheme);
112 
113  if ( i.opposite_l().is_border_l()) break;
114  } while ( (i = i.opposite_l().prev_l()) != h);
115 
116  // Note that the normal vector is not normalized.
117  set_normal( vid, d);
118  }
119 }

121 // Evaluate vertex normals as average of face normals of all incident faces.
122 void
123 RFC_Window_overlay::evaluate_normals() {
124  // std::cout << "eval normals" << std::endl;
125  // First, evaluate normals for each pane
126  Pane_set::iterator it, iend=_pane_set.end();
127  for ( it=_pane_set.begin() ; it != iend; ++it) {
128  if ( it->second->is_master())
129  ((RFC_Pane_overlay*)it->second)->evaluate_normals();
130  }
131 
132  // std::cout << "reducing" << std::endl;
133  // Second, perform reduction over all vertices
134  reduce_normals_to_all( MPI_SUM);
135  // std::cout << "reducing done" << std::endl;
136 
137  // Evaluate the one-sided normals for the halfedge edges incident
138  // on sharp features.
139  for ( it=_pane_set.begin() ; it != iend; ++it) {
140  RFC_Pane_overlay &pane = (RFC_Pane_overlay &)*it->second;
141  free_vector( pane._f_nrmls); free_vector( pane._f_tngts);
142  free_vector( pane._f_n_index); free_vector( pane._f_t_index);
143  int num_hedgs=4*pane.size_of_faces() + pane.size_of_border_edges(); // pane.hds().size_of_halfedges();
144  pane._f_n_index.resize( num_hedgs, -1);
145  pane._f_t_index.resize( num_hedgs, -1);
146  }
147 
148  Overlay_primitives op;
149  Point_2 one(1,0);
150  // Loop throught the sharp halfedges in 1-features
151  for ( Feature_list_1::iterator it = _f_list_1.begin(), iend=_f_list_1.end();
152  it != iend; ++it) {
153  for ( Feature_1::iterator j=it->begin(); j!=it->end(); ++j) {
154  HEdge k = *j, kopp=k.opposite_g();
155  do {
156  if (!k.is_border_l() && k.pane()->
157  _f_n_index[k.pane()->get_index(k)] == -1) {
158  // First, compute the vector
159  Vector_3 v = op.get_face_normal( k, one, scheme);
160  HEdge h0 = k, h = acc.get_next_around_destination(k);
161  do {
162  if ( is_feature_1( h)) break;
163  RFC_assertion( !h.is_border_l());
164  v += op.get_face_normal( h, one, scheme);
165  } while ( (h=acc.get_next_around_destination(h)) != h0);
166 
167  // Second, insert the vector into the table
168  normalize(v);
169  swap( h, h0);
170  do {
171  RFC_Pane_overlay &pane = *h.pane();
172  if ( pane._f_nrmls.size() && v == pane._f_nrmls.back()) {
173  pane._f_n_index[ pane.get_index( h)] = pane._f_nrmls.size()-1;
174  }
175  else {
176  pane._f_n_index[ pane.get_index( h)] = pane._f_nrmls.size();
177  pane._f_nrmls.push_back( v);
178  }
179  } while ( (h=acc.get_next_around_destination(h)) != h0);
180  }
181  } while ( (k=(k==kopp)?*j:kopp) != *j);
182 
183  // Compute the tangent
184  Vector_3 tng = ( k.destination_l().point() - kopp.destination_l().point());
185  k.pane()->add_tangent( k, tng);
186  if ( !is_feature_0( k.destination_g())) {
187  Feature_1::iterator t = j; ++t;
188  if ( t==it->end()) t = it->begin();
189  HEdge h = (*t).opposite_g();
190  h.pane()->add_tangent( h, tng);
191  }
192  kopp.pane()->add_tangent( kopp, tng);
193  if ( !is_feature_0( kopp.destination_g())) {
194  Feature_1::iterator t = (j==it->begin()) ? it->end() : j;
195  HEdge h = *(--t);
196  h.pane()->add_tangent( h, tng);
197  }
198  }
199  }
200 
201  // Loop through the vertices in 0-features
202  for ( Feature_list_0::iterator it = _f_list_0.begin(), iend=_f_list_0.end();
203  it != iend; ++it) {
204  HEdge h = it->vertex().halfedge_g();
205  RFC_Pane_overlay &pane = *h.pane();
206 
207  if ( !h.is_border_l() && pane._f_n_index[ pane.get_index( h)] == -1) {
208  HEdge h0 = h;
209  do {
210  Vector_3 v = op.get_face_normal( h, one, scheme)+
211  op.get_face_normal( h.opposite_g(), Point_2(0,0), scheme);
212  normalize(v);
213 
214  if ( pane._f_nrmls.size() && v == pane._f_nrmls.back()) {
215  pane._f_n_index[ pane.get_index( h)] = pane._f_nrmls.size()-1;
216  }
217  else {
218  pane._f_n_index[ pane.get_index( h)] = pane._f_nrmls.size();
219  pane._f_nrmls.push_back( v);
220  }
221  } while ( (h=acc.get_next_around_destination(h)) != h0);
222  }
223  }
224 
225  // Finally, normalize normals and tangents
226  for ( it=_pane_set.begin() ; it != iend; ++it) {
227  RFC_Pane_overlay &pane = (RFC_Pane_overlay &)*it->second;
228 
229  if ( pane.is_master()) {
230  for ( int i=0, size = pane._nrmls.size(); i<size; ++i)
231  normalize( pane._nrmls[i]);
232  for ( int i=0, size = pane._f_tngts.size(); i<size; ++i)
233  normalize( pane._f_tngts[i]);
234  }
235  }
236  // std::cout << "eval normals done." << std::endl;
237 }

240 const Vector_3 &RFC_Pane_overlay::get_normal( HEdge h, const Node &v) const {
241  RFC_assertion( h.pane() == this && v.pane()==this);
242 
243  if ( v != h.destination_l()) {
244  h = h.prev_l();
245  RFC_assertion( v == h.destination_l());
246  }
247  if ( h.id().is_border())
248  h = h.opposite_g().prev_l();
249 
250  int j=_f_n_index[get_index( h)];
251  if ( j < 0)
252  return _nrmls[ get_index(v)];
253  else
254  return _f_nrmls[j];

272 // Determine counterparts of border edges.
273 void
274 RFC_Pane_overlay::determine_counterparts() {
275  typedef set< pair<int,int> > V2h;
276  typedef map<int, V2h> V2h_table;
277  V2h_table _v2h_table;
278 
279  //`Loop through the B2v_table
280  for ( B2v_table::iterator
281  it=_b2v_table.begin(), iend=_b2v_table.end(); it != iend; ++it) {
282  for ( int i=0, size=it->second.size(); i<size; ++i) {
283  pair<int,int> p(it->first, i);
284  if ( it->first==id()) {
285  if ( i&1) --p.second; else ++p.second;
286  }
287  _v2h_table[it->second[i]].insert( p);
288  }
289  }
290 
291  _cntrs_hedge.resize( size_of_border_edges());
292 
293  // Loop through all border edges.
294  for ( int i=0; i<size_of_border_edges(); ++i) {
295  HEdge h( this, Edge_ID( i+1, Edge_ID::BndID));
296 
297  Node v = h.destination_l();
298  V2h_table::iterator it = _v2h_table.find( get_lid(v));
299  if ( it == _v2h_table.end()) continue;
300 
301  V2h_table::iterator i0=_v2h_table.find(get_lid( h.origin_l()));
302 
303  // If the origin not incident with any, there is no counterpart.
304  if ( i0==_v2h_table.end()) continue;
305 
306  // check whether the two vertices are overlapping.
307  V2h::iterator j0=i0->second.begin(), j1=it->second.begin();
308 
309  while ( j0 != i0->second.end() && j1 != it->second.end()) {
310  if ( j0->first == j1->first) {
311  bool found=false;
312  V2h::iterator k0, k1;
313  for ( k0=j0; k0!=i0->second.end()&&k0->first==j0->first; ++k0) {
314  if (found) continue;
315  for ( k1=j1; k1!=it->second.end()&&k1->first==j1->first; ++k1) {
316  if (found) continue;
317  // Check whether it is an edge in that pane
318  const RFC_Pane_overlay &p = _window->pane(k0->first);
319  const B2v &b2v = p._b2v_table.find(id())->second;
320  pair<int,int> e( b2v[k0->second], b2v[k1->second]);
321  RFC_assertion( e.first>0 && e.second>0);
322 
323  map< pair<int,int>, HEdge>::const_iterator
324  b2e_it = p._bv2edges.find( e);
325 
326  if ( b2e_it != p._bv2edges.end()) {
327  _cntrs_hedge[i] = b2e_it->second;
328  found=true;
329  }
330  else if ( k0->first != id()) {
331  swap( e.first, e.second);
332  if (p._bv2edges.find( e) != p._bv2edges.end()) {
333  std::cerr << "ERROR: Inconsistency in window \""
334  << _window->name() << "\"!!!! \n\tPanes " << id()
335  << " and " << j0->first
336  << " have opposite orientation!" << std::endl
337  << "\tComputation can not continue. Aborting..."
338  << std::endl;
339  RFC_assertion( p._bv2edges.find( e) == p._bv2edges.end());
340  abort();
341  }
342  }
343  }
344  }
345  if ( found) { j0 = k0; j1 = k1; }
346  else { ++j0; ++j1; }
347  }
348  else
349  if ( j0->first < j1->first) ++j0; else ++j1;
350  }
351  }
352 
353  // Fix physical border ones to have them point to themselves
354  for ( int i=0; i<size_of_border_edges(); ++i) {
355  HEdge h( this, Edge_ID( i+1, Edge_ID::BndID));
356 
357  if ( _cntrs_hedge[i].pane() == NULL) _cntrs_hedge[i] = h;
358  }
359 
360  // Determine the primaries of the border vertices
361  _primnodes.resize( size_of_border_edges());
362  for ( int i=1; i<=size_of_border_edges(); ++i) {
363  HEdge h( this, Edge_ID( i, Edge_ID::BndID));
364 
365  int vid = get_lid( h.destination_l());
366  pair<Self*,int> t=_window->get_primary( this, vid);
367 
368  Node n(this, vid);
369  _primnodes[ get_border_index( n.halfedge_l())] =
370  t.first->get_node_from_id( t.second);
371  }
372 }
