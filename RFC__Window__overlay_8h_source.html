517 HEdge Node::halfedge_g() const {
518  Edge_ID eID =_pm->get_incident_real_edge( _vID);
519 
520  // the current node (vertex) is in general the origin of the edge or
521  // the edge center, unless the node is isolated.
522  if (!eID.is_border())
523  return HEdge( _pm, _pm->get_opposite_real_edge(eID));
524 
525  HEdge h(_pm, _pm->get_opposite_real_edge( eID)), h0=h;
526  do {
527  if (h.is_border_l()) break;
528  } while ( (h=h.next_g().opposite_g()) != h0);
529 
530  assert( h.destination_g() == get_primary());
531  return h;
532 }
