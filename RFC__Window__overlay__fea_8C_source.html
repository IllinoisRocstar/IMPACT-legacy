121 float
122 RFC_Window_overlay::cos_face_angle( const HEdge &h, const HEdge &hopp) {
123  // Precondition: Face angles are initialized to HUGE_VALF.
124  RFC_Pane_overlay *p1=h.pane();
125  float &t = p1->get_cos_face_angle(h);
126  if ( t<HUGE_VALF) return t;
127 
128  if ( h.is_border_g() || hopp.is_border_g())
129  t = -1;
130  else {
131  Overlay_primitives op;
132 
133  Vector_3 v1 = op.get_face_normal( h, Point_2(0.5,0));
134  Vector_3 v2 = op.get_face_normal( hopp, Point_2(0.5,0));
135 
136  t=std::min(v1 * v2 / sqrt( (v1*v1) * (v2*v2)),1.);
137  RFC_Pane_overlay *p2=hopp.pane();
138  if ( p1 != p2) p2->get_cos_face_angle( hopp) = t;
139  }
140  return t;
141 }

146 float RFC_Window_overlay::
147 cos_edge_angle( const HEdge &h1, const HEdge &h2) {
148  Vector_3 v1 = get_tangent( h1);
149  Vector_3 v2 = get_tangent( h2);
150 
151  float d=std::max( -1., std::min( (v1*v2)/sqrt((v1*v1)*(v2*v2)), 1.));
152  if ( h1.destination_g() == h2.destination_g() ||
153  h1.origin_g() == h2.origin_g())
154  return -d;
155  else
156  return d;

207 float
208 RFC_Window_overlay::comp_angle_defect( const Node &v) {
209  const float pi = 3.1415926535;
210 
211  // Precondition: d was initialized to HUGE_VALF
212  float &t=v.pane()->get_angle_defect(v), d=t;
213  if ( d < HUGE_VALF) return d;
214 
215  float angle_sum = 0.;
216 
217  HEdge h = v.halfedge_g(), h0=h;
218  bool is_border = false;
219  do {
220  if ( !h.is_border_g())
221  angle_sum += acos( -cos_edge_angle( h, h.next_l()));
222  else
223  is_border = true;
224  } while ( (h=acc.get_next_around_destination( h)) != h0);
225 
226  return t = abs((2-is_border)*pi - angle_sum);
227 }

236 bool
237 RFC_Window_overlay::is_strong_ad( const Node &v) {
238  // Determine whether the edge is theta-strong
239  float d = comp_angle_defect( v);
240  if ( d >= _ud) {
241  return true;
242  }
243  else if ( d >= _ld) {
244  float max_ad=0;
245  HEdge h0 = v.halfedge_g(), h1=h0;
246  // Loop through incident edges of v
247  do {
248  float a = comp_angle_defect( h1.origin_g());
249  max_ad = std::max( max_ad, a);
250  } while ( (h1=acc.get_next_around_destination(h1))!=h0);
251 
252  if ( d >= _rd*max_ad)
253  return true;
254  }
255  return false;
256 }

312 bool
313 RFC_Window_overlay::check_false_strong_1( Feature_1 &f1) {
314  if ( cos_face_angle( f1.front())==-1) return false;
315 
316  Node src = f1.front().origin_g();
317  Node dst = f1.back().destination_g();
318 
319  // is a loop or strong at both ends
320  if ( src == dst || is_on_feature( src) && is_on_feature( dst)
321  && (!_f0_ranks.empty() || is_feature_0(src) || is_feature_0(dst)) ||
322  cos_face_angle(f1.front())<0 && cos_face_angle(f1.back())<0)
323  return false;
324 
325  // Is it weak ended? If so, pop out the weak edges
326  if ( _strong_ended) {
327  if ( !f1.empty()) for (;;) {
328  HEdge h=f1.front(), hopp=h.opposite_g();
329 
330  if ( is_on_feature( src) || cos_face_angle( h, hopp) <= _cos_uf || is_strong_ad( src))
331  break;
332  else {
333  h.pane()->unset_strong_edge(h);
334  hopp.pane()->unset_strong_edge( hopp);
335  f1.pop_front();
336  }
337  if ( !f1.empty())
338  src = f1.front().origin_g();
339  else
340  return true;
341  }
342 
343  if ( !f1.empty()) for (;;) {
344  HEdge h=f1.back();
345  HEdge hopp=h.opposite_g();
346 
347  if ( is_on_feature( dst) || cos_face_angle( h, hopp) <= _cos_uf
348  || is_strong_ad( src))
349  break;
350  else {
351  h.pane()->unset_strong_edge(h);
352  hopp.pane()->unset_strong_edge( hopp);
353  f1.pop_back();
354  }
355  if ( !f1.empty())
356  dst = f1.back().destination_g();
357  else
358  return true;
359  }
360  }
361 
362  // is it too short?
363  if ( _min_1f_len) {
364  int c = !(is_strong_ad( src) || is_feature_0( src)) +
365  !(is_strong_ad( dst) || is_feature_0( dst));
366  if ( int(f1.size()) <= c*_min_1f_len)
367  return true;
368  }
369 
370  // is it too close to another features?
371  if ( _long_falseness_check) {
372  if ( _f_list_1.empty() ||
373  (is_strong_ad( src) || is_feature_0( src)) &&
374  (is_strong_ad( dst) || is_feature_0( dst)))
375  return false;
376  for ( Feature_1::const_iterator it=++f1.begin(); it!=f1.end(); ++it) {
377  Node v = it->origin_g();
378  HEdge h0 = v.halfedge_g(), h1=h0;
379  // Loop through incident edges of v
380  do {
381  if ( is_on_feature( h1.origin_g())) { v=Node(); break; }
382  } while ( (h1=acc.get_next_around_destination(h1))!=h0);
383  if (v.pane()==NULL) continue;
384  else return false;
385  }
386  return true;
387  }
388  return false;
389 }

394 void RFC_Window_overlay::
395 subdiv_feature_curve( const Feature_1 &f1,
396  Feature_list_1 &new_flist, int &dropped) {
397 
398  // We loop through all vertices in the curve to locate the 0-features
399  list< Feature_1::const_iterator> divs;
400  Node src = f1.front().origin_g();
401  Node trg = f1.back().destination_g();
402  bool isloop = (src==trg);
403 
404  // A 1-feature is dangling if one of its end point is a terminus
405  bool isdangling = !isloop &&
406  (_f0_ranks.find( src) == _f0_ranks.end() ||
407  _f0_ranks.find( trg) == _f0_ranks.end());
408 
409  float cos_min_fa_before=1, cos_min_fa_after=1;
410  Feature_1::const_iterator f1end = f1.end(), hfirst = f1end, hlast=f1end;
411 
412  for (Feature_1::const_iterator hprev=f1.begin(),hi=++f1.begin(),hnext=hi;
413  hi!=f1end; hprev=hi,hi=hnext) {
414  ++hnext;
415  Node v = hi->origin_g();
416  if ( is_feature_0( v)) { RFC_assertion( hi==f1.begin()); continue; }
417 
418  float cos_ea = cos_edge_angle( f1, hi, isloop);
419  if ( cos_ea <= _cos_ue ||
420  is_rstrong_ea( f1, hprev, hnext, cos_ea, isloop)) {
421  divs.push_back( hi);
422  // When breaking loops, mark breakpoints as corners
423  set_feature_0( v);
424  }
425  else if ( isdangling) {
426  // Mark transitions between strong and week edges for danging 1-features
427  // as 0-feature
428  float dpre = cos_face_angle( *hprev, hprev->opposite_g());
429  float d = cos_face_angle( *hi, hi->opposite_g());
430 
431  if ( hfirst == f1end && dpre < cos_min_fa_before)
432  cos_min_fa_before = dpre;
433 
434  if ( std::min( dpre,d) <= _cos_weakend &&
435  std::max( dpre,d) > _cos_weakend ) {
436  if ( hfirst == f1end) hfirst = hi;
437  hlast = hi; cos_min_fa_after = 1.;
438  }
439 
440  if ( d < cos_min_fa_before) cos_min_fa_after = d;
441  }
442  }
443 
444  if ( isdangling && divs.empty() ) {
445  if ( hfirst!=f1end && cos_min_fa_before>_cos_weakend &&
446  _f0_ranks.find( src) == _f0_ranks.end() )
447  { divs.push_back( hfirst); set_feature_0( hfirst->origin_g()); }
448 
449  if ( hlast!=f1end && hlast!=hfirst && cos_min_fa_after>_cos_weakend &&
450  _f0_ranks.find( trg) == _f0_ranks.end())
451  { divs.push_back( hlast); set_feature_0( hlast->origin_g()); }
452  }
453 
454  if ( isloop) {
455  // Finally, we process end vertices
456  Feature_1::const_iterator hi = f1.begin();
457  float cos_ea = cos_edge_angle( f1, hi, isloop);
458  if ( cos_ea <= _cos_ue ||
459  is_rstrong_ea( f1,f1end,++f1.begin(),cos_ea,isloop)) {
460  // When breaking loops, mark breakpoints as corners
461  divs.push_back( f1end); set_feature_0( hi->origin_g());
462  }
463  }
464  else
465  divs.push_back( f1end);
466 
467  // Now subdivide the curve into sub-curves
468  Feature_list_1 subcur;
469  if ( divs.size()==0)
470  subcur.push_back( f1);
471  else {
472  subcur.push_back( Feature_1( f1.begin(),divs.front()));
473  list< Feature_1::const_iterator>::const_iterator dit=divs.begin();
474  for ( list< Feature_1::const_iterator>::const_iterator
475  dinext=dit; ++dinext != divs.end(); dit=dinext)
476  subcur.push_back( Feature_1( *dit, *dinext));
477 
478  if ( isloop) {
479  Feature_1 &newf=subcur.front();
480  newf.insert( newf.begin(), *dit, f1end);
481  }
482  }
483 
484  // Check the sub-curves to filter out false-strongness.
485  for (Feature_list_1::iterator sit=subcur.begin();sit!=subcur.end();++sit) {
486  if ( check_false_strong_1( *sit)) {
487  for ( Feature_1::const_iterator i=sit->begin(); i!=sit->end(); ++i) {
488  i->pane()->unset_strong_edge(*i); ++dropped;
489  HEdge hopp=i->opposite_g();
490  hopp.pane()->unset_strong_edge( hopp);
491  }
492  }
493  else {
494  // Set all vertices on-feature
495  set_on_feature( sit->front().origin_g());
496  for ( Feature_1::const_iterator i=sit->begin(); i!=sit->end(); ++i)
497  set_on_feature( i->destination_g());
498 
499  Node src = sit->front().origin_g();
500  Node dst = sit->back().destination_g();
501 
502  if ( src!=dst) { set_feature_0( src); set_feature_0( dst); }
503  new_flist.push_back( Feature_1()); new_flist.back().swap(*sit);
504  }
505  }
506 }

710 RFC_Window_overlay::detect_features() {
711  int size_edges=0, dropped=0;
712 
713  // Initializing data arrays.
714  Pane_set::iterator it=_pane_set.begin(), iend=_pane_set.end();
715  for ( ; it != iend; ++it) {
716  RFC_Pane_overlay &pane = (RFC_Pane_overlay &)*it->second;
717  int num_hedgs=4*pane.size_of_faces() + pane.size_of_border_edges();
718  int num_verts=pane.size_of_nodes();
719  size_edges+=num_hedgs/2;
720 
721  pane.init_face_angle(); pane.init_angle_defect(); pane.init_edge_angle();
722  pane._is_f_1.clear(); pane._is_f_1.resize( num_hedgs, false);
723  pane._is_f_0.clear(); pane._is_f_0.resize( num_verts, false);
724  pane._is_on_f.clear(); pane._is_on_f.resize( num_verts, false);
725  }
726  std::vector< pair<float, HEdge> > tstrong_edges, rstrong_edges;
727  tstrong_edges.reserve(size_edges/10);
728  rstrong_edges.reserve(size_edges/10);
729 
730  float t0=get_wtime(), totaltime=0;
731  // loop through all panes and primary halfedges in each pane
732  for ( it=_pane_set.begin(); it != iend; ++it) {
733  RFC_Pane_overlay &pane = (RFC_Pane_overlay &)*it->second;
734 
735  // Detect theta-strong edges
736  for ( int i=0, s=pane.size_of_faces(); i<s; ++i) {
737  HEdge h( &pane, Edge_ID(i+1,0)), h0=h;
738 
739  do {
740  HEdge lopp = h.opposite_l();
741 
742  if ( lopp.is_border_l() || h.id() < lopp.id()) {
743  HEdge hopp = h.opposite_g();
744  if ( pane.id() <= hopp.pane()->id()) {
745  // Determine whether the edge is theta-strong
746  float d = cos_face_angle( h, hopp);
747 
748  if ( d < _cos_uf)
749  tstrong_edges.push_back( make_pair(d,h));
750  }
751  }
752  } while ( (h=h.next_l()) != h0);
753  }
754  }
755  float t1=get_wtime(); totaltime+=t1-t0;
756  if ( verb >= 3) {
757  std::cout << "\tIdentified " << tstrong_edges.size()
758  << " theta-strong edges with Theta=" << acos( _cos_uf)*r2d
759  << " in " << t1-t0 << " sec." << std::endl;
760  t0=get_wtime();
761  }
762 
763  sort(tstrong_edges.begin(),tstrong_edges.end());
764  t1=get_wtime(); totaltime+=t1-t0;
765  if ( verb >= 3) {
766  std::cout << "\tSorted theta-strong edges in "
767  << t1-t0 << " sec." << std::endl;
768  }
769 
770  vector<std::pair<float,HEdge> > iedges; iedges.reserve(16);
771  _f_list_1.clear();
772 
773  // Sort the strong edges into strong curves
774  unmark_alledges();
775  float min_fa_r=HUGE_VALF;
776  t0=get_wtime();
777  for ( vector< pair<float,HEdge> >::iterator
778  it=tstrong_edges.begin(),iend=tstrong_edges.end(); it!=iend; ++it){
779  HEdge h=it->second;
780  if ( acc.marked( h)) continue;
781  HEdge hopp=h.opposite_g();
782 
783  // Create a new list
784  Feature_1 f1;
785  f1.push_back( h);
786 
787  // Mark the edge and its opposite
788  h.pane()->set_strong_edge(h); acc.mark( h);
789  hopp.pane()->set_strong_edge(hopp); acc.mark( hopp);
790 
791  Node src = f1.front().origin_g();
792  Node dst = f1.back().destination_g();
793 
794  // Traverse forwards and backwards along the curve, and append an
795  // r-strong edge to the curve until we reach a non-strong curve.
796  for (int c=0; c<2; ++c) {
797  for (;;) {
798  HEdge h = ((c==0)?f1.back():f1.front());
799  if ( src == dst || c==0&&is_on_feature(dst) || c&&is_on_feature(src))
800  break;
801 
802  // Get the unmarked 1-feature with mimimum angle
803  iedges.clear();
804  HEdge h0=h.opposite_g();
805  float d0=cos_face_angle( h, h0);
806  pair<float, HEdge> t0(d0,(c==0)?h0:h);
807  pair<float, HEdge> cos_max(HUGE_VALF,HEdge());
808 
809  HEdge h1=acc.get_next_around_origin(t0.second);
810  do {
811  HEdge h1o=h1.opposite_g();
812  float d = cos_face_angle(h1, h1o);
813  pair<float, HEdge> t(d,h1);
814  iedges.push_back( t);
815 
816  if ( t < cos_max) cos_max = t;
817  } while ( (h1=acc.get_next_around_origin(h1)) != t0.second);
818 
819  bool is_strong=true;
820  const Vector_3 v1=get_tangent( t0.second);
821  if ( cos_max.first>_cos_uf && iedges.size()>1) {
822  for ( int i=0,s=iedges.size(); i<s; ++i) {
823  const Vector_3 v2=get_tangent( iedges[i].second);
824  const Real t = std::max(-v1*v2/sqrt((v1*v1)*(v2*v2)),0.);
825  iedges[i].first = acos(iedges[i].first)*t;
826  }
827  sort(iedges.rbegin(),iedges.rend());
828  cos_max = iedges[0];
829  if ( cos(cos_max.first)>_cos_lf) break;
830 
831  is_strong = cos_max.first >= iedges[1].first*_rf ||
832  is_on_feature( cos_max.second.destination_g()) ||
833  is_strong_ad( cos_max.second.destination_g());
834  min_fa_r=std::min(min_fa_r,cos_max.first/iedges[1].first);
835  }
836  const Vector_3 v2=get_tangent( cos_max.second);
837  float a = -v1*v2/sqrt((v1*v1)*(v2*v2));
838  if ( a<_cos_ue || a<cos_face_angle(cos_max.second)) break;
839 
840  rstrong_edges.push_back(make_pair(cos_face_angle(cos_max.second),
841  cos_max.second));
842 
843  h = cos_max.second;
844  if ( acc.marked( h)) break;
845  HEdge hopp = h.opposite_g();
846 
847  acc.mark( h); h.pane()->set_strong_edge( h);
848  acc.mark( hopp); hopp.pane()->set_strong_edge( hopp);
849  if (c==0) {
850  f1.push_back( h);
851  dst = h.destination_g();
852  }
853  else {
854  f1.push_front( hopp);
855  src = hopp.origin_g();
856  }
857  if ( !is_strong) break;
858  }
859  }
860 
861  Feature_list_1 flist;
862  subdiv_feature_curve( f1, flist, dropped);
863  _f_list_1.splice( _f_list_1.end(), flist);
864  }
865 
866  t1=get_wtime(); totaltime+=t1-t0;
867  if ( verb >= 3) {
868  std::cout << "\tIdentified " << rstrong_edges.size()
869  << " r-strong edges with r=" << _rf
870  << " in " << t1-t0 << " sec." << std::endl;
871  sort(rstrong_edges.begin(), rstrong_edges.end());
872  dump_strong_edges( tstrong_edges, rstrong_edges);
873  t0=get_wtime();
874  }
875 
876  identify_features_0();
877 
878  totaltime+=t1-t0;
879  if ( verb >= 1) {
880  std::cout << "\tFound " << _f_list_1.size() << " ridges and "
881  << _f_list_0.size() << " corners and dropped "
882  << dropped << " false-strong edges.\n"
883  << "\tDone in " << totaltime << " sec." << std::endl;
884 #ifndef DEBUG
885  if ( verb >= 2)
886 #endif
887  print_features();
888  }
889 
890  // Compute the bounding boxes of the 1-features
891  for ( Feature_list_1::iterator it=_f_list_1.begin();
892  it != _f_list_1.end(); ++it) {
893  it->bbox += Bbox_3(it->front().origin_g().point());
894  for (Feature_1::iterator hi=it->begin(), hiend=it->end(); hi!=hiend; ++hi){
895  it->bbox += Bbox_3(hi->destination_g().point());
896  }
897  }
898 
899  // Finalization
900  unmark_alledges();
901 
902  free_vector( tstrong_edges); rstrong_edges.clear();
903  for ( it=_pane_set.begin(); it != _pane_set.end(); ++it) {
904  RFC_Pane_overlay &pane = (RFC_Pane_overlay &)*it->second;
905 
906  free_vector( pane._fd_1);
907  free_vector( pane._ad_0);
908  free_vector( pane._ea_0);
909  }
910 }
