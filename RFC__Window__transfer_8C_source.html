77 // Initialize buffer spaces for data transfer
78 void RFC_Window_transfer::
79 init_nodal_buffers( const Nodal_data &nd, int size, bool init_emm) {
80  _buf_dim = nd.dimension();
81 
82  // Loop through the panes
83  for (Pane_set::iterator pi=_pane_set.begin(); pi != _pane_set.end(); ++pi) {
84  RFC_Pane_transfer &pane = (RFC_Pane_transfer&)*pi->second;
85 
86  pane._buffer.resize( size);
87 
88  for ( int i=0; i<size; ++i)
89  pane._buffer[i].resize(pane._base->size_of_nodes()*_buf_dim);
90 
91  int n=pane.size_of_faces();
92 
93  if ( !init_emm) continue;
94 
95  pane._emm_offset.resize( n+1);
96  // Initialize space for the mass matrix
97  pane._emm_offset[0] = 0;
98 
99  int k=0;
100  // Loop through the connectivity table to mark their nodes as false.
101  if ( pane._base->is_structured()) {
102  for ( int i=0; i<n; ++i, ++k) {
103  pane._emm_offset[k+1] = pane._emm_offset[k] + 16;
104  }
105  }
106  else {
107  std::vector< const COM::Connectivity*> elems;
108  pane._base->elements( elems);
109 
110  for ( std::vector<const COM::Connectivity*>::const_iterator
111  it = elems.begin(); it != elems.end(); ++it) {
112  const int nn = (*it)->size_of_nodes_pe();
113  for ( int i=1,size=(*it)->size_of_elements(); i<=size; ++i, ++k) {
114  pane._emm_offset[k+1] = pane._emm_offset[k] + nn*nn;
115  }
116  }
117  }
118 
119  pane._emm_buffer.resize( pane._emm_offset[n], 0);
120  }
121 }

184 void
185 RFC_Window_transfer::incident_panes( std::vector<int> &pane_ids) {
186  std::set<int> ids;
187 
188  // Loop through the panes to detect incident panes
189  for (Pane_set::iterator pi=_pane_set.begin(); pi != _pane_set.end(); ++pi) {
190  RFC_Pane_transfer &pane = (RFC_Pane_transfer &)*pi->second;
191 
192  for ( int i=0, size=pane._subface_counterparts.size(); i<size; ++i) {
193  ids.insert( pane._subface_counterparts[i].pane_id);
194  }
195  for ( int i=0, size=pane._subnode_counterparts.size(); i<size; ++i) {
196  RFC_assertion(ids.find( pane._subnode_counterparts[i].pane_id) != ids.end());
197  }
198  }
199 
200  pane_ids.resize(0); pane_ids.reserve( ids.size());
201  pane_ids.insert( pane_ids.end(), ids.begin(), ids.end());
202 }
