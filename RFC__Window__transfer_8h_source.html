47  Real *pointer( int i) {
48  if ( i>=0) {
49  if ( is_master())
50  return Base::pointer(i);
51  else {
52  RFC_assertion( _data_buf_id == i);
53  return &_data_buf.front();
54  }
55  }

58  }

62  if ( is_master())
63  return Base::coordinates();
64  else {
65  return &_coor_buf.front();
66  }
67  }
68  bool need_recv( int i) const {
69  // Check whether an element is on the receiver list
70  if ( _to_recv == NULL) return true;
71  RFC_assertion( i>=1 && i<=size_of_faces());
72  return _to_recv[ i-1];
73  }
