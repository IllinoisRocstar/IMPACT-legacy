270 template <int attr_type>
271 inline int Rocblas::get_stride( const DataItem *attr) {
272  if ( attr_type == BLAS_SCNE || attr_type == BLAS_VEC2D) {
273  if ( attr->size_of_items()>1)
274  return attr->stride();
275  else
276  return 0;
277  }
278  else if ( attr_type == BLAS_VEC)
279  return 0;
280  else
281  return 1;
282 }

284 template <class data_type, int attr_type, bool is_staggered>
285 inline data_type &
286 Rocblas::getref( data_type *base, const int r, const int c, const int nc) {
287  if ( attr_type == BLAS_SCNE || attr_type == BLAS_VEC2D && is_staggered)
288  return base[r*nc];
289  else if ( attr_type == BLAS_VEC2D)
290  return base[r*nc+c];
291  else if ( attr_type == BLAS_VEC)
292  return base[c];
293  else
294  return *base;
295 }
