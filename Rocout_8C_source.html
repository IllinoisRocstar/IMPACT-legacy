330 void Rocout::write_rocin_control_file(const char* window_name,
331  const char* file_prefixes,
332  const char* control_file_name)
333 {
334  const MPI_Comm default_comm = COM_get_default_communicator();
335  const MPI_Comm comm_null = MPI_COMM_NULL;
336  const MPI_Comm* myComm = COMMPI_Initialized() ? &default_comm : &comm_null;
337 
338 
339  // Obtain process rank
340  int flag = 0, rank = 0, size = 1;
341  MPI_Initialized(&flag);
342 
343  if (flag) {
344  MPI_Comm_rank(*myComm, &rank);
345  MPI_Comm_size(*myComm, &size);
346  }
347 
348  if (rank == 0) {
349  std::vector<std::vector<int> > paneIds(size);
350 
351  /* Get list of panes local each process */
352  int i;
353  for (i=0; i<size; ++i)
354  COM_get_panes(window_name, paneIds[i], i);
355 
356  int rw, pw;
357  {
358  std::istringstream sin(_options["rankwidth"]);
359  sin >> rw;
360  }
361  {
362  std::istringstream sin(_options["pnidwidth"]);
363  sin >> pw;
364  }
365 
366 /*
367  std::ostringstream sout;
368  sout << "@Files:";
369 
370  // Write out file name pattern only if number of panes is nonzero.
371  std::istringstream sin(file_prefixes);
372  while (!sin.eof()) {
373  std::string prefix;
374  sin >> prefix;
375  if (prefix.empty())
376  continue;
377 
378  sout << ' ' << prefix;
379  if (rw > 0)
380  sout << "%0" << rw << 'p';
381  if (pw > 0) {
382  if (rw > 0)
383  sout << _options["separator"];
384  sout << "%0" << pw << 'i';
385  }
386 
387  const std::string fmt = _options["format"];
388  if ( fmt == "HDF4" || fmt == "HDF")
389  sout << ".hdf";
390  else if (fmt == "HDF5")
391  sout << ".hdf5";
392  else if (fmt == "CGNS")
393  sout << ".cgns";
394  }
395 */
396 
397  /* Now write data into control file */
398  std::ofstream fout(control_file_name);
399  COM_assertion_msg( fout.is_open(),
400  (std::string("Rocout cannot open control file:")+control_file_name+" for writing.\n").c_str());
401 
402  const std::string fmt = _options["format"];
403  for (i=0; i<size; i++) {
404  fout << "@Proc: " << i << std::endl;
405  if ( !paneIds[i].empty()) {
406  std::ostringstream sout;
407  sout << "@Files:";
408 
409  // Write out file name pattern only if number of panes is nonzero.
410  std::istringstream sin(file_prefixes);
411  while (!sin.eof()) {
412  std::string prefix;
413  sin >> prefix;
414  if (prefix.empty())
415  continue;
416 
417  sout << ' ';
418  // write output file in <rank> dir
419  if (_options["rankdir"] == "on") {
420  std::ostringstream rank_prefix;
421  rank_prefix << i << "/";
422  sout << rank_prefix.str();
423  }
424  sout << prefix;
425  if (rw > 0)
426  sout << "%0" << rw << 'p';
427  if (pw > 0) {
428  if (rw > 0)
429  sout << _options["separator"];
430  sout << "%0" << pw << 'i';
431  }
432 
433  if ( fmt == "HDF4" || fmt == "HDF")
434  sout << ".hdf";
435  else if (fmt == "HDF5")
436  sout << ".hdf5";
437  else if (fmt == "CGNS")
438  sout << ".cgns";
439  }
440  fout << sout.str() << std::endl;
441  }
442  else // Write out an empty block
443  fout << "@Files: " << std::endl;
444 
445  fout << "@Panes:";
446 
447  /* Write all the panes of this process to control file */
448  std::vector<int>::iterator p;
449  for (p=paneIds[i].begin(); p!=paneIds[i].end(); ++p)
450  fout << ' ' << *p;
451  fout << '\n' << std::endl;
452  }
453 
454  fout.close();
455  }

497 static bool is_option_value(const std::string& name, const std::string& val)
498 {
499  return ((name == "format"
500  && (val == "HDF" || val == "HDF4" || val == "HDF5" || val == "CGNS"))
501  || (name == "async"
502  && (val == "on" || val == "off"))
503  || (name == "mode"
504  && (val == "w" || val == "a"))
505  || (name == "localdir" /* && is_valid_path(val) */ )
506  || ((name == "rankwidth" || name == "pnidwidth") && is_whole(val))
507  || (name == "rankdir"
508  && (val == "on" || val == "off"))
509  || (name == "errorhandle"
510  && (val == "abort" || val == "ignore" || val == "warn"))
511  || (name == "ghosthandle"
512  && (val == "write" || val == "ignore")));

549 void Rocout::read_control_file( const char* filename)
550 {
551  // Attempt to open the control file.
552  std::ifstream fin(filename);
553  if (!fin.is_open()) {
554  ERROR_MSG("Rocout::read_control_file(): could not open file \""
555  << filename << "\".");
556  return;
557  }
558 
559  // Each line of the control file should be of the form "option_name=value",
560  // e.g. "format=HDF4". White space at the beginning, end, and around the
561  // '=' is ignored, as are empty lines and comments (lines that start with
562  // '#').
563  int lineNum = 0;
564  char buffer[256];
565  std::string line, name, val;
566  std::string::size_type eq;
567  while (!fin.eof()) {
568  ++lineNum;
569  // Read a line.
570  fin.getline(buffer, sizeof(buffer));
571  line = buffer;
572 
573  // Skip empty lines and comments.
574  if (line.empty() || line.find_first_not_of(" \t\n") == std::string::npos
575  || line[line.find_first_not_of(" \t\n")] == '#')
576  continue;
577 
578  // Find the '='. It's mandatory.
579  eq = line.find('=');
580  if (eq == std::string::npos) {
581  ERROR_MSG("Rocout::read_control_file(): option without value at line "
582  << lineNum << " of " << filename << '.');
583  continue;
584  }
585 
586  // Split the line into an option name and a value.
587  name = line.substr(0, eq);
588  val = line.substr(eq + 1);
589 
590  // Remove leading whitespace from the option name.
591  if (name.find_first_not_of(" \t\n") != 0)
592  name.erase(0, name.find_first_not_of(" \t\n"));
593 
594  // Check for an empty option name.
595  if (name.empty()) {
596  ERROR_MSG("Rocout::read_control_file(): missing option name at line "
597  << lineNum << " of " << filename << '.');
598  continue;
599  }
600 
601  // Remove trailing whitespace from the option name.
602  if (name.find_last_not_of(" \t\n") != name.length() - 1)
603  name.erase(name.find_last_not_of(" \t\n") + 1);
604 
605  // Make sure it is a valid option name.
606  if (!is_option_name(name)) {
607  ERROR_MSG("Rocout::read_control_file(): unknown option name \"" << name
608  << "\" at line " << lineNum << " of " << filename << '.');
609  continue;
610  }
611 
612  // Remove leading whitespace from the option value.
613  if (val.find_first_not_of(" \t\n") != 0)
614  val.erase(0, val.find_first_not_of(" \t\n"));
615 
616  // Check for an empty option value.
617  if (val.empty()) {
618  ERROR_MSG("Rocout::read_control_file(): option without value at line "
619  << lineNum << " of " << filename << '.');
620  continue;
621  }
622 
623  // Remove trailing whitespace from the option value.
624  if (val.find_last_not_of(" \t\n") != val.length() - 1)
625  val.erase(val.find_last_not_of(" \t\n") + 1);
626 
627  // Make sure it is a valid value for the named option.
628  if (!is_option_value(name, val)) {
629  ERROR_MSG("Rocout::read_control_file(): invalid value \"" << val
630  << "\" for option \"" << name << "\" at line " << lineNum
631  << " of " << filename << '.');
632  continue;
633  }
634 
635  set_option(name.c_str(), val.c_str());
636  }

639 void* Rocout::write_dataitem_internal(void* attrInfo)
640 {
641  WriteAttrInfo* ai = static_cast<WriteAttrInfo*>(attrInfo);
642  const DataItem* attr = ai->m_attr;
643  Window* tempWin = NULL;
644 
645  if (ai->m_cloned) {
646 #ifdef USE_PTHREADS
647  _writesem.Wait();
648 #endif // USE_PTHREADS
649  tempWin = new Window(attr->window()->name(),
650  attr->window()->get_communicator());
651  attr = tempWin->inherit(const_cast<DataItem*>(attr), attr->name(),
652  Pane::INHERIT_CLONE, true, NULL, 0);
653  }
654 
655  int flag = 0; MPI_Initialized(&flag);
656 
657  // Obtain process rank
658  int rank;
659  if ( flag) {
660  MPI_Comm wcomm;
661  const MPI_Comm* pComm;
662 
663  if ( ai->m_pComm)
664  pComm = ai->m_pComm;
665  else {
666  wcomm = attr->window()->get_communicator();
667  pComm = &wcomm;
668  }
669 
670  if (*pComm != MPI_COMM_NULL)
671  MPI_Comm_rank(*pComm, &rank);
672  else
673  rank = 0;
674  }
675  else
676  rank = 0;
677 
678  int append = ai->m_append;
679  if (append < 0) {
680  if (ai->m_rout->_options["mode"] == "w")
681  append = 0;
682  else
683  append = 1;
684  }
685 
686  std::vector<int> paneIds;
687  COM_get_panes(attr->window()->name().c_str(), paneIds);
688 
689  std::vector<int>::iterator begin = paneIds.begin(), end = paneIds.end(), p;
690  if (ai->m_pPaneId != NULL && *(ai->m_pPaneId) > 0) {
691  begin = std::find(begin, end, *(ai->m_pPaneId));
692  if (begin != end)
693  end = begin + 1;
694  }
695 
696  std::set<std::string> written;
697  for (p=begin; p!=end; ++p) {
698  std::string fname, mfile;
699  fname = ai->m_rout->get_fname(ai->m_prefix, rank, *p, true);
700  if (!ai->m_meshPrefix.empty())
701  mfile = ai->m_rout->get_fname(ai->m_meshPrefix, rank, *p);
702 
703  int ap = append + written.count(fname);
704  written.insert(fname);
705 
706  const std::string fmt = ai->m_rout->_options["format"];

742 std::string Rocout::get_fname(const std::string& prefix,
743  int rank /* = -1 */,
744  int paneId /* = 0 */,
745  bool check /* = false */)
746 {
747  // Modify the prefix using the "localdir" option.
748  std::string pre(_options["localdir"]);
749  if (!pre.empty()) {
750  // Make sure there's exactly one '/' between the localdir and given prefix.
751  if (prefix[0] == '/') {
752  if (pre[pre.length()-1] == '/')
753  pre.erase(pre.length() - 1);
754  } else {
755  if (pre[pre.length()-1] != '/')
756  pre += '/';
757  }
758  }
759  pre += prefix;
760 
761  if (_options["rankdir"] == "on") { // write output file in <rank> dir
762  std::ostringstream rank_prefix;
763  rank_prefix << "/" << rank;
764  std::string::size_type s = pre.find_last_of('/');
765  if (s != std::string::npos)
766  pre.insert(s, rank_prefix.str());
767  }
768 
769  // Make sure the directory exists. Ignore any errors except for the last.
770  int result = 0;
771  std::string::size_type s = pre.find('/', 1);
772  while (s != std::string::npos) {
773  result = mkdir(pre.substr(0, s).c_str(), 0755);
774  s = pre.find('/', s + 1);
775  }
776  if (result < 0 && errno != EEXIST) {
777  ERROR_MSG("Rocout::write_dataitem(): could not create directory '"
778  << pre.substr(0, pre.rfind('/') + 1) << "'.");
779  }
780 
781  if ( pre.find(".hdf") == pre.size()-4) {
782  _options["format"] = "HDF4";
783  return pre;
784  }
785  else if (pre.find(".cgns") == pre.size()-5) {
786  _options["format"] = "CGNS";
787  return pre;
788  }
789 
790  if (rank < 0) {
791  int flag = 0;
792  MPI_Initialized(&flag);
793  if (flag)
794  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
795  else
796  rank = 0;
797  }
798 
799  std::string name;
800 
801  if (!pre.empty()) {
802  int rw, pw;
803  {
804  std::istringstream sin(_options["rankwidth"]);
805  sin >> rw;
806  }
807  {
808  std::istringstream sin(_options["pnidwidth"]);
809  sin >> pw;
810  }
811 
812  std::ostringstream sout;
813  sout << pre;
814  if (rw > 0)
815  sout << std::setw(rw) << std::setfill('0') << rank;
816  if (pw > 0 && paneId > 0) {
817  if (rw > 0)
818  sout << _options["separator"];
819  sout << std::setw(pw) << std::setfill('0') << paneId;
820  }
821 
822  const std::string fmt = _options["format"];
823  if ( fmt == "HDF" || fmt == "HDF4")
824  sout << ".hdf";
825  else if (fmt == "HDF5")
826  sout << ".hdf5";
827  else if (fmt == "CGNS")
828  sout << ".cgns";
829 
830  name = sout.str();
831 
832  if (check) {
833  std::FILE* f;
834  if ((f = std::fopen(name.c_str(), "r")) != NULL)
835  std::fclose(f);
836  else if ((f = std::fopen(name.c_str(), "w")) != NULL) {
837  std::fclose(f);
838  std::remove(name.c_str());
839  } else {
840  // Change the directory to current directory
841  std::string::size_type n = name.rfind('/');
842  if (n != std::string::npos)
843  name.erase(0, n + 1);
844  }
845  }
846  }
847 
848  return name;
849 }
