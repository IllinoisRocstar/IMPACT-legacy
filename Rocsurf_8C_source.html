70 void Rocsurf::compute_edge_lengths( double *lave, double *lmin, double *lmax) {
71 
72  Window_manifold_2::PM_iterator it=_wm->pm_begin(), iend=_wm->pm_end();
73 
74  double local_lmin=HUGE_VAL, local_lmax=0, local_lsum=0, local_weights=0;
75 
76  // Loop through the panes to identify strong edges
77  for ( ; it!=iend; ++it) {
78  // for (int i=0, nf=it->size_of_real_faces(); i<nf; ++i) {
79  // Halfedge h=Halfedge( &*it, Edge_ID( i+1, 0), ACROSS_PANE), h0=h;
80  for (int i=0, nf=(*it)->size_of_real_faces(); i<nf; ++i) {
81  Halfedge h=Halfedge( *it, Edge_ID( i+1, 0), ACROSS_PANE), h0=h;
82  do {
83  Halfedge hopp=h.opposite();
84 
85  if ( hopp.is_border() || h < hopp) {
86  double l = h.tangent().norm();
87 
88  local_lmin = std::min( local_lmin, l);
89  local_lmax = std::max( local_lmax, l);
90  local_lsum += l;
91  local_weights += 1;
92  }
93  } while ( (h=h.next())!=h0);
94  }
95  }
96 
97  double global_lmin=local_lmin, global_lmax=local_lmax,
98  global_lsum=local_lsum, global_weighs=local_weights;
99 
100  if ( COMMPI_Initialized()) {
101  MPI_Comm comm = _wm->window()->get_communicator();
102  MPI_Allreduce( &local_lmin, &global_lmin, 1, MPI_DOUBLE, MPI_MIN, comm);
103  MPI_Allreduce( &local_lmax, &global_lmax, 1, MPI_DOUBLE, MPI_MAX, comm);
104  MPI_Allreduce( &local_lsum, &global_lsum, 1, MPI_DOUBLE, MPI_SUM, comm);
105  MPI_Allreduce( &local_weights, &global_weighs, 1, MPI_DOUBLE, MPI_SUM, comm);
106  }
107 
108  if ( lave) *lave = (global_weighs>0)? (global_lsum / global_weighs) : 0.;
109  if ( lmin) *lmin = global_lmin;
110  if ( lmax) *lmax = global_lmax;
111 }

176

190 
191  types[1] = types[2] = types[3] = COM_DOUBLE;
