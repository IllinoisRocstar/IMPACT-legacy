251 void Scheduler::topological_sort()
252 {
253  ActionList pool;
254  pool = actions;
255  while (1) {
256  ActionList::iterator aitem = pool.begin();
257  for (; aitem!=pool.end(); aitem++) {
258  if ((*aitem)->n_read() == 0) { break; }
259  // check every input
260  int flag = 1;
261  for (unsigned int i=0; i<(*aitem)->n_read(); i++) {
262  ActionItem *in = (*aitem)->input[i];
263  // search in sort
264  ActionList::iterator s;
265  for (s = sort.begin(); s!=sort.end(); ++s)
266  if ((*s)->action() == in->action()) break;
267  if (s == sort.end()) { flag = 0; break; }
268  }
269  if (flag) break;
270  }
271  if (aitem != pool.end()) {
272  sort.push_back(*aitem);
273  pool.erase(aitem);
274  }
275  else
276  break;
277  }
278 
279  if (pool.size() != 0) {
280  printf("ERROR in sorting!\n");
281  exit (1);
282  }
283 }
