27 class Facet_ID {
28 public:
29  enum { nDigits=4, BndID=15 };
30 
31  Facet_ID( ) : _code(0) {}
32  Facet_ID( int eid, char lid) : _code( (eid<<nDigits)+(BndID&lid)) {}
33 
35  int eid() const { return _code>>nDigits; }
37  char lid() const { return _code&BndID; }
38 
40  bool is_border() const { return lid()==BndID; }
41 
43  bool operator==( const Facet_ID e) const
44  { return _code==e._code; }
45  bool operator!=( const Facet_ID e) const
46  { return _code!=e._code; }
47  bool operator<( const Facet_ID e) const
48  { return lid()<e.lid() || lid()==e.lid() && _code<e._code; }
49 private:
50  int _code;
51 };
