71 void SolidAgent::load_module()
72 {
73  MAN_DEBUG(1, ("[%d] Rocman: SolidAgent::load_module %s %s.\n", comm_rank, rocmod_name.c_str(), mod_instance.c_str()));
74 
75 #ifdef STATIC_LINK
76  if (rocmod_name == "Rocsolid")
77  COM_F_FUNC2( rocsolid_load_module, ROCSOLID_LOAD_MODULE)( mod_instance.c_str(), mod_instance.length());
78 #ifdef ROCFRAC3
79  else if (rocmod_name == "Rocfrac3")
80  COM_F_FUNC2( rocfrac3_load_module, ROCFRAC3_LOAD_MODULE)( mod_instance.c_str(), mod_instance.length());
81 #else
82  else if (rocmod_name == "Rocfrac")
83  COM_F_FUNC2( rocfrac_load_module, ROCFRAC_LOAD_MODULE)( mod_instance.c_str(), mod_instance.length());
84 #endif
85  else
86  COM_assertion_msg(0, "Unknown SolidAgent mod!");
87 #else
88  // dynamic loading
89  COM_assertion_msg(rocmod_name == "Rocsolid" || rocmod_name == "Rocfrac"
90  || rocmod_name == "Rocfrac3",
91  (std::string("Unknown SolidAgent module:")+rocmod_name).c_str());
92  COM_load_module(rocmod_name.c_str(), mod_instance.c_str());
93 #endif
94 
95  init_function_handles(); // defined in Agent
96 }

319

321  y_hdl = COM_get_dataitem_handle( solidBuf+".nc");
322 
323  if (!get_coupling()->initial_start()) read_restart_data();
