16 // This integrator can integrate over subset of an element.
17 // Note that the integral is added to the current values of v and area.
18 template < class _Data>
19 void
20 Transfer_base::integrate_subface( const RFC_Pane_transfer *p_src,
21  RFC_Pane_transfer *p_trg,
22  const _Data &data_s,
23  ENE &ene_src,
24  ENE &ene_trg,
25  int sfid_src,
26  int sfid_trg,
27  const Real alpha,
28  Facial_data &tDF,
29  Facial_data &tBF,
30  int doa)
31 {
32  // Initialize natural cooredinates of the subnodes of the subface in its
33  // parent source and target faces. Compute ncs_s only if alpha!=1.
34  Point_3 ps_s[Generic_element::MAX_SIZE], ps_t[Generic_element::MAX_SIZE];
35  Point_2 ncs_s[Generic_element::MAX_SIZE], ncs_t[Generic_element::MAX_SIZE];
36 
37  Generic_element e_s(ene_src.pane()?ene_src.size_of_edges():3,
38  ene_src.pane()?ene_src.size_of_nodes():3);
39 
40  // Initialize physical and natural coordinates for source element
41  if ( is_nodal(data_s) || alpha!=1) {
42  for ( int i=0; i<3; ++i)
43  p_src->get_nat_coor_in_element( sfid_src, i, ncs_s[i]);
44 
45  if ( alpha!=1.) {
46  Nodal_coor_const nc;
47  Element_coor_const pnts_s( nc, p_src->coordinates(), ene_src);
48 
49  for ( int i=0; i<3; ++i) e_s.interpolate( pnts_s, ncs_s[i], &ps_s[i]);
50  }
51  }
52  else {
53  std::fill_n( &ncs_s[0][0], 6, QUIET_NAN);
54  std::fill_n( &ps_s[0][0], Generic_element::MAX_SIZE*3, QUIET_NAN);
55  }
56 
57  { // Initialize physical and natural coordinates for target element
58  for ( int i=0; i<3; ++i)
59  p_trg->get_nat_coor_in_element( sfid_trg, i, ncs_t[i]);
60 
61  Nodal_coor_const nc;
62  Generic_element e_t( ene_trg.size_of_edges(), ene_trg.size_of_nodes());
63  Element_coor_const pnts_t( nc, p_trg->coordinates(), ene_trg);
64 
65  for ( int i=0; i<3; ++i) e_t.interpolate( pnts_t, ncs_t[i], &ps_t[i]);
66  }
67 
68  // Loop throught the quadrature points of the subfacet.
69  Vector_n vt(data_s.dimension(), 0);
70  Array_n t( vt.begin(), vt.end()), v=tDF.get_value( p_trg, ene_trg.id());
71  Real &area = tBF.get_value( p_trg, ene_trg.id())[0];
72 
73  Generic_element sub_e( 3);
74  Point_2 sub_nc, nc_s;
75 
76  if ( is_nodal( data_s)) {
77  if ( doa == 0) doa = 2; // Set the default degree of accuracy
78  }
79  else
80  doa=1;
81 
82  for ( int i=0, n = sub_e.get_num_gp(doa); i<n; ++i) {
83  sub_e.get_gp_nat_coor( i, sub_nc, doa);
84 
85  if ( is_nodal( data_s))
86  sub_e.interpolate( ncs_s, sub_nc, &nc_s);
87 
88  interpolate( e_s, data_s, nc_s, t);
89 
90  Real a=sub_e.get_gp_weight(i, doa);
91  a *= sub_e.Jacobian_det( ps_s, ps_t, alpha, sub_nc);
92 
93  t *= a;
94  v += t;
95  area += a;
96  }
97 }

99 template < class _SDF>
100 void
101 Transfer_base::transfer_2f( const _SDF &sDF,
102  Facial_data &tDF,
103  const Real alpha,
104  int doa,
105  bool verbose)
106 {
107  double t0=0.;
108  if ( verbose) {
109  trg.barrier(); t0=get_wtime();
110  }
111 
112  src.replicate_data( sDF, alpha!=1);
113  // First, create buffer space for the target window and initialize
114  // the entries of the target mesh to zero.
115  trg.init_facial_buffers( tDF, 1);
116  Facial_data tBF ( trg.facial_buffer( 0));
117 
118  for ( Pane_iterator pit=trg_ps.begin(); pit!=trg_ps.end(); ++pit) {
119  Real *trg_data = (*pit)->pointer( tDF.id());
120  Real *trg_buf = (*pit)->pointer( tBF.id());
121 
122  std::fill( trg_data, trg_data+(*pit)->size_of_faces()*tDF.dimension(), 0);
123  std::fill( trg_buf, trg_buf+(*pit)->size_of_faces(), 0);
124  }
125 
126  // Second, compute the integral over the target meshes by looping through
127  // the subfaces of the target window
128  ENE ene_src, ene_trg;
129  const RFC_Pane_transfer *p_src = NULL;
130  for ( Pane_iterator pit=trg_ps.begin(); pit!=trg_ps.end(); ++pit) {
131  // Loop through the subfaces of the target window
132  for ( int i=1, size=(*pit)->size_of_subfaces(); i<=size; ++i) {
133  (*pit)->get_host_element_of_subface( i, ene_trg);
134  if ( !(*pit)->need_recv( ene_trg.id())) continue;
135 
136  const Face_ID &fid = (*pit)->get_subface_counterpart(i);
137  if ( !p_src || p_src->id()!=fid.pane_id)
138  p_src = get_src_pane( fid.pane_id);
139  if ( alpha!=1 || is_nodal( sDF.tag()) )
140  p_src->get_host_element_of_subface( fid.face_id, ene_src);
141 
142  if ( is_nodal( sDF.tag()))
143  integrate_subface( p_src, *pit,
144  make_field(sDF, p_src, ene_src),
145  ene_src, ene_trg, fid.face_id, i, alpha, tDF, tBF, doa);
146  else {
147  int id = p_src->get_parent_face( fid.face_id);
148  integrate_subface( p_src, *pit,
149  make_field(sDF, p_src, id),
150  ene_src, ene_trg, fid.face_id, i, alpha, tDF, tBF, doa);
151  }
152  }
153  }
154 
155  // Loop through the panes of the target mesh
156  for ( Pane_iterator pit=trg_ps.begin(); pit!=trg_ps.end(); ++pit) {
157  Real *trg_data = (*pit)->pointer( tDF.id());
158  Real *trg_buf = (*pit)->pointer( tBF.id());
159 
160  for ( int i=1, size=(*pit)->size_of_faces(); i<=size; ++i) {
161  if ( !(*pit)->need_recv(i)) continue;
162  tDF.get_value( trg_data, i) /= tBF.get_value( trg_buf, i)[0];
163  }
164  }
165 
166  // Clean up the transfer buffers
167  trg.delete_facial_buffers();
168  src.clear_replicated_data();
169 
170  if ( verbose) {
171  trg.barrier();
172  if ( trg.is_root()) {
173  std::cout << "ROCFACE: Transfer to faces done in "
174  << get_wtime()-t0 << " seconds." << std::endl;
175  }
176  }
177 }
