18 // This function obtains the initial guess for the unknowns
19 // by transfering them from the source mesh.
20 template <class _SDF>
21 void Transfer_base::
22 interpolate_fe( const _SDF &sDF, Nodal_data &tDF, bool verbose)
23 {
24  double t0=0.;
25  if ( verbose) {
26  trg.barrier(); t0=get_wtime();
27  }
28 
29  std::vector<bool> flags;
30  // Loop through all the panes of target window
31  for ( Pane_iterator pit=trg_ps.begin(); pit!=trg_ps.end(); ++pit) {
32  RFC_Pane_transfer *p_trg = *pit;
33  Real *p = p_trg->pointer( tDF.id());
34  std::fill( p, p+p_trg->size_of_nodes()*tDF.dimension(), Real(0));
35  flags.clear(); flags.resize(p_trg->size_of_nodes()+1, false);
36 
37  // Loop through the faces to mark the ones that expect values.
38  ENE ene( p_trg->base(), 1);
39  for ( int k=1, size=p_trg->size_of_faces(); k<=size; ++k, ene.next()) {
40  if ( !p_trg->need_recv(k)) continue;
41 
42  for ( int i=0, n=ene.size_of_nodes(); i<n; ++i) flags[ene[i]] = true;
43  }
44 
45  ENE ene_src;
46  Point_2 nc;
47  const RFC_Pane_transfer *p_src = NULL;
48  int count=0, nnodes=p_trg->size_of_nodes()-p_trg->size_of_isolated_nodes();
49 
50  // Loop through the subnodes of the target window
51  for ( int i=1, size=p_trg->size_of_subnodes(); i<=size; ++i) {
52 
53  int svid_trg = i;
54  if ( p_trg->parent_type_of_subnode( svid_trg) != PARENT_VERTEX)
55  { if ( count >= nnodes) break; else continue; }
56  else ++count;
57 
58  int pvid_trg = p_trg->get_parent_node( svid_trg);
59  if ( !flags[pvid_trg]) continue;
60 
61  const Node_ID &SVID_src = p_trg->get_subnode_counterpart( svid_trg);
62  if ( !p_src || p_src->id()!=SVID_src.pane_id)
63  p_src = get_src_pane( SVID_src.pane_id);
64 
65  int svid_src = SVID_src.node_id;
66  p_src->get_host_element_of_subnode(svid_src, ene_src, nc);
67 
68  Array_n v = tDF.get_value( p, pvid_trg);
69 
70  if ( is_nodal(sDF.tag())) {
71  Generic_element e( ene_src.size_of_edges(), ene_src.size_of_nodes());
72  interpolate( e, make_field( sDF, p_src, ene_src), nc, v);
73  }
74  else {
75  interpolate( Generic_element(3),
76  make_field( sDF, p_src, ene_src), nc, v);
77  }
78  }
79 
80  // If linear, we are done with this pane
81  if ( !p_trg->is_quadratic()) continue;
82 
83  // Otherwise, we must interpolate values to other nodes.
84  // We now loop through the faces of the pane.
85  ene = ENE( p_trg->base(), 1);
86  for ( int k=1, size=p_trg->size_of_faces(); k<=size; ++k, ene.next()) {
87  if ( !p_trg->need_recv(k)) continue; // Skip the face if not receiving
88  Element_var f( tDF, p_trg->pointer( tDF.id()), ene);
89 
90  switch ( ene.size_of_nodes()) {
91  case 6:
92  f[3] = 0.5*(f[0]+f[1]);
93  f[4] = 0.5*(f[1]+f[2]);
94  f[5] = 0.5*(f[2]+f[0]);
95  break;
96  case 9:
97  f[8] = 0.25*(f[0]+f[1]+f[2]+f[3]); // Then continue as 8-nodes
98  case 8:
99  f[4] = 0.5*(f[0]+f[1]);
100  f[5] = 0.5*(f[1]+f[2]);
101  f[6] = 0.5*(f[2]+f[3]);
102  f[7] = 0.5*(f[3]+f[0]);
103  break;
104  }
105  }
106  }
107 
108  trg.reduce_maxabs_to_all( tDF);
109 
110  if ( verbose) {
111  // Output timing information
112  trg.barrier();
113  if ( trg.is_root())
114  std::cout << "ROCFACE: Interpolation done in "
115  << get_wtime()-t0 << " seconds." << std::endl;
116  }
117 }

119 // Computes the element-wise load vector and mass matrix.
120 template < class _Data, class _Points, class _Loads, class Tag>
121 void Transfer_base::
122 element_load_vector( const _Data &data_s,
123  const Generic_element &e_s,
124  const Generic_element &e_t,
125  const _Points &pnts_s,
126  const _Points &pnts_t,
127  const Point_2 *ncs_s,
128  const Point_2 *ncs_t,
129  const Real alpha,
130  const int sne,
131  const Tag &tag,
132  _Loads loads,
133  _Loads diag,
134  Real *emm_out,
135  int doa,
136  bool lump)
137 {
138  const unsigned int n = e_t.size_of_nodes();
139 
140  Generic_element sub_e( sne);
141  Vector_n vt(data_s.dimension(), 0); Array_n v( vt.begin(), vt.end());
142 
143  // Interpolate the coordinates.
144  Point_3 ps_s[Generic_element::MAX_SIZE];
145  Point_3 ps_t[Generic_element::MAX_SIZE];
146  for ( int i=0; i<sne; ++i) e_t.interpolate( pnts_t, ncs_t[i], &ps_t[i]);
147  if ( alpha!=1.)
148  for ( int i=0; i<sne; ++i) e_s.interpolate( pnts_s, ncs_s[i], &ps_s[i]);
149 
150  // Set the default degree of accuracy for the quadrature rules.
151  if ( is_nodal( data_s))
152  { if ( doa == 0) doa = std::max(e_t.order(), e_s.order())==1? 2 : 4; }
153  else doa=1;
154 
155  // Create a local buffer for element-mass-matrix. Note that only
156  // lower triangle matrix is computed, as emm is symmetric.
157  Real emm[Generic_element::MAX_SIZE*Generic_element::MAX_SIZE];
158  bool compute_mass = diag.dimension()>0;
159  if ( compute_mass) std::fill(emm, emm+n*n, 0.);
160 
161 #ifdef SOBOLEV
162  // Evaluate the gradient as well for Sobolev minimization
163  std::vector<Real> esf(n*n,0.);
164  std::vector<Vector_n> l_t(n,Vector_n(data_s.dimension(),0));
165 #endif
166 
167  Point_2 sub_nc, nc_s, nc_t;
168  Real N[Generic_element::MAX_SIZE];
169  // Loop through the quadrature points of the subelement
170  for ( int i=0, ni=sub_e.get_num_gp(doa); i < ni; ++i) {
171  sub_e.get_gp_nat_coor( i, sub_nc, doa);
172  sub_e.interpolate( ncs_s, sub_nc, &nc_s);
173 
174  // Interploate to the quadrature point
175  interpolate( e_s, data_s, nc_s, v);
176 
177  // Compute area of subelement in target element
178  Real a_t=sub_e.get_gp_weight(i, doa), a_s=a_t;
179  a_t *= sub_e.Jacobian_det( ps_t, sub_nc);
180 
181  // Compute area of subelement in source element. Use target area if alpha=1
182  if ( alpha!=1.) a_s *= sub_e.Jacobian_det( ps_s, sub_nc);
183  else a_s = a_t;
184 
185  v *= a_s;
186  sub_e.interpolate( ncs_t, sub_nc, &nc_t);
187  e_t.shape_func( nc_t, N);
188 
189 #if SOBOLEV
190  Vector_3 grads_t[Generic_element::MAX_SIZE];
191  e_t.gradients( pnts_t, nc_t, grads_t);
192 
193  std::vector<Vector_3> load_prime( data_s.dimension(), NULL_VECTOR);
194 
195  Vector_3 grads_s[Generic_element::MAX_SIZE];
196  if ( alpha!=1)
197  e_s.gradients( pnts_s, nc_s, grads_s);
198  else
199  std::copy(grads_t,grads_t+n, grads_s);
200  compute_load_prime( e_s, data_s, grads_s, &load_prime[0], tag);
201 #endif
202 
203  for ( unsigned int j=0; j<n; ++j) {
204  loads[j] += N[j] * v;
205  if ( compute_mass) {
206  for ( unsigned int k=0; k<=j; ++k)
207  emm[j*n+k] += N[j]*N[k]*a_t;
208  }
209 #if SOBOLEV
210  for ( unsigned int k=0; k<data_s.dimension(); ++k)
211  l_t[j][k] += grads_t[j]*load_prime[k]*area;
212 
213  if ( !lump && compute_mass) {
214  for ( unsigned int k=0; k<=j; ++k) {
215  // Weigh esf by the stretch ratio.
216  esf[j*n+k] += grads_t[j]*grads_t[k]*area;
217  }
218  }
219 #endif
220  }
221  }
222 
223 #if SOBOLEV
224  Real mu=-1;
225  if ( mu<0 && !lump && compute_mass) {
226  Real t1=0, t2=0;
227  // Evaluate mu
228  for ( unsigned int j=0; j<n; ++j) {
229  for ( unsigned int k=0; k<j; ++k) {
230  t1 += emm[j*n+k]*esf[j*n+k]; t2 += esf[j*n+k]*esf[j*n+k];
231  }
232  }
233  mu = -t1 / t2;
234  }
235 
236  if ( mu > 0) {
237  // Add the stiffness to the mass matrix.
238  for ( unsigned int j=0; j<n; ++j) {
239  loads[j] += mu * l_t[j];
240 
241  if ( emm_out) {
242  for ( unsigned int k=0; k<=j; ++k) {
243  emm[j*n+k] += mu*esf[j*n+k];
244  }
245  }
246  }
247  }
248 #endif
249 
250  // Add emm onto emm_out and/or diag. Note that emm only saves lower triangle
251  if ( compute_mass) {
252  for ( unsigned int j=0; j<n; ++j) {
253  // Update the diagonal
254  if ( lump) {
255  // Lump the mass matrix
256  for ( unsigned int k=0; k<n; ++k)
257  diag[j][0] += emm[std::max(j,k)*n+std::min(j,k)];
258  }
259  else // Otherwise, add only the diagonal entries
260  diag[j][0] += emm[j*n+j];
261  }
262 
263  // Update the element mass matrix. Adding lower triangle to the
264  // diagonal and upper triangle as well.
265  if ( emm_out) {
266  for ( unsigned int j=0; j<n; ++j)
267  for ( unsigned int k=0; k<n; ++k)
268  emm_out[j*n+k] += emm[std::max(j,k)*n+std::min(j,k)];
269  }
270  }
271 }

274 void Transfer_base::
275 compute_load_vector_wra( const RFC_Pane_transfer *p_src,
276  RFC_Pane_transfer *p_trg,
277  const _SDF &sDF,
278  ENE &ene_src,
279  ENE &ene_trg,
280  int sfid_src,
281  int sfid_trg,
282  const Real alpha,
283  Nodal_data &rhs,
284  Nodal_data &diag,
285  int doa,
286  bool lump)
287 {
288  // Construct generic elements in parent source and target elements
289  Generic_element e_src( ene_src.size_of_edges(), ene_src.size_of_nodes());
290  Generic_element e_trg( ene_trg.size_of_edges(), ene_trg.size_of_nodes());
291 
292  // Obtain the local coordinates in source and target elements
293  Point_2 ncs_src[3], ncs_trg[3];
294  for ( int i=0; i<3; ++i) {
295  p_src->get_nat_coor_in_element( sfid_src, i, ncs_src[i]);
296  p_trg->get_nat_coor_in_element( sfid_trg, i, ncs_trg[i]);
297  }
298 
299  // Construct enumerators for nodal vertices in source and target elements
300  Nodal_coor_const nc;
301  Element_coor_const pnts_s( nc, p_src->coordinates(), ene_src);
302  Element_coor_const pnts_t( nc, p_trg->coordinates(), ene_trg);
303 
304  // Initialize pointer to element mass matrix. If a lumped mass matrix
305  // is desired, then use a local buffer to store the element matrix.
306  Real *pemm=NULL;
307 
308  bool needs_diag = diag.dimension()>0;
309  // Compute the element mass matrix only needs_diag is true.
310  if ( needs_diag)
311  pemm = lump ? (Real*)NULL : p_trg->get_emm( ene_trg.id());
312 
313  // Invoke the implementation to compute load vector and mass matrix.
314  element_load_vector( make_field( sDF, p_src, ene_src),
315  e_src, e_trg, pnts_s, pnts_t, ncs_src, ncs_trg,
316  alpha, 3, sDF.tag(),
317  make_field( rhs, p_trg, ene_trg),
318  make_field( diag, p_trg, ene_trg),
319  pemm, doa, lump);
320 }

322 template < class _SDF>
323 void Transfer_base::
324 init_load_vector( const _SDF &sDF,
325  const Real alpha,
326  Nodal_data &rhs,
327  Nodal_data &diag,
328  int doa,
329  bool lump)
330 {
331  bool needs_diag = diag.dimension()>0;
332 
333  // First, initialize the entries of the target mesh to zero.
334  for ( Pane_iterator pit=trg_ps.begin(); pit!=trg_ps.end(); ++pit) {
335  Real *p = (*pit)->pointer( rhs.id());
336  std::fill( p, p+(*pit)->size_of_nodes()*rhs.dimension(), Real(0));
337 
338  if ( needs_diag) {
339  p = (*pit)->pointer( diag.id());
340  std::fill( p, p+(*pit)->size_of_nodes()*diag.dimension(), Real(0));
341  }
342  }
343 
344  // Second, compute the integral over the target meshes by looping through
345  // the subfaces of the target window
346  ENE ene_src, ene_trg;
347  const RFC_Pane_transfer *p_src = NULL;
348  for ( Pane_iterator pit=trg_ps.begin(); pit!=trg_ps.end(); ++pit) {
349  // Loop through the subfaces of the target window
350  for ( int i=1, size=(*pit)->size_of_subfaces(); i<=size; ++i) {
351  (*pit)->get_host_element_of_subface( i, ene_trg);
352  if ( !(*pit)->need_recv( ene_trg.id())) continue;
353 
354  const Face_ID &fid = (*pit)->get_subface_counterpart(i);
355  if ( !p_src || p_src->id()!=fid.pane_id)
356  p_src = get_src_pane( fid.pane_id);
357  p_src->get_host_element_of_subface( fid.face_id, ene_src);
358 
359  compute_load_vector_wra( p_src, *pit, sDF, ene_src, ene_trg,
360  fid.face_id, i, alpha, rhs, diag, doa, lump);
361  }
362  }
363 
364  trg.reduce_to_all( rhs, MPI_SUM);
365  if ( needs_diag) trg.reduce_to_all( diag, MPI_SUM);
366 }

368 template < class _SDF>
369 void Transfer_base::
370 transfer_2n( const _SDF &sDF, Nodal_data &tDF, const Real alpha,
371  Real *tol, int *iter, int doa, bool verbose)
372 {
373  double t0(0);
374 
375  if ( verbose) {
376  trg.barrier(); t0 = get_wtime();
377  }
378 
379  // Allocate buffers
380  trg.init_nodal_buffers( tDF, (*iter>0)?7:3, (*iter>0));
381  Nodal_data b( trg.nodal_buffer(0));
382  Nodal_data z( trg.nodal_buffer(1));
383  Nodal_data diag( trg.nodal_buffer(2));
384 
385  bool needs_source_coor = alpha!=1.;
386  // Replicate the data of the source mesh (including coordinates if alpha!=1)
387  src.replicate_data( sDF, needs_source_coor);
388 
389  bool lump = *iter<=0; // whether to lump mass matrix
390 
391  // Initialize the load vector and the diagonal vector.
392  init_load_vector( sDF, alpha, b, diag, doa, lump);
393 
394  // Obtaining an initial guess by interpolation.
395  if ( !lump) interpolate_fe( sDF, tDF, false);
396 
397  // Clear up replicated data after obtaining the load vector and interpolation
398  src.clear_replicated_data();
399 
400  if ( *iter>0) {
401  Nodal_data p( trg.nodal_buffer(3));
402  Nodal_data q( trg.nodal_buffer(4));
403  Nodal_data r( trg.nodal_buffer(5));
404  Nodal_data s( trg.nodal_buffer(6));
405 
406  int ierr=pcg( tDF, b, p, q, r, s, z, diag, tol, iter);
407 
408  if (ierr) {
409  std::cerr << "***ROCFACE::WARNING: PCG did not converge after "
410  << *iter << " iterations and relative error is "
411  << *tol << std::endl;
412  }
413  }
414  else {
415  precondition_Jacobi( b, diag, tDF);
416  }
417 
418  trg.reduce_maxabs_to_all( tDF);
419 
420  // Delete buffer spaces
421  trg.delete_nodal_buffers();
422 
423  if ( verbose) {
424  trg.barrier();
425  if ( trg.is_root()) {
426  std::cout << "ROCFACE: Transfer to nodes done in "
427  << get_wtime()-t0 << " seconds";
428  if ( *iter > 0)
429  std:: cout << " with relative error "
430  << *tol << " after " << *iter << " iterators"
431  << std::endl;
432  else
433  std:: cout << "." << std::endl;
434  }
435  }
436 }
