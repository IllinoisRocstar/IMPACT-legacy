196  template < class _Data>
197  void compute_load_prime( const Generic_element &e_s,
198  const _Data &data_s, const Vector_3 *grads_s,
199  Vector_3 *load_prime, Tag_nodal ) {
200  Vector_3 NULL_VECTOR(0,0,0);
201  for ( unsigned int k=0; k<data_s.dimension(); ++k) {
202  load_prime[k] = NULL_VECTOR;
203  for ( unsigned int j=0; j<e_s.size_of_nodes(); ++j) {
204  load_prime[k] += data_s[j][k] * grads_s[j];
205  }
206  }
207  }
