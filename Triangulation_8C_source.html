45 static bool do_intersect( const Point_2 &p1, const Point_2 &p2,
46  const Point_2 &q1, const Point_2 &q2) {
47  Real u1[2], v21[2], v22[2], u2[2], v11[2];
48  Real a11, a12, a21, a22, sqn1;
49  Real v12[2], uref[2];
50  Real sqn2;
51  static Real dtemp_2[2] = {0.0, 1.0};
52  int i;
53 
54  /* Compute the signed areas a11, a12, a21, and a22. */
55  u1[0] = p2[0] - p1[0];
56  u1[1] = p2[1] - p1[1];
57 
58  v21[0] = q1[0] - p1[0];
59  v21[1] = q1[1] - p1[1];
60  v22[0] = q2[0] - p1[0];
61  v22[1] = q2[1] - p1[1];
62 
63  a11 = -u1[1] * v21[0];
64  a11 = a11 + u1[0] * v21[1];
65  a12 = -u1[1] * v22[0];
66  a12 = a12 + u1[0] * v22[1];
67 
68  u2[0] = q2[0] - q1[0];
69  u2[1] = q2[1] - q1[1];
70 
71  v11[0] = p1[0] - q1[0];
72  v11[1] = p1[1] - q1[1];
73  v12[0] = p2[0] - q1[0];
74  v12[1] = p2[1] - q1[1];
75 
76  a21 = -u2[1] * v11[0];
77  a21 = a21 + u2[0] * v11[1];
78  a22 = - u2[1] * v12[0];
79  a22 = a22 + u2[0] * v12[1];
80 
81  if (a11 == 0.0 && a12 == 0.0) {
82  /* Two line segments are colinear. Use intersect_intervals. */
83  sqn1 = 0.0;
84  sqn2 = 0.0;
85  for (i=0; i<=1; i+=1) {
86  sqn1 = sqn1 + u1[i] * u1[i];
87  sqn2 = sqn2 + u2[i] * u2[i];
88  }
89  if (sqn1 > sqn2) {
90  uref[0] = u1[0] / sqn1;
91  uref[1] = u1[1] / sqn1;
92 
93  Real dtemp_0[2];
94  dtemp_0[0] = v21[0] * uref[0] + v21[1] * uref[1];
95  dtemp_0[1] = v22[0] * uref[0] + v22[1] * uref[1];
96  return do_intersect_intervals(dtemp_2, dtemp_0);
97  }
98  else if (sqn2 > 0.0) {
99  uref[0] = u2[0] / sqn2;
100  uref[1] = u2[1] / sqn2;
101 
102  Real dtemp_1[2];
103  dtemp_1[0] = v11[0] * uref[0] + v11[1] * uref[1];
104  dtemp_1[1] = v12[0] * uref[0] + v12[1] * uref[1];
105  return do_intersect_intervals(dtemp_1, dtemp_2);
106  }
107  else {
108  for (i=0; i<=1; i+=2) {
109  if ( p1[i] != q1[i] || p1[1 + i] != q1[1 + i])
110  return false;
111  }
112  return true;
113  }
114  }
115  else if (a12 <= 0.0 && a11 >= 0.0 || a12 >= 0.0 && a11 <= 0.0) {
116  return a22 <= 0.0 && a21 >= 0.0 || a22 >= 0.0 && a21 <= 0.0;
117  }
118  else {
119  return false;
120  }
121 }

125 bool Triangulation::in_cone( Node_const_iterator u, Node_const_iterator v) const {
126  Node_const_iterator a0 = get_prev( u);
127  Node_const_iterator a1 = get_next( u);
128 
129  if ( !rightturn( get_point(*u), get_point(*a1), get_point(*a0))) {
130  return ( leftturn(get_point(*u), get_point(*v), get_point(*a0)) &&
131  leftturn(get_point(*v), get_point(*u), get_point(*a1)));
132  }
133 
134  return ( !rightturn(get_point(*v), get_point(*u), get_point(*a1)) &&
135  !rightturn(get_point(*a0), get_point(*u), get_point(*v)));

151 void Triangulation::triangulate( const Point_2 *ps, int n, Connectivity *tri) {
152  _pnts = ps;
153  _nodes.clear();
154  for ( int i=0; i<n; ++i) _nodes.push_back( Node(i));
155 
156  // Initialize ears
157  for ( Node_iterator it=_nodes.begin(); it!=_nodes.end(); ++it) {
158  it->set_ear( is_diagonal( get_prev(it), get_next(it)));
159  }
160 
161  tri->resize(0); tri->reserve( n-2);
162  // Loop through to remove ears.
163  for ( ; n>3; --n) {
164  // The inner loop searches for an ear
165  for ( Node_iterator it=_nodes.begin(); it!=_nodes.end(); ++it) {
166  if ( it->is_ear()) {
167  Node_iterator v1=get_prev( it), v0=get_prev( v1);
168  Node_iterator v3=get_next( it), v4=get_next( v3);
169 
170  // Output the ear
171  tri->push_back( Triangle( v1->id(), it->id(), v3->id()));
172 
173  // update neighbor vertices
174  v1->set_ear( is_diagonal( v0, v3));
175  v3->set_ear( is_diagonal( v1, v4));
176 
177  // Cut off the ear
178  _nodes.erase( it);
179  break;
180  }
181  }
182  }
183  RFC_assertion( n==3);
184 
185  tri->push_back( Triangle( _nodes.front().id(), (++_nodes.begin())->id(),
186  _nodes.back().id()));
187  _nodes.clear();
188 }
