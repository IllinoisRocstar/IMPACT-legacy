13 // vector of size n.

111  Self &operator=( const Array_n_const& v) {
112  if ( _start != v._start) { // Check not assigning to itself.

122  RFC_assertion( _end-_start == v1._end-v1._start);
123  const Real *q=v1._start;
124  for ( Real *p=_start; p!=_end; ++p, ++q) *p += *q;
125  return *this;
126  }

137  }

140  RFC_assertion( _end-_start == v1._end-v1._start);
141  const Real *q=v1._start;
142  for ( Real *p=_start; p!=_end; ++p, ++q) *p *= *q;
143  return *this;
144  }
145 
146  Self& operator/=( Real t) {
147  for ( Real *p=_start; p!=_end; ++p) { *p = (t==0) ? 1. : *p / t; }
148  return *this;
149  }
150

152  RFC_assertion( _end-_start == v1._end-v1._start);
153  const Real *q=v1._start;
154  for ( Real *p=_start; p!=_end; ++p, ++q) { *p = (*q==0) ? 1. : *p / *q; }
155  return *this;
156  }
157 
158  Self& invert() {
159  for ( Real *p=_start; p!=_end; ++p) { *p = (*p==0) ? 0. : 1. / *p; }
160  return *this;
161  }
162 
163 protected:
164  Real *_start, *_end;
