10 // Performs the operation: z = a*x op y
11 template <class data_type, int atype>
12 void Rocblas::axpy_gen( const void *ain, const DataItem *x,
13  const DataItem *y, DataItem *z) {
14  COM_assertion_msg(!x->is_windowed() && !y->is_windowed() && !z->is_windowed(),
15  "Unsupported dataitem type");
16  COM_assertion_msg( COM_compatible_types(x->data_type(),y->data_type()),
17  (std::string("Incompatible data types between ")+
18  x->fullname()+" and "+y->fullname()).c_str());
19  COM_assertion_msg( COM_compatible_types(x->data_type(),z->data_type()),
20  (std::string("Incompatible data types between ")+
21  x->fullname()+" and "+z->fullname()).c_str());
22 
23  int num_dims = z->size_of_components();
24  COM_assertion_msg(num_dims == x->size_of_components() &&
25  num_dims == y->size_of_components(),
26  (std::string("Numbers of components do not match between ")+
27  x->fullname()+" and "+y->fullname()).c_str());
28 
29  std::vector<const Pane*> apanes, xpanes, ypanes;
30  std::vector<Pane*> zpanes;
31 
32  x->window()->panes(xpanes);
33  y->window()->panes(ypanes);
34  z->window()->panes(zpanes);
35 
36  COM_assertion_msg( xpanes.size() == ypanes.size(),
37  (std::string("Numbers of panes do not match between ")+
38  x->window()->name()+" and "+y->window()->name()).c_str());
39  COM_assertion_msg( xpanes.size() == zpanes.size(),
40  (std::string("Numbers of panes do not match between ")+
41  x->window()->name()+" and "+z->window()->name()).c_str());
42 
43  std::vector<Pane*>::iterator zit, zend;
44  std::vector<const Pane*>::const_iterator xit, yit;
45 
46  const Pane** ait=NULL;
47  const DataItem *a=NULL;
48  const data_type *aval=NULL;
49 
50  if ( atype == BLAS_VOID)
51  aval = reinterpret_cast<const data_type *>( ain);
52  else {
53  a = reinterpret_cast<const DataItem*>(ain);
54 
55  if ( !a->is_windowed()) {
56  a->window()->panes( apanes);
57  COM_assertion_msg(xpanes.size() == ypanes.size(),
58  (std::string("Numbers of panes do not match between ")+
59  x->window()->name()+" and "+y->window()->name()).c_str());
60  ait = &apanes[0];
61  }
62  else {
63  COM_assertion_msg( a->size_of_items()==1, "Size of items do not match");
64  aval = reinterpret_cast<const data_type *>( a->pointer());
65 
66  COM_assertion_msg( aval, (std::string("Caught NULL pointer in ")+
67  a->fullname()).c_str());
68  }
69  }
70 
71  const int anum_dims = atype!=BLAS_VOID ? a->size_of_components() : 0;
72  COM_assertion_msg( atype==BLAS_VOID || (anum_dims==1 || anum_dims==num_dims),
73  (std::string("Numbers of components do not match between ")+
74  a->fullname()+" and "+z->fullname()).c_str());
75 
76  for(zit=zpanes.begin(),zend=zpanes.end(),xit=xpanes.begin(),yit=ypanes.begin();
77  zit != zend; ++zit, ait+=(atype!=BLAS_VOID&&ait), ++xit, ++yit) {
78  DataItem *pz = (*zit)->dataitem(z->id());
79  const int length = pz->size_of_items();
80  int zstrd=get_stride<BLAS_VEC2D>(pz);
81  const bool zstg = num_dims!=zstrd;
82 
83  const DataItem *px = (*xit)->dataitem(x->id());
84  int xstrd = get_stride<BLAS_VEC2D>(px);
85  COM_assertion_msg(length == px->size_of_items() || xstrd==0,
86  (std::string("Numbers of items do not match between ")+
87  x->fullname()+" and "+z->fullname()+
88  " on pane "+to_str((*zit)->id())).c_str());
89  const bool xstg = num_dims!=xstrd || xstrd==0;
90 
91  const DataItem *py = (*yit)->dataitem(y->id());
92  int ystrd = get_stride<BLAS_VEC2D>(py);
93  COM_assertion_msg(length == py->size_of_items() || ystrd==0,
94  (std::string("Numbers of items do not match between ")+
95  y->fullname()+" and "+z->fullname()+
96  " on pane "+to_str((*zit)->id())).c_str());
97 
98  const bool ystg = num_dims!=ystrd || ystrd==0;
99 
100  const DataItem *pa = (atype!=BLAS_VOID&&ait)?(*ait)->dataitem(a->id()):a;
101  int astrd = get_stride<atype>(pa);
102  const bool astg = pa && (anum_dims!=num_dims || anum_dims!=astrd);
103  COM_assertion_msg( atype!=BLAS_SCNE && atype!=BLAS_VEC2D ||
104  length == int(pa->size_of_items()) || astrd==0,
105  (std::string("Numbers of items do not match between ")+
106  a->fullname()+" and "+z->fullname()+
107  " on pane "+to_str((*zit)->id())).c_str());
108 
109  // Optimized version for contiguous dataitems
110  if ( !xstg && !zstg && !ystg && !astg && atype != BLAS_VEC
111  && (atype != BLAS_SCNE || num_dims==1)) {
112  const data_type *xval = (const data_type *)px->pointer();
113  const data_type *yval = (const data_type *)py->pointer();
114  data_type *zval = (data_type *)pz->pointer();
115 
116  // Get address for a if a is not window dataitem
117  if ( atype != BLAS_VOID && ait)
118  aval = reinterpret_cast<const data_type *>(pa->pointer());
119 
120  //Loop for each element/node and for each dimension
121  for(Size i = 0, s = length*num_dims; i<s; ++i, ++zval, ++xval, ++yval)
122  *zval = getref<data_type,atype,0>(aval,i,0,1)* *xval+ *yval;
123  }
124  else { // General version
125  //Loop for each dimension.
126  for(int i = 0; i < num_dims; ++i) {
127  DataItem *pz_i = num_dims==1?pz:(*zit)->dataitem(z->id()+i+1);
128  data_type *zval = (data_type *)pz_i->pointer();
129  zstrd=get_stride<BLAS_VEC2D>(pz_i);
130 
131  const DataItem *px_i = num_dims==1?px:(*xit)->dataitem(x->id()+i+1);
132  const data_type *xval = (const data_type *)px_i->pointer();
133  xstrd=get_stride<BLAS_VEC2D>(px_i);
134 
135  const DataItem *py_i = num_dims==1?py:(*yit)->dataitem(y->id()+i+1);
136  const data_type *yval = (const data_type *)py_i->pointer();
137  ystrd=get_stride<BLAS_VEC2D>(py_i);
138 
139  if ( atype != BLAS_VOID && ait) {
140  const DataItem *pa_i=anum_dims==1?pa:(*ait)->dataitem(a->id()+i+1);
141  aval = reinterpret_cast<const data_type *>( pa_i->pointer());
142  astrd = get_stride<atype>( pa_i);
143  }
144 
145  //Loop for each element/node.
146  for(int j=0; j<length; ++j, xval+=xstrd, zval+=zstrd, yval+=ystrd)
147  *zval = getref<data_type,atype,1>(aval,j,i,astrd) * *xval + *yval;
148  }
149  } // end if
150  } // end for
151 }

154 //Operation wrapper for z = ax + y.
155 void Rocblas::axpy( const DataItem *a, const DataItem *x, const DataItem *y,
156  DataItem *z) {
157  COM_Type att_type = z->data_type();
158  const int ancomp = a->size_of_components();
159 
160  if(att_type == COM_INT || att_type == COM_INTEGER) {
161  if ( a->is_windowed()) {
162  if ( ancomp == 1)
163  axpy_gen<int,BLAS_SCALAR>(a, x, y, z);
164  else
165  axpy_gen<int,BLAS_VEC>(a, x, y, z);
166  }
167  else {
168  if ( ancomp == 1)
169  axpy_gen<int,BLAS_SCNE>(a, x, y, z);
170  else
171  axpy_gen<int,BLAS_VEC2D>(a, x, y, z);
172  }
173  }
174  else {
175  COM_assertion_msg( att_type==COM_DOUBLE || att_type==COM_DOUBLE_PRECISION,
176  (std::string("Unsupported data type in ")+
177  z->fullname()).c_str());
178 
179  if ( a->is_windowed()) {
180  if ( ancomp == 1)
181  axpy_gen<double,BLAS_SCALAR>(a, x, y, z);
182  else
183  axpy_gen<double,BLAS_VEC>(a, x, y, z);
184  }
185  else {
186  if ( ancomp == 1)
187  axpy_gen<double,BLAS_SCNE>(a, x, y, z);
188  else
189  axpy_gen<double,BLAS_VEC2D>(a, x, y, z);
190  }
191  }
192 }

195 void Rocblas::axpy_scalar( const void *a, const DataItem *x, const DataItem *y,
196  DataItem *z) {

198 
199  if(att_type == COM_INT || att_type == COM_INTEGER)
200  axpy_gen<int,BLAS_VOID>(a, x, y, z);

202  axpy_gen<double,BLAS_VOID>(a, x, y, z);
203 }
