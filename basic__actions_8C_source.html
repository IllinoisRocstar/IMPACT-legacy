227  MAN_DEBUG(2, ("Rocman: ComputeFluidLoad_ALE::run() with t:%e dt:%e.\n", t, dt));
228  // ts = tf + ( mdot*Vs

230  // part 1 (POST_UPDATE_FLUID in fluid_agent.f90)
231  if (traction_mode != NO_SHEER) {
232  COM_call_function( RocBlas::copy, &f_tf_hdl, &f_ts_hdl);
233 
234  // Compute tfmts = t_s-t_f (see developers guide).
235  // Here fb_mdot_tmp is used as buffer space.
236  COM_call_function( RocBlas::div, &fb_mdot_hdl, &fb_rhof_alp_hdl, &fb_mdot_tmp_hdl);

238  if ( sagent->withALE) {
239  COM_call_function( RocBlas::sub, &fb_mdot_tmp_hdl, &b_rb_hdl, &fb_mdot_tmp_hdl);

242  COM_call_function( RocBlas::mul, &fb_mdot_tmp_hdl, &fb_mdot_hdl, &fb_mdot_tmp_hdl);
243  COM_call_function( RocBlas::mul, &fb_nf_alp_hdl, &fb_mdot_tmp_hdl, &fb_ts_hdl);
244 
245  // Compute ts = tf - (tf-ts)
246  COM_call_function( RocBlas::sub, &fb_tf_hdl, &fb_ts_hdl, &fb_ts_hdl);
247  }
248  else { // ts is a scalar
249 
250  COM_call_function( RocBlas::copy, &f_pf_hdl, &f_ts_hdl);
251 
252  // Compute tsmpf = p_f-t_s (see developers guide).
253  // Here fb_mdot_tmp is used as buffer space.
254  COM_call_function( RocBlas::div, &fb_mdot_hdl, &fb_rhof_alp_hdl, &fb_mdot_tmp_hdl);
255 
256  if ( sagent->withALE) {
257  COM_call_function( RocBlas::sub, &fb_mdot_tmp_hdl, &b_rb_hdl, &fb_mdot_tmp_hdl);

489 void FluidPropagateSurface::run( double t, double dt, double alpha_dummy)
490 {
491  if (zoom <= 0.0 || fagent->get_coupling()->initial_start() ) return;
492 
493  int rank = 0;
494  MPI_Comm mycomm = fagent->get_communicator();
495  MPI_Comm_rank(mycomm,&rank);
496  MAN_DEBUG(2, ("Rocman: FluidPropagateSurface::run() with t:%e dt:%e zoom:%f p_cnstr_type:%d.\n", t, dt, zoom, p_cnstr_type));
497 
498  double zero = 0.0;
499  COM_call_function( RocBlas::copy_scalar, &zero, &p_rb_hdl);
500 
501  COM_call_function( RocBlas::copy, &b_rb_hdl, &fb_rb_hdl);
502 
503  // Burnout strategy:
504  //
505  // integer array (bflag) lives on interacting boundary
506  // faces. bflag indicates whether the boundary face
507  // is burning/reacting/injecting, etc. In the current
508  // implementation, bflag can be 1 or 0 for burning or
509  // not-burning.
510  //
511  // integer array (cflag) lives on boundary nodes.
512  // cflag indicates the status of one or more geometrical
513  // or other imposed constraints on the deformation or
514  // solution space of the domain.
515  // In the current implementation cflag = 1 for nodes
