705 template <int dim, COM_Type type, Access_mode mode>
706 void
707 com_obtain_array_helper( const std::string &waname, int pane_id, void *addr,
708  int &strd, int &cap, int offset, int l=0) {
709 
710  if ( mode != AM_RESIZE && strd<0) strd=0;
711 
712  COM_base::Pointer_descriptor ptr(NULL, dim);
713  if ( mode==AM_ALLOC) {
714  COM_get_com()->allocate_array( waname, pane_id, NULL, strd, cap);
715  COM_get_com()->get_array( waname, pane_id, ptr);
716  }
717  else if ( mode==AM_RESIZE) {
718  COM_get_com()->resize_array( waname, pane_id, NULL, strd, cap);
719  COM_get_com()->get_array( waname, pane_id, ptr);
720  }
721  else if ( mode == AM_COPY) {
722  if ( dim==0) ptr.ptr = addr; // If dim==0, it is not a pointer
723  else ptr.ptr = com_get_address_f<dim,type>( addr, l);
724 
725  COM_get_com()->copy_array( waname, pane_id, ptr.ptr,
726  strd, cap, offset);
727  }
728  else {
729  COM_get_com()->get_array( waname, pane_id, ptr, &strd,
730  &cap, mode==AM_GETC);
731  }
732 
733  if ( mode != AM_COPY)
734  com_set_address_f<dim, type>( ptr, addr, l);
735 }
