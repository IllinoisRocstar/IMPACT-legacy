46 Real get_face_volume(Point_3<Real> ps_face[], int ne) {
47 
48  Generic_element_2 e(ne);
49  Vector_2<Real> nc(0,0);
50  Vector_3<Real> J[2];
51  int order = 1;
52 
53  int size = e.get_num_gp(order);
54  Real volume = 0;
55  for (int k = 0; k<size; k++){
56  Real weight = e.get_gp_weight( k, order);
57  e.get_gp_nat_coor(k, nc, order);
58 
59  Point_3<Real> x;
60  e.interpolate(ps_face,nc,&x);
61 
62  e.Jacobian( ps_face,nc,J);
63  // Get the normal to the face
64  Vector_3<Real> n = Vector_3<Real>::cross_product(J[0],J[1]);
65 
66  Real t = weight * (n*(Vector_3<Real>&)x);
67 
68  // compute the volume, and add to running total
69  volume += t;
70  }
71  return volume;
72 }

87 void Rocsurf::compute_bounded_volumes( const COM::DataItem *old_location,
88  const COM::DataItem *new_location,
89  COM::DataItem *element_volume,
90  void *flag) {
91 
92  assert( element_volume != NULL && element_volume ->size_of_components() == 1);
93  assert( element_volume->window() == old_location->window());
94  assert( element_volume->window() == new_location->window());
95 
96  std::vector< COM:: Pane*> panes;
97  element_volume->window() -> panes( panes);
98 
99  Point_3<Real> ps_face4[4];
100  Point_3<Real> ps_face3[3];
101  Point_3<Real> ps[8];
102 
103  // ps is used as a container for both the old and new coordinates
104  // the following ordering convention is used:
105  // the coordinates of the new surface occupy indexes 0 to 3
106  // the coordinates of the old surface occupy indexes 4 to 7
107  // both surfaces have coordinates listed in the same order
108  // i.e. ps[0] and ps[4] are new and old coordinates for the same node
109  // under these conventions, ps[3] and ps[7] are not used for triangular faces
110 
111  std::vector< COM::Pane*>::const_iterator it = panes.begin();
112  // Loop through the elements of the pane.
113  for (int i=0, local_npanes = panes.size(); i<local_npanes; ++i, ++it){
114  const COM::Pane &pane = **it;
115  Real *ptr_ev = (Real *)(pane.dataitem( element_volume->id())->pointer());
116  const Point_3<Real> *ptr_new =
117  (const Point_3<Real>*)(pane.dataitem( new_location->id())->pointer());
118  const Point_3<Real> *ptr_old =
119  (const Point_3<Real>*)(pane.dataitem( old_location->id())->pointer());
120 
121  Element_node_vectors_k_const<Point_3<Real> > ps_new, ps_old;
122 
123  Element_node_enumerator ene( &pane, 1);
124  for ( int j=pane.size_of_elements(); j>0; --j, ene.next(), ++ptr_ev) {
125  if ( flag!=NULL && *ptr_ev==0.) continue;
126 
127  Real volume = 0;
128  // if mesh is triangular, we have two triangular faces, 3 quadrilaterals
129  ps_new.set( ptr_new, ene, 1);
130  ps_old.set( ptr_old, ene, 1);
131 
132  ps[0]=ps_new[0];ps[1]=ps_new[1];ps[2]=ps_new[2];
133  ps[4]=ps_old[0];ps[5]=ps_old[1];ps[6]=ps_old[2];
134 
135  if (ene.size_of_edges()==3) {
136 
137  if ( ps[0]==ps[4] && ps[1]==ps[5] && ps[2]==ps[6])
138  { *ptr_ev = 0.; continue; }
139  ps[3] = Point_3<Real>(0,0,0);
140  normalize_coor( ps, 7, 6);
141 
142  // solid has two triangular faces
143  arrange(ps_face3,ps,0,2,1);
144  volume += get_face_volume(ps_face3, 3);
145  arrange(ps_face3,ps,4,5,6);
146  volume += get_face_volume(ps_face3, 3);
147  // quadrilateral faces
148  arrange(ps_face4,ps,0,4,6,2);
149  volume += get_face_volume(ps_face4, 4);
150  }
151  else{ // if not triangular, mesh is quadrilateral
152  ps[3]=ps_new[3]; ps[7]=ps_old[3];
153  if ( ps[0]==ps[4] && ps[1]==ps[5] && ps[2]==ps[6] && ps[3]==ps[7])
154  { *ptr_ev = 0.; continue; }
155  normalize_coor( ps, 8, 8);
156 
157  arrange(ps_face4,ps,0,3,2,1);
158  volume += get_face_volume(ps_face4, 4);
159  arrange(ps_face4,ps,4,5,6,7);
160  volume += get_face_volume(ps_face4, 4);
161  arrange(ps_face4,ps,3,7,6,2);
162  volume += get_face_volume(ps_face4, 4);
163  arrange(ps_face4,ps,0,4,7,3);
164  volume += get_face_volume(ps_face4, 4);
165  }
166  // Two quarilateral faces are the same in either case
167  arrange(ps_face4,ps,1,2,6,5);
168  volume += get_face_volume(ps_face4, 4);
169  arrange(ps_face4,ps,0,1,5,4);
170  volume += get_face_volume(ps_face4, 4);
171 
172  volume /= 3.;
173  *ptr_ev = -volume;
174  }
175  }

270 void Rocsurf::compute_center( const COM::DataItem *mesh,
271  Vector_3<Real> &cnt) {
272  std::vector< const COM:: Pane*> panes;
273  mesh->window() -> panes( panes);
274 
275  struct Accumulator {
276  Accumulator() : cnt(Vector_3<Real>(0,0,0)), count(0) {}
277 
278  Vector_3<Real> cnt;
279  double count;
280  };
281  Accumulator global;
282 
283  std::vector< const COM::Pane*>::const_iterator it = panes.begin();
284  // Loop through the elements of the pane.
285  for (int i=0, local_npanes = panes.size(); i<local_npanes; ++i, ++it){
286  const COM::Pane &pane = **it;
287  const Vector_3<Real> *ptr =
288  (const Vector_3<Real>*)(pane.dataitem( COM::COM_NC)->pointer());
289 
290  for ( int j=pane.size_of_real_nodes(); j>0; --j, ++ptr) {
291  global.cnt += *ptr;
292  global.count += 1.;
293  }
294  }
295 
296  // Perform reduction on cnt and count
297  Accumulator local=global;
298 
299  if ( COMMPI_Initialized()) {
300  MPI_Allreduce( &local.cnt[0], &global.cnt[0], 4, MPI_DOUBLE, MPI_SUM,
301  mesh->window()->get_communicator());
302  }
303  cnt = global.cnt/global.count;
304 }

327 
328  Element_node_enumerator ene( &pane, 1);
329  for ( int j=pane.size_of_elements(); j>0; --j, ene.next()) {
330  ps.set( ptr, ene, 1);
331 
332  normalized_face[0]=ps[0]-cnt;
333  normalized_face[1]=ps[1]-cnt;
