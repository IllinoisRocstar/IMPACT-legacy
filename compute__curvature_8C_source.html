18 // Helper function for compute_nodal_mcn.
19 // Compute weights of LB-operator about the first vertex of a given
20 // quadrilateral. Also returns the area of the element.
21 static double
22 compute_lbop_weights( Point_3<Real> *ps, Real *ws) {
23  ws[0] = ws[1] = ws[2] = 0.;
24 
25  Generic_element_2 e(4);
26  int size = e.get_num_gp();
27 
28  double area = 0;
29  for (int i=0; i<size; i++){
30  // Compute tangents at quadrature points.
31  Vector_2<Real> nc;
32  e.get_gp_nat_coor(i, nc);
33 
34  const Real &xi=nc[0], &eta=nc[1];
35 
36  const Vector_3<Real> u = (1-eta)*(ps[1]-ps[0])+eta*(ps[2]-ps[3]);
37  const Vector_3<Real> v = (1-xi)* (ps[3]-ps[0])+xi* (ps[2]-ps[1]);
38 
39  // Obtain the weights and local area of quadrature point
40  const Real weight = e.get_gp_weight( i);
41 
42  // Evaluate the weights for Laplace-Beltrami operator
43  const Real jacobi_det = u.cross_product(u,v).norm();
44  const Real uv = u*v;
45 
46  const Real su = weight * ((1-eta)*uv-(1-xi)*(u*u))/jacobi_det;
47  const Real sv = weight * ((1-xi)*uv-(1-eta)*(v*v))/jacobi_det;
48 
49  ws[0] += (1-eta)*sv -xi*su;
50  ws[1] += xi*su + eta*sv;
51  ws[2] += (1-xi)*su - eta*sv;;
52  area += weight * jacobi_det;
53  }
54 
55  return area;
56 }

58 // Compute mean-curvature normals and their Laplace-Beltrami operator.
59 // Algorithm based on Y. Zhang, C. Bajaj, and G. Xu, "Surface Smoothing
60 // and quality improvement of quadrilateral/hexahedral meshes with
61 // geometric flow", International Meshing Roundtable, 2005.
62 void Window_manifold_2::
63 compute_mcn( COM::DataItem *mcn_in, COM::DataItem *lbmcn_in) {
64 
65  COM_assertion( mcn_in != NULL && mcn_in ->size_of_components() == 3 &&
66  mcn_in->is_nodal());
67  COM_assertion( lbmcn_in != NULL && lbmcn_in->size_of_components() == 3 &&
68  lbmcn_in->is_nodal());
69 
70  COM::DataItem *mcn, *lbmcn;
71  // Inherit mcn and lbmcn onto buffer window
72  if ( mcn_in->window() != _buf_window)
73  mcn = _buf_window->inherit( mcn_in, "mcn__MCNTEMP",
74  false, true, NULL, 0);
75  else
76  mcn = mcn_in;
77 
78  if ( lbmcn_in->window() != _buf_window)
79  lbmcn = _buf_window->inherit( lbmcn_in, "lbmcn__MCNTEMP",
80  false, true, NULL, 0);
81  else
82  lbmcn = lbmcn_in;
83 
84  // Allocate buffer spaces for weights and areas
85  COM::DataItem *areas=NULL, *weights=NULL;
86  areas = _buf_window->new_dataitem( "areas__MCNTEMP", 'n',
87  COM_DOUBLE, 1, "");
88  _buf_window->resize_array( areas, NULL);
89 
90  weights = _buf_window->new_dataitem( "weights__MCNTEMP", 'e',
91  COM_DOUBLE, 12, "");
92  _buf_window->resize_array( weights, NULL);
93  _buf_window->init_done( false);
94 
95  // Initialize dataitems to 0s
96  double zero = 0.;
97  Rocblas::copy_scalar( &zero, mcn);
98  Rocblas::copy_scalar( &zero, areas);
99  Rocblas::copy_scalar( &zero, weights);
100 
101  std::vector< COM::Pane*> panes;
102  mcn->window()->panes( panes);
103  Point_3<Real> ps[4];
104 
105  std::vector< COM::Pane*>::const_iterator it = panes.begin();
106 
107  // First, compute the mean-curvature normal, and save the weights
108  // for Laplace-Beltrami operator.
109  for (int i=0, local_npanes = panes.size(); i<local_npanes; ++i, ++it){
110  const COM::Pane &pane = **it;
111 
112  const Point_3<Real> *pnts =
113  reinterpret_cast<const Point_3<Real>*>(pane.coordinates());
114  Vector_3<Real> *mcn_ptr = (Vector_3<Real> *)
115  (pane.dataitem( mcn->id())->pointer());
116  Vector_3<Real> *ws_ptr = (Vector_3<Real> *)
117  (pane.dataitem( weights->id())->pointer());
118  Real *area_ptr = (Real *)(pane.dataitem( areas->id())->pointer());
119 
120  // Loop through elements of the pane
121  Element_node_enumerator ene( &pane, 1);
122 
123  for ( int j=0, jsize=pane.size_of_elements(); j<jsize; ++j, ene.next()) {
124  Point_3<Real> cnt(0,0,0);
125  int ne=ene.size_of_edges();
126  for ( int kk=0; kk<ne; ++kk)
127  (Vector_3<Real>&)cnt += (const Vector_3<Real>&)pnts[ene[kk]-1];
128  (Vector_3<Real>&)cnt /= ne;
129 
130  // Loop through each vertex
131  for ( int k=0; k<ne; ++k) {
132  // Assign points for the quadrilateral.
133  ps[0] = pnts[ene[k]-1];
134  ps[1] = ps[0]+0.5*(pnts[ene[(k+1)%ne]-1]-ps[0]);
135  ps[2] = cnt;
136  ps[3] = ps[0]+0.5*(pnts[ene[(k+ne-1)%ne]-1]-ps[0]);
137 
138  // Compute weights of LB-operator for the first vertex and
139  // increment the area.
140  Vector_3<Real> &ws = ws_ptr[4*j+k];
141  area_ptr[ ene[k]-1] += compute_lbop_weights( ps, &ws[0]);
142 
143  Vector_3<Real> dA = (ws[0]*(ps[1]-ps[0])+ws[1]*(ps[2]-ps[0])+
144  ws[2]*(ps[3]-ps[0]));
145 
146  mcn_ptr[ ene[k]-1] += dA;
147  }
148  }
149  }
150 
151  // Reduce on the area and mcn and divide mcn by areas.
152  reduce_on_shared_nodes( areas, MPI_SUM);
153  reduce_on_shared_nodes( mcn, MPI_SUM);
154  Rocblas::div( mcn, areas, mcn);
155 
156  // Second, compute the Laplace-Beltrami of the mean-curvature.
157  Rocblas::copy_scalar( &zero, lbmcn);
158 
159  it = panes.begin();
160  for (int i=0, local_npanes = panes.size(); i<local_npanes; ++i, ++it) {
161  const COM::Pane &pane = **it;
162 
163  const Vector_3<Real> *mcn_ptr = (const Vector_3<Real> *)
164  (pane.dataitem( mcn->id())->pointer());
165  const Vector_3<Real> *ws_ptr = (const Vector_3<Real> *)
166  (pane.dataitem( weights->id())->pointer());
167 
168  Vector_3<Real> *lbmcn_ptr = (Vector_3<Real> *)
169  (pane.dataitem( lbmcn->id())->pointer());
170 
171  // Loop through elements of the pane
172  Element_node_enumerator ene( &pane, 1);
173  double fs[4];
174 
175  for ( int j=0, jsize=pane.size_of_elements(); j<jsize; ++j, ene.next()) {
176  int ne=ene.size_of_edges();
177 
178  // Loop through each vertex
179  for ( int k=0; k<ne; ++k) {
180 
181  // Compute the magnitude and vector of current vertex
182  int ii0=ene[k]-1;
183  fs[0] = mcn_ptr[ii0].norm();
184  Vector_3<Real> vec=(mcn_ptr[ii0]); if (fs[0]>0) vec/=fs[0];
185 
186  // Compute length of other vertices mcn corresponding to the
187  // direction of mcn at the current vertex
188  for ( int kk=1; kk<ne; ++kk) {
189  int ii=ene[(k+kk)%ne]-1;
190 
191  fs[kk] = sign(mcn_ptr[ii]*vec)*mcn_ptr[ii].norm();
192  }
193 
194  const Vector_3<Real> &ws = ws_ptr[4*j+k];
195  lbmcn_ptr[ ene[k]-1] += 4.*(ws[0]*(fs[1]-fs[0])+ws[1]*(fs[2]-fs[0])+
196  ws[2]*(fs[3]-fs[0]))*vec;
197  }
198  }
199  }
200 
201  // Reduce on lbmcn and divide it by areas.
202  reduce_on_shared_nodes( lbmcn, MPI_SUM);
203  Rocblas::div( lbmcn, areas, lbmcn);
204 
205  // Delete buffer space in reverse order
206  _buf_window->delete_dataitem( weights->name());
207  _buf_window->delete_dataitem( areas->name());
208  if ( lbmcn_in->window() != _buf_window)
209  _buf_window->delete_dataitem( lbmcn->name());
210  if ( mcn_in->window() != _buf_window)
211  _buf_window->delete_dataitem( mcn->name());
212 
213  _buf_window->init_done(false);
214 }
