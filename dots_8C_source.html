12 void Rocblas::calcDot( void *yout, const DataItem *x, const DataItem *z,
13  const MPI_Comm *comm, const DataItem *mults) {
14 
15  COM_assertion_msg( !x->is_windowed() && !z->is_windowed(),
16  "Unsupported dataitem type");
17  COM_assertion_msg( COM_compatible_types(x->data_type(),z->data_type()),
18  (std::string("Incompatible data types between ")+
19  x->fullname()+" and "+z->fullname()).c_str());
20 
21  int num_dims = z->size_of_components();
22  COM_assertion_msg( num_dims == x->size_of_components(),
23  (std::string("Numbers of components do not match between ")+
24  x->fullname()+" and "+z->fullname()).c_str());
25 
26  std::vector<const Pane*> xpanes, zpanes, mpanes;
27  std::vector<Pane*> ypanes;
28 
29  z->window()->panes(zpanes);
30  x->window()->panes(xpanes);
31 
32  COM_assertion_msg( xpanes.size() == zpanes.size(),
33  (std::string("Numbers of panes do not match between ")+
34  x->window()->name()+" and "+z->window()->name()).c_str());
35 
36  std::vector<const Pane*>::const_iterator zit, zend, xit;
37  Pane **yit=NULL;
38 
39  DataItem *y=NULL;
40  data_type *yval=NULL;
41 
42  if ( ytype == BLAS_VOID) {
43  yval = reinterpret_cast<data_type *>( yout);
44  *yval = data_type(0);
45  }
46  else {
47  y = reinterpret_cast<DataItem *>(yout);
48 
49  if ( !y->is_windowed()) {
50  y->window()->panes(ypanes);
51  COM_assertion_msg(xpanes.size() == ypanes.size(),
52  (std::string("Numbers of panes do not match between ")+
53  x->fullname()+" and "+y->fullname()).c_str());
54  yit = &ypanes[0];
55  }
56  else {
57  COM_assertion_msg( y->size_of_items()==1,
58  (std::string("Numbers of items do not match between ")+
59  x->fullname()+" and "+y->fullname()).c_str());
60  yval = reinterpret_cast<data_type *>( y->pointer());
61 
62  COM_assertion_msg( yval, (std::string("Caught NULL pointer in ")+
63  y->fullname()).c_str());
64  }
65  }
66 
67  const int ynum_dims = ytype != BLAS_VOID?y->size_of_components():0;
68  COM_assertion_msg( ytype==BLAS_VOID || (ynum_dims==1 || ynum_dims==num_dims),
69  (std::string("Numbers of components do not match between ")+
70  z->fullname()+" and "+y->fullname()).c_str());
71 
72  // Initialize pointer to multiplicities
73  const Pane ** mit=NULL;
74  const int *mval=NULL;
75  if ( mults != NULL) {
76  COM_assertion_msg( COM_compatible_types( COM_INT, mults->data_type()) &&
77  mults->size_of_components()==1,
78  (std::string("Multiplier ")+mults->fullname()+
79  "must be integer scalars.").c_str());
80 
81  mults->window()->panes(mpanes);
82  COM_assertion_msg(xpanes.size() == mpanes.size(),
83  (std::string("Numbers of panes do not match between ")+
84  x->window()->name()+" and "+
85  mults->window()->name()).c_str());
86  mit = &mpanes[0];
87  }
88 
89  // Initialize y to 0 for BLAS_VOID or window dataitem
90  if ( ytype == BLAS_VOID) {
91  *yval = data_type(0);
92  }
93  else {
94  if ( y->is_windowed()) {
95  for ( int i=0; i<ynum_dims; ++i) yval[i] = data_type(0);
96  }
97  }
98 
99  for( zit=zpanes.begin(), zend=zpanes.end(), xit=xpanes.begin();
100  zit!=zend; ++zit, ++xit, yit+=( ytype!=BLAS_VOID && yit)) {
101  const DataItem *pz = (*zit)->dataitem(z->id());
102  const int length = pz->size_of_items();
103  int zstrd=get_stride<BLAS_VEC2D>(pz);
104  const bool zstg = num_dims!=zstrd;
105 
106  const DataItem *px = (*xit)->dataitem(x->id());
107  int xstrd = get_stride<BLAS_VEC2D>(px);
108  COM_assertion_msg(length == px->size_of_items(),
109  (std::string("Numbers of items do not match between ")+
110  x->fullname()+" and "+z->fullname()+
111  " on pane "+to_str((*zit)->id())).c_str());
112 
113  const bool xstg = num_dims!=xstrd;
114 
115  DataItem *py = y;
116  if ( ytype != BLAS_VOID && !y->is_windowed()) {
117  // Obtain py and initialize to 0
118  py = (*yit)->dataitem(y->id());
119  const int ynum_comp = py->size_of_components();
120  const int ylen = py->size_of_items();
121  COM_assertion_msg( ylen==1 || ylen==length,
122  (std::string("Numbers of items do not match between ")+
123  y->fullname()+" and "+z->fullname()+
124  " on pane "+to_str((*zit)->id())).c_str());
125 
126  yval = reinterpret_cast<data_type *>( py->pointer());
127 
128  if ( py->stride()==ynum_comp) {
129  for ( int i=0, ni=ynum_comp*ylen; i<ni; ++i) yval[i] = data_type(0);
130  }
131  else {
132  // Loop through the number of components
133  for (int i=0; i<ynum_comp; ++i) {
134  DataItem *py_i = ynum_comp>1?(*yit)->dataitem( y->id()+i+1):py;
135  data_type *yv_i = reinterpret_cast<data_type*>(py_i->pointer());
136  const int strd=get_stride<BLAS_VEC2D>(py_i);
137  // loop through the stride
138  for ( int j=0, nj=ylen*strd; j<nj; j+=strd) yv_i[j] = data_type(0);
139  }
140  }
141  }
142  int ystrd = get_stride<ytype>(py);
143  const bool ystg = py && (ynum_dims!=num_dims || ynum_dims!=ystrd);
144 
145  // Obtain the multiplier
146  if ( mults != NULL) {
147  const DataItem *pm = (*mit)->dataitem(mults->id());
148  COM_assertion_msg( pm->size_of_items()==length,
149  (std::string("Numbers of items do not match between ")+
150  mults->fullname()+" and "+z->fullname()+
151  " on pane "+to_str((*zit)->id())).c_str());
152 
153  mval = reinterpret_cast<const int*>( pm->pointer());
154  ++mit;
155  }
156 
157  // Optimized version for contiguous dataitems
158  if ( !xstg && !zstg && !ystg && mval == NULL) {
159  const data_type *xval = (const data_type *)px->pointer();
160  const data_type *zval = (const data_type *)pz->pointer();
161 
162  //Loop for each element/node and for each dimension
163  for(Size i = 0, s = length*num_dims; i<s; ++i, ++xval, ++zval)
164  getref<data_type,ytype,0>(yval,i,0,1) += *xval* *zval;
165  }
166  else { // General version
167  //Loop for each dimension.
168  for(int i = 0; i < num_dims; ++i) {
169  const DataItem *pz_i = num_dims==1?pz:(*zit)->dataitem(z->id()+i+1);
170  const data_type *zval = (const data_type *)pz_i->pointer();
171  zstrd=get_stride<BLAS_VEC2D>(pz_i);
172 
173  const DataItem *px_i = num_dims==1?px:(*xit)->dataitem(x->id()+i+1);
174  const data_type *xval = (const data_type *)px_i->pointer();
175  xstrd=get_stride<BLAS_VEC2D>(px_i);
176 
177  if ( ytype != BLAS_VOID && !y->is_windowed()) {
178  DataItem *py_i=ynum_dims==1?py:(*yit)->dataitem(y->id()+i+1);
179  yval = reinterpret_cast<data_type *>( py_i->pointer());
180  ystrd = get_stride<ytype>( py_i);
181  }
182 
183  if ( mval != NULL) {
184  //Loop for each element/node.
185  for(int j=0; j<length; ++j, xval+=xstrd,zval+=zstrd)
186  getref<data_type,ytype,1>( yval,j,i,ystrd) += *xval* *zval / mval[j];
187  }
188  else
189  for(int j=0; j<length; ++j, xval+=xstrd,zval+=zstrd)
190  getref<data_type,ytype,1>( yval,j,i,ystrd) += *xval* *zval;
191  }
192  }
193  }
194 
195  if ( ( ytype==BLAS_VOID || ytype==BLAS_SCALAR || ytype==BLAS_VEC)
196  && comm && *comm!=MPI_COMM_NULL && COMMPI_Initialized()) {
197  int n = (ytype == BLAS_VEC) ? num_dims : 1;
198  COM_Type att_type = x->data_type();
199 
200  data_type t = *yval;
201  if ( att_type == COM_INT || att_type == COM_INTEGER) {
202  MPI_Allreduce( &t, yval, n, MPI_INT, MPI_SUM, *comm);
203  }
204  else {
205  MPI_Allreduce( &t, yval, n, MPI_DOUBLE, MPI_SUM, *comm);
206  }
207  }

210 //Wrapper for dot product that produces a single scalar answer.
211 void Rocblas::dot_MPI(const DataItem *x, const DataItem *y, DataItem *z,
212  const MPI_Comm *comm, const DataItem *mults) {
213  COM_Type att_type = x->data_type();
214  const int zncomp = z->size_of_components();
215 
216  if ( att_type == COM_INT || att_type == COM_INTEGER) {
217  if ( z->is_windowed()) {
218  if ( zncomp == 1)
219  calcDot<int,BLAS_SCALAR>(z, x, y, comm, mults);
220  else
221  calcDot<int,BLAS_VEC>(z, x, y, comm, mults);
222  }
223  else {
224  if ( zncomp == 1)
225  calcDot<int,BLAS_SCNE>(z, x, y, comm, mults);
226  else
227  calcDot<int,BLAS_VEC2D>(z, x, y, comm, mults);
228  }
229  }
230  else {
231  COM_assertion_msg( att_type==COM_DOUBLE || att_type==COM_DOUBLE_PRECISION,
232  (std::string("Unsupported data type in ")+
233  x->fullname()).c_str());
234  if ( z->is_windowed()) {
235  if ( zncomp == 1)
236  calcDot<double,BLAS_SCALAR>(z, x, y, comm, mults);
237  else
238  calcDot<double,BLAS_VEC>(z, x, y, comm, mults);
239  }
240  else {
241  if ( zncomp == 1)
242  calcDot<double,BLAS_SCNE>(z, x, y, comm, mults);
243  else
244  calcDot<double,BLAS_VEC2D>(z, x, y, comm, mults);
245  }
246  }
