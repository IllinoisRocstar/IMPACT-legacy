20 void init_structured_mesh( double coors[][3], int nrow, int ncol,
21  int rank, int nproc) {
22  // consider the processors as a 2*(nproc/2) grid
23  int proc_col=nproc;
24  if (nproc%2 ==0) {
25  proc_col=nproc/2;
26  }
27  else {
28  proc_col=nproc;
29  }
30  int row=rank/proc_col, col=rank%proc_col;
31 
32  const double width=100., length=100.;
33 
34  for (int i=0; i<nrow; ++i) {
35  for (int j=0; j<ncol; ++j) {
36  coors[i*ncol+j][0]=col*length+length/(ncol-1)*j;
37  coors[i*ncol+j][1]=row*width+width/(nrow-1)*i;
38  coors[i*ncol+j][2]=0.01;
39  }
40  }
41 }

43 void init_unstructure_mesh( double coors[][3], int elmts[][4],
44  int nrow, int ncol, int rank, int nproc) {
45  // consider the processors as a 2*(nproc/2) grid
46  int proc_col=nproc;
47  if (nproc%2 ==0) {
48  proc_col=nproc/2;
49  }
50  else {
51  proc_col=nproc;
52  }
53 
54  int row=rank/proc_col, col=rank%proc_col;
55 
56  const double width=100., length=100.;
57 
58  for (int i=0; i<nrow; ++i) {
59  for (int j=0; j<ncol; ++j) {
60  coors[i*ncol+j][0]=col*length+length/(ncol-1)*j;
61  coors[i*ncol+j][1]=row*width+width/(nrow-1)*i;
62  coors[i*ncol+j][2]=0;
63  }
64  }
65 
66  // computes elmts
67  for (int i=0; i<nrow-1; ++i) {
68  for (int j=0; j<ncol-1; ++j) {
69  elmts[i*(ncol-1)+j][0]=i*ncol+j+1;
70  elmts[i*(ncol-1)+j][1]=i*ncol+j+ncol+1;
71  elmts[i*(ncol-1)+j][2]=i*ncol+j+ncol+2;
72  elmts[i*(ncol-1)+j][3]=i*ncol+j+2;
73  }
74  }
75 }
