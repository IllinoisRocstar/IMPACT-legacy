81 // Interpolate node-centered data x to element-centered data z.
82 // Now support only double precision data.
83 void Rocsurf::interpolate_to_centers( const COM::DataItem *x,
84  COM::DataItem *z) {
85  assert( x->is_nodal() && z->is_elemental()
86  && x->size_of_components() == z->size_of_components());
87  assert( x->data_type() == COM_DOUBLE ||
88  x->data_type() == COM_DOUBLE_PRECISION);
89  assert( z->data_type() == COM_DOUBLE ||
90  z->data_type() == COM_DOUBLE_PRECISION);
91 
92  std::vector<const COM::Pane*> xpanes;
93  std::vector<COM::Pane*> zpanes;
94 
95  x->window()->panes(xpanes);
96  z->window()->panes(zpanes);
97 
98  std::vector<COM::Pane*>::iterator zit, zend;
99  std::vector<const COM::Pane*>::const_iterator xit;
100 
101  const int ndim = x->size_of_components();
102  // Loop through all the panes.
103  for( zit = zpanes.begin(), zend = zpanes.end(), xit = xpanes.begin();
104  zit != zend; ++zit, ++xit) {
105  //Loop for each dimension.
106  for(int i = 0; i < ndim; ++i) {
107  COM::DataItem *z_pa = (*zit)->dataitem(z->id()+((ndim>1)?i+1:0));
108  Real *zval = reinterpret_cast<Real *>(z_pa->pointer());
109  int zstrd=z_pa->stride();
110 
111  const COM::DataItem *x_pa = (*xit)->dataitem(x->id()+((ndim>1)?i+1:0));
112  const Real *xval = reinterpret_cast<const Real *>(x_pa->pointer());
113 
114  Nodal_scalar_const_2d<Real> X(x_pa->stride(), 0);
115  Element_node_enumerator ene( *xit, 1);
116  Field<Nodal_scalar_const_2d<Real> > f( X, xval, ene);
117 
118  for (int j=(*xit)->size_of_elements(); j>0; --j, ene.next(), zval+=zstrd)
119  Generic_element_2( ene.size_of_edges(), ene.size_of_nodes()).
120  interpolate_to_center( f, zval);
121  }
122  }
123 }
