21 // Read in an unstructed triangular mesh in obj format
22 int read_obj( istream &is, vector<double> &coors,
23  vector<int> &elems) {
24  const int MAXLEN=255;
25  char buf[MAXLEN];
26  int face_type = 3;
27 
28  is.getline( buf, MAXLEN);
29  if ( strncmp( buf, "#OBJ", 4)==0 && !is.eof()) {
30  face_type = buf[4] - '0';
31  assert( face_type == 3 || face_type == 4 || face_type == 6);
32  do { is.getline( buf, MAXLEN); } while ( buf[0]=='#' && !is.eof());
33  }
34  int np=0, nf=0;
35  if ( strcmp( buf, "off")==0 || strcmp( buf, "OFF")==0 )
36  do { is.getline( buf, MAXLEN); } while ( buf[0]=='#' && !is.eof());
37  sscanf( buf, "%d %d", &np, &nf);
38 
39  // Allocate space for points and faces.
40  coors.reserve( np*3); elems.reserve( nf*face_type);
41 
42  // Read in the coordinates
43  for ( int i=0; i<np; ++i) {
44  do { is.getline( buf, MAXLEN); } while ( buf[0]=='#' && !is.eof());
45  double p0, p1, p2;
46  sscanf( buf, "%lf %lf %lf", &p0, &p1, &p2);
47  coors.push_back( p0); coors.push_back( p1); coors.push_back( p2);
48  }
49 
50  // Read in faces
51  vector<int> f( face_type);
52  while( is.peek() == '#' && !is.eof()) is.getline( buf, MAXLEN);
53  for ( int i=0; i<nf; ++i) {
54  for ( int j=0; j<face_type; ++j) {
55  is >> f[j]; assert( f[j] >= 1 && f[j] <= np);
56  elems.push_back( f[j]);
57  }
58  }
59 
60  return face_type;
61 }
