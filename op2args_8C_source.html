97 // Performs the operation: y op z
98 template <class FuncType, int ytype>
99 void Rocblas::gen2arg( DataItem *z, void *yin, FuncType opp) {
100  typedef typename FuncType::argument_type argument_type;
101  typedef typename FuncType::result_type result_type;
102 
103  int num_dims = z->size_of_components();
104 
105  std::vector<Pane*> zpanes;
106  std::vector<Pane*>::iterator zit, zend;
107  z->window()->panes(zpanes);
108 
109  std::vector<Pane*> ypanes;
110  Pane **yit=NULL;
111  DataItem *y=NULL;
112  argument_type *yval=NULL;
113 
114  if ( ytype == BLAS_VOID) {
115  yval = reinterpret_cast<argument_type *>( yin);
116 
117  COM_assertion_msg( yval, std::string("Caught NULL pointer in scalar \
118 operand when processing "+z->fullname()).c_str());
119  }
120  else {
121  y = reinterpret_cast<DataItem *>(yin);
122  bool type_coercion = compare_types<argument_type,result_type>();
123 
124  COM_assertion_msg( type_coercion ||
125  COM_compatible_types(z->data_type(), y->data_type()),
126  (std::string("Incompatible data types between ")+
127  z->fullname()+" and "+y->fullname()).c_str());
128 
129  if ( !y->is_windowed()) {
130  y->window()->panes(ypanes);
131  COM_assertion_msg(zpanes.size() == ypanes.size(),
132  (std::string("Numbers of panes do not match between ")+
133  y->window()->name()+" and "+z->window()->name()).c_str());
134  yit = &ypanes[0];
135  }
136  else {
137  if ( y->size_of_items()!=1) {
138  std::cout << "Rocbals Error: The size-of-items of dataitem "
139  << y->fullname() << " on pane " << y->pane()->id()
140  << " is " << y->size_of_items() << std::endl;
141  COM_assertion_msg( y->size_of_items()==1, "If an argument is a window \
142 dataitem, then its number of items must be 1.");
143  }
144 
145  yval = reinterpret_cast<argument_type *>( y->pointer());
146 
147  COM_assertion_msg( yval, (std::string("Caught NULL pointer in ")+
148  y->fullname()).c_str());
149  }
150  }
151 
152  const int ynum_dims = ytype != BLAS_VOID?y->size_of_components():0;
153  COM_assertion_msg( ytype==BLAS_VOID || (ynum_dims==1 || ynum_dims==num_dims),
154  (std::string("Numbers of components do not match between ")+
155  y->fullname()+" and "+z->fullname()).c_str());
156 
157  if ( z->is_windowed()) {
158  COM_assertion_msg( ytype == BLAS_VOID || y->is_windowed(),
159  (std::string("Wrong type of operand")+
160  y->fullname()).c_str());
161  int zs=get_stride<BLAS_VEC2D>(z);
162  int ys=get_stride<ytype>(y);
163  result_type *zval = (result_type *)z->pointer();
164  int length = z->size_of_items();
165 
166  COM_assertion_msg( length==0 || ytype == BLAS_VOID || zval && yval,
167  (std::string("Caught NULL pointer in w-dataitem ")+
168  z->fullname()+" or "+y->fullname()).c_str());
169 
170  if ( (length==1 || num_dims == zs) && ytype != BLAS_VEC) {
171  for ( Size i = 0, s = num_dims*length; i < s; ++i, ++zval)
172  opp( *zval, getref<argument_type,ytype,0>(yval,i,0,1));
173  }
174  else {
175  for(int i = 0; i < num_dims; ++i) {
176  DataItem *z_i = num_dims==1?z:z+i+1;
177  zval = (result_type *)z_i->pointer();
178  zs = get_stride<BLAS_VEC2D>(z_i);
179 
180  if ( ytype != BLAS_VOID) {
181  DataItem *y_i=ynum_dims==1?y:y+i+1;
182  yval = reinterpret_cast<argument_type *>( y_i->pointer());
183  ys = get_stride<ytype>( y_i);
184  }
185 
186  for(int j=0; j<length; ++j, zval+=zs)
187  opp( *zval, getref<argument_type,ytype,1>(yval,j,i,ys));
188  }
189  }
190  return;
191  }
192  // otherwise:
193 
194  for( zit=zpanes.begin(), zend=zpanes.end(); zit!=zend;
195  ++zit, yit+=( ytype!=BLAS_VOID&&yit)) {
196  DataItem *pz = (*zit)->dataitem(z->id());
197  const int length = pz->size_of_items();
198  int zstrd=get_stride<BLAS_VEC2D>(pz);
199  const bool zstg = length>1 && num_dims!=zstrd;
200 
201  DataItem *py = ytype!=BLAS_VOID&&yit?(*yit)->dataitem(y->id()):y;
202  int ystrd = get_stride<ytype>(py);
203  const bool ystg = py && (ynum_dims!=num_dims || ynum_dims!=ystrd);
204  COM_assertion_msg( ytype!=BLAS_SCNE && ytype!=BLAS_VEC2D ||
205  length == int(py->size_of_items()) || ystrd==0,
206  (std::string("Numbers of items do not match between ")+
207  y->fullname()+" and "+z->fullname()+
208  " on pane "+to_str((*zit)->id())).c_str());
209 
210  // Optimized version for contiguous dataitems
211  if ( !zstg && !ystg && ytype != BLAS_VEC &&
212  (ytype != BLAS_SCNE || num_dims==1)) {
213  result_type *zval = reinterpret_cast<result_type *>(pz->pointer());
214 
215  if ( ytype != BLAS_VOID && yit)
216  yval = reinterpret_cast<argument_type *>(py->pointer());
217 
218  COM_assertion_msg( length==0 || ytype == BLAS_VOID || zval && yval,
219  (std::string("Caught NULL pointer in ")+
220  z->fullname()+" or "+y->fullname()+" on pane "+
221  to_str( (*zit)->id())).c_str());
222 
223  if ( zval)
224  //Loop for each element/node and for each dimension
225  for(int i = 0, s = length*num_dims; i < s; ++i, ++zval)
226  opp(*zval, getref<argument_type,ytype,0>(yval,i,0,1));
227  }
228  else { // General version
229  //Loop for each dimension.
230  for(int i = 0; i < num_dims; ++i) {
231  DataItem *pz_i = num_dims==1?pz:(*zit)->dataitem(z->id()+i+1);
232  result_type *zval = (result_type *)pz_i->pointer();
233  zstrd=get_stride<BLAS_VEC2D>(pz_i);
234 
235  if ( ytype != BLAS_VOID && yit) {
236  DataItem *py_i=ynum_dims==1?py:(*yit)->dataitem(y->id()+i+1);
237  yval = reinterpret_cast<argument_type *>( py_i->pointer());
238  ystrd = get_stride<ytype>( py_i);
239  }
240 
241  COM_assertion_msg( length==0 || ytype == BLAS_VOID || zval && yval,
242  (std::string("Caught NULL pointer in ")+
243  z->fullname()+" or "+y->fullname()+" on pane "+
244  to_str( (*zit)->id())).c_str());
245 
246  if ( zval)
247  for(int j=0; j < length; ++j, zval+=zstrd)
248  opp( *zval, getref<argument_type,ytype,1>(yval,j,i,ystrd));
249  }
250  }
251  }
252 }

255 //Wrapper for swap.
256 void Rocblas::swap(DataItem *x, DataItem *y) {
257  switch ( x->id()) {
258  case COM::COM_DATA: {
259  COM_assertion_msg( y->id() == COM::COM_DATA,
260  "Aggregate dataitems for swap must match.");
261 
262  // Obtain all the dataitems of their corresponding windows
263  std::vector< DataItem*> x_attrs;
264  x->window()->dataitems( x_attrs);
265 
266  std::vector< DataItem*> y_attrs;
267  y->window()->dataitems( y_attrs);
268 
269  COM_assertion_msg( x_attrs.size() == y_attrs.size(),
270  (std::string("Numbers of dataitems do not match between ")+
271  x->window()->name()+" and "+y->window()->name()).c_str());
272  // Copy all the individual dataitems
273  for ( int i=x_attrs.size()-1; i>=0; --i)
274  swap( x_attrs[i], y_attrs[i]);
275  return;
276  }
277  case COM::COM_MESH:
278  case COM::COM_PMESH:
279  case COM::COM_CONN:
280  case COM::COM_ALL:
281  COM_assertion_msg(false, "Only the aggregate dataitem atts is supported");
282  default: ;
283  }
284 
285  COM_Type att_type = x->data_type();
286 
287  if(att_type == COM_INT || att_type == COM_INTEGER)
288  gen2arg<swapp<int>,BLAS_VEC2D>(x, y, swapp<int>());
289  else if(att_type == COM_CHAR)
290  gen2arg<swapp<char>,BLAS_VEC2D>(x, y, swapp<char>());
291  else {
292  COM_assertion_msg(att_type == COM_DOUBLE || att_type == COM_DOUBLE_PRECISION,
293  (std::string("Unsupported data type in ")+
294  x->fullname()).c_str());
295  gen2arg<swapp<double>,BLAS_VEC2D>(x, y, swapp<double>());
296  }
297 }

615 void Rocblas::reduce_MPI(const DataItem *x, DataItem *z,

628

630  std::vector<int> t((int*)z->pointer(), ((int*)z->pointer())+zncomp);
631  MPI_Allreduce( &t[0], z->pointer(), zncomp, MPI_INT,
632  convert2mpiop(OPMPI), *comm);
633  }
634  }
635  else {
636  if ( zncomp == 1)
637  gen2arg<OPint,BLAS_SCNE>(const_cast<DataItem*>(x), z, OPint());
638  else

640  }
641  }

643  copy_scalar( &initialI, z);
644 
645  if ( z->is_windowed()) {
646  if ( zncomp == 1)
647  gen2arg<OPint,BLAS_SCALAR>(const_cast<DataItem*>(x), z, OPint());
648  else
649  gen2arg<OPint,BLAS_VEC>(const_cast<DataItem*>(x), z, OPint());
650

655  }
656  }
657  else {
