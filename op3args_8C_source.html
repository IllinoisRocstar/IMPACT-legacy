12 //Function object that implements a limit1 operation.
13 template <class T>
14 struct Rocblas::limit1v : std::binary_function<T,T,T> {
15  T operator()(T x, T y) {
16  if ( x>=0&&y>=0 || x<=0&&y<=0)
17  return std::abs(x)<std::abs(y)?x:y;
18  else
19  return 0;
20  }
21 };

33 template <class FuncType, int ytype>
34 void Rocblas::calc( DataItem *z, const DataItem *x, const void *yin,
35  FuncType opp, bool swap)
36 {
37  typedef typename FuncType::result_type data_type;
38 
39  COM_assertion_msg( !x->is_windowed() && !z->is_windowed(),
40  (std::string("Unsupported data type in ")+
41  x->fullname()).c_str());
42  COM_assertion_msg( COM_compatible_types(x->data_type(),z->data_type()),
43  (std::string("Incompatible data types between ")+
44  x->fullname()+" and "+z->fullname()).c_str());
45 
46  int num_dims = z->size_of_components();
47  COM_assertion_msg( num_dims == x->size_of_components(),
48  (std::string("Numbers of components do not match between ")+
49  x->fullname()+" and "+z->fullname()).c_str());
50 
51  std::vector<Pane*> zpanes;
52  std::vector<const Pane*> xpanes, ypanes;
53 
54  z->window()->panes(zpanes);
55  x->window()->panes(xpanes);
56 
57  COM_assertion_msg( xpanes.size() == zpanes.size(),
58  (std::string("Numbers of panes do not match between ")+
59  x->window()->name()+" and "+z->window()->name()).c_str());
60 
61  std::vector<Pane*>::iterator zit, zend;
62  std::vector<const Pane*>::const_iterator xit;
63  const Pane **yit=NULL;
64 
65  const DataItem *y=NULL;
66  const data_type *yval=NULL;
67 
68  // Obtain yval for BLAS_VOID or window dataitems
69  if ( ytype==BLAS_VOID)
70  yval = reinterpret_cast<const data_type *>( yin);
71  else {
72  y = reinterpret_cast<const DataItem *>(yin);
73 
74  if ( !y->is_windowed()) {
75  y->window()->panes(ypanes);
76  COM_assertion_msg(xpanes.size() == ypanes.size(),
77  (std::string("Numbers of panes do not match between ")+
78  x->window()->name()+" and "+
79  y->window()->name()).c_str());
80  yit = &ypanes[0];
81  }
82  else {
83  COM_assertion_msg( y->size_of_items()==1,
84  (std::string("Numbers of items do not match between ")+
85  x->fullname()+" and "+y->fullname()).c_str());
86 
87  yval = reinterpret_cast<const data_type *>( y->pointer());
88 
89  COM_assertion_msg( yval, (std::string("Caught NULL pointer in ")+
90  y->fullname()).c_str());
91  }
92  }
93 
94  const int ynum_dims = ytype!=BLAS_VOID ? y->size_of_components() : 0;
95  COM_assertion_msg( ytype==BLAS_VOID || (ynum_dims==1 || ynum_dims==num_dims),
96  (std::string("Numbers of components do not match between ")+
97  z->fullname()+" and "+y->fullname()).c_str());
98 
99  for( zit=zpanes.begin(), zend=zpanes.end(), xit=xpanes.begin(); zit!=zend;
100  ++zit, ++xit, yit+=( ytype!=BLAS_VOID && yit!=NULL)) {
101  DataItem *pz = (*zit)->dataitem(z->id());
102  const int length = pz->size_of_items();
103  int zstrd=get_stride<BLAS_VEC2D>(pz);
104  const bool zstg = num_dims!=zstrd;
105 
106  const DataItem *px = (*xit)->dataitem(x->id());
107  int xstrd = get_stride<BLAS_VEC2D>(px);
108  COM_assertion_msg(length == px->size_of_items() || xstrd==0,
109  (std::string("Numbers of items do not match between ")+
110  x->fullname()+" and "+z->fullname()+
111  " on pane "+to_str((*zit)->id())).c_str());
112 
113  const bool xstg = num_dims!=xstrd || xstrd==0;
114 
115  const DataItem *py = (ytype!=BLAS_VOID&&yit)?(*yit)->dataitem(y->id()):y;
116  int ystrd = get_stride<ytype>(py);
117  COM_assertion_msg( ytype!=BLAS_SCNE && ytype!=BLAS_VEC2D ||
118  (length == int(py->size_of_items()) || ystrd==0),
119  (std::string("Numbers of items do not match between ")+
120  y->fullname()+" and "+z->fullname()+
121  " on pane "+to_str((*zit)->id())).c_str());
122 
123  const bool ystg = py && (ynum_dims!=num_dims || ynum_dims!=ystrd);
124 
125  // Optimized version for contiguous dataitems
126  if ( !xstg && !zstg && !ystg && ytype != BLAS_VEC
127  && (ytype != BLAS_SCNE || num_dims==1)) {
128  const data_type *xval = (const data_type *)px->pointer();
129  data_type *zval = (data_type *)pz->pointer();
130 
131  // Get address for y if y is not window dataitem
132  if ( ytype != BLAS_VOID && yit)
133  yval = reinterpret_cast<const data_type *>(py->pointer());
134 
135  //Loop for each element/node and for each dimension
136  if ( swap == false)
137  for(Size i = 0, s = length*num_dims; i<s; ++i, ++zval, ++xval)
138  *zval = opp( *xval, getref<data_type,ytype,0>(yval,i,0,1));
139  else
140  for(Size i = 0, s = length*num_dims; i<s; ++i, ++zval, ++xval)
141  *zval = opp( getref<data_type,ytype,0>(yval,i,0,1), *xval);
142  }
143  else { // General version
144  //Loop for each dimension.
145  for(int i = 0; i < num_dims; ++i) {
146  DataItem *pz_i = num_dims==1?pz:(*zit)->dataitem(z->id()+i+1);
147  data_type *zval = (data_type *)pz_i->pointer();
148  zstrd=get_stride<BLAS_VEC2D>(pz_i);
149 
150  const DataItem *px_i = num_dims==1?px:(*xit)->dataitem(x->id()+i+1);
151  const data_type *xval = (const data_type *)px_i->pointer();
152  xstrd=get_stride<BLAS_VEC2D>(px_i);
153 
154  if ( ytype != BLAS_VOID && yit) {
155  const DataItem *py_i=ynum_dims==1?py:(*yit)->dataitem(y->id()+i+1);
156  yval = reinterpret_cast<const data_type *>( py_i->pointer());
157  ystrd = get_stride<ytype>( py_i);
158  }
159 
160  // Loop for each element/node.
161  if ( swap == false) {
162  for(int j = 0; j < length; ++j, zval+=zstrd, xval+=xstrd)
163  *zval = opp( *xval, getref<data_type,ytype,1>(yval,j,i,ystrd));
164  }
165  else {
166  for(int j = 0; j < length; ++j, zval+=zstrd, xval+=xstrd)
167  *zval = opp( getref<data_type,ytype,1>(yval,j,i,ystrd), *xval);
168  }
169  } // end for i
170  } // end if
171  } // end for
172 }

175 //Chooses which calc function to call based on type of y.
176 template <class FuncType>
177 void Rocblas::calcChoose(const DataItem *x, const DataItem *y, DataItem *z,
178  FuncType opp)
179 {
180  typedef typename FuncType::result_type data_type;
181 
182  if (x->is_windowed()) {
183  if ( x->size_of_components() == 1)
184  calc<FuncType,BLAS_SCALAR>(z, y, x, opp, true);
185  else
186  calc<FuncType,BLAS_VEC>(z, y, x, opp, true);
187  }
188  else if (y->is_windowed()) {
189  if ( y->size_of_components() == 1)
190  calc<FuncType,BLAS_SCALAR>(z, x, y, opp, false);
191  else
192  calc<FuncType,BLAS_VEC>(z, x, y, opp, false);
193  }
194  else if ( x->size_of_components()<y->size_of_components()) {
195  calc<FuncType,BLAS_SCNE>(z, y, x, opp, true);
196  }
197  else {
198  if ( y->size_of_components() == 1)
199  calc<FuncType,BLAS_SCNE>(z, x, y, opp, false);
200  else
201  calc<FuncType,BLAS_VEC2D>(z, x, y, opp, false);
202  }
203 }
