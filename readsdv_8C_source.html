31 // A sample program for reading an output file of Rocface for a single pane.
32 // Arguments: fname: the file name of the CGNS file
33 // prefix: is the window name (also the base material name) in CGNS files
34 // pid: the pane ID of the window.
35 // sd: the output structure contained in an Subdiv object.
36 void read_pane_sp( std::string &fname,
37  const std::string &prefix,
38  int pid,
39  Subdiv &sd) {
40 
41  // Load Rocin into Roccom
42  SimIN_load_module( "SDV_IN");
43 
44  // Obtain function handles to Rocin functions
45  int hdl_read = COM_get_function_handle( "SDV_IN.read_windows");
46  int hdl_obtain = COM_get_function_handle( "SDV_IN.obtain_dataitem");
47 
48  // Define the base-window and sdv-window names
49  std::string bufprefix = "__BUF";
50  std::string sdv_material = prefix+"_sdv";
51  std::string sdv_wname = bufprefix+sdv_material;
52 
53  // Read the pane from the given file. Note that the file contains both
54  // the parent and the subdivided windows. Read only the subdivided one.
55  MPI_Comm comm_self = MPI_COMM_SELF;
56  COM_call_function( hdl_read, fname.c_str(), bufprefix.c_str(),
57  sdv_material.c_str(), &comm_self);
58  int hdl_all = COM_get_dataitem_handle( (sdv_wname+".all").c_str());
59  COM_call_function( hdl_obtain, &hdl_all, &hdl_all, &pid);
60 
61  // Obtain number of sub-nodes, sub-faces, nodes, and faces
62  int nsubn, nsubf, nn, nf;
63  COM_get_size( (sdv_wname+".sn_parent_fcID").c_str(), pid, &nsubn);
64  COM_get_size( (sdv_wname+".:t3").c_str(), pid, &nsubf);
65 
66  COM_get_size( (sdv_wname+".sn_subID").c_str(), pid, &nn);
67  COM_get_size( (sdv_wname+".sf_offset").c_str(), pid, &nf);
68 
69  // Obtain the connectivity
70  sd.subfaces.resize( 3*nsubf);
71  COM_copy_array( (sdv_wname+".:t3").c_str(), pid, &sd.subfaces[0], 3);
72  // NOTE: The last argument (3) indicates that three sub-node IDs are stored
73  // consecutively for each sub-face. Use the number one (1) if the first
74  // sub-nodes of all sub-faces are stored together followed by the second
75  // sub-nodes and then third.
76 
77 
78  // Obtain subnode_parents, subnode_ncs, and subnode_counterparts
79  sd.subnode_parents.resize( nsubn);
80  COM_copy_array( (sdv_wname+".sn_parent_fcID").c_str(),
81  pid, &sd.subnode_parents[0]);
82 
83  sd.subnode_ncs.resize( 2*nsubn);
84  COM_copy_array( (sdv_wname+".sn_parent_ncs").c_str(), pid, &sd.subnode_ncs[0], 2);
85  // NOTE: The last argument (2) indicates that the two local coordinates are
86  // stored consecutively. Use the number one (1) if the first local parameter
87  // (xi) of all sub-nodes are stored together followed by the second one (eta).
88 
89  sd.subnode_subIDs.resize( nn);
90  COM_copy_array( (sdv_wname+".sn_subID").c_str(), pid, &sd.subnode_subIDs[0]);
91 
92  sd.subnode_counterparts.resize( nsubn);
93  COM_copy_array( (sdv_wname+".sn_cntpt_ndID").c_str(),
94  pid, &sd.subnode_counterparts[0]);
95 
96  sd.subface_parents.resize( nsubf);
97  COM_copy_array( (sdv_wname+".sf_parent").c_str(), pid, &sd.subface_parents[0]);
98 
99  sd.subface_nat_coors.resize( nsubf*6);
100  COM_copy_array( (sdv_wname+".sf_ncs").c_str(), pid,
101  &sd.subface_nat_coors[0], 6);
102  // NOTE: The last argument (6) indicates that the local coordinates are
103  // stored consecutively (xi1, eta1, xi2, eta2, xi3, eta3). Use the number
104  // one (1) if the xi1 for all nodes are stored together and then xi2, etc..
105 
106  // Obtain subface_offsets
107  sd.subface_offsets.resize( nf);
108  COM_copy_array( (sdv_wname+".sf_offset").c_str(), pid, &sd.subface_offsets[0]);
109 
110  sd.subface_counterparts.resize( nsubf);
111  COM_copy_array( (sdv_wname+".sf_cntpt_fcID").c_str(), pid,
112  &sd.subface_counterparts[0]);
113 
114  // Delete the window created by Rocin.
115  COM_delete_window( sdv_wname.c_str());
116 
117  // Unload Rocin from Roccom.
118  SimIN_unload_module( "SDV_IN");
119 }
