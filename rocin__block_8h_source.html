38 struct GridInfo_HDF4 {
40  inline GridInfo_HDF4(int32* size, int ng)
41  : m_name(":st0:"),
42  m_numElements(size[2] - 1
43  * std::max(int32(1), (size[1] - 1))
44  * std::max(int32(1), (size[0] - 1))),

114 struct GridInfo_CGNS {
116  inline GridInfo_CGNS(int* size, int cellDim, int ng)
117  : m_name(":st0:"), m_numElements(size[cellDim]),
118  m_numGhostElements(ng)
119  { m_name[3] += cellDim; m_size[0] = size[0]; m_size[1] = m_size[2] = 1;
120  if (cellDim > 1) { m_size[1] = size[1]; m_numElements *= size[cellDim+1];
