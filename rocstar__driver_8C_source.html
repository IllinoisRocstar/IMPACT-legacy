72 Coupling *create_coupling( Control_parameters &param, const RocmanControl_parameters &rocman_param) {
73  const std::string &name = param.coupling_scheme;
74 
75  if ( name == "FluidAlone")
76  return new FluidAlone( param.fluid_module, param.communicator, &param, &rocman_param);
77  else if ( name == "FluidBurnAlone")
78  return new FluidBurnAlone( param.fluid_module, param.burn_module, param.communicator, &param, &rocman_param);
79  else if ( name == "SolidAlone")
80  return new SolidAlone( param.solid_module, param.communicator, &param, &rocman_param);
81  else if ( name == "SolidBurn")
82  return new SolidBurn( param.solid_module, param.burn_module, param.communicator, &param, &rocman_param);
83  else if ( name == "SolidFluidSPC")
84  return new SolidFluidSPC( param.fluid_module, param.solid_module, param.communicator, &param, &rocman_param);
85  else if ( name == "FluidSolidISS")
86  return new FluidSolidISS( param.fluid_module, param.solid_module, param.communicator, &param, &rocman_param);
87  else if ( name == "SolidFluidBurnSPC")
88  return new SolidFluidBurnSPC( param.fluid_module, param.solid_module, param.burn_module, param.communicator, &param, &rocman_param);
89  else if ( name == "SolidFluidBurnEnergySPC")
90  return new SolidFluidBurnEnergySPC( param.fluid_module, param.solid_module, param.burn_module, param.communicator, &param, &rocman_param);
91  else {
92  std::cout << "Rocman> ERROR: Unknown coupling scheme: " << name << std::endl;
93  std::cout << std::endl;
94  std::cout << "Rocman> Supported coupling schemes: " << std::endl;
95  std::cout << " FluidAlone: Fluid alone with no burn" << std::endl;
96  std::cout << " FluidBurnAlone: Fluid alone with burn" << std::endl;
97  std::cout << " SolidAlone: Solid alone with no burn" << std::endl;
98  std::cout << " SolidFluidBurnSPC: FullyCoupled simple staggered scheme with P-C" << std::endl;
99  std::cout << " SolidFluidSPC: FullyCoupled no burn" << std::endl;
100  std::cout << " FluidSolidISS: FullyCoupled no burn with improved staggered scheme" << std::endl;
101  std::cout << " SolidFluidBurnEnergySPC: FullyCoupled with burn energy" << std::endl;
102  COM_assertion_msg(0, "ERROR: Unknown coupling scheme!");
103  return NULL;
104  }
105 }

433 bool reached_simulation_time( const Control_parameters &param) {
434  // if exceed max num of steps
435  if (param.cur_step>=param.maxNumTimeSteps) {
436  MAN_DEBUG(1, ("\n\nRocman: reached_simulation_time returns TRUE with cur_step: %d \n\n", param.cur_step));
437  return true;
438  }
439 
440  // if exceed max wall time
441  double startTimeLoopTime;
442  if (param.myRank == 0) startTimeLoopTime = MPI_Wtime();
443  MPI_Bcast( &startTimeLoopTime, 1, MPI_DOUBLE, 0, param.communicator);
444  const double elaped_time = startTimeLoopTime - param.startTime;
445  if (elaped_time >= param.maxwalltime ) {
446  if (param.myRank == 0)
447  printf("ROCSTAR: Quitting; elapsed wall clock time = %f\n", elaped_time);
448  return true;
449  }
450  else {
451  if (param.myRank == 0) {
452  MAN_DEBUG(1, ("\n\nRocman: reached_simulation_time: elapsed wall time = %f \n\n", elaped_time));
453  }
454  }
455 
456  // if exceed MaximumTime the simulation time
457  if (param.current_time >= param.simue_time) { // MaximumTime
458  if (param.myRank == 0)
459  printf("ROCSTAR: Quitting; CurrentTime = %e, MaximumTime = %e\n", param.current_time, param.simue_time);
460  return true;
461  }
462 
463  // if exceed the Maximum Number of dumps
464  if( (param.maxNumDumps > 0 ) && (param.current_dump >= param.maxNumDumps)){
465  if(param.myRank == 0)
466  printf("ROCSTAR: Quitting; Reached the maximum number of dumps for this run. (%d)\n",param.current_dump);
467  return true;
468  }
469 
470  return false;
471 }

515 void rocstar_driver( int verb, int remeshed) {
516  // Read in config files to initialize param
517  Control_parameters *param_ptr = new Control_parameters;
518  Control_parameters &param = *param_ptr;
519  param.read();
520 
521  int comm_rank = param.myRank;
522 
523  // Read in rocman config files to initialize param
524  RocmanControl_parameters *rocman_param_ptr = new RocmanControl_parameters;
525  RocmanControl_parameters &rocman_param = *rocman_param_ptr;
526  rocman_param.read( param.communicator, comm_rank);
527  rocman_param.remeshed = remeshed;
528 
529  if ( comm_rank == 0 ) {
530  printf(" Rocstar version %s-%04d\n", VERSION,BUILD_NUM);
531  printf(" Rocstar last auto-built was at: %s\n", CI_DATE);
532 
533  printf("\n");
534  printf("ROCSTAR: CALL CouplingInitialize\n");
535 
536  param.print();
537  rocman_param.print();
538 
539  COM_set_verbose( verb); // shift by 2 to depress first level printing
540  }
541 
542  MPI_Barrier(MPI_COMM_WORLD);
543  // if(!comm_rank)
544  // std::cout << "Rocstar: creating coupling." << std::endl;
545 
546  // Construct coupling object and schedule the actions
547  Coupling *coup = create_coupling( param, rocman_param);
548  coup->schedule();
549 
550  // std::cout << "Rocstar(" << comm_rank << "): coupling created." << std::endl;
551  MPI_Barrier(MPI_COMM_WORLD);
552  // if(!comm_rank)
553  // std::cout << "Rocstar: All couplings created." << std::endl;
554  // visualization
555  // std::string fname = std::string(param.coupling_scheme) + ".gdl";
556  // coup->print(fname.c_str());
557 
558  // Load other service modules
559  RocBlas::init();
560  MPI_Barrier(MPI_COMM_WORLD);
561  // if(!comm_rank)
562  // std::cout << "Rocstar: Simpal initd." << std::endl;
563 
564  // Have Roccom profile all the calls to COM_call_function.
565  COM_set_profiling( 1);
566  MPI_Barrier(MPI_COMM_WORLD);
567  // if(!comm_rank)
568  // std::cout << "Rocstar: profiling set." << std::endl;
569 
570  int with_pciter = (param.maxNumPredCorrCycles > 1);
571  if ( comm_rank == 0) {
572  std::cout << "ROCSTAR: The maximum number of PC-iterations is " << param.maxNumPredCorrCycles << std::endl;
573  }
574 
575  // in case of restarting, reset cur_step and current_time from Restart.txt
