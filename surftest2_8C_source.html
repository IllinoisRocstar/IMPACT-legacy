22 void init_structured_mesh( double coors[][3], int nrow, int ncol,
23  int rank, int nproc, int check=1) {
24  // consider the processors as a 2*(nproc/2) grid
25  int proc_col=nproc;
26  if (nproc%2 ==0) {
27  proc_col=nproc/2;
28  }
29  else {
30  proc_col=nproc;
31  }
32  int row=rank/proc_col, col=rank%proc_col;
33 
34  const double width=300./(nrow-1), length=300./(ncol-1);
35 
36  double x0 = col*300, y0=row*300, y1=y0;
37  for (int i=0; i<nrow; ++i) {
38  double x1 = x0;
39  for (int j=0; j<ncol; ++j) {
40  if ( check<0)
41  assert( coors[i*ncol+j][0]==x1 && coors[i*ncol+j][1]==y1 &&
42  coors[i*ncol+j][2]==0.5);
43  else {
44  coors[i*ncol+j][0]=x1*check;
45  coors[i*ncol+j][1]=y1*check;
46  coors[i*ncol+j][2]=0.5;
47  }
48  x1 += length;
49  }
50  y1 += width;
51  }
52 }

54 void init_unstructure_mesh( double coors[][3], int elmts[][4], int nrow,
55  int ncol, int rank, int nproc, int check=1) {
56  // consider the processors as a 2*(nproc/2) grid
57  int proc_col=nproc;
58  if (nproc%2 ==0) {
59  proc_col=nproc/2;
60  }
61  else {
62  proc_col=nproc;
63  }
64 
65  int row=rank/proc_col, col=rank%proc_col;
66 
67  const double width=300./(nrow-1), length=300./(ncol-1);
68 
69  double x0 = col*300, y0=row*300, y1=y0;
70  for (int i=0; i<nrow; ++i) {
71  double x1 = x0;
72  for (int j=0; j<ncol; ++j) {
73  if ( check<0)
74  assert( coors[i*ncol+j][0]==x1 && coors[i*ncol+j][1]==y1 &&
75  coors[i*ncol+j][2]==0.0);
76  else {
77  coors[i*ncol+j][0]=x1*check;
78  coors[i*ncol+j][1]=y1*check;
79  coors[i*ncol+j][2]=0.0;
80  }
81  x1 += length;
82  }
83  y1 += width;
84  }
85 
86  // computes elmts
87  for (int i=0; i<nrow-1; ++i) {
88  for (int j=0; j<ncol-1; ++j) {
89  if ( check<0) {
90  assert( elmts[i*(ncol-1)+j][0]==i*ncol+j+1 &&
91  elmts[i*(ncol-1)+j][1]==i*ncol+j+ncol+1 &&
92  elmts[i*(ncol-1)+j][2]==i*ncol+j+ncol+2 &&
93  elmts[i*(ncol-1)+j][3]==i*ncol+j+2);
94  }
95  else {
96  elmts[i*(ncol-1)+j][0]=check*i*ncol+j+1;
97  elmts[i*(ncol-1)+j][1]=check*i*ncol+j+ncol+1;
98  elmts[i*(ncol-1)+j][2]=check*i*ncol+j+ncol+2;
99  elmts[i*(ncol-1)+j][3]=check*i*ncol+j+2;
100  }
101  }
102  }
103 }
